49  /**
50   * Displays text in panes covering mostly-static information. Uptime is
51   * refreshed every second.
52   */
53  public class OsHwTextPanel extends OshiJPanel { // NOSONAR squid:S110
54  
55      private static final long serialVersionUID = 1L;
56  
57      private static final String OPERATING_SYSTEM = "Operating System";
58      private static final String HARDWARE_INFORMATION = "Hardware Information";
59      private static final String PROCESSOR = "Processor";
60      private static final String DISPLAYS = "Displays";
61      private String osPrefix;
62  
63      public OsHwTextPanel(SystemInfo si) {
64          super();
65          init(si);
66      }
67  
68      private void init(SystemInfo si) {
69          osPrefix = getOsPrefix(si);
70  
71          GridBagConstraints osLabel = new GridBagConstraints();
72          GridBagConstraints osConstraints = new GridBagConstraints();
73          osConstraints.gridy = 1;
74          osConstraints.fill = GridBagConstraints.BOTH;
75          osConstraints.insets = new Insets(0, 0, 15, 15); // T,L,B,R
76  
77          GridBagConstraints procLabel = (GridBagConstraints) osLabel.clone();
78          procLabel.gridy = 2;
79          GridBagConstraints procConstraints = (GridBagConstraints) osConstraints.clone();
80          procConstraints.gridy = 3;
81  
82          GridBagConstraints displayLabel = (GridBagConstraints) procLabel.clone();
83          displayLabel.gridy = 4;
84          GridBagConstraints displayConstraints = (GridBagConstraints) osConstraints.clone();
85          displayConstraints.gridy = 5;
86          displayConstraints.insets = new Insets(0, 0, 0, 15); // T,L,B,R
87  
88          GridBagConstraints csLabel = (GridBagConstraints) osLabel.clone();
89          csLabel.gridx = 1;
90          GridBagConstraints csConstraints = new GridBagConstraints();
91          csConstraints.gridx = 1;
92          csConstraints.gridheight = 6;
93          csConstraints.fill = GridBagConstraints.BOTH;
94  
95          JPanel oshwPanel = new JPanel();
96          oshwPanel.setLayout(new GridBagLayout());
97  
98          JTextArea osArea = new JTextArea(0, 0);
99          osArea.setText(updateOsData(si));
100         oshwPanel.add(new JLabel(OPERATING_SYSTEM), osLabel);
101         oshwPanel.add(osArea, osConstraints);
102 
103         JTextArea procArea = new JTextArea(0, 0);
104         procArea.setText(getProc(si));
105         oshwPanel.add(new JLabel(PROCESSOR), procLabel);
106         oshwPanel.add(procArea, procConstraints);
107 
108         JTextArea displayArea = new JTextArea(0, 0);
109         displayArea.setText(getDisplay(si));
110         oshwPanel.add(new JLabel(DISPLAYS), displayLabel);
111         oshwPanel.add(displayArea, displayConstraints);
112 
113         JTextArea csArea = new JTextArea(0, 0);
114         csArea.setText(getHw(si));
115         oshwPanel.add(new JLabel(HARDWARE_INFORMATION), csLabel);
116         oshwPanel.add(csArea, csConstraints);
117 
118         add(oshwPanel, BorderLayout.CENTER);
119 
120         // Update up time every second
121         Timer timer = new Timer(Config.REFRESH_FAST, e -> osArea.setText(updateOsData(si)));
122         timer.start();
123     }
124 
125     private static String getOsPrefix(SystemInfo si) {
126         StringBuilder sb = new StringBuilder(OPERATING_SYSTEM);
127 
128         OperatingSystem os = si.getOperatingSystem();
129         sb.append(String.valueOf(os));
130         sb.append("\n\n").append("Booted: ").append(Instant.ofEpochSecond(os.getSystemBootTime())).append('\n')
131                 .append("Uptime: ");
132         return sb.toString();
133     }
134 
135     private static String getHw(SystemInfo si) {
136         StringBuilder sb = new StringBuilder();
137         ObjectMapper mapper = new ObjectMapper();
138         ComputerSystem computerSystem = si.getHardware().getComputerSystem();
139         try {
140             sb.append(mapper.writerWithDefaultPrettyPrinter().writeValueAsString(computerSystem));
141         } catch (JsonProcessingException e) {
142             sb.append(e.getMessage());
143         }
144         return sb.toString();
145     }
146 
147     private static String getProc(SystemInfo si) {
148         StringBuilder sb = new StringBuilder();
149         CentralProcessor proc = si.getHardware().getProcessor();
150         sb.append(proc.toString());
151 
152         return sb.toString();
153     }
154 
155     private static String getDisplay(SystemInfo si) {
156         StringBuilder sb = new StringBuilder();
157         List<Display> displays = si.getHardware().getDisplays();
158         if (displays.isEmpty()) {
159             sb.append("None detected.");
160         } else {
161             int i = 0;
162             for (Display display : displays) {
163                 byte[] edid = display.getEdid();
164                 byte[][] desc = EdidUtil.getDescriptors(edid);
165                 String name = "Display " + i;
166                 for (byte[] b : desc) {
167                     if (EdidUtil.getDescriptorType(b) == 0xfc) {
168                         name = EdidUtil.getDescriptorText(b);
169                     }
170                 }
171                 if (i++ > 0) {
172                     sb.append('\n');
173                 }
174                 sb.append(name).append(": ");
175                 int hSize = EdidUtil.getHcm(edid);
176                 int vSize = EdidUtil.getVcm(edid);
177                 sb.append(String.format("%d x %d cm (%.1f x %.1f in)", hSize, vSize, hSize / 2.54, vSize / 2.54));
178             }
179         }
180         return sb.toString();
181     }
182 
183     private String updateOsData(SystemInfo si) {
184         return osPrefix + FormatUtil.formatElapsedSecs(si.getOperatingSystem().getSystemUptime());
185     }
