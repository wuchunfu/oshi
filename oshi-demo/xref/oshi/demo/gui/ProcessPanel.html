52  /**
53   * Displays a process list, such as ps or task manager. This performs more like
54   * Windows Task Manger with current CPU as measured between polling intervals,
55   * while PS uses a cumulative CPU value.
56   */
57  public class ProcessPanel extends OshiJPanel { // NOSONAR squid:S110

59      private static final long serialVersionUID = 1L;

61      private static final String PROCESSES = "Processes";
62      private static final String[] COLUMNS = { "PID", "PPID", "Threads", "% CPU", "Cumulative", "VSZ", "RSS", "% Memory",
63              "Process Name" };
64      private static final double[] COLUMN_WIDTH_PERCENT = { 0.07, 0.07, 0.07, 0.07, 0.09, 0.1, 0.1, 0.08, 0.35 };

66      private transient Map<Integer, OSProcess> priorSnapshotMap = new HashMap<>();

68      private transient ButtonGroup cpuOption = new ButtonGroup();
69      private transient JRadioButton perProc = new JRadioButton("of one Processor");
70      private transient JRadioButton perSystem = new JRadioButton("of System");

72      private transient ButtonGroup sortOption = new ButtonGroup();
73      private transient JRadioButton cpuButton = new JRadioButton("CPU %");
74      private transient JRadioButton cumulativeCpuButton = new JRadioButton("Cumulative CPU");
75      private transient JRadioButton memButton = new JRadioButton("Memory %");

77      public ProcessPanel(SystemInfo si) {
78          super();
79          init(si);
80      }

82      private void init(SystemInfo si) {
83          OperatingSystem os = si.getOperatingSystem();
84          JLabel procLabel = new JLabel(PROCESSES);
85          add(procLabel, BorderLayout.NORTH);
86  
87          JPanel settings = new JPanel();
88  
89          JLabel cpuChoice = new JLabel("          CPU %:");
90          settings.add(cpuChoice);
91          cpuOption.add(perProc);
92          settings.add(perProc);
93          cpuOption.add(perSystem);
94          settings.add(perSystem);
95          if (SystemInfo.getCurrentPlatform().equals(PlatformEnum.WINDOWS)) {
96              perSystem.setSelected(true);
97          } else {
98              perProc.setSelected(true);
99          }
100 
101         JLabel sortChoice = new JLabel("          Sort by:");
102         settings.add(sortChoice);
103         sortOption.add(cpuButton);
104         settings.add(cpuButton);
105         sortOption.add(cumulativeCpuButton);
106         settings.add(cumulativeCpuButton);
107         sortOption.add(memButton);
108         settings.add(memButton);
109         cpuButton.setSelected(true);
110 
111         TableModel model = new DefaultTableModel(parseProcesses(os.getProcesses(null, null, 0), si), COLUMNS);
112         JTable procTable = new JTable(model);
113         JScrollPane scrollV = new JScrollPane(procTable);
114         scrollV.setVerticalScrollBarPolicy(ScrollPaneConstants.VERTICAL_SCROLLBAR_AS_NEEDED);
115         resizeColumns(procTable.getColumnModel());
116 
117         add(scrollV, BorderLayout.CENTER);
118         add(settings, BorderLayout.SOUTH);
119 
120         Timer timer = new Timer(Config.REFRESH_SLOW, e -> {
121             DefaultTableModel tableModel = (DefaultTableModel) procTable.getModel();
122             Object[][] newData = parseProcesses(os.getProcesses(null, null, 0), si);
123             int rowCount = tableModel.getRowCount();
124             for (int row = 0; row < newData.length; row++) {
125                 if (row < rowCount) {
126                     // Overwrite row
127                     for (int col = 0; col < newData[row].length; col++) {
128                         tableModel.setValueAt(newData[row][col], row, col);
129                     }
130                 } else {
131                     // Add row
132                     tableModel.addRow(newData[row]);
133                 }
134             }
135             // Delete any extra rows
136             for (int row = rowCount - 1; row >= newData.length; row--) {
137                 tableModel.removeRow(row);
138             }
139         });
140         timer.start();
141     }

143     private Object[][] parseProcesses(List<OSProcess> list, SystemInfo si) {
144         long totalMem = si.getHardware().getMemory().getTotal();
145         int cpuCount = si.getHardware().getProcessor().getLogicalProcessorCount();
146         // Build a map with a value for each process to control the sort
147         Map<OSProcess, Double> processSortValueMap = new HashMap<>();
148         for (OSProcess p : list) {
149             int pid = p.getProcessID();
150             // Ignore the Idle process on Windows
151             if (pid > 0 || !SystemInfo.getCurrentPlatform().equals(PlatformEnum.WINDOWS)) {
152                 // Set up for appropriate sort
153                 if (cpuButton.isSelected()) {
154                     processSortValueMap.put(p, p.getProcessCpuLoadBetweenTicks(priorSnapshotMap.get(pid)));
155                 } else if (cumulativeCpuButton.isSelected()) {
156                     processSortValueMap.put(p, p.getProcessCpuLoadCumulative());
157                 } else {
158                     processSortValueMap.put(p, (double) p.getResidentSetSize());
159                 }
160             }
161         }
162         // Now sort the list by the values
163         List<Entry<OSProcess, Double>> procList = new ArrayList<>(processSortValueMap.entrySet());
164         procList.sort(Entry.comparingByValue());
165         // Insert into array in reverse order (lowest sort value last)
166         int i = procList.size();
167         Object[][] procArr = new Object[i][COLUMNS.length];
168         // These are in descending CPU order
169         for (Entry<OSProcess, Double> e : procList) {
170             OSProcess p = e.getKey();
171             // Matches order of COLUMNS field
172             i--;
173             int pid = p.getProcessID();
174             procArr[i][0] = pid;
175             procArr[i][1] = p.getParentProcessID();
176             procArr[i][2] = p.getThreadCount();
177             if (perProc.isSelected()) {
178                 procArr[i][3] = String.format("%.1f",
179                         100d * p.getProcessCpuLoadBetweenTicks(priorSnapshotMap.get(pid)) / cpuCount);
180                 procArr[i][4] = String.format("%.1f", 100d * p.getProcessCpuLoadCumulative() / cpuCount);
181             } else {
182                 procArr[i][3] = String.format("%.1f",
183                         100d * p.getProcessCpuLoadBetweenTicks(priorSnapshotMap.get(pid)));
184                 procArr[i][4] = String.format("%.1f", 100d * p.getProcessCpuLoadCumulative());
185             }
186             procArr[i][5] = FormatUtil.formatBytes(p.getVirtualSize());
187             procArr[i][6] = FormatUtil.formatBytes(p.getResidentSetSize());
188             procArr[i][7] = String.format("%.1f", 100d * p.getResidentSetSize() / totalMem);
189             procArr[i][8] = p.getName();
190         }
191         // Re-populate snapshot map
192         priorSnapshotMap.clear();
193         for (OSProcess p : list) {
194             priorSnapshotMap.put(p.getProcessID(), p);
195         }
196         return procArr;
197     }

199     private static void resizeColumns(TableColumnModel tableColumnModel) {
200         TableColumn column;
201         int tW = tableColumnModel.getTotalColumnWidth();
202         int cantCols = tableColumnModel.getColumnCount();
203         for (int i = 0; i < cantCols; i++) {
204             column = tableColumnModel.getColumn(i);
205             int pWidth = (int) Math.round(COLUMN_WIDTH_PERCENT[i] * tW);
206             column.setPreferredWidth(pWidth);
207         }
208     }
