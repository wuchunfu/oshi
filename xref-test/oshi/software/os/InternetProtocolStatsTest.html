49  @TestInstance(Lifecycle.PER_CLASS)
50  class InternetProtocolStatsTest {
51  
52      private InternetProtocolStats ipStats = null;
53  
54      @BeforeAll
55      void setUp() {
56          SystemInfo si = new SystemInfo();
57          this.ipStats = si.getOperatingSystem().getInternetProtocolStats();
58      }

60      @Test
61      void testTCPStats() {
62          TcpStats tcp4 = ipStats.getTCPv4Stats();
63          assertThat("IPV4 TCP connections should be 0 or higher", tcp4.getConnectionsEstablished(),
64                  is(greaterThanOrEqualTo(0L)));
65          assertThat("IPV4 TCP active connections should be 0 or higher", tcp4.getConnectionsActive(),
66                  is(greaterThanOrEqualTo(0L)));
67          assertThat("IPV4 TCP passive connections should be 0 or higher", tcp4.getConnectionsPassive(),
68                  is(greaterThanOrEqualTo(0L)));
69          assertThat("IPV4 TCP connection failures should be 0 or higher", tcp4.getConnectionFailures(),
70                  is(greaterThanOrEqualTo(0L)));
71          assertThat("IPV4 TCP connections reset should be 0 or higher", tcp4.getConnectionsReset(),
72                  is(greaterThanOrEqualTo(0L)));
73          assertThat("IPV4 TCP segments received should be 0 or higher", tcp4.getSegmentsReceived(),
74                  is(greaterThanOrEqualTo(0L)));
75          assertThat("IPV4 TCP segments sent should be 0 or higher", tcp4.getSegmentsSent(),
76                  is(greaterThanOrEqualTo(0L)));
77          assertThat("IPV4 TCP segments retransmitted should be 0 or higher", tcp4.getSegmentsRetransmitted(),
78                  is(greaterThanOrEqualTo(0L)));
79          assertThat("IPV4 TCP errors should be or higher ", tcp4.getInErrors(), is(greaterThanOrEqualTo(0L)));
80          assertThat("IPV4 TCP segments transmitted with the reset flag should be 0 or higher", tcp4.getOutResets(),
81                  is(greaterThanOrEqualTo(0L)));
82  
83          TcpStats tcp6 = ipStats.getTCPv6Stats();
84          assertThat("IPV6 TCP connections should be 0 or higher", tcp6.getConnectionsEstablished(),
85                  is(greaterThanOrEqualTo(0L)));
86          assertThat("IPV6 TCP active connections should be 0 or higher", tcp6.getConnectionsActive(),
87                  is(greaterThanOrEqualTo(0L)));
88          assertThat("IPV6 TCP passive connections should be 0 or higher", tcp6.getConnectionsPassive(),
89                  is(greaterThanOrEqualTo(0L)));
90          assertThat("IPV6 TCP connection failures should be 0 or higher", tcp6.getConnectionFailures(),
91                  is(greaterThanOrEqualTo(0L)));
92          assertThat("IPV6 TCP connections reset should be 0 or higher", tcp6.getConnectionsReset(),
93                  is(greaterThanOrEqualTo(0L)));
94          assertThat("IPV6 TCP segments received should be 0 or higher", tcp6.getSegmentsReceived(),
95                  is(greaterThanOrEqualTo(0L)));
96          assertThat("IPV6 TCP segments sent should be 0 or higher", tcp6.getSegmentsSent(),
97                  is(greaterThanOrEqualTo(0L)));
98          assertThat("IPV6 TCP segments retransmitted should be 0 or higher", tcp6.getSegmentsRetransmitted(),
99                  is(greaterThanOrEqualTo(0L)));
100         assertThat("IPV6 TCP errors should be or higher ", tcp6.getInErrors(), is(greaterThanOrEqualTo(0L)));
101         assertThat("IPV6 TCP segments transmitted with the reset flag should be 0 or higher", tcp6.getOutResets(),
102                 is(greaterThanOrEqualTo(0L)));
103 
104         assertThat("IPV4 TCP stats shouldn't be null", tcp4.toString(), is(notNullValue()));
105         assertThat("IPV6 TCP stats shouldn't be null", tcp6.toString(), is(notNullValue()));
106     }
107 
108     @Test
109     void testUDPStats() {
110         UdpStats udp4 = ipStats.getUDPv4Stats();
111         assertThat("IPV4 UDP datagrams delivered to UDP users should be 0 or higher", udp4.getDatagramsReceived(),
112                 is(greaterThanOrEqualTo(0L)));
113         assertThat("IPV4 UDP datagrams sent should be 0 or higher", udp4.getDatagramsSent(),
114                 is(greaterThanOrEqualTo(0L)));
115         assertThat("IPV4 UDP datagrams received with no application at the destination port should be 0 or higher",
116                 udp4.getDatagramsNoPort(), is(greaterThanOrEqualTo(0L)));
117         assertThat(
118                 "IPV4 UDP datagrams received that could not be delivered for reasons other than lack of application at the destination port should be 0 or higher",
119                 udp4.getDatagramsReceivedErrors(), is(greaterThanOrEqualTo(0L)));
120 
121         UdpStats udp6 = ipStats.getUDPv6Stats();
122         assertThat("IPV6 UDP datagrams delivered to UDP users should be 0 or higher", udp6.getDatagramsReceived(),
123                 is(greaterThanOrEqualTo(0L)));
124         assertThat("IPV6 UDP datagrams sent should be 0 or higher", udp6.getDatagramsSent(),
125                 is(greaterThanOrEqualTo(0L)));
126         assertThat("IPV6 UDP datagrams received with no application at the destination port should be 0 or higher",
127                 udp6.getDatagramsNoPort(), is(greaterThanOrEqualTo(0L)));
128         assertThat(
129                 "IPV6 UDP datagrams received that could not be delivered for reasons other than lack of application at the destination port should be 0 or higher",
130                 udp6.getDatagramsReceivedErrors(), is(greaterThanOrEqualTo(0L)));
131 
132         assertThat("IPV4 UDP stats shouldn't be null", udp4.toString(), is(notNullValue()));
133         assertThat("IPV6 UDP stats shouldn't be null", udp6.toString(), is(notNullValue()));
134     }

136     @Test
137     void testGetConnections() {
138         for (IPConnection conn : ipStats.getConnections()) {
139             assertThat("Protocol name is not null or empty", conn.getType(),
140                     allOf(is(not(emptyString())), is(notNullValue())));
141             if (conn.getType().contains("6")) {
142                 assertThat("Local address array size should be 0, or 16", conn.getLocalAddress().length,
143                         anyOf(is(0), is(16)));
144                 assertThat("Foreign address rray size should be 0, or 16", conn.getForeignAddress().length,
145                         anyOf(is(0), is(16)));
146             } else {
147                 assertThat("Local address array size should be 0, or 4", conn.getLocalAddress().length,
148                         anyOf(is(0), is(4)));
149                 assertThat("Foreign address rray size should be 0, or 4", conn.getForeignAddress().length,
150                         anyOf(is(0), is(4)));
151             }
152             assertThat("Local port must be a 16 bit value", conn.getLocalPort(),
153                     allOf(greaterThanOrEqualTo(0), lessThanOrEqualTo(0xffff)));
154             assertThat("Foreign port must be a 16 bit value", conn.getLocalPort(),
155                     allOf(greaterThanOrEqualTo(0), lessThanOrEqualTo(0xffff)));
156             assertThat("Transmit queue msut be nonnegative", conn.getTransmitQueue(), greaterThanOrEqualTo(0));
157             assertThat("Receive queue msut be nonnegative", conn.getReceiveQueue(), greaterThanOrEqualTo(0));
158             assertThat("Connection state must not be null", conn.getState(), is(notNullValue()));
159             assertThat("Owning Process ID must be -1 or higher", conn.getowningProcessId(), greaterThanOrEqualTo(-1));
160             assertThat("Connection toString must not be null", conn.toString(), is(notNullValue()));
161         }
162     }
