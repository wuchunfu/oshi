88                  is(lessThan(System.currentTimeMillis() / 1000L)));

96                  "The current process' permissions (if has sudo or Administrator privileges) should be determined correctly",

181     @Test
182     void testProcessQueryByList() {
183         SystemInfo si = new SystemInfo();
184         OperatingSystem os = si.getOperatingSystem();
185         assertThat("OS family shouldn't be null", os.getFamily(), is(notNullValue()));
186         assertThat("OS manufacturer shouldn't be null", os.getManufacturer(), is(notNullValue()));
187         OSVersionInfo versionInfo = os.getVersionInfo();
188         assertThat("OS version info shouldn't be null", versionInfo, is(notNullValue()));
189 
190         assertThat("OS currently running processes should be 1 or higher", os.getProcessCount(), is(greaterThan(0)));
191         assertThat("OS thread count should be 1 or higher", os.getThreadCount(), is(greaterThan(0)));
192         assertThat("OS current running process id should be 0 or higher", os.getProcessId(),
193                 is(greaterThanOrEqualTo(0)));
194 
195         List<OSProcess> processes = os.getProcesses(null, null, 0);
196         assertThat("Currently running processes shouldn't be null", processes, is(notNullValue()));
197         assertThat("every OS should have at least one process running on it", processes, is(not(empty())));
198         // the list of pids we want info on
199         List<Integer> pids = new ArrayList<>();
200         for (OSProcess p : processes) {
201             pids.add(p.getProcessID());
202         }
203         // query for just those processes
204         Collection<OSProcess> processes1 = os.getProcesses(pids);
205         // there's a potential for a race condition here, if a process we
206         // queried for initially wasn't running during the second query. In this case,
207         // try again with the shorter list
208         while (processes1.size() < pids.size()) {
209             pids.clear();
210             for (OSProcess p : processes1) {
211                 pids.add(p.getProcessID());
212             }
213             // query for just those processes
214             processes1 = os.getProcesses(pids);
215         }
216         assertThat("OS processes should match processes with pids we want info on", pids, hasSize(processes1.size()));
217 
218     }

220     /**
221      * Tests child and dependent process getter
222      */
223     @Test
224     void testGetChildAndDependentProcesses() {
225         // Testing child processes is tricky because we don't really know a priori what
226         // processes might have children, and if we do test the full list vs. individual
227         // processes, we run into a race condition where child processes can start or
228         // stop before we measure a second time. So we can't really test for one-to-one
229         // correspondence of child process lists.
230         //
231         // We can expect code logic failures to occur all/most of the time for
232         // categories of processes, however, and allow occasional differences due to
233         // race conditions. So we will test three categories of processes: Those with 0
234         // children, those with exactly 1 child process, and those with multiple child
235         // processes. On the second poll, we expect at least half of processes in those
236         // categories to still be in the same category.
237         //
238         SystemInfo si = new SystemInfo();
239         OperatingSystem os = si.getOperatingSystem();
240         List<OSProcess> processes = os.getProcesses(null, null, 0);
241         Set<Integer> zeroChildSet = new HashSet<>();
242         Set<Integer> oneChildSet = new HashSet<>();
243         Set<Integer> manyChildSet = new HashSet<>();
244         // Initialize all processes with no children
245         for (OSProcess p : processes) {
246             zeroChildSet.add(p.getProcessID());
247         }
248         // Move parents with 1 or more children to other set
249         for (OSProcess p : processes) {
250             if (zeroChildSet.contains(p.getParentProcessID())) {
251                 // Zero to One
252                 zeroChildSet.remove(p.getParentProcessID());
253                 oneChildSet.add(p.getParentProcessID());
254             } else if (oneChildSet.contains(p.getParentProcessID())) {
255                 // One to many
256                 oneChildSet.remove(p.getParentProcessID());
257                 manyChildSet.add(p.getParentProcessID());
258             }
259         }
260         // Now test that majority of each set is in same category
261         // Zero
262         int matchedChild = 0;
263         int matchedDescendant = 0;
264         int descendantNotLessThanChild = 0;
265         int total = 0;
266         for (Integer i : zeroChildSet) {
267             List<OSProcess> children = os.getChildProcesses(i, null, null, 0);
268             List<OSProcess> descendants = os.getDescendantProcesses(i, null, null, 0);
269             if (children.size() == 0) {
270                 matchedChild++;
271             }
272             if (descendants.size() == 0) {
273                 matchedDescendant++;
274             }
275             // Quit if enough to test
276             if (++total > 9) {
277                 break;
278             }
279         }
280         if (total > 5) {
281             assertThat("Most processes with no children should not suddenly have them.", matchedChild,
282                     is(greaterThan(total / 4)));
283             assertThat("Most processes with no children should not suddenly have descendants.", matchedDescendant,
284                     is(greaterThan(total / 4)));
285         }
286         // One child
287         matchedChild = 0;
288         matchedDescendant = 0;
289         descendantNotLessThanChild = 0;
290         total = 0;
291         for (Integer i : oneChildSet) {
292             List<OSProcess> children = os.getChildProcesses(i, null, null, 0);
293             List<OSProcess> descendants = os.getDescendantProcesses(i, null, null, 0);
294             if (children.size() == 1) {
295                 matchedChild++;
296             }
297             if (descendants.size() >= 1) {
298                 matchedDescendant++;
299             }
300             if (descendants.size() >= children.size()) {
301                 descendantNotLessThanChild++;
302             }
303             // Quit if enough to test
304             if (++total > 9) {
305                 break;
306             }
307         }
308         if (total > 5) {
309             assertThat("Most processes with one child should not suddenly have zero or more than one.", matchedChild,
310                     is(greaterThan(total / 4)));
311             assertThat("Most processes with one child should not suddenly have zero descendants.", matchedDescendant,
312                     is(greaterThan(total / 4)));
313             assertThat("Most processes with one child should have no more children than descendants",
314                     descendantNotLessThanChild, is(greaterThan(total / 4)));
315         }
316         // Many children
317         matchedChild = 0;
318         matchedDescendant = 0;
319         descendantNotLessThanChild = 0;
320         total = 0;
321         for (Integer i : manyChildSet) {
322             // Use a non-null sorting for test purposes
323             List<OSProcess> children = os.getChildProcesses(i, ProcessFiltering.ALL_PROCESSES, ProcessSorting.CPU_DESC,
324                     Integer.MAX_VALUE);
325             List<OSProcess> descendants = os.getDescendantProcesses(i, ProcessFiltering.ALL_PROCESSES,
326                     ProcessSorting.CPU_DESC, Integer.MAX_VALUE);
327             if (children.size() > 1) {
328                 matchedChild++;
329             }
330             if (descendants.size() > 1) {
331                 matchedDescendant++;
332             }
333             if (descendants.size() >= children.size()) {
334                 descendantNotLessThanChild++;
335             }
336             // Quit if enough to test
337             if (++total > 9) {
338                 break;
339             }
340         }
341         if (total > 5) {
342             assertThat("Most processes with more than one child should not suddenly have one or less.", matchedChild,
343                     is(greaterThan(total / 4)));
344             assertThat("Most processes with more than one child should not suddenly have one or less descendants.",
345                     matchedDescendant, is(greaterThan(total / 4)));
346             assertThat("Most processes with more than one child should have no more children than descendants",
347                     descendantNotLessThanChild, is(greaterThan(total / 4)));
348         }
349     }

398         SystemInfo si = new SystemInfo();
399         OperatingSystem os = si.getOperatingSystem();
400         int stopped = 0;
401         int running = 0;
402         for (OSService svc : os.getServices()) {
403             assertThat(svc.getName(), is(not(emptyString())));
404             switch (svc.getState()) {
405             case STOPPED:
406                 stopped++;
407                 break;
408             case RUNNING:
409                 running++;
410                 break;
411             default:
412                 break;
413             }
414         }
415         // Should be at least one of each
416         assertThat("There should be at least 1 stopped service", stopped, is(greaterThan(0)));
417         assertThat("There should be at least 1 running service", running, is(greaterThan(0)));
418     }

442      * Test get desktop windows
443      */
444     @Test
445     void testGetDesktopWindows() {
446         SystemInfo si = new SystemInfo();
447         OperatingSystem os = si.getOperatingSystem();
448         List<OSDesktopWindow> allWindows = os.getDesktopWindows(false);
449         List<OSDesktopWindow> visibleWindows = os.getDesktopWindows(true);
450         assertThat("Visible should be a subset of all windows", visibleWindows.size(),
451                 is(lessThanOrEqualTo(allWindows.size())));
452         Set<Long> windowIds = new HashSet<>();
453         for (OSDesktopWindow dw : visibleWindows) {
454             assertThat("Visible window should be visible", dw.isVisible(), is(true));
455             assertThat("Height should be nononegative", dw.getLocAndSize().height, is(greaterThanOrEqualTo(0)));
456             assertThat("Width should be nononegative", dw.getLocAndSize().width, is(greaterThanOrEqualTo(0)));
457             windowIds.add(dw.getWindowId());
458         }
459         assertThat("Window IDs should be unique", windowIds.size(), is(visibleWindows.size()));
