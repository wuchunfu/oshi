24  package oshi.driver.linux.proc;
25  
26  import static org.hamcrest.MatcherAssert.assertThat;
27  import static org.hamcrest.Matchers.greaterThanOrEqualTo;
28  
29  import org.junit.jupiter.api.Test;
30  
31  import com.sun.jna.Platform;
32  
33  import oshi.SystemInfo;
34  import oshi.hardware.CentralProcessor;
35  import oshi.hardware.HardwareAbstractionLayer;
36  
37  /**
38   * Tests for {@link CpuStat}.
39   */
40  class CpuStatTest {
41  
42      @Test
43      void testSystemCpuLoadTicks() {
44          if (Platform.isLinux()) {
45              long[] systemCpuLoadTicks = CpuStat.getSystemCpuLoadTicks();
46              for (long systemCpuTick : systemCpuLoadTicks) {
47                  assertThat("CPU tick should be greater than or equal to 0", systemCpuTick, greaterThanOrEqualTo(0L));
48              }
49          }
50      }
51  
52      @Test
53      void testGetProcessorCpuLoadTicks() {
54          if (Platform.isLinux()) {
55              SystemInfo si = new SystemInfo();
56              HardwareAbstractionLayer hal = si.getHardware();
57              CentralProcessor processor = hal.getProcessor();
58              int logicalProcessorCount = processor.getLogicalProcessorCount();
59              long[][] processorCpuLoadTicks = CpuStat.getProcessorCpuLoadTicks(logicalProcessorCount);
60              for (long[] cpuTicks : processorCpuLoadTicks) {
61                  for (long cpuTick : cpuTicks) {
62                      assertThat("CPU tick should be greater than or equal to 0", cpuTick, greaterThanOrEqualTo(0L));
63                  }
64              }
65          }
66      }
67  
68      @Test
69      void testGetContextSwitches() {
70          if (Platform.isLinux()) {
71              assertThat("Context switches should be greater than or equal to -1", CpuStat.getContextSwitches(),
72                      greaterThanOrEqualTo(-1L));
73          }
74      }
75  
76      @Test
77      void testGetInterrupts() {
78          if (Platform.isLinux()) {
79              assertThat("Interrupts should be greater than or equal to -1", CpuStat.getInterrupts(),
80                      greaterThanOrEqualTo(-1L));
81          }
82      }
83  
84      @Test
85      void testGetBootTime() {
86          if (Platform.isLinux()) {
87              assertThat("Boot time should be greater than or equal to 0", CpuStat.getBootTime(),
88                      greaterThanOrEqualTo(0L));
89          }
90      }
91  }
