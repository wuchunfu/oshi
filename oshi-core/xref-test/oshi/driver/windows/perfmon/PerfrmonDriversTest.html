85      @Test
86      void testQueryPageSwaps() {
87          if (Platform.isWindows()) {
88              assertThat("Failed Combined queryPageSwaps", MemoryInformation.queryPageSwaps(),
89                      is(aMapWithSize(PageSwapProperty.values().length)));
90              assertThat("Failed PDH queryPageSwaps", PerfCounterQuery.queryValuesFromPDH(PageSwapProperty.class, MEMORY),
91                      is(aMapWithSize(PageSwapProperty.values().length)));

93                      PerfCounterQuery.queryValuesFromWMI(PageSwapProperty.class, WIN32_PERF_RAW_DATA_PERF_OS_MEMORY),
94                      is(aMapWithSize(PageSwapProperty.values().length)));
95          }
96      }

217     private <T extends PdhCounterWildcardProperty> void testWildcardCounters(String s,
218             Pair<List<String>, Map<T, List<Long>>> counterData) {
219         int instanceCount = counterData.getA().size();
220         Map<T, List<Long>> counters = counterData.getB();
221         for (List<Long> counter : counters.values()) {
222             assertThat("Failed " + s + " instance count", counter.size(), is(instanceCount));
223         }
224 
225         @SuppressWarnings("unchecked")
226         int expectedSize = ((Class<T>) counters.keySet().iterator().next().getClass()).getEnumConstants().length - 1;
227         assertThat("Failed " + s + " map size", counters, is(aMapWithSize(expectedSize)));
228     }
