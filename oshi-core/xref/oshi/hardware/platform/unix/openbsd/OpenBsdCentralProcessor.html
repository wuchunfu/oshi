69      private static final Pattern DMESG_CPU = Pattern.compile("cpu(\\d+): smt (\\d+), core (\\d+), package (\\d+)");
70  
71      @Override
72      protected ProcessorIdentifier queryProcessorId() {
73          String cpuVendor = OpenBsdSysctlUtil.sysctl("machdep.cpuvendor", "");
74          int[] mib = new int[2];
75          mib[0] = CTL_HW;
76          mib[1] = HW_MODEL;
77          String cpuName = OpenBsdSysctlUtil.sysctl(mib, "");
78          // CPUID: first 32 bits is cpufeature, last 32 bits is cpuid
79          int cpuid = ParseUtil.hexStringToInt(OpenBsdSysctlUtil.sysctl("machdep.cpuid", ""), 0);
80          int cpufeature = ParseUtil.hexStringToInt(OpenBsdSysctlUtil.sysctl("machdep.cpufeature", ""), 0);
81          Triplet<Integer, Integer, Integer> cpu = cpuidToFamilyModelStepping(cpuid);
82          String cpuFamily = cpu.getA().toString();
83          String cpuModel = cpu.getB().toString();
84          String cpuStepping = cpu.getC().toString();
85          long cpuFreq = ParseUtil.parseHertz(cpuName);
86          if (cpuFreq < 0) {
87              cpuFreq = queryMaxFreq();
88          }
89          mib[1] = HW_MACHINE;
90          String machine = OpenBsdSysctlUtil.sysctl(mib, "");
91          boolean cpu64bit = machine != null && machine.contains("64")
92                  || ExecutingCommand.getFirstAnswer("uname -m").trim().contains("64");
93          String processorID = String.format("%08x%08x", cpufeature, cpuid);
94  
95          return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
96                  cpuFreq);
97      }
98  
99      private static Triplet<Integer, Integer, Integer> cpuidToFamilyModelStepping(int cpuid) {
100         // family is bits 27:20 | 11:8
101         int family = cpuid >> 16 & 0xff0 | cpuid >> 8 & 0xf;
102         // model is bits 19:16 | 7:4
103         int model = cpuid >> 12 & 0xf0 | cpuid >> 4 & 0xf;
104         // stepping is bits 3:0
105         int stepping = cpuid & 0xf;
106         return new Triplet<>(family, model, stepping);
107     }

109     @Override
110     protected long queryMaxFreq() {
111         return queryCurrentFreq()[0];
112     }

114     @Override
115     protected long[] queryCurrentFreq() {
116         long[] freq = new long[1];
117         int[] mib = new int[2];
118         mib[0] = CTL_HW;
119         mib[1] = HW_CPUSPEED;
120         freq[0] = OpenBsdSysctlUtil.sysctl(mib, 0L) * 1_000_000L;
121         return freq;
122     }

124     @Override
125     protected List<LogicalProcessor> initProcessorCounts() {
126         // Iterate dmesg, look for lines:
127         // cpu0: smt 0, core 0, package 0
128         // cpu1: smt 0, core 1, package 0
129         Map<Integer, Integer> coreMap = new HashMap<>();
130         Map<Integer, Integer> packageMap = new HashMap<>();
131         for (String line : ExecutingCommand.runNative("dmesg")) {
132             Matcher m = DMESG_CPU.matcher(line);
133             if (m.matches()) {
134                 int cpu = ParseUtil.parseIntOrDefault(m.group(1), 0);
135                 coreMap.put(cpu, ParseUtil.parseIntOrDefault(m.group(3), 0));
136                 packageMap.put(cpu, ParseUtil.parseIntOrDefault(m.group(4), 0));
137             }
138         }
139         // native call seems to fail here, use fallback
140         int logicalProcessorCount = OpenBsdSysctlUtil.sysctl("hw.ncpuonline", 1);
141         // If we found more procs in dmesg, update
142         if (logicalProcessorCount < coreMap.keySet().size()) {
143             logicalProcessorCount = coreMap.keySet().size();
144         }
145         List<LogicalProcessor> logProcs = new ArrayList<>(logicalProcessorCount);
146         for (int i = 0; i < logicalProcessorCount; i++) {
147             logProcs.add(new LogicalProcessor(i, coreMap.getOrDefault(i, 0), packageMap.getOrDefault(i, 0)));
148         }
149         return logProcs;
150     }

172     private static Pair<Long, Long> queryVmStats() {
173         long contextSwitches = 0L;
174         long interrupts = 0L;
175         List<String> vmstat = ExecutingCommand.runNative("vmstat -s");
176         for (String line : vmstat) {
177             if (line.endsWith("cpu context switches")) {
178                 contextSwitches = ParseUtil.getFirstIntValue(line);
179             } else if (line.endsWith("interrupts")) {
180                 interrupts = ParseUtil.getFirstIntValue(line);
181             }
182         }
183         return new Pair<>(contextSwitches, interrupts);
184     }

186     /**
187      * Get the system CPU load ticks
188      *
189      * @return The system CPU load ticks
190      */
191     @Override
192     protected long[] querySystemCpuLoadTicks() {
193         long[] ticks = new long[TickType.values().length];
194         int[] mib = new int[2];
195         mib[0] = CTL_KERN;
196         mib[1] = KERN_CPTIME;
197         Memory m = OpenBsdSysctlUtil.sysctl(mib);
198         // array of 5 or 6 native longs
199         long[] cpuTicks = cpTimeToTicks(m, false);
200         if (cpuTicks.length >= 5) {
201             ticks[TickType.USER.getIndex()] = cpuTicks[CP_USER];
202             ticks[TickType.NICE.getIndex()] = cpuTicks[CP_NICE];
203             ticks[TickType.SYSTEM.getIndex()] = cpuTicks[CP_SYS];
204             int offset = cpuTicks.length > 5 ? 1 : 0;
205             ticks[TickType.IRQ.getIndex()] = cpuTicks[CP_INTR + offset];
206             ticks[TickType.IDLE.getIndex()] = cpuTicks[CP_IDLE + offset];
207         }
208         return ticks;
209     }

211     /**
212      * Get the processor CPU load ticks
213      *
214      * @return The processor CPU load ticks
215      */
216     @Override
217     protected long[][] queryProcessorCpuLoadTicks() {
218         long[][] ticks = new long[getLogicalProcessorCount()][TickType.values().length];
219         int[] mib = new int[3];
220         mib[0] = CTL_KERN;
221         mib[1] = KERN_CPTIME2;
222         for (int cpu = 0; cpu < getLogicalProcessorCount(); cpu++) {
223             mib[2] = cpu;
224             Memory m = OpenBsdSysctlUtil.sysctl(mib);
225             // array of 5 or 6 longs
226             long[] cpuTicks = cpTimeToTicks(m, true);
227             if (cpuTicks.length >= 5) {
228                 ticks[cpu][TickType.USER.getIndex()] = cpuTicks[CP_USER];
229                 ticks[cpu][TickType.NICE.getIndex()] = cpuTicks[CP_NICE];
230                 ticks[cpu][TickType.SYSTEM.getIndex()] = cpuTicks[CP_SYS];
231                 int offset = cpuTicks.length > 5 ? 1 : 0;
232                 ticks[cpu][TickType.IRQ.getIndex()] = cpuTicks[CP_INTR + offset];
233                 ticks[cpu][TickType.IDLE.getIndex()] = cpuTicks[CP_IDLE + offset];
234             }
235         }
236         return ticks;
237     }

239     /**
240      * Parse memory buffer returned from sysctl kern.cptime or kern.cptime2 to an
241      * array of 5 or 6 longs depending on version.
242      * <p>
243      * Versions 6.4 and later have a 6-element array while earlier versions have
244      * only 5 elements. Additionally kern.cptime uses a native-sized long (32- or
245      * 64-bit) value while kern.cptime2 is always a 64-bit value.
246      *
247      * @param m
248      *            A buffer containing the array.
249      * @param force64bit
250      *            True if the buffer is filled with 64-bit longs, false if native
251      *            long sized values
252      * @return The array
253      */
254     private static long[] cpTimeToTicks(Memory m, boolean force64bit) {
255         long longBytes = force64bit ? 8L : Native.LONG_SIZE;
256         int arraySize = m == null ? 0 : (int) (m.size() / longBytes);
257         if (force64bit && m != null) {
258             return m.getLongArray(0, arraySize);
259         }
260         long[] ticks = new long[arraySize];
261         for (int i = 0; i < arraySize; i++) {
262             ticks[i] = m.getNativeLong(i * longBytes).longValue();
263         }
264         return ticks;
265     }

267     /**
268      * Returns the system load average for the number of elements specified, up to
269      * 3, representing 1, 5, and 15 minutes. The system load average is the sum of
270      * the number of runnable entities queued to the available processors and the
271      * number of runnable entities running on the available processors averaged over
272      * a period of time. The way in which the load average is calculated is
273      * operating system specific but is typically a damped time-dependent average.
274      * If the load average is not available, a negative value is returned. This
275      * method is designed to provide a hint about the system load and may be queried
276      * frequently.
277      * <p>
278      * The load average may be unavailable on some platforms (e.g., Windows) where
279      * it is expensive to implement this method.
280      *
281      * @param nelem
282      *            Number of elements to return.
283      * @return an array of the system load averages for 1, 5, and 15 minutes with
284      *         the size of the array specified by nelem; or negative values if not
285      *         available.
286      */
287     @Override
288     public double[] getSystemLoadAverage(int nelem) {
289         if (nelem < 1 || nelem > 3) {
290             throw new IllegalArgumentException("Must include from one to three elements.");
291         }
292         double[] average = new double[nelem];
293         int retval = OpenBsdLibc.INSTANCE.getloadavg(average, nelem);
294         if (retval < nelem) {
295             Arrays.fill(average, -1d);
296         }
297         return average;
298     }
