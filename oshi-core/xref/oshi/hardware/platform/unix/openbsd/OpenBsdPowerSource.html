39  /**
40   * A Power Source
41   */
42  @ThreadSafe
43  public final class OpenBsdPowerSource extends AbstractPowerSource {

45      public OpenBsdPowerSource(String psName, String psDeviceName, double psRemainingCapacityPercent,
46              double psTimeRemainingEstimated, double psTimeRemainingInstant, double psPowerUsageRate, double psVoltage,
47              double psAmperage, boolean psPowerOnLine, boolean psCharging, boolean psDischarging,
48              CapacityUnits psCapacityUnits, int psCurrentCapacity, int psMaxCapacity, int psDesignCapacity,
49              int psCycleCount, String psChemistry, LocalDate psManufactureDate, String psManufacturer,
50              String psSerialNumber, double psTemperature) {
51          super(psName, psDeviceName, psRemainingCapacityPercent, psTimeRemainingEstimated, psTimeRemainingInstant,
52                  psPowerUsageRate, psVoltage, psAmperage, psPowerOnLine, psCharging, psDischarging, psCapacityUnits,
53                  psCurrentCapacity, psMaxCapacity, psDesignCapacity, psCycleCount, psChemistry, psManufactureDate,
54                  psManufacturer, psSerialNumber, psTemperature);
55      }
56  
57      /**
58       * Gets Battery Information
59       *
60       * @return An array of PowerSource objects representing batteries, etc.
61       */
62      public static List<PowerSource> getPowerSources() {
63          Set<String> psNames = new HashSet<>();
64          for (String line : ExecutingCommand.runNative("systat -ab sensors")) {
65              if (line.contains(".amphour") || line.contains(".watthour")) {
66                  int dot = line.indexOf('.');
67                  psNames.add(line.substring(0, dot));
68              }
69          }
70          List<PowerSource> psList = new ArrayList<>();
71          for (String name : psNames) {
72              psList.add(getPowerSource(name));
73          }
74          return psList;
75      }
76  
77      private static OpenBsdPowerSource getPowerSource(String name) {
78          String psName = name.startsWith("acpi") ? name.substring(4) : name;
79          double psRemainingCapacityPercent = 1d;
80          double psTimeRemainingEstimated = -1d; // -1 = unknown, -2 = unlimited
81          double psPowerUsageRate = 0d;
82          double psVoltage = -1d;
83          double psAmperage = 0d;
84          boolean psPowerOnLine = false;
85          boolean psCharging = false;
86          boolean psDischarging = false;
87          CapacityUnits psCapacityUnits = CapacityUnits.RELATIVE;
88          int psCurrentCapacity = 0;
89          int psMaxCapacity = 1;
90          int psDesignCapacity = 1;
91          int psCycleCount = -1;
92          LocalDate psManufactureDate = null;
93  
94          double psTemperature = 0d;
95  
96          for (String line : ExecutingCommand.runNative("systat -ab sensors")) {
97              String[] split = ParseUtil.whitespaces.split(line);
98              if (split.length > 1 && split[0].startsWith(name)) {
99                  if (split[0].contains("volt0") || split[0].contains("volt") && line.contains("current")) {
100                     psVoltage = ParseUtil.parseDoubleOrDefault(split[1], -1d);
101                 } else if (split[0].contains("current0")) {
102                     psAmperage = ParseUtil.parseDoubleOrDefault(split[1], 0d);
103                 } else if (split[0].contains("temp0")) {
104                     psTemperature = ParseUtil.parseDoubleOrDefault(split[1], 0d);
105                 } else if (split[0].contains("watthour") || split[0].contains("amphour")) {
106                     psCapacityUnits = split[0].contains("watthour") ? CapacityUnits.MWH : CapacityUnits.MAH;
107                     if (line.contains("remaining")) {
108                         psCurrentCapacity = (int) (1000d * ParseUtil.parseDoubleOrDefault(split[1], 0d));
109                     } else if (line.contains("full")) {
110                         psMaxCapacity = (int) (1000d * ParseUtil.parseDoubleOrDefault(split[1], 0d));
111                     } else if (line.contains("new") || line.contains("design")) {
112                         psDesignCapacity = (int) (1000d * ParseUtil.parseDoubleOrDefault(split[1], 0d));
113                     }
114                 }
115             }
116         }
117 
118         int state = ParseUtil.parseIntOrDefault(ExecutingCommand.getFirstAnswer("apm -b"), 255);
119         // state 0=high, 1=low, 2=critical, 3=charging, 4=absent, 255=unknown
120         if (state < 4) {
121             psPowerOnLine = true;
122             if (state == 3) {
123                 psCharging = true;
124             } else {
125                 int time = ParseUtil.parseIntOrDefault(ExecutingCommand.getFirstAnswer("apm -m"), -1);
126                 // time is in minutes
127                 psTimeRemainingEstimated = time < 0 ? -1d : 60d * time;
128                 psDischarging = true;
129             }
130         }
131         // life is in percent
132         int life = ParseUtil.parseIntOrDefault(ExecutingCommand.getFirstAnswer("apm -l"), -1);
133         if (life > 0) {
134             psRemainingCapacityPercent = life / 100d;
135         }
136         if (psMaxCapacity < psDesignCapacity && psMaxCapacity < psCurrentCapacity) {
137             psMaxCapacity = psDesignCapacity;
138         } else if (psDesignCapacity < psMaxCapacity && psDesignCapacity < psCurrentCapacity) {
139             psDesignCapacity = psMaxCapacity;
140         }
141 
142         String psDeviceName = Constants.UNKNOWN;
143         String psSerialNumber = Constants.UNKNOWN;
144         String psChemistry = Constants.UNKNOWN;
145         String psManufacturer = Constants.UNKNOWN;
146 
147         double psTimeRemainingInstant = psTimeRemainingEstimated;
148         if (psVoltage > 0) {
149             if (psAmperage > 0 && psPowerUsageRate == 0) {
150                 psPowerUsageRate = psAmperage * psVoltage;
151             } else if (psAmperage == 0 && psPowerUsageRate > 0) {
152                 psAmperage = psPowerUsageRate / psVoltage;
153             }
154         }
155 
156         return new OpenBsdPowerSource(psName, psDeviceName, psRemainingCapacityPercent, psTimeRemainingEstimated,
157                 psTimeRemainingInstant, psPowerUsageRate, psVoltage, psAmperage, psPowerOnLine, psCharging,
158                 psDischarging, psCapacityUnits, psCurrentCapacity, psMaxCapacity, psDesignCapacity, psCycleCount,
159                 psChemistry, psManufactureDate, psManufacturer, psSerialNumber, psTemperature);
160     }
