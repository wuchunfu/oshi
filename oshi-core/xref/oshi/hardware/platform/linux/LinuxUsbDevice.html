47      private static final String SUBSYSTEM_USB = "usb";
48      private static final String DEVTYPE_USB_DEVICE = "usb_device";
49      private static final String ATTR_PRODUCT = "product";
50      private static final String ATTR_MANUFACTURER = "manufacturer";
51      private static final String ATTR_VENDOR_ID = "idVendor";
52      private static final String ATTR_PRODUCT_ID = "idProduct";
53      private static final String ATTR_SERIAL = "serial";
54  
55      public LinuxUsbDevice(String name, String vendor, String vendorId, String productId, String serialNumber,
56              String uniqueDeviceId, List<UsbDevice> connectedDevices) {
57          super(name, vendor, vendorId, productId, serialNumber, uniqueDeviceId, connectedDevices);
58      }

60      /**
61       * Instantiates a list of {@link oshi.hardware.UsbDevice} objects, representing
62       * devices connected via a usb port (including internal devices).
63       * <p>
64       * If the value of {@code tree} is true, the top level devices returned from
65       * this method are the USB Controllers; connected hubs and devices in its device
66       * tree share that controller's bandwidth. If the value of {@code tree} is
67       * false, USB devices (not controllers) are listed in a single flat list.
68       *
69       * @param tree
70       *            If true, returns a list of controllers, which requires recursive
71       *            iteration of connected devices. If false, returns a flat list of
72       *            devices excluding controllers.
73       * @return a list of {@link oshi.hardware.UsbDevice} objects.
74       */
75      public static List<UsbDevice> getUsbDevices(boolean tree) {
76          List<UsbDevice> devices = getUsbDevices();
77          if (tree) {
78              return devices;
79          }
80          List<UsbDevice> deviceList = new ArrayList<>();
81          // Top level is controllers; they won't be added to the list, but all
82          // their connected devices will be
83          for (UsbDevice device : devices) {
84              deviceList.add(new LinuxUsbDevice(device.getName(), device.getVendor(), device.getVendorId(),
85                      device.getProductId(), device.getSerialNumber(), device.getUniqueDeviceId(),
86                      Collections.emptyList()));
87              addDevicesToList(deviceList, device.getConnectedDevices());
88          }
89          return deviceList;
90      }
91  
92      private static List<UsbDevice> getUsbDevices() {
93          // Build a list of devices with no parent; these will be the roots
94          List<String> usbControllers = new ArrayList<>();
95  
96          // Maps to store information using device syspath as the key
97          Map<String, String> nameMap = new HashMap<>();
98          Map<String, String> vendorMap = new HashMap<>();
99          Map<String, String> vendorIdMap = new HashMap<>();
100         Map<String, String> productIdMap = new HashMap<>();
101         Map<String, String> serialMap = new HashMap<>();
102         Map<String, List<String>> hubMap = new HashMap<>();
103 
104         // Enumerate all usb devices and build information maps
105         Udev.UdevContext udev = Udev.INSTANCE.udev_new();
106         try {
107             UdevEnumerate enumerate = udev.enumerateNew();
108             try {
109                 enumerate.addMatchSubsystem(SUBSYSTEM_USB);
110                 enumerate.scanDevices();
111 
112                 // For each item enumerated, store information in the maps
113                 for (UdevListEntry entry = enumerate.getListEntry(); entry != null; entry = entry.getNext()) {
114                     String syspath = entry.getName();
115                     UdevDevice device = udev.deviceNewFromSyspath(syspath);
116                     if (device != null) {
117                         try {
118                             // Only include usb_device devtype, skipping usb_interface
119                             if (DEVTYPE_USB_DEVICE.equals(device.getDevtype())) {
120                                 String value = device.getSysattrValue(ATTR_PRODUCT);
121                                 if (value != null) {
122                                     nameMap.put(syspath, value);
123                                 }
124                                 value = device.getSysattrValue(ATTR_MANUFACTURER);
125                                 if (value != null) {
126                                     vendorMap.put(syspath, value);
127                                 }
128                                 value = device.getSysattrValue(ATTR_VENDOR_ID);
129                                 if (value != null) {
130                                     vendorIdMap.put(syspath, value);
131                                 }
132                                 value = device.getSysattrValue(ATTR_PRODUCT_ID);
133                                 if (value != null) {
134                                     productIdMap.put(syspath, value);
135                                 }
136                                 value = device.getSysattrValue(ATTR_SERIAL);
137                                 if (value != null) {
138                                     serialMap.put(syspath, value);
139                                 }
140 
141                                 UdevDevice parent = device.getParentWithSubsystemDevtype(SUBSYSTEM_USB,
142                                         DEVTYPE_USB_DEVICE);
143                                 if (parent == null) {
144                                     // This is a controller with no parent, add to list
145                                     usbControllers.add(syspath);
146                                 } else {
147                                     // Add child syspath to parent's path
148                                     String parentPath = parent.getSyspath();
149                                     hubMap.computeIfAbsent(parentPath, x -> new ArrayList<>()).add(syspath);
150                                 }
151                             }
152                         } finally {
153                             device.unref();
154                         }
155                     }
156                 }
157             } finally {
158                 enumerate.unref();
159             }
160         } finally {
161             udev.unref();
162         }
163 
164         // Build tree and return
165         List<UsbDevice> controllerDevices = new ArrayList<>();
166         for (String controller : usbControllers) {
167             controllerDevices.add(getDeviceAndChildren(controller, "0000", "0000", nameMap, vendorMap, vendorIdMap,
168                     productIdMap, serialMap, hubMap));
169         }
170         return controllerDevices;
171     }
172 
173     private static void addDevicesToList(List<UsbDevice> deviceList, List<UsbDevice> list) {
174         for (UsbDevice device : list) {
175             deviceList.add(device);
176             addDevicesToList(deviceList, device.getConnectedDevices());
177         }
178     }

180     /**
181      * Recursively creates LinuxUsbDevices by fetching information from maps to
182      * populate fields
183      *
184      * @param devPath
185      *            The device node path.
186      * @param vid
187      *            The default (parent) vendor ID
188      * @param pid
189      *            The default (parent) product ID
190      * @param nameMap
191      *            the map of names
192      * @param vendorMap
193      *            the map of vendors
194      * @param vendorIdMap
195      *            the map of vendorIds
196      * @param productIdMap
197      *            the map of productIds
198      * @param serialMap
199      *            the map of serial numbers
200      * @param hubMap
201      *            the map of hubs
202      * @return A LinuxUsbDevice corresponding to this device
203      */
204     private static LinuxUsbDevice getDeviceAndChildren(String devPath, String vid, String pid,
205             Map<String, String> nameMap, Map<String, String> vendorMap, Map<String, String> vendorIdMap,
206             Map<String, String> productIdMap, Map<String, String> serialMap, Map<String, List<String>> hubMap) {
207         String vendorId = vendorIdMap.getOrDefault(devPath, vid);
208         String productId = productIdMap.getOrDefault(devPath, pid);
209         List<String> childPaths = hubMap.getOrDefault(devPath, new ArrayList<>());
210         List<UsbDevice> usbDevices = new ArrayList<>();
211         for (String path : childPaths) {
212             usbDevices.add(getDeviceAndChildren(path, vendorId, productId, nameMap, vendorMap, vendorIdMap,
213                     productIdMap, serialMap, hubMap));
214         }
215         Collections.sort(usbDevices);
216         return new LinuxUsbDevice(nameMap.getOrDefault(devPath, vendorId + ":" + productId),
217                 vendorMap.getOrDefault(devPath, ""), vendorId, productId, serialMap.getOrDefault(devPath, ""), devPath,
218                 usbDevices);
219     }
