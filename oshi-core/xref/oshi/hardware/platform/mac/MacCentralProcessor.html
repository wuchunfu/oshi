67      private static final int ROSETTA_CPUTYPE = 0x00000007;
68      private static final int ROSETTA_CPUFAMILY = 0x573b5eec;
69      private static final int M1_CPUTYPE = 0x0100000C;
70      private static final int M1_CPUFAMILY = 0x1b588bb3;

72      @Override
73      protected ProcessorIdentifier queryProcessorId() {
74          String cpuName = SysctlUtil.sysctl("machdep.cpu.brand_string", "");
75          String cpuVendor;
76          String cpuStepping;
77          String cpuModel;
78          String cpuFamily;
79          String processorID;
80          long cpuFreq = 0L;
81          if (cpuName.startsWith("Apple")) {
82              // Processing an M1 chip
83              cpuVendor = vendor.get();
84              cpuStepping = "0"; // No correlation yet
85              cpuModel = "0"; // No correlation yet
86              int type = SysctlUtil.sysctl("hw.cputype", 0);
87              int family = SysctlUtil.sysctl("hw.cpufamily", 0);
88              // M1 should have hw.cputype 0x0100000C (ARM64) and hw.cpufamily 0x1b588bb3 for
89              // an ARM SoC. However, under Rosetta 2, low level cpuid calls in the translated
90              // environment report hw.cputype for x86 (0x00000007) and hw.cpufamily for an
91              // Intel Westmere chip (0x573b5eec), family 6, model 44, stepping 0.
92              // Test if under Rosetta and generate correct chip
93              if (family == ROSETTA_CPUFAMILY) {
94                  type = typeFamilyFreq.get().getA();
95                  family = typeFamilyFreq.get().getB();
96              }
97              cpuFreq = typeFamilyFreq.get().getC();
98              // Translate to output
99              cpuFamily = String.format("0x%08x", family);
100             // Processor ID is an intel concept but CPU type + family conveys same info
101             processorID = String.format("%08x%08x", type, family);
102         } else {
103             // Processing an Intel chip
104             cpuVendor = SysctlUtil.sysctl("machdep.cpu.vendor", "");
105             int i = SysctlUtil.sysctl("machdep.cpu.stepping", -1);
106             cpuStepping = i < 0 ? "" : Integer.toString(i);
107             i = SysctlUtil.sysctl("machdep.cpu.model", -1);
108             cpuModel = i < 0 ? "" : Integer.toString(i);
109             i = SysctlUtil.sysctl("machdep.cpu.family", -1);
110             cpuFamily = i < 0 ? "" : Integer.toString(i);
111             long processorIdBits = 0L;
112             processorIdBits |= SysctlUtil.sysctl("machdep.cpu.signature", 0);
113             processorIdBits |= (SysctlUtil.sysctl("machdep.cpu.feature_bits", 0L) & 0xffffffff) << 32;
114             processorID = String.format("%016x", processorIdBits);
115         }
116         if (cpuFreq == 0) {
117             cpuFreq = SysctlUtil.sysctl("hw.cpufrequency", 0L);
118         }
119         boolean cpu64bit = SysctlUtil.sysctl("hw.cpu64bit_capable", 0) != 0;
120 
121         return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
122                 cpuFreq);
123     }

125     @Override
126     protected List<LogicalProcessor> initProcessorCounts() {
127         int logicalProcessorCount = SysctlUtil.sysctl("hw.logicalcpu", 1);
128         int physicalProcessorCount = SysctlUtil.sysctl("hw.physicalcpu", 1);
129         int physicalPackageCount = SysctlUtil.sysctl("hw.packages", 1);
130         List<LogicalProcessor> logProcs = new ArrayList<>(logicalProcessorCount);
131         for (int i = 0; i < logicalProcessorCount; i++) {
132             logProcs.add(new LogicalProcessor(i, i * physicalProcessorCount / logicalProcessorCount,
133                     i * physicalPackageCount / logicalProcessorCount));
134         }
135         return logProcs;
136     }

138     @Override
139     public long[] querySystemCpuLoadTicks() {
140         long[] ticks = new long[TickType.values().length];
141         int machPort = SystemB.INSTANCE.mach_host_self();
142         HostCpuLoadInfo cpuLoadInfo = new HostCpuLoadInfo();
143         if (0 != SystemB.INSTANCE.host_statistics(machPort, SystemB.HOST_CPU_LOAD_INFO, cpuLoadInfo,
144                 new IntByReference(cpuLoadInfo.size()))) {
145             LOG.error("Failed to get System CPU ticks. Error code: {} ", Native.getLastError());
146             return ticks;
147         }
148 
149         ticks[TickType.USER.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_USER];
150         ticks[TickType.NICE.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_NICE];
151         ticks[TickType.SYSTEM.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_SYSTEM];
152         ticks[TickType.IDLE.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_IDLE];
153         // Leave IOWait and IRQ values as 0
154         return ticks;
155     }

169     @Override
170     public double[] getSystemLoadAverage(int nelem) {
171         if (nelem < 1 || nelem > 3) {
172             throw new IllegalArgumentException("Must include from one to three elements.");
173         }
174         double[] average = new double[nelem];
175         int retval = SystemB.INSTANCE.getloadavg(average, nelem);
176         if (retval < nelem) {
177             Arrays.fill(average, -1d);
178         }
179         return average;
180     }

182     @Override
183     public long[][] queryProcessorCpuLoadTicks() {
184         long[][] ticks = new long[getLogicalProcessorCount()][TickType.values().length];
185 
186         int machPort = SystemB.INSTANCE.mach_host_self();
187 
188         IntByReference procCount = new IntByReference();
189         PointerByReference procCpuLoadInfo = new PointerByReference();
190         IntByReference procInfoCount = new IntByReference();
191         if (0 != SystemB.INSTANCE.host_processor_info(machPort, SystemB.PROCESSOR_CPU_LOAD_INFO, procCount,
192                 procCpuLoadInfo, procInfoCount)) {
193             LOG.error("Failed to update CPU Load. Error code: {}", Native.getLastError());
194             return ticks;
195         }
196 
197         int[] cpuTicks = procCpuLoadInfo.getValue().getIntArray(0, procInfoCount.getValue());
198         for (int cpu = 0; cpu < procCount.getValue(); cpu++) {
199             int offset = cpu * SystemB.CPU_STATE_MAX;
200             ticks[cpu][TickType.USER.getIndex()] = FormatUtil.getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_USER]);
201             ticks[cpu][TickType.NICE.getIndex()] = FormatUtil.getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_NICE]);
202             ticks[cpu][TickType.SYSTEM.getIndex()] = FormatUtil
203                     .getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_SYSTEM]);
204             ticks[cpu][TickType.IDLE.getIndex()] = FormatUtil.getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_IDLE]);
205         }
206         return ticks;
207     }

225     private static String platformExpert() {
226         String manufacturer = null;
227         IORegistryEntry platformExpert = IOKitUtil.getMatchingService("IOPlatformExpertDevice");
228         if (platformExpert != null) {
229             // Get manufacturer from IOPlatformExpertDevice
230             byte[] data = platformExpert.getByteArrayProperty("manufacturer");
231             if (data != null) {
232                 manufacturer = Native.toString(data, StandardCharsets.UTF_8);
233             }
234             platformExpert.release();
235         }
236         return Util.isBlank(manufacturer) ? "Apple Inc." : manufacturer;
237     }

239     private static Triplet<Integer, Integer, Long> queryArmCpu() {
240         int type = ROSETTA_CPUTYPE;
241         int family = ROSETTA_CPUFAMILY;
242         long freq = 0L;
243         // All CPUs are an IOPlatformDevice
244         // Iterate each CPU and save frequency and "compatible" strings
245         IOIterator iter = IOKitUtil.getMatchingServices("IOPlatformDevice");
246         if (iter != null) {
247             Set<String> compatibleStrSet = new HashSet<>();
248             IORegistryEntry cpu = iter.next();
249             while (cpu != null) {
250                 if (cpu.getName().startsWith("cpu")) {
251                     // Accurate CPU vendor frequency in kHz as little-endian byte array
252                     byte[] data = cpu.getByteArrayProperty("clock-frequency");
253                     if (data != null) {
254                         long cpuFreq = ParseUtil.byteArrayToLong(data, data.length, false) * 1000L;
255                         if (cpuFreq > freq) {
256                             freq = cpuFreq;
257                         }
258                     }
259                     // Compatible key is null-delimited C string array in byte array
260                     data = cpu.getByteArrayProperty("compatible");
261                     if (data != null) {
262                         for (String s : new String(data, StandardCharsets.UTF_8).split("\0")) {
263                             if (!s.isEmpty()) {
264                                 compatibleStrSet.add(s);
265                             }
266                         }
267                     }
268                 }
269                 cpu.release();
270                 cpu = iter.next();
271             }
272             iter.release();
273             // Match strings in "compatible" field with expectation for M1 chip
274             // Hard coded for M1 for now. Need to update and make more configurable for M1X,
275             // M2, etc.
276             List<String> m1compatible = Arrays.asList("ARM,v8", "apple,firestorm", "apple,icestorm");
277             compatibleStrSet.retainAll(m1compatible);
278             if (compatibleStrSet.size() == m1compatible.size()) {
279                 type = M1_CPUTYPE;
280                 family = M1_CPUFAMILY;
281             }
282         }
283         return new Triplet<>(type, family, freq);
284     }
