64      private static final boolean X64 = Platform.is64Bit();

67      private static final int BATTERY_IS_SHORT_TERM = 0x20000000;
68      private static final int BATTERY_POWER_ON_LINE = 0x00000001;
69      private static final int BATTERY_DISCHARGING = 0x00000002;
70      private static final int BATTERY_CHARGING = 0x00000004;
71      private static final int BATTERY_CAPACITY_RELATIVE = 0x40000000;

98      private static WindowsPowerSource getPowerSource(String name) {
99          String psName = name;
100         String psDeviceName = Constants.UNKNOWN;
101         double psRemainingCapacityPercent = 1d;
102         double psTimeRemainingEstimated = -1d; // -1 = unknown, -2 = unlimited
103         double psTimeRemainingInstant = 0d;
104         int psPowerUsageRate = 0;
105         double psVoltage = -1d;
106         double psAmperage = 0d;
107         boolean psPowerOnLine = false;
108         boolean psCharging = false;
109         boolean psDischarging = false;
110         CapacityUnits psCapacityUnits = CapacityUnits.RELATIVE;
111         int psCurrentCapacity = 0;
112         int psMaxCapacity = 1;
113         int psDesignCapacity = 1;
114         int psCycleCount = -1;
115         String psChemistry = Constants.UNKNOWN;
116         LocalDate psManufactureDate = null;
117         String psManufacturer = Constants.UNKNOWN;
118         String psSerialNumber = Constants.UNKNOWN;
119         double psTemperature = 0d;
120 
121         // windows PowerSource information comes from two sources: the PowrProf's
122         // CallNTPowerInformation function which returns information for a single
123         // object, and DeviceIoControl with each battery's (if more than one) handle
124         // (which, in theory, return an array of objects but in most cases should return
125         // one).
126         //
127         // We start by fetching the PowrProf information, which will be replicated
128         // across all IOCTL entries if there are more than one.
129 
130         int size = new SystemBatteryState().size();
131         Memory mem = new Memory(size);
132         if (0 == PowrProf.INSTANCE.CallNtPowerInformation(POWER_INFORMATION_LEVEL.SystemBatteryState, null, 0, mem,
133                 size)) {
134             SystemBatteryState batteryState = new SystemBatteryState(mem);
135             if (batteryState.batteryPresent > 0) {
136                 if (batteryState.acOnLine == 0 && batteryState.charging == 0 && batteryState.discharging > 0) {
137                     psTimeRemainingEstimated = batteryState.estimatedTime;
138                 } else if (batteryState.charging > 0) {
139                     psTimeRemainingEstimated = -2d;
140                 }
141                 psMaxCapacity = batteryState.maxCapacity;
142                 psCurrentCapacity = batteryState.remainingCapacity;
143                 psRemainingCapacityPercent = Math.min(1d, (double) psCurrentCapacity / psMaxCapacity);
144                 psPowerUsageRate = batteryState.rate;
145             }
146         }
147 
148         // Enumerate batteries and ask each one for information
149         // Ported from:
150         // https://docs.microsoft.com/en-us/windows/win32/power/enumerating-battery-devices
151 
152         HANDLE hdev = SetupApi.INSTANCE.SetupDiGetClassDevs(GUID_DEVCLASS_BATTERY, null, null,
153                 SetupApi.DIGCF_PRESENT | SetupApi.DIGCF_DEVICEINTERFACE);
154         if (!WinBase.INVALID_HANDLE_VALUE.equals(hdev)) {
155             boolean batteryFound = false;
156             // Limit search to 100 batteries max
157             for (int idev = 0; !batteryFound && idev < 100; idev++) {
158                 SP_DEVICE_INTERFACE_DATA did = new SP_DEVICE_INTERFACE_DATA();
159                 did.cbSize = did.size();
160 
161                 if (SetupApi.INSTANCE.SetupDiEnumDeviceInterfaces(hdev, null, GUID_DEVCLASS_BATTERY, idev, did)) {
162                     IntByReference requiredSize = new IntByReference(0);
163                     SetupApi.INSTANCE.SetupDiGetDeviceInterfaceDetail(hdev, did, null, 0, requiredSize, null);
164                     if (WinError.ERROR_INSUFFICIENT_BUFFER == Kernel32.INSTANCE.GetLastError()) {
165                         // PSP_DEVICE_INTERFACE_DETAIL_DATA: int size + TCHAR array
166                         Memory pdidd = new Memory(requiredSize.getValue());
167                         // pdidd->cbSize is defined as sizeof(*pdidd)
168                         // On 64 bit, cbSize is 8. On 32-bit it's 5 or 6 based on char size
169                         // This must be set properly for the method to work but is otherwise ignored
170                         pdidd.setInt(0, Integer.BYTES + (X64 ? 4 : CHAR_WIDTH));
171                         // Regardless of this setting the string portion starts after one byte
172                         if (SetupApi.INSTANCE.SetupDiGetDeviceInterfaceDetail(hdev, did, pdidd, (int) pdidd.size(),
173                                 requiredSize, null)) {
174                             // Enumerated a battery. Ask it for information.
175                             String devicePath = CHAR_WIDTH > 1 ? pdidd.getWideString(Integer.BYTES)
176                                     : pdidd.getString(Integer.BYTES);
177                             HANDLE hBattery = Kernel32.INSTANCE.CreateFile(devicePath, // pdidd->DevicePath
178                                     WinNT.GENERIC_READ | WinNT.GENERIC_WRITE,
179                                     WinNT.FILE_SHARE_READ | WinNT.FILE_SHARE_WRITE, null, WinNT.OPEN_EXISTING,
180                                     WinNT.FILE_ATTRIBUTE_NORMAL, null);
181                             if (!WinBase.INVALID_HANDLE_VALUE.equals(hBattery)) {
182                                 // Ask the battery for its tag.
183                                 BATTERY_QUERY_INFORMATION bqi = new PowrProf.BATTERY_QUERY_INFORMATION();
184                                 IntByReference dwWait = new IntByReference(0);
185                                 IntByReference dwTag = new IntByReference();
186                                 IntByReference dwOut = new IntByReference();
187 
188                                 if (Kernel32.INSTANCE.DeviceIoControl(hBattery, IOCTL_BATTERY_QUERY_TAG,
189                                         dwWait.getPointer(), Integer.BYTES, dwTag.getPointer(), Integer.BYTES, dwOut,
190                                         null)) {
191                                     bqi.BatteryTag = dwTag.getValue();
192                                     if (bqi.BatteryTag > 0) {
193                                         // With the tag, you can query the battery info.
194                                         bqi.InformationLevel = PowrProf.BATTERY_QUERY_INFORMATION_LEVEL.BatteryInformation
195                                                 .ordinal();
196                                         bqi.write();
197                                         BATTERY_INFORMATION bi = new BATTERY_INFORMATION();
198                                         if (Kernel32.INSTANCE.DeviceIoControl(hBattery, IOCTL_BATTERY_QUERY_INFORMATION,
199                                                 bqi.getPointer(), bqi.size(), bi.getPointer(), bi.size(), dwOut,
200                                                 null)) {
201                                             // Only non-UPS system batteries count
202                                             bi.read();
203                                             if (0 != (bi.Capabilities & BATTERY_SYSTEM_BATTERY)
204                                                     && 0 == (bi.Capabilities & BATTERY_IS_SHORT_TERM)) {
205                                                 // Capabilities flags non-mWh units
206                                                 if (0 == (bi.Capabilities & BATTERY_CAPACITY_RELATIVE)) {
207                                                     psCapacityUnits = CapacityUnits.MWH;
208                                                 }
209                                                 psChemistry = Native.toString(bi.Chemistry, StandardCharsets.US_ASCII);
210                                                 psDesignCapacity = bi.DesignedCapacity;
211                                                 psMaxCapacity = bi.FullChargedCapacity;
212                                                 psCycleCount = bi.CycleCount;
213 
214                                                 // Query the battery status.
215                                                 PowrProf.BATTERY_WAIT_STATUS bws = new PowrProf.BATTERY_WAIT_STATUS();
216                                                 bws.BatteryTag = bqi.BatteryTag;
217                                                 bws.write();
218                                                 PowrProf.BATTERY_STATUS bs = new PowrProf.BATTERY_STATUS();
219                                                 if (Kernel32.INSTANCE.DeviceIoControl(hBattery,
220                                                         IOCTL_BATTERY_QUERY_STATUS, bws.getPointer(), bws.size(),
221                                                         bs.getPointer(), bs.size(), dwOut, null)) {
222                                                     bs.read();
223                                                     if (0 != (bs.PowerState & BATTERY_POWER_ON_LINE)) {
224                                                         psPowerOnLine = true;
225                                                     }
226                                                     if (0 != (bs.PowerState & BATTERY_DISCHARGING)) {
227                                                         psDischarging = true;
228                                                     }
229                                                     if (0 != (bs.PowerState & BATTERY_CHARGING)) {
230                                                         psCharging = true;
231                                                     }
232                                                     psCurrentCapacity = bs.Capacity;
233                                                     psVoltage = bs.Voltage > 0 ? bs.Voltage / 1000d : bs.Voltage;
234                                                     psPowerUsageRate = bs.Rate;
235                                                     if (psVoltage > 0) {
236                                                         psAmperage = psPowerUsageRate / psVoltage;
237                                                     }
238                                                 }
239                                             }
240 
241                                             psDeviceName = batteryQueryString(hBattery, dwTag.getValue(),
242                                                     PowrProf.BATTERY_QUERY_INFORMATION_LEVEL.BatteryDeviceName
243                                                             .ordinal());
244                                             psManufacturer = batteryQueryString(hBattery, dwTag.getValue(),
245                                                     PowrProf.BATTERY_QUERY_INFORMATION_LEVEL.BatteryManufactureName
246                                                             .ordinal());
247                                             psSerialNumber = batteryQueryString(hBattery, dwTag.getValue(),
248                                                     PowrProf.BATTERY_QUERY_INFORMATION_LEVEL.BatterySerialNumber
249                                                             .ordinal());
250 
251                                             bqi.InformationLevel = PowrProf.BATTERY_QUERY_INFORMATION_LEVEL.BatteryManufactureDate
252                                                     .ordinal();
253                                             bqi.write();
254                                             BATTERY_MANUFACTURE_DATE bmd = new BATTERY_MANUFACTURE_DATE();
255                                             if (Kernel32.INSTANCE.DeviceIoControl(hBattery,
256                                                     IOCTL_BATTERY_QUERY_INFORMATION, bqi.getPointer(), bqi.size(),
257                                                     bmd.getPointer(), bmd.size(), dwOut, null)) {
258                                                 bmd.read();
259                                                 // If failed, returns -1 for each field
260                                                 if (bmd.Year > 1900 && bmd.Month > 0 && bmd.Day > 0) {
261                                                     psManufactureDate = LocalDate.of(bmd.Year, bmd.Month, bmd.Day);
262                                                 }
263                                             }
264 
265                                             bqi.InformationLevel = PowrProf.BATTERY_QUERY_INFORMATION_LEVEL.BatteryTemperature
266                                                     .ordinal();
267                                             bqi.write();
268                                             IntByReference tempK = new IntByReference(); // 1/10 degree K
269                                             if (Kernel32.INSTANCE.DeviceIoControl(hBattery,
270                                                     IOCTL_BATTERY_QUERY_INFORMATION, bqi.getPointer(), bqi.size(),
271                                                     tempK.getPointer(), Integer.BYTES, dwOut, null)) {
272                                                 psTemperature = tempK.getValue() / 10d - 273.15;
273                                             }
274 
275                                             // Put last because we change the AtRate field
276                                             bqi.InformationLevel = PowrProf.BATTERY_QUERY_INFORMATION_LEVEL.BatteryEstimatedTime
277                                                     .ordinal();
278                                             if (psPowerUsageRate != 0) {
279                                                 bqi.AtRate = psPowerUsageRate;
280                                             }
281                                             bqi.write();
282                                             IntByReference tr = new IntByReference();
283                                             if (Kernel32.INSTANCE.DeviceIoControl(hBattery,
284                                                     IOCTL_BATTERY_QUERY_INFORMATION, bqi.getPointer(), bqi.size(),
285                                                     tr.getPointer(), Integer.BYTES, dwOut, null)) {
286                                                 psTimeRemainingInstant = tr.getValue();
287                                             }
288                                             // Fallback
289                                             if (psTimeRemainingInstant < 0 && psPowerUsageRate != 0) {
290                                                 psTimeRemainingInstant = (psMaxCapacity - psCurrentCapacity) * 3600d
291                                                         / psPowerUsageRate;
292                                                 if (psTimeRemainingInstant < 0) {
293                                                     psTimeRemainingInstant *= -1;
294                                                 }
295                                             }
296                                             // Exit loop
297                                             batteryFound = true;
298                                         }
299                                     }
300                                 }
301                                 Kernel32.INSTANCE.CloseHandle(hBattery);
302                             }
303                         }
304                     }
305                 } else if (WinError.ERROR_NO_MORE_ITEMS == Kernel32.INSTANCE.GetLastError()) {
306                     break; // Enumeration failed - perhaps we're out of items
307                 }
308             }
309             SetupApi.INSTANCE.SetupDiDestroyDeviceInfoList(hdev);
310         }
311 
312         return new WindowsPowerSource(psName, psDeviceName, psRemainingCapacityPercent, psTimeRemainingEstimated,
313                 psTimeRemainingInstant, psPowerUsageRate, psVoltage, psAmperage, psPowerOnLine, psCharging,
314                 psDischarging, psCapacityUnits, psCurrentCapacity, psMaxCapacity, psDesignCapacity, psCycleCount,
315                 psChemistry, psManufactureDate, psManufacturer, psSerialNumber, psTemperature);
316     }

318     private static String batteryQueryString(HANDLE hBattery, int tag, int infoLevel) {
319         BATTERY_QUERY_INFORMATION bqi = new PowrProf.BATTERY_QUERY_INFORMATION();
320         bqi.BatteryTag = tag;
321         bqi.InformationLevel = infoLevel;
322         bqi.write();
323         IntByReference dwOut = new IntByReference();
324         boolean ret = false;
325         long bufSize = 0;
326         Memory nameBuf;
327         do {
328             // First increment is probably enough
329             bufSize += 256;
330             nameBuf = new Memory(bufSize);
331             ret = Kernel32.INSTANCE.DeviceIoControl(hBattery, IOCTL_BATTERY_QUERY_INFORMATION, bqi.getPointer(),
332                     bqi.size(), nameBuf, (int) nameBuf.size(), dwOut, null);
333         } while (!ret && bufSize < 4096);
334         return CHAR_WIDTH > 1 ? nameBuf.getWideString(0) : nameBuf.getString(0);
335     }
