163     private static Map<LwpsInfoT, Integer> initLwpsOffsets() {
164         Map<LwpsInfoT, Integer> infoMap = new EnumMap<>(LwpsInfoT.class);
165         int offset = 0;
166         for (LwpsInfoT field : LwpsInfoT.values()) {
167             infoMap.put(field, offset);
168             offset += field.size;
169         }
170         return infoMap;
171     }

173     private static Map<PsInfoT, Integer> initPsOffsets() {
174         Map<PsInfoT, Integer> infoMap = new EnumMap<>(PsInfoT.class);
175         int offset = 0;
176         for (PsInfoT field : PsInfoT.values()) {
177             infoMap.put(field, offset);
178             offset += field.size;
179         }
180         return infoMap;

183     /**
184      * Reads the pr_argc, pr_argv, pr_envp, and pr_dmodel fields from
185      * /proc/pid/psinfo
186      *
187      * @param pid
188      *            The process ID
189      * @return A quartet containing the argc, argv, envp and dmodel values, or null
190      *         if unable to read
191      */
192     public static Quartet<Integer, Long, Long, Byte> queryArgsEnvAddrs(int pid) {
193         File procpsinfo = new File("/proc/" + pid + "/psinfo");
194         try (RandomAccessFile psinfo = new RandomAccessFile(procpsinfo, "r");
195                 FileChannel chan = psinfo.getChannel();
196                 ByteArrayOutputStream out = new ByteArrayOutputStream()) {
197             int bufferSize = psInfoOffsets.get(PsInfoT.SIZE);
198             if (bufferSize > chan.size()) {
199                 bufferSize = (int) chan.size();
200             }
201 
202             ByteBuffer buf = ByteBuffer.allocate(bufferSize);
203             if (chan.read(buf) > psInfoOffsets.get(PsInfoT.PR_DMODEL)) {
204                 // We've read bytes up to at least the dmodel
205                 if (IS_LITTLE_ENDIAN) {
206                     buf.order(ByteOrder.LITTLE_ENDIAN);
207                 }
208                 // Now grab the elements we want
209                 int argc = buf.getInt(psInfoOffsets.get(PsInfoT.PR_ARGC));
210                 // Must have at least one argc (the command itself) so failure here means exit
211                 if (argc > 0) {
212                     long argv = Native.POINTER_SIZE == 8 ? buf.getLong(psInfoOffsets.get(PsInfoT.PR_ARGV))
213                             : buf.getInt(psInfoOffsets.get(PsInfoT.PR_ARGV));
214                     long envp = Native.POINTER_SIZE == 8 ? buf.getLong(psInfoOffsets.get(PsInfoT.PR_ENVP))
215                             : buf.getInt(psInfoOffsets.get(PsInfoT.PR_ENVP));
216                     // Process data model 1 = 32 bit, 2 = 64 bit
217                     byte dmodel = buf.get(psInfoOffsets.get(PsInfoT.PR_DMODEL));
218                     // Sanity check
219                     if (dmodel * 4 != (envp - argv) / (argc + 1)) {
220                         LOG.trace("Failed data model and offset increment sanity check: dm={} diff={}", dmodel,
221                                 envp - argv);
222                         return null;
223                     }
224                     return new Quartet<>(argc, argv, envp, dmodel);
225                 }
226                 LOG.trace("No permission to read file: {} ", procpsinfo);
227             }
228         } catch (IOException e) {
229             LOG.debug("Failed to read file: {} ", procpsinfo);
230         }
231         return null;
232     }

234     /**
235      * Read the argument and environment strings from process address space
236      *
237      * @param pid
238      *            the process id
239      * @return A pair containing a list of the arguments and a map of environment
240      *         variables
241      */
242     public static Pair<List<String>, Map<String, String>> queryArgsEnv(int pid) {
243         List<String> args = new ArrayList<>();
244         Map<String, String> env = new LinkedHashMap<>();
245 
246         // Get the arg count and list of env vars
247         Quartet<Integer, Long, Long, Byte> addrs = queryArgsEnvAddrs(pid);
248         if (addrs != null) {
249             // Open a file descriptor to the address space
250             String procas = "/proc/" + pid + "/as";
251             int fd = LIBC.open(procas, 0);
252             if (fd < 0) {
253                 LOG.trace("No permission to read file: {} ", procas);
254                 return new Pair<>(args, env);
255             }
256             try {
257                 // Non-null addrs means argc > 0
258                 int argc = addrs.getA();
259                 long argv = addrs.getB();
260                 long envp = addrs.getC();
261                 long increment = addrs.getD() * 4L;
262 
263                 // Reusable buffer
264                 long bufStart = 0;
265                 Memory buffer = new Memory(PAGE_SIZE * 2);
266                 size_t bufSize = new size_t(buffer.size());
267 
268                 // Read the pointers to the arg strings
269                 // We know argc so we can count them
270                 long[] argp = new long[argc];
271                 long offset = argv;
272                 for (int i = 0; i < argc; i++) {
273                     bufStart = conditionallyReadBufferFromStartOfPage(fd, buffer, bufSize, bufStart, offset);
274                     argp[i] = bufStart == 0 ? 0 : getOffsetFromBuffer(buffer, offset - bufStart, increment);
275                     offset += increment;
276                 }
277 
278                 // Also read the pointers to the env strings
279                 // We don't know how many, so stop when we get to null pointer
280                 List<Long> envPtrList = new ArrayList<>();
281                 offset = envp;
282                 long addr = 0;
283                 int limit = 500; // sane max env strings to stop at
284                 do {
285                     bufStart = conditionallyReadBufferFromStartOfPage(fd, buffer, bufSize, bufStart, offset);
286                     addr = bufStart == 0 ? 0 : getOffsetFromBuffer(buffer, offset - bufStart, increment);
287                     if (addr != 0) {
288                         envPtrList.add(addr);
289                     }
290                     offset += increment;
291                 } while (addr != 0 && --limit > 0);
292 
293                 // Now read the arg strings from the buffer
294                 for (int i = 0; i < argp.length && argp[i] != 0; i++) {
295                     bufStart = conditionallyReadBufferFromStartOfPage(fd, buffer, bufSize, bufStart, argp[i]);
296                     if (bufStart != 0) {
297                         String argStr = buffer.getString(argp[i] - bufStart);
298                         if (!argStr.isEmpty()) {
299                             args.add(argStr);
300                         }
301                     }
302                 }
303 
304                 // And now read the env strings from the buffer
305                 for (Long envPtr : envPtrList) {
306                     bufStart = conditionallyReadBufferFromStartOfPage(fd, buffer, bufSize, bufStart, envPtr);
307                     if (bufStart != 0) {
308                         String envStr = buffer.getString(envPtr - bufStart);
309                         int idx = envStr.indexOf('=');
310                         if (idx > 0) {
311                             env.put(envStr.substring(0, idx), envStr.substring(idx + 1));
312                         }
313                     }
314                 }
315             } finally {
316                 LIBC.close(fd);
317             }
318         }
319         return new Pair<>(args, env);
320     }

322     /**
323      * Reads the page containing addr into buffer, unless the buffer already
324      * contains that page (as indicated by the bufStart address), in which case
325      * nothing is changed.
326      *
327      * @param fd
328      *            The file descriptor for the address space
329      * @param buffer
330      *            An allocated buffer, possibly with data reread from bufStart
331      * @param bufSize
332      *            The size of the buffer
333      * @param bufStart
334      *            The start of data currently in bufStart, or 0 if uninitialized
335      * @param addr
336      *            THe address whose page to read into the buffer
337      * @return The new starting pointer for the buffer
338      */
339     private static long conditionallyReadBufferFromStartOfPage(int fd, Memory buffer, size_t bufSize, long bufStart,
340             long addr) {
341         // If we don't have the right buffer, update it
342         if (addr < bufStart || addr - bufStart > PAGE_SIZE) {
343             long newStart = Math.floorDiv(addr, PAGE_SIZE) * PAGE_SIZE;
344             ssize_t result = LIBC.pread(fd, buffer, bufSize, new NativeLong(newStart));
345             // May return less than asked but should be at least a full page
346             if (result.longValue() < PAGE_SIZE) {
347                 LOG.debug("Failed to read page from address space: {} bytes read", result.longValue());
348                 return 0;
349             }
350             return newStart;
351         }
352         return bufStart;
353     }
354 
355     private static long getOffsetFromBuffer(Memory buffer, long offset, long increment) {
356         return increment == 8 ? buffer.getLong(offset) : buffer.getInt(offset);
357     }
