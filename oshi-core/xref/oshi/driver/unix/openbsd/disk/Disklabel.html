24  package oshi.driver.unix.openbsd.disk;
25  
26  import java.util.ArrayList;
27  import java.util.List;
28  
29  import oshi.annotation.concurrent.ThreadSafe;
30  import oshi.hardware.HWPartition;
31  import oshi.util.Constants;
32  import oshi.util.ExecutingCommand;
33  import oshi.util.ParseUtil;
34  import oshi.util.tuples.Pair;
35  import oshi.util.tuples.Quartet;
36  
37  /**
38   * Utility class parsing partition information from disklabel command
39   */
40  @ThreadSafe
41  public final class Disklabel {
42  
43      private Disklabel() {
44      }
45  
46      /**
47       * Gets disk and partition information
48       *
49       * @param diskName
50       *            The disk to fetch partition information from
51       * @return A quartet containing the disk's name/label, DUID, size, and a list of
52       *         partitions
53       */
54      public static Quartet<String, String, Long, List<HWPartition>> getDiskParams(String diskName) {
55          // disklabel (requires root) supports 15 configurable partitions, `a' through
56          // `p', excluding `c'.
57          // The `c' partition describes the entire physical disk.
58          // By convention, the `a' partition of the boot disk is the root
59          // partition, and the `b' partition of the boot disk is the swap partition,
60          // and the 'i' partition is usually the boot record
61  
62          // Create a list for all the other partitions
63          List<HWPartition> partitions = new ArrayList<>();
64          // Save some values to return to the caller to populate HWDiskStore values
65          String totalMarker = "total sectors:";
66          long totalSectors = 1L;
67          String bpsMarker = "bytes/sector:";
68          int bytesPerSector = 1;
69          String labelMarker = "label:";
70          String label = "";
71          String duidMarker = "duid:";
72          String duid = "";
73          for (String line : ExecutingCommand.runNative("disklabel -n " + diskName)) {
74              // Check for values in the header we need for the HWDiskstore
75              // # /dev/rsd1c:
76              // type: SCSI
77              // disk: SCSI disk
78              // label: Storage Device
79              // duid: 0000000000000000
80              // flags:
81              // bytes/sector: 512
82              // sectors/track: 63
83              // tracks/cylinder: 255
84              // sectors/cylinder: 16065
85              // cylinders: 976
86              // total sectors: 15693824
87              // boundstart: 0
88              // boundend: 15693824
89              // drivedata: 0
90              if (line.contains(totalMarker)) {
91                  totalSectors = ParseUtil.getFirstIntValue(line);
92              } else if (line.contains(bpsMarker)) {
93                  bytesPerSector = ParseUtil.getFirstIntValue(line);
94              } else if (line.contains(labelMarker)) {
95                  label = line.split(labelMarker)[1].trim();
96              } else if (line.contains(duidMarker)) {
97                  duid = line.split(duidMarker)[1].trim();
98              }
99              /*-
100             16 partitions:
101             #                size           offset  fstype [fsize bsize   cpg]
102               a:          2097152             1024  4.2BSD   2048 16384 12958 # /
103               b:         17023368          2098176    swap                    # none
104               c:        500118192                0  unused
105               d:          8388576         19121568  4.2BSD   2048 16384 12958 # /tmp
106               e:         41386752         27510144  4.2BSD   2048 16384 12958 # /var
107               f:          4194304         68896896  4.2BSD   2048 16384 12958 # /usr
108               g:          2097152         73091200  4.2BSD   2048 16384 12958 # /usr/X11R6
109               h:         20971520         75188352  4.2BSD   2048 16384 12958 # /usr/local
110               i:              960               64   MSDOS
111               j:          4194304         96159872  4.2BSD   2048 16384 12958 # /usr/src
112               k:         12582912        100354176  4.2BSD   2048 16384 12958 # /usr/obj
113               l:        387166336        112937088  4.2BSD   4096 32768 26062 # /home
114              Note size is in sectors
115              */
116             if (line.trim().indexOf(':') == 1) {
117                 // partition table values have a single letter followed by a colon
118                 String[] split = ParseUtil.whitespaces.split(line.trim(), 9);
119                 String name = split[0].substring(0, 1);
120                 // get major and minor from stat
121                 Pair<Integer, Integer> majorMinor = getMajorMinor(diskName, name);
122                 // add partitions
123                 if (split.length > 4) {
124                     partitions.add(new HWPartition(diskName + name, name, split[3], duid + "." + name,
125                             ParseUtil.parseLongOrDefault(split[1], 0L) * bytesPerSector, majorMinor.getA(),
126                             majorMinor.getB(), split.length > 5 ? split[split.length - 1] : ""));
127                 }
128             }
129         }
130         if (partitions.isEmpty()) {
131             return getDiskParamsNoRoot(diskName);
132         }
133         return new Quartet<>(label, duid, totalSectors * bytesPerSector, partitions);
134     }
135 
136     private static Quartet<String, String, Long, List<HWPartition>> getDiskParamsNoRoot(String diskName) {
137         List<HWPartition> partitions = new ArrayList<>();
138         for (String line : ExecutingCommand.runNative("df")) {
139             if (line.startsWith("/dev/" + diskName)) {
140                 String[] split = ParseUtil.whitespaces.split(line);
141                 String name = split[0].substring(5 + diskName.length());
142                 Pair<Integer, Integer> majorMinor = getMajorMinor(diskName, name);
143                 if (split.length > 5) {
144                     long partSize = ParseUtil.parseLongOrDefault(split[1], 1L) * 512L;
145                     partitions.add(new HWPartition(split[0], split[0].substring(5), Constants.UNKNOWN,
146                             Constants.UNKNOWN, partSize, majorMinor.getA(), majorMinor.getB(), split[5]));
147                 }
148             }
149         }
150         return new Quartet<>(Constants.UNKNOWN, Constants.UNKNOWN, 0L, partitions);
151     }
152 
153     private static Pair<Integer, Integer> getMajorMinor(String diskName, String name) {
154         int major = 0;
155         int minor = 0;
156         String majorMinor = ExecutingCommand.getFirstAnswer("stat -f %Hr,%Lr /dev/" + diskName + name);
157         int comma = majorMinor.indexOf(',');
158         if (comma > 0 && comma < majorMinor.length()) {
159             major = ParseUtil.parseIntOrDefault(majorMinor.substring(0, comma), 0);
160             minor = ParseUtil.parseIntOrDefault(majorMinor.substring(comma + 1), 0);
161         }
162         return new Pair<>(major, minor);
163     }
