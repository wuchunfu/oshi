77      /**
78       * Query the serial number from dmidecode
79       *
80       * @return The serial number if available, null otherwise
81       */
82      public static String querySerialNumber() {
83          // If root privileges this will work
84          String marker = "Serial Number:";
85          for (String checkLine : ExecutingCommand.runNative("dmidecode -t system")) {
86              if (checkLine.contains(marker)) {
87                  return checkLine.split(marker)[1].trim();
88              }
89          }
90          return null;
91      }
92  
93      /**
94       * Query the UUID from dmidecode
95       *
96       * @return The UUID if available, null otherwise
97       */
98      public static String queryUUID() {
99          // If root privileges this will work
100         String marker = "UUID:";
101         for (String checkLine : ExecutingCommand.runNative("dmidecode -t system")) {
102             if (checkLine.contains(marker)) {
103                 return checkLine.split(marker)[1].trim();
104             }
105         }
106         return null;
107     }
108 
109     /**
110      * Query the name and revision from dmidecode
111      *
112      * @return The a pair containing the name and revision if available, null values
113      *         in the pair otherwise
114      */
115     public static Pair<String, String> queryBiosNameRev() {
116         String biosName = null;
117         String revision = null;
118 
119         final String biosMarker = "SMBIOS";
120         final String revMarker = "Bios Revision:";
121 
122         // Requires root, may not return anything
123         for (final String checkLine : ExecutingCommand.runNative("dmidecode -t bios")) {
124             if (checkLine.contains(biosMarker)) {
125                 String[] biosArr = ParseUtil.whitespaces.split(checkLine);
126                 if (biosArr.length >= 2) {
127                     biosName = biosArr[0] + " " + biosArr[1];
128                 }
129             }
130             if (checkLine.contains(revMarker)) {
131                 revision = checkLine.split(revMarker)[1].trim();
132                 // SMBIOS should be first line so if we're here we are done iterating
133                 break;
134             }
135         }
136         return new Pair<>(biosName, revision);
137     }
