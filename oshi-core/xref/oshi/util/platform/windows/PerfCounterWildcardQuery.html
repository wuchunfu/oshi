50  /**
51   * Enables queries of Performance Counters using wild cards to filter instances
52   */
53  @ThreadSafe
54  public final class PerfCounterWildcardQuery {
55  
56      private static final Logger LOG = LoggerFactory.getLogger(PerfCounterWildcardQuery.class);
57  
58      // Use a thread safe set to cache failed pdh queries
59      private static final Set<String> FAILED_QUERY_CACHE = ConcurrentHashMap.newKeySet();
60  
61      private PerfCounterWildcardQuery() {
62      }
63  
64      /**
65       * Query the a Performance Counter using PDH, with WMI backup on failure, for
66       * values corresponding to the property enum.
67       *
68       * @param <T>
69       *            The enum type of {@code propertyEnum}
70       * @param propertyEnum
71       *            An enum which implements
72       *            {@link oshi.util.platform.windows.PerfCounterQuery.PdhCounterProperty}
73       *            and contains the WMI field (Enum value) and PDH Counter string
74       *            (instance and counter)
75       * @param perfObject
76       *            The PDH object for this counter; all counters on this object will
77       *            be refreshed at the same time
78       * @param perfWmiClass
79       *            The WMI PerfData_RawData_* class corresponding to the PDH object
80       * @return An pair containing a list of instances and an {@link EnumMap} of the
81       *         corresponding values indexed by {@code propertyEnum} on success, or
82       *         an empty list and empty map if both PDH and WMI queries failed.
83       */
84      public static <T extends Enum<T>> Pair<List<String>, Map<T, List<Long>>> queryInstancesAndValues(
85              Class<T> propertyEnum, String perfObject, String perfWmiClass) {
86          if (!FAILED_QUERY_CACHE.contains(perfObject)) {
87              Pair<List<String>, Map<T, List<Long>>> instancesAndValuesMap = queryInstancesAndValuesFromPDH(propertyEnum,
88                      perfObject);
89              if (!instancesAndValuesMap.getA().isEmpty()) {
90                  return instancesAndValuesMap;
91              }
92              // If we are here, query failed
93              LOG.warn("Disabling further attempts to query {}.", perfObject);
94              FAILED_QUERY_CACHE.add(perfObject);
95          }
96          return queryInstancesAndValuesFromWMI(propertyEnum, perfWmiClass);
97      }
98  
99      /**
100      * Query the a Performance Counter using PDH for values corresponding to the
101      * property enum.
102      *
103      * @param <T>
104      *            The enum type of {@code propertyEnum}
105      * @param propertyEnum
106      *            An enum which implements
107      *            {@link oshi.util.platform.windows.PerfCounterQuery.PdhCounterProperty}
108      *            and contains the WMI field (Enum value) and PDH Counter string
109      *            (instance and counter)
110      * @param perfObject
111      *            The PDH object for this counter; all counters on this object will
112      *            be refreshed at the same time
113      * @return An pair containing a list of instances and an {@link EnumMap} of the
114      *         corresponding values indexed by {@code propertyEnum} on success, or
115      *         an empty list and empty map if the PDH query failed.
116      */
117     public static <T extends Enum<T>> Pair<List<String>, Map<T, List<Long>>> queryInstancesAndValuesFromPDH(
118             Class<T> propertyEnum, String perfObject) {
119         T[] props = propertyEnum.getEnumConstants();
120         if (props.length < 2) {
121             throw new IllegalArgumentException("Enum " + propertyEnum.getName()
122                     + " must have at least two elements, an instance filter and a counter.");
123         }
124         String instanceFilter = ((PdhCounterWildcardProperty) propertyEnum.getEnumConstants()[0]).getCounter()
125                 .toLowerCase();
126         // If pre-Vista, localize the perfObject
127         String perfObjectLocalized = PerfCounterQuery.localizeIfNeeded(perfObject);
128 
129         // Get list of instances
130         final PdhEnumObjectItems objectItems;
131         try {
132             objectItems = PdhUtil.PdhEnumObjectItems(null, null, perfObjectLocalized, 100);
133         } catch (PdhException e) {
134             return new Pair<>(Collections.emptyList(), Collections.emptyMap());
135         }
136         List<String> instances = objectItems.getInstances();
137         // Filter out instances not matching filter
138         instances.removeIf(i -> !Util.wildcardMatch(i.toLowerCase(), instanceFilter));
139         EnumMap<T, List<Long>> valuesMap = new EnumMap<>(propertyEnum);
140         try (PerfCounterQueryHandler pdhQueryHandler = new PerfCounterQueryHandler()) {
141             // Set up the query and counter handles
142             EnumMap<T, List<PerfCounter>> counterListMap = new EnumMap<>(propertyEnum);
143             // Start at 1, first counter defines instance filter
144             for (int i = 1; i < props.length; i++) {
145                 T prop = props[i];
146                 List<PerfCounter> counterList = new ArrayList<>(instances.size());
147                 for (String instance : instances) {
148                     PerfCounter counter = PerfDataUtil.createCounter(perfObject, instance,
149                             ((PdhCounterWildcardProperty) prop).getCounter());
150                     if (!pdhQueryHandler.addCounterToQuery(counter)) {
151                         return new Pair<>(Collections.emptyList(), Collections.emptyMap());
152                     }
153                     counterList.add(counter);
154                 }
155                 counterListMap.put(prop, counterList);
156             }
157             // And then query. Zero timestamp means update failed
158             if (0 < pdhQueryHandler.updateQuery()) {
159                 // Start at 1, first counter defines instance filter
160                 for (int i = 1; i < props.length; i++) {
161                     T prop = props[i];
162                     List<Long> values = new ArrayList<>();
163                     for (PerfCounter counter : counterListMap.get(prop)) {
164                         values.add(pdhQueryHandler.queryCounter(counter));
165                     }
166                     valuesMap.put(prop, values);
167                 }
168             }
169         }
170         return new Pair<>(instances, valuesMap);
171     }

173     /**
174      * Query the a Performance Counter using WMI for values corresponding to the
175      * property enum.
176      *
177      * @param <T>
178      *            The enum type of {@code propertyEnum}
179      * @param propertyEnum
180      *            An enum which implements
181      *            {@link oshi.util.platform.windows.PerfCounterQuery.PdhCounterProperty}
182      *            and contains the WMI field (Enum value) and PDH Counter string
183      *            (instance and counter)
184      * @param wmiClass
185      *            The WMI PerfData_RawData_* class corresponding to the PDH object
186      * @return An pair containing a list of instances and an {@link EnumMap} of the
187      *         corresponding values indexed by {@code propertyEnum} on success, or
188      *         an empty list and empty map if the WMI query failed.
189      */
190     public static <T extends Enum<T>> Pair<List<String>, Map<T, List<Long>>> queryInstancesAndValuesFromWMI(
191             Class<T> propertyEnum, String wmiClass) {
192         List<String> instances = new ArrayList<>();
193         EnumMap<T, List<Long>> valuesMap = new EnumMap<>(propertyEnum);
194         WmiQuery<T> query = new WmiQuery<>(wmiClass, propertyEnum);
195         WmiResult<T> result = Objects.requireNonNull(WmiQueryHandler.createInstance()).queryWMI(query);
196         if (result.getResultCount() > 0) {
197             for (T prop : propertyEnum.getEnumConstants()) {
198                 // First element is instance name
199                 if (prop.ordinal() == 0) {
200                     for (int i = 0; i < result.getResultCount(); i++) {
201                         instances.add(WmiUtil.getString(result, prop, i));
202                     }
203                 } else {
204                     List<Long> values = new ArrayList<>();
205                     for (int i = 0; i < result.getResultCount(); i++) {
206                         switch (result.getCIMType(prop)) {
207                         case Wbemcli.CIM_UINT16:
208                             values.add((long) WmiUtil.getUint16(result, prop, i));
209                             break;
210                         case Wbemcli.CIM_UINT32:
211                             values.add(WmiUtil.getUint32asLong(result, prop, i));
212                             break;
213                         case Wbemcli.CIM_UINT64:
214                             values.add(WmiUtil.getUint64(result, prop, i));
215                             break;
216                         case Wbemcli.CIM_DATETIME:
217                             values.add(WmiUtil.getDateTime(result, prop, i).toInstant().toEpochMilli());
218                             break;
219                         default:
220                             throw new ClassCastException("Unimplemented CIM Type Mapping.");
221                         }
222                     }
223                     valuesMap.put(prop, values);
224                 }
225             }
226         }
227         return new Pair<>(instances, valuesMap);
228     }

230     /**
231      * Contract for Counter Property Enums
232      */
233     public interface PdhCounterWildcardProperty {
234         /**
235          * @return Returns the counter. The first element of the enum will return the
236          *         instance filter rather than a counter.
237          */
238         String getCounter();
239     }
