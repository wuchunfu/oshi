50  /**
51   * Provides access to SMC calls on macOS
52   */
53  @ThreadSafe
54  public final class SmcUtil {
55  
56      private static final Logger LOG = LoggerFactory.getLogger(SmcUtil.class);
57  
58      private static final IOKit IO = IOKit.INSTANCE;
59  
60      /**
61       * Thread-safe map for caching info retrieved by a key necessary for subsequent
62       * calls.
63       */
64      private static Map<Integer, SMCKeyDataKeyInfo> keyInfoCache = new ConcurrentHashMap<>();
65  
66      /**
67       * Byte array used for matching return type
68       */
69      private static final byte[] DATATYPE_SP78 = ParseUtil.asciiStringToByteArray("sp78", 5);
70      private static final byte[] DATATYPE_FPE2 = ParseUtil.asciiStringToByteArray("fpe2", 5);
71      private static final byte[] DATATYPE_FLT = ParseUtil.asciiStringToByteArray("flt ", 5);
72  
73      public static final String SMC_KEY_FAN_NUM = "FNum";
74      public static final String SMC_KEY_FAN_SPEED = "F%dAc";
75      public static final String SMC_KEY_CPU_TEMP = "TC0P";
76      public static final String SMC_KEY_CPU_VOLTAGE = "VC0C";
77  
78      public static final byte SMC_CMD_READ_BYTES = 5;
79      public static final byte SMC_CMD_READ_KEYINFO = 9;
80      public static final int KERNEL_INDEX_SMC = 2;
81  
82      private SmcUtil() {
83      }
84  
85      /**
86       * Open a connection to SMC.
87       *
88       * @return The connection if successful, null if failure
89       */
90      public static IOConnect smcOpen() {
91          IOService smcService = IOKitUtil.getMatchingService("AppleSMC");
92          if (smcService != null) {
93              PointerByReference connPtr = new PointerByReference();
94              int result = IO.IOServiceOpen(smcService, SystemB.INSTANCE.mach_task_self(), 0, connPtr);
95              smcService.release();
96              if (result == 0) {
97                  return new IOConnect(connPtr.getValue());
98              } else if (LOG.isErrorEnabled()) {
99                  LOG.error(String.format("Unable to open connection to AppleSMC service. Error: 0x%08x", result));
100             }
101         } else {
102             LOG.error("Unable to locate AppleSMC service");
103         }
104         return null;
105     }

107     /**
108      * Close connection to SMC.
109      *
110      * @param conn
111      *            The connection
112      *
113      * @return 0 if successful, nonzero if failure
114      */
115     public static int smcClose(IOConnect conn) {
116         return IO.IOServiceClose(conn);
117     }
118 
119     /**
120      * Get a value from SMC which is in a floating point datatype (SP78, FPE2, FLT)
121      *
122      * @param conn
123      *            The connection
124      * @param key
125      *            The key to retrieve
126      * @return Double representing the value
127      */
128     public static double smcGetFloat(IOConnect conn, String key) {
129         SMCVal val = new SMCVal();
130         int result = smcReadKey(conn, key, val);
131         if (result == 0 && val.dataSize > 0) {
132             if (Arrays.equals(val.dataType, DATATYPE_SP78) && val.dataSize == 2) {
133                 // First bit is sign, next 7 bits are integer portion, last 8 bits are
134                 // fractional portion
135                 return val.bytes[0] + val.bytes[1] / 256d;
136             } else if (Arrays.equals(val.dataType, DATATYPE_FPE2) && val.dataSize == 2) {
137                 // First E (14) bits are integer portion last 2 bits are fractional portion
138                 return ParseUtil.byteArrayToFloat(val.bytes, val.dataSize, 2);
139             } else if (Arrays.equals(val.dataType, DATATYPE_FLT) && val.dataSize == 4) {
140                 // Standard 32-bit floating point
141                 return ByteBuffer.wrap(val.bytes).order(ByteOrder.LITTLE_ENDIAN).getFloat();
142             }
143         }
144         // Read failed
145         return 0d;
146     }
147 
148     /**
149      * Get a 64-bit integer value from SMC
150      *
151      * @param conn
152      *            The connection
153      * @param key
154      *            The key to retrieve
155      * @return Long representing the value
156      */
157     public static long smcGetLong(IOConnect conn, String key) {
158         SMCVal val = new SMCVal();
159         int result = smcReadKey(conn, key, val);
160         if (result == 0) {
161             return ParseUtil.byteArrayToLong(val.bytes, val.dataSize);
162         }
163         // Read failed
164         return 0;
165     }
166 
167     /**
168      * Get cached keyInfo if it exists, or generate new keyInfo
169      *
170      * @param conn
171      *            The connection
172      * @param inputStructure
173      *            Key data input
174      * @param outputStructure
175      *            Key data output
176      * @return 0 if successful, nonzero if failure
177      */
178     public static int smcGetKeyInfo(IOConnect conn, SMCKeyData inputStructure, SMCKeyData outputStructure) {
179         if (keyInfoCache.containsKey(inputStructure.key)) {
180             SMCKeyDataKeyInfo keyInfo = keyInfoCache.get(inputStructure.key);
181             outputStructure.keyInfo.dataSize = keyInfo.dataSize;
182             outputStructure.keyInfo.dataType = keyInfo.dataType;
183             outputStructure.keyInfo.dataAttributes = keyInfo.dataAttributes;
184         } else {
185             inputStructure.data8 = SMC_CMD_READ_KEYINFO;
186             int result = smcCall(conn, KERNEL_INDEX_SMC, inputStructure, outputStructure);
187             if (result != 0) {
188                 return result;
189             }
190             SMCKeyDataKeyInfo keyInfo = new SMCKeyDataKeyInfo();
191             keyInfo.dataSize = outputStructure.keyInfo.dataSize;
192             keyInfo.dataType = outputStructure.keyInfo.dataType;
193             keyInfo.dataAttributes = outputStructure.keyInfo.dataAttributes;
194             keyInfoCache.put(inputStructure.key, keyInfo);
195         }
196         return 0;
197     }
198 
199     /**
200      * Read a key from SMC
201      *
202      * @param conn
203      *            The connection
204      * @param key
205      *            Key to read
206      * @param val
207      *            Structure to receive the result
208      * @return 0 if successful, nonzero if failure
209      */
210     public static int smcReadKey(IOConnect conn, String key, SMCVal val) {
211         SMCKeyData inputStructure = new SMCKeyData();
212         SMCKeyData outputStructure = new SMCKeyData();
213 
214         inputStructure.key = (int) ParseUtil.strToLong(key, 4);
215         int result = smcGetKeyInfo(conn, inputStructure, outputStructure);
216         if (result == 0) {
217             val.dataSize = outputStructure.keyInfo.dataSize;
218             val.dataType = ParseUtil.longToByteArray(outputStructure.keyInfo.dataType, 4, 5);
219 
220             inputStructure.keyInfo.dataSize = val.dataSize;
221             inputStructure.data8 = SMC_CMD_READ_BYTES;
222 
223             result = smcCall(conn, KERNEL_INDEX_SMC, inputStructure, outputStructure);
224             if (result == 0) {
225                 System.arraycopy(outputStructure.bytes, 0, val.bytes, 0, val.bytes.length);
226                 return 0;
227             }
228         }
229         return result;
230     }

232     /**
233      * Call SMC
234      *
235      * @param conn
236      *            The connection
237      * @param index
238      *            Kernel index
239      * @param inputStructure
240      *            Key data input
241      * @param outputStructure
242      *            Key data output
243      * @return 0 if successful, nonzero if failure
244      */
245     public static int smcCall(IOConnect conn, int index, SMCKeyData inputStructure, SMCKeyData outputStructure) {
246         return IO.IOConnectCallStructMethod(conn, index, inputStructure, new NativeLong(inputStructure.size()),
247                 outputStructure, new NativeLongByReference(new NativeLong(outputStructure.size())));
248     }
