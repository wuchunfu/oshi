79      private static final int ARGMAX;
80      static {
81          int[] mib = new int[2];
82          mib[0] = 1; // CTL_KERN
83          mib[1] = 8; // KERN_ARGMAX
84          Memory m = new Memory(Integer.BYTES);
85          size_t.ByReference size = new size_t.ByReference(new size_t(Integer.BYTES));
86          if (OpenBsdLibc.INSTANCE.sysctl(mib, mib.length, m, size, null, size_t.ZERO) == 0) {
87              ARGMAX = m.getInt(0);
88          } else {
89              LOG.warn("Failed sysctl call for process arguments max size (kern.argmax). Error code: {}",
90                      Native.getLastError());
91              ARGMAX = 0;
92          }
93      }

123     public OpenBsdOSProcess(int pid, Map<PsKeywords, String> psMap) {
124         super(pid);
125         // OpenBSD does not maintain a compatibility layer.
126         // Process bitness is OS bitness
127         this.bitness = Native.LONG_SIZE * 8;
128         updateThreadCount();
129         updateAttributes(psMap);
130     }

157     private List<String> queryArguments() {
158         if (ARGMAX > 0) {
159             // Get arguments via sysctl(3)
160             int[] mib = new int[4];
161             mib[0] = 1; // CTL_KERN
162             mib[1] = 55; // KERN_PROC_ARGS
163             mib[2] = getProcessID();
164             mib[3] = 1; // KERN_PROC_ARGV
165             // Allocate memory for arguments
166             Memory m = new Memory(ARGMAX);
167             size_t.ByReference size = new size_t.ByReference(new size_t(ARGMAX));
168             // Fetch arguments
169             if (OpenBsdLibc.INSTANCE.sysctl(mib, mib.length, m, size, null, size_t.ZERO) == 0) {
170                 // Returns a null-terminated list of pointers to the actual data
171                 List<String> args = new ArrayList<>();
172                 // To iterate the pointer-list
173                 long offset = 0;
174                 // Get the data base address to calculate offsets
175                 long baseAddr = Pointer.nativeValue(m);
176                 long maxAddr = baseAddr + size.getValue().longValue();
177                 // Get the address of the data. If null (0) we're done iterating
178                 long argAddr = Pointer.nativeValue(m.getPointer(offset));
179                 while (argAddr > baseAddr && argAddr < maxAddr) {
180                     args.add(m.getString(argAddr - baseAddr));
181                     offset += Native.POINTER_SIZE;
182                     argAddr = Pointer.nativeValue(m.getPointer(offset));
183                 }
184                 return Collections.unmodifiableList(args);
185             }
186         }
187         return Collections.emptyList();
188     }

195     private Map<String, String> queryEnvironmentVariables() {
196         // Get environment variables via sysctl(3)
197         int[] mib = new int[4];
198         mib[0] = 1; // CTL_KERN
199         mib[1] = 55; // KERN_PROC_ARGS
200         mib[2] = getProcessID();
201         mib[3] = 3; // KERN_PROC_ENV
202         // Allocate memory for environment variables
203         Memory m = new Memory(ARGMAX);
204         size_t.ByReference size = new size_t.ByReference(new size_t(ARGMAX));
205         // Fetch environment variables
206         if (OpenBsdLibc.INSTANCE.sysctl(mib, mib.length, m, size, null, size_t.ZERO) == 0) {
207             // Returns a null-terminated list of pointers to the actual data
208             Map<String, String> env = new LinkedHashMap<>();
209             // To iterate the pointer-list
210             long offset = 0;
211             // Get the data base address to calculate offsets
212             long baseAddr = Pointer.nativeValue(m);
213             long maxAddr = baseAddr + size.longValue();
214             // Get the address of the data. If null (0) we're done iterating
215             long argAddr = Pointer.nativeValue(m.getPointer(offset));
216             while (argAddr > baseAddr && argAddr < maxAddr) {
217                 String envStr = m.getString(argAddr - baseAddr);
218                 int idx = envStr.indexOf('=');
219                 if (idx > 0) {
220                     env.put(envStr.substring(0, idx), envStr.substring(idx + 1));
221                 }
222                 offset += Native.POINTER_SIZE;
223                 argAddr = Pointer.nativeValue(m.getPointer(offset));
224             }
225             return Collections.unmodifiableMap(env);
226         }
227         return Collections.emptyMap();
228     }

325     @Override
326     public long getAffinityMask() {
327         long bitMask = 0L;
328         // Would prefer to use native cpuset_getaffinity call but variable sizing is
329         // kernel-dependent and requires C macros, so we use commandline instead.
330         String cpuset = ExecutingCommand.getFirstAnswer("cpuset -gp " + getProcessID());
331         // Sample output:
332         // pid 8 mask: 0, 1
333         // cpuset: getaffinity: No such process
334         String[] split = cpuset.split(":");
335         if (split.length > 1) {
336             String[] bits = split[1].split(",");
337             for (String bit : bits) {
338                 int bitToSet = ParseUtil.parseIntOrDefault(bit.trim(), -1);
339                 if (bitToSet >= 0) {
340                     bitMask |= 1L << bitToSet;
341                 }
342             }
343         }
344         return bitMask;
345     }

347     @Override
348     public List<OSThread> getThreadDetails() {
349         List<OSThread> threads = new ArrayList<>();
350         String psCommand = "ps -aHwwxo " + PS_THREAD_COLUMNS;
351         if (getProcessID() >= 0) {
352             psCommand += " -p " + getProcessID();
353         }
354         List<String> threadList = ExecutingCommand.runNative(psCommand);
355         if (threadList.isEmpty() || threadList.size() < 2) {
356             return threads;
357         }
358         // remove header row
359         threadList.remove(0);
360         // Fill list
361         for (String thread : threadList) {
362             Map<PsThreadColumns, String> threadMap = ParseUtil.stringToEnumMap(PsThreadColumns.class, thread.trim(),
363                     ' ');
364             if (threadMap.containsKey(PsThreadColumns.ARGS)) {
365                 threads.add(new OpenBsdOSThread(getProcessID(), threadMap));
366             }
367         }
368         return threads;
369     }

386     @Override
387     public boolean updateAttributes() {
388         // 'ps' does not provide threadCount or kernelTime on OpenBSD
389         String psCommand = "ps -awwxo " + OpenBsdOperatingSystem.PS_COMMAND_ARGS + " -p " + getProcessID();
390         List<String> procList = ExecutingCommand.runNative(psCommand);
391         if (procList.size() > 1) {
392             // skip header row
393             Map<PsKeywords, String> psMap = ParseUtil.stringToEnumMap(PsKeywords.class, procList.get(1).trim(), ' ');
394             // Check if last (thus all) value populated
395             if (psMap.containsKey(PsKeywords.ARGS)) {
396                 updateThreadCount();
397                 return updateAttributes(psMap);
398             }
399         }
400         this.state = INVALID;
401         return false;
402     }
403 
404     private boolean updateAttributes(Map<PsKeywords, String> psMap) {
405         long now = System.currentTimeMillis();
406         switch (psMap.get(PsKeywords.STATE).charAt(0)) {
407         case 'R':
408             this.state = RUNNING;
409             break;
410         case 'I':
411         case 'S':
412             this.state = SLEEPING;
413             break;
414         case 'D':
415         case 'L':
416         case 'U':
417             this.state = WAITING;
418             break;
419         case 'Z':
420             this.state = ZOMBIE;
421             break;
422         case 'T':
423             this.state = STOPPED;
424             break;
425         default:
426             this.state = OTHER;
427             break;
428         }
429         this.parentProcessID = ParseUtil.parseIntOrDefault(psMap.get(PsKeywords.PPID), 0);
430         this.user = psMap.get(PsKeywords.USER);
431         this.userID = psMap.get(PsKeywords.UID);
432         this.group = psMap.get(PsKeywords.GROUP);
433         this.groupID = psMap.get(PsKeywords.GID);
434         this.priority = ParseUtil.parseIntOrDefault(psMap.get(PsKeywords.PRI), 0);
435         // These are in KB, multiply
436         this.virtualSize = ParseUtil.parseLongOrDefault(psMap.get(PsKeywords.VSZ), 0) * 1024;
437         this.residentSetSize = ParseUtil.parseLongOrDefault(psMap.get(PsKeywords.RSS), 0) * 1024;
438         // Avoid divide by zero for processes up less than a second
439         long elapsedTime = ParseUtil.parseDHMSOrDefault(psMap.get(PsKeywords.ETIME), 0L);
440         this.upTime = elapsedTime < 1L ? 1L : elapsedTime;
441         this.startTime = now - this.upTime;
442         this.userTime = ParseUtil.parseDHMSOrDefault(psMap.get(PsKeywords.CPUTIME), 0L);
443         // kernel time is included in user time
444         this.kernelTime = 0L;
445         this.path = psMap.get(PsKeywords.COMM);
446         this.name = this.path.substring(this.path.lastIndexOf('/') + 1);
447         this.minorFaults = ParseUtil.parseLongOrDefault(psMap.get(PsKeywords.MINFLT), 0L);
448         this.majorFaults = ParseUtil.parseLongOrDefault(psMap.get(PsKeywords.MAJFLT), 0L);
449         long nonVoluntaryContextSwitches = ParseUtil.parseLongOrDefault(psMap.get(PsKeywords.NIVCSW), 0L);
450         long voluntaryContextSwitches = ParseUtil.parseLongOrDefault(psMap.get(PsKeywords.NVCSW), 0L);
451         this.contextSwitches = voluntaryContextSwitches + nonVoluntaryContextSwitches;
452         this.commandLineBackup = psMap.get(PsKeywords.ARGS);
453         return true;
454     }
455 
456     private void updateThreadCount() {
457         List<String> threadList = ExecutingCommand.runNative("ps -axHo tid -p " + getProcessID());
458         if (!threadList.isEmpty()) {
459             // Subtract 1 for header
460             this.threadCount = threadList.size() - 1;
461         }
462         this.threadCount = 1;
463     }
