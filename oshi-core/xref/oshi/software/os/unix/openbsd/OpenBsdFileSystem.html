41  /**
42   * The FreeBSD File System contains {@link oshi.software.os.OSFileStore}s which
43   * are a storage pool, device, partition, volume, concrete file system or other
44   * implementation specific means of file storage.
45   */
46  @ThreadSafe
47  public class OpenBsdFileSystem extends AbstractFileSystem {
48  
49      public static final String OSHI_OPENBSD_FS_PATH_EXCLUDES = "oshi.os.openbsd.filesystem.path.excludes";
50      public static final String OSHI_OPENBSD_FS_PATH_INCLUDES = "oshi.os.openbsd.filesystem.path.includes";
51      public static final String OSHI_OPENBSD_FS_VOLUME_EXCLUDES = "oshi.os.openbsd.filesystem.volume.excludes";
52      public static final String OSHI_OPENBSD_FS_VOLUME_INCLUDES = "oshi.os.openbsd.filesystem.volume.includes";
53  
54      private static final List<PathMatcher> FS_PATH_EXCLUDES = FileSystemUtil
55              .loadAndParseFileSystemConfig(OSHI_OPENBSD_FS_PATH_EXCLUDES);
56      private static final List<PathMatcher> FS_PATH_INCLUDES = FileSystemUtil
57              .loadAndParseFileSystemConfig(OSHI_OPENBSD_FS_PATH_INCLUDES);
58      private static final List<PathMatcher> FS_VOLUME_EXCLUDES = FileSystemUtil
59              .loadAndParseFileSystemConfig(OSHI_OPENBSD_FS_VOLUME_EXCLUDES);
60      private static final List<PathMatcher> FS_VOLUME_INCLUDES = FileSystemUtil
61              .loadAndParseFileSystemConfig(OSHI_OPENBSD_FS_VOLUME_INCLUDES);
62  
63      @Override
64      public List<OSFileStore> getFileStores(boolean localOnly) {
65          return getFileStoreMatching(null, localOnly);
66      }
67  
68      // Called by OpenBsdOSFileStore
69      static List<OSFileStore> getFileStoreMatching(String nameToMatch) {
70          return getFileStoreMatching(nameToMatch, false);
71      }
72  
73      private static List<OSFileStore> getFileStoreMatching(String nameToMatch, boolean localOnly) {
74          List<OSFileStore> fsList = new ArrayList<>();
75  
76          // Get inode usage data
77          Map<String, Long> inodeFreeMap = new HashMap<>();
78          Map<String, Long> inodeUsedlMap = new HashMap<>();
79          String command = "df -i" + (localOnly ? " -l" : "");
80          for (String line : ExecutingCommand.runNative(command)) {
81              /*- Sample Output:
82               $ df -i
83              Filesystem  512-blocks      Used     Avail Capacity iused   ifree  %iused  Mounted on
84              /dev/wd0a      2149212    908676   1133076    45%    8355  147163     5%   /
85              /dev/wd0e      4050876        36   3848300     0%      10  285108     0%   /home
86              /dev/wd0d      6082908   3343172   2435592    58%   27813  386905     7%   /usr
87              */
88              if (line.startsWith("/")) {
89                  String[] split = ParseUtil.whitespaces.split(line);
90                  if (split.length > 6) {
91                      inodeUsedlMap.put(split[0], ParseUtil.parseLongOrDefault(split[5], 0L));
92                      inodeFreeMap.put(split[0], ParseUtil.parseLongOrDefault(split[6], 0L));
93                  }
94              }
95          }
96  
97          // Get mount table
98          for (String fs : ExecutingCommand.runNative("mount -v")) { // NOSONAR squid:S135
99              /*-
100              Sample Output:
101              /dev/wd0a (d1c342b6965d372c.a) on / type ffs (rw, local, ctime=Sun Jan  3 18:03:00 2021)
102              /dev/wd0e (d1c342b6965d372c.e) on /home type ffs (rw, local, nodevl, nosuid, ctime=Sun Jan  3 18:02:56 2021)
103              /dev/wd0d (d1c342b6965d372c.d) on /usr type ffs (rw, local, nodev, wxallowed, ctime=Sun Jan  3 18:02:56 2021)
104              */
105             String[] split = ParseUtil.whitespaces.split(fs, 7);
106             if (split.length == 7) {
107                 // 1st field is volume name [0-index] + partition letter
108                 // 2nd field is disklabel UUID (DUID) + partition letter after the dot
109                 // 4th field is mount point
110                 // 6rd field is fs type
111                 // 7th field is options
112                 String volume = split[0];
113                 String uuid = split[1];
114                 String path = split[3];
115                 String type = split[5];
116                 String options = split[6];
117 
118                 // Skip non-local drives if requested, and exclude pseudo file systems
119                 if ((localOnly && NETWORK_FS_TYPES.contains(type)) || !path.equals("/")
120                         && (PSEUDO_FS_TYPES.contains(type) || FileSystemUtil.isFileStoreExcluded(path, volume,
121                                 FS_PATH_INCLUDES, FS_PATH_EXCLUDES, FS_VOLUME_INCLUDES, FS_VOLUME_EXCLUDES))) {
122                     continue;
123                 }
124 
125                 String name = path.substring(path.lastIndexOf('/') + 1);
126                 // Special case for /, pull last element of volume instead
127                 if (name.isEmpty()) {
128                     name = volume.substring(volume.lastIndexOf('/') + 1);
129                 }
130 
131                 if (nameToMatch != null && !nameToMatch.equals(name)) {
132                     continue;
133                 }
134                 File f = new File(path);
135                 long totalSpace = f.getTotalSpace();
136                 long usableSpace = f.getUsableSpace();
137                 long freeSpace = f.getFreeSpace();
138 
139                 String description;
140                 if (volume.startsWith("/dev") || path.equals("/")) {
141                     description = "Local Disk";
142                 } else if (volume.equals("tmpfs")) {
143                     // dynamic size in memory FS
144                     description = "Ram Disk (dynamic)";
145                 } else if (volume.equals("mfs")) {
146                     // fixed size in memory FS
147                     description = "Ram Disk (fixed)";
148                 } else if (NETWORK_FS_TYPES.contains(type)) {
149                     description = "Network Disk";
150                 } else {
151                     description = "Mount Point";
152                 }
153 
154                 fsList.add(new OpenBsdOSFileStore(name, volume, name, path, options, uuid, "", description, type,
155                         freeSpace, usableSpace, totalSpace, inodeFreeMap.getOrDefault(volume, 0L),
156                         inodeUsedlMap.getOrDefault(volume, 0L) + inodeFreeMap.getOrDefault(volume, 0L)));
157             }
158         }
159         return fsList;
160     }
161 
162     @Override
163     public long getOpenFileDescriptors() {
164         return OpenBsdSysctlUtil.sysctl("kern.nfiles", 0);
165     }
166 
167     @Override
168     public long getMaxFileDescriptors() {
169         return OpenBsdSysctlUtil.sysctl("kern.maxfiles", 0);
170     }
