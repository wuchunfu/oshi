80      enum PsKeywords {
81          ST, PID, PPID, USER, UID, GROUP, GID, THCOUNT, PRI, VSIZE, RSSIZE, ETIME, TIME, COMM, PAGEIN, ARGS;
82          // ARGS must always be last
83      }

86              .collect(Collectors.joining(","));

93      @Override
94      public Pair<String, OSVersionInfo> queryFamilyVersionInfo() {
95          perfstat_partition_config_t cfg = config.get();
96  
97          String systemName = System.getProperty("os.name");
98          String archName = System.getProperty("os.arch");
99          String versionNumber = System.getProperty("os.version");
100         if (Util.isBlank(versionNumber)) {
101             versionNumber = ExecutingCommand.getFirstAnswer("oslevel");
102         }
103         String releaseNumber = Native.toString(cfg.OSBuild);
104         if (Util.isBlank(releaseNumber)) {
105             releaseNumber = ExecutingCommand.getFirstAnswer("oslevel -s");
106         } else {
107             // strip leading date
108             int idx = releaseNumber.lastIndexOf(' ');
109             if (idx > 0 && idx < releaseNumber.length()) {
110                 releaseNumber = releaseNumber.substring(idx + 1);
111             }
112         }
113         return new Pair<>(systemName, new OSVersionInfo(versionNumber, archName, releaseNumber));
114     }

116     @Override
117     protected int queryBitness(int jvmBitness) {
118         if (jvmBitness == 64) {
119             return 64;
120         }
121         // 9th bit of conf is 64-bit kernel
122         return (config.get().conf & 0x0080_0000) > 0 ? 64 : 32;
123     }

140     @Override
141     public List<OSProcess> queryChildProcesses(int parentPid) {
142         List<OSProcess> allProcs = queryAllProcesses();
143         Set<Integer> descendantPids = getChildrenOrDescendants(allProcs, parentPid, false);
144         return allProcs.stream().filter(p -> descendantPids.contains(p.getProcessID())).collect(Collectors.toList());
145     }

154     @Override
155     public OSProcess getProcess(int pid) {
156         List<OSProcess> procs = getProcessListFromPS("ps -o " + PS_COMMAND_ARGS + " -p ", pid);
157         if (procs.isEmpty()) {
158             return null;
159         }
160         return procs.get(0);
161     }
162 
163     private List<OSProcess> getProcessListFromPS(String psCommand, int pid) {
164         perfstat_process_t[] perfstat = procCpu.get();
165         List<String> procList = ExecutingCommand.runNative(psCommand + (pid < 0 ? "" : pid));
166         if (procList.isEmpty() || procList.size() < 2) {
167             return Collections.emptyList();
168         }
169         // Parse array to map of user/system times
170         Map<Integer, Pair<Long, Long>> cpuMap = new HashMap<>();
171         for (perfstat_process_t stat : perfstat) {
172             cpuMap.put((int) stat.pid, new Pair<>((long) stat.ucpu_time, (long) stat.scpu_time));
173         }
174         // remove header row
175         procList.remove(0);
176         // Fill list
177         List<OSProcess> procs = new ArrayList<>();
178         for (String proc : procList) {
179             Map<PsKeywords, String> psMap = ParseUtil.stringToEnumMap(PsKeywords.class, proc.trim(), ' ');
180             // Check if last (thus all) value populated
181             if (psMap.containsKey(PsKeywords.ARGS)) {
182                 procs.add(new AixOSProcess(pid < 0 ? ParseUtil.parseIntOrDefault(psMap.get(PsKeywords.PID), 0) : pid,
183                         psMap, cpuMap, procCpu));
184             }
185         }
186         return procs;
187     }

218     private static long querySystemBootTimeMillis() {
219         long bootTime = Who.queryBootTime();
220         if (bootTime >= 1000L) {
221             return bootTime;
222         }
223         return System.currentTimeMillis() - Uptime.queryUpTime();
224     }

231     @Override
232     public List<OSService> getServices() {
233         List<OSService> services = new ArrayList<>();
234         // Get system services from lssrc command
235         /*-
236          Output:
237          Subsystem         Group            PID          Status
238             platform_agent                    2949214      active
239             cimsys                            2490590      active
240             snmpd            tcpip            2883698      active
241             syslogd          ras              2359466      active
242             sendmail         mail             3145828      active
243             portmap          portmap          2818188      active
244             inetd            tcpip            2752656      active
245             lpd              spooler                       inoperative
246                         ...
247          */
248         List<String> systemServicesInfoList = ExecutingCommand.runNative("lssrc -a");
249         if (systemServicesInfoList.size() > 1) {
250             systemServicesInfoList.remove(0); // remove header
251             for (String systemService : systemServicesInfoList) {
252                 String[] serviceSplit = ParseUtil.whitespaces.split(systemService.trim());
253                 if (systemService.contains("active")) {
254                     if (serviceSplit.length == 4) {
255                         services.add(new OSService(serviceSplit[0], ParseUtil.parseIntOrDefault(serviceSplit[2], 0),
256                                 RUNNING));
257                     } else if (serviceSplit.length == 3) {
258                         services.add(new OSService(serviceSplit[0], ParseUtil.parseIntOrDefault(serviceSplit[1], 0),
259                                 RUNNING));
260                     }
261                 } else if (systemService.contains("inoperative")) {
262                     services.add(new OSService(serviceSplit[0], 0, STOPPED));
263                 }
264             }
265         }
266         // Get installed services from /etc/rc.d/init.d
267         File dir = new File("/etc/rc.d/init.d");
268         File[] listFiles;
269         if (dir.exists() && dir.isDirectory() && (listFiles = dir.listFiles()) != null) {
270             for (File file : listFiles) {
271                 String installedService = ExecutingCommand.getFirstAnswer(file.getAbsolutePath() + " status");
272                 // Apache httpd daemon is running with PID 3997858.
273                 if (installedService.contains("running")) {
274                     services.add(new OSService(file.getName(), ParseUtil.parseLastInt(installedService, 0), RUNNING));
275                 } else {
276                     services.add(new OSService(file.getName(), 0, STOPPED));
277                 }
278             }
279         }
280         return services;
281     }
