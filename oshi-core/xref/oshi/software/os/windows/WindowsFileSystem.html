49  /**
50   * The Windows File System contains {@link oshi.software.os.OSFileStore}s which
51   * are a storage pool, device, partition, volume, concrete file system or other
52   * implementation specific means of file storage. In Windows, these are
53   * represented by a drive letter, e.g., "A:\" and "C:\"
54   */

100     private static final long MAX_WINDOWS_HANDLES;
101     static {
102         // Determine whether 32-bit or 64-bit handle limit, although both are
103         // essentially infinite for practical purposes. See
104         // https://blogs.technet.microsoft.com/markrussinovich/2009/09/29/pushing-the-limits-of-windows-handles/
105         if (System.getenv("ProgramFiles(x86)") == null) {
106             MAX_WINDOWS_HANDLES = 16_777_216L - 32_768L;
107         } else {
108             MAX_WINDOWS_HANDLES = 16_777_216L - 65_536L;
109         }
110     }

122     @Override
123     public List<OSFileStore> getFileStores(boolean localOnly) {
124         // Create list to hold results
125         ArrayList<OSFileStore> result;
126 
127         // Begin with all the local volumes
128         result = getLocalVolumes(null);
129 
130         // Build a map of existing mount point to OSFileStore
131         Map<String, OSFileStore> volumeMap = new HashMap<>();
132         for (OSFileStore volume : result) {
133             volumeMap.put(volume.getMount(), volume);
134         }
135 
136         // Iterate through volumes in WMI and update description (if it exists)
137         // or add new if it doesn't (expected for network drives)
138         for (OSFileStore wmiVolume : getWmiVolumes(null, localOnly)) {
139             if (volumeMap.containsKey(wmiVolume.getMount())) {
140                 // If the volume is already in our list, update the name field
141                 // using WMI's more verbose name and update label if needed
142                 OSFileStore volume = volumeMap.get(wmiVolume.getMount());
143                 result.remove(volume);
144                 result.add(new WindowsOSFileStore(wmiVolume.getName(), volume.getVolume(),
145                         volume.getLabel().isEmpty() ? wmiVolume.getLabel() : volume.getLabel(), volume.getMount(),
146                         volume.getOptions(), volume.getUUID(), "", volume.getDescription(), volume.getType(),
147                         volume.getFreeSpace(), volume.getUsableSpace(), volume.getTotalSpace(), 0, 0));
148             } else if (!localOnly) {
149                 // Otherwise add the new volume in its entirety
150                 result.add(wmiVolume);
151             }
152         }
153         return result;
154     }

156     /**
157      * Package private method for getting all mounted local drives.
158      *
159      * @param volumeToMatch
160      *            an optional string to filter match, null otherwise
161      * @return A list of {@link OSFileStore} objects representing all local mounted
162      *         volumes
163      */
164     static ArrayList<OSFileStore> getLocalVolumes(String volumeToMatch) {
165         ArrayList<OSFileStore> fs;
166         String volume;
167         String strFsType;
168         String strName;
169         String strMount;
170         WinNT.HANDLE hVol;
171         WinNT.LARGE_INTEGER userFreeBytes;
172         WinNT.LARGE_INTEGER totalBytes;
173         WinNT.LARGE_INTEGER systemFreeBytes;
174         char[] aVolume;
175         char[] fstype;
176         char[] name;
177         char[] mount;
178         IntByReference pFlags;
179 
180         fs = new ArrayList<>();
181         aVolume = new char[BUFSIZE];
182 
183         hVol = Kernel32.INSTANCE.FindFirstVolume(aVolume, BUFSIZE);
184         if (WinBase.INVALID_HANDLE_VALUE.equals(hVol)) {
185             return fs;
186         }
187         try {
188             do {
189                 fstype = new char[16];
190                 name = new char[BUFSIZE];
191                 mount = new char[BUFSIZE];
192                 pFlags = new IntByReference();
193 
194                 userFreeBytes = new WinNT.LARGE_INTEGER(0L);
195                 totalBytes = new WinNT.LARGE_INTEGER(0L);
196                 systemFreeBytes = new WinNT.LARGE_INTEGER(0L);
197 
198                 volume = Native.toString(aVolume);
199                 Kernel32.INSTANCE.GetVolumeInformation(volume, name, BUFSIZE, null, null, pFlags, fstype, 16);
200                 final int flags = pFlags.getValue();
201                 Kernel32.INSTANCE.GetVolumePathNamesForVolumeName(volume, mount, BUFSIZE, null);
202 
203                 strMount = Native.toString(mount);
204                 if (!strMount.isEmpty() && (volumeToMatch == null || volumeToMatch.equals(volume))) {
205                     strName = Native.toString(name);
206                     strFsType = Native.toString(fstype);
207 
208                     StringBuilder options = new StringBuilder((FILE_READ_ONLY_VOLUME & flags) == 0 ? "rw" : "ro");
209                     String moreOptions = OPTIONS_MAP.entrySet().stream().filter(e -> (e.getKey() & flags) > 0)
210                             .map(Map.Entry::getValue).collect(Collectors.joining(","));
211                     if (!moreOptions.isEmpty()) {
212                         options.append(',').append(moreOptions);
213                     }
214                     Kernel32.INSTANCE.GetDiskFreeSpaceEx(volume, userFreeBytes, totalBytes, systemFreeBytes);
215                     // Parse uuid from volume name
216                     String uuid = ParseUtil.parseUuidOrDefault(volume, "");
217 
218                     fs.add(new WindowsOSFileStore(String.format("%s (%s)", strName, strMount), volume, strName,
219                             strMount, options.toString(), uuid, "", getDriveType(strMount), strFsType,
220                             systemFreeBytes.getValue(), userFreeBytes.getValue(), totalBytes.getValue(), 0, 0));
221                 }
222             } while (Kernel32.INSTANCE.FindNextVolume(hVol, aVolume, BUFSIZE));
223             return fs;
224         } finally {
225             Kernel32.INSTANCE.FindVolumeClose(hVol);
226         }
227     }

229     /**
230      * Package private method for getting logical drives listed in WMI.
231      *
232      * @param nameToMatch
233      *            an optional string to filter match, null otherwise
234      * @param localOnly
235      *            Whether to only search local drives
236      * @return A list of {@link OSFileStore} objects representing all network
237      *         mounted volumes
238      */
239     static List<OSFileStore> getWmiVolumes(String nameToMatch, boolean localOnly) {
240         long free;
241         long total;
242         List<OSFileStore> fs = new ArrayList<>();
243         WmiResult<LogicalDiskProperty> drives = Win32LogicalDisk.queryLogicalDisk(nameToMatch, localOnly);
244         for (int i = 0; i < drives.getResultCount(); i++) {
245             free = WmiUtil.getUint64(drives, LogicalDiskProperty.FREESPACE, i);
246             total = WmiUtil.getUint64(drives, LogicalDiskProperty.SIZE, i);
247             String description = WmiUtil.getString(drives, LogicalDiskProperty.DESCRIPTION, i);
248             String name = WmiUtil.getString(drives, LogicalDiskProperty.NAME, i);
249             String label = WmiUtil.getString(drives, LogicalDiskProperty.VOLUMENAME, i);
250             String options = WmiUtil.getUint16(drives, LogicalDiskProperty.ACCESS, i) == 1 ? "ro" : "rw";
251             int type = WmiUtil.getUint32(drives, LogicalDiskProperty.DRIVETYPE, i);
252             String volume;
253             if (type != 4) {
254                 char[] chrVolume = new char[BUFSIZE];
255                 Kernel32.INSTANCE.GetVolumeNameForVolumeMountPoint(name + "\\", chrVolume, BUFSIZE);
256                 volume = Native.toString(chrVolume);
257             } else {
258                 volume = WmiUtil.getString(drives, LogicalDiskProperty.PROVIDERNAME, i);
259                 String[] split = volume.split("\\\\");
260                 if (split.length > 1 && split[split.length - 1].length() > 0) {
261                     description = split[split.length - 1];
262                 }
263             }
264             fs.add(new WindowsOSFileStore(String.format("%s (%s)", description, name), volume, label, name + "\\",
265                     options, "", "", getDriveType(name), WmiUtil.getString(drives, LogicalDiskProperty.FILESYSTEM, i),
266                     free, free, total, 0, 0));
267         }
268         return fs;
269     }

271     /**
272      * Private method for getting mounted drive type.
273      *
274      * @param drive
275      *            Mounted drive
276      * @return A drive type description
277      */
278     private static String getDriveType(String drive) {
279         switch (Kernel32.INSTANCE.GetDriveType(drive)) {
280         case 2:
281             return "Removable drive";
282         case 3:
283             return "Fixed drive";
284         case 4:
285             return "Network drive";
286         case 5:
287             return "CD-ROM";
288         case 6:
289             return "RAM drive";
290         default:
291             return "Unknown drive type";
292         }
293     }

295     @Override
296     public long getOpenFileDescriptors() {
297         Map<HandleCountProperty, List<Long>> valueListMap = ProcessInformation.queryHandles().getB();
298         List<Long> valueList = valueListMap.get(HandleCountProperty.HANDLECOUNT);
299         long descriptors = 0L;
300         if (valueList != null) {
301             for (int i = 0; i < valueList.size(); i++) {
302                 descriptors += valueList.get(i);
303             }
304         }
305         return descriptors;
306     }
307 
308     @Override
309     public long getMaxFileDescriptors() {
310         return MAX_WINDOWS_HANDLES;
311     }
