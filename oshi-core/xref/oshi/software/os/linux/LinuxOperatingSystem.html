118     @Override
119     public Pair<String, OSVersionInfo> queryFamilyVersionInfo() {
120         Triplet<String, String, String> familyVersionCodename = queryFamilyVersionCodenameFromReleaseFiles();
121         String buildNumber = null;
122         List<String> procVersion = FileUtil.readFile(ProcPath.VERSION);
123         if (!procVersion.isEmpty()) {
124             String[] split = ParseUtil.whitespaces.split(procVersion.get(0));
125             for (String s : split) {
126                 if (!"Linux".equals(s) && !"version".equals(s)) {
127                     buildNumber = s;
128                     break;
129                 }
130             }
131         }
132         OSVersionInfo versionInfo = new OSVersionInfo(familyVersionCodename.getB(), familyVersionCodename.getC(),
133                 buildNumber);
134         return new Pair<>(familyVersionCodename.getA(), versionInfo);
135     }

174     @Override
175     public List<OSProcess> queryChildProcesses(int parentPid) {
176         File[] pidFiles = ProcessStat.getPidFiles();
177         if (parentPid >= 0) {
178             // Only return descendants
179             return queryProcessList(getChildrenOrDescendants(getParentPidsFromProcFiles(pidFiles), parentPid, false));
180         }
181         Set<Integer> descendantPids = new HashSet<>();
182         // Put everything in the "descendant" set
183         for (File procFile : pidFiles) {
184             int pid = ParseUtil.parseIntOrDefault(procFile.getName(), -2);
185             if (pid != -2) {
186                 descendantPids.add(pid);
187             }
188         }
189         return queryProcessList(descendantPids);
190     }
191 
192     @Override
193     public List<OSProcess> queryDescendantProcesses(int parentPid) {
194         File[] pidFiles = ProcessStat.getPidFiles();
195         return queryProcessList(getChildrenOrDescendants(getParentPidsFromProcFiles(pidFiles), parentPid, true));
196     }

198     private static List<OSProcess> queryProcessList(Set<Integer> descendantPids) {
199         List<OSProcess> procs = new ArrayList<>();
200         for (int pid : descendantPids) {
201             OSProcess proc = new LinuxOSProcess(pid);
202             if (!proc.getState().equals(State.INVALID)) {
203                 procs.add(proc);
204             }
205         }
206         return procs;
207     }

209     private static Map<Integer, Integer> getParentPidsFromProcFiles(File[] pidFiles) {
210         Map<Integer, Integer> parentPidMap = new HashMap<>();
211         for (File procFile : pidFiles) {
212             int pid = ParseUtil.parseIntOrDefault(procFile.getName(), 0);
213             parentPidMap.put(pid, getParentPidFromProcFile(pid));
214         }
215         return parentPidMap;
216     }

218     private static int getParentPidFromProcFile(int pid) {
219         String stat = FileUtil.getStringFromFile(String.format("/proc/%d/stat", pid));
220         // A race condition may leave us with an empty string
221         if (stat.isEmpty()) {
222             return 0;
223         }
224         // Grab PPID
225         long[] statArray = ParseUtil.parseStringToLongArray(stat, PPID_INDEX, ProcessStat.PROC_PID_STAT_LENGTH, ' ');
226         return (int) statArray[0];
227     }

241         try {
242             Sysinfo info = new Sysinfo();
243             if (0 != LibC.INSTANCE.sysinfo(info)) {
244                 LOG.error("Failed to get process thread count. Error code: {}", Native.getLastError());
245                 return 0;
246             }
247             return info.procs;
248         } catch (UnsatisfiedLinkError | NoClassDefFoundError e) {
249             LOG.error("Failed to get procs from sysinfo. {}", e.getMessage());
250         }
251         return 0;
252     }

269     private static Triplet<String, String, String> queryFamilyVersionCodenameFromReleaseFiles() {
270         Triplet<String, String, String> familyVersionCodename;
271         // There are two competing options for family/version information.
272         // Newer systems are adopting a standard /etc/os-release file:
273         // https://www.freedesktop.org/software/systemd/man/os-release.html
274         //
275         // Some systems are still using the lsb standard which parses a
276         // variety of /etc/*-release files and is most easily accessed via
277         // the commandline lsb_release -a, see here:
278         // https://linux.die.net/man/1/lsb_release
279         // In this case, the /etc/lsb-release file (if it exists) has
280         // optional overrides to the information in the /etc/distrib-release
281         // files, which show: "Distributor release x.x (Codename)"
282 
283         // Attempt to read /etc/system-release which has more details than
284         // os-release on (CentOS and Fedora)
285         if ((familyVersionCodename = readDistribRelease("/etc/system-release")) != null) {
286             // If successful, we're done. this.family has been set and
287             // possibly the versionID and codeName
288             return familyVersionCodename;
289         }
290 
291         // Attempt to read /etc/os-release file.
292         if ((familyVersionCodename = readOsRelease()) != null) {
293             // If successful, we're done. this.family has been set and
294             // possibly the versionID and codeName
295             return familyVersionCodename;
296         }
297 
298         // Attempt to execute the `lsb_release` command
299         if ((familyVersionCodename = execLsbRelease()) != null) {
300             // If successful, we're done. this.family has been set and
301             // possibly the versionID and codeName
302             return familyVersionCodename;
303         }
304 
305         // The above two options should hopefully work on most
306         // distributions. If not, we keep having fun.
307         // Attempt to read /etc/lsb-release file
308         if ((familyVersionCodename = readLsbRelease()) != null) {
309             // If successful, we're done. this.family has been set and
310             // possibly the versionID and codeName
311             return familyVersionCodename;
312         }
313 
314         // If we're still looking, we search for any /etc/*-release (or
315         // similar) filename, for which the first line should be of the
316         // "Distributor release x.x (Codename)" format or possibly a
317         // "Distributor VERSION x.x (Codename)" format
318         String etcDistribRelease = getReleaseFilename();
319         if ((familyVersionCodename = readDistribRelease(etcDistribRelease)) != null) {
320             // If successful, we're done. this.family has been set and
321             // possibly the versionID and codeName
322             return familyVersionCodename;
323         }
324         // If we've gotten this far with no match, use the distrib-release
325         // filename (defaults will eventually give "Unknown")
326         String family = filenameToFamily(etcDistribRelease.replace("/etc/", "").replace("release", "")
327                 .replace("version", "").replace("-", "").replace("_", ""));
328         return new Triplet<>(family, Constants.UNKNOWN, Constants.UNKNOWN);
329     }

331     /**
332      * Attempts to read /etc/os-release
333      *
334      * @return a triplet with the parsed family, versionID and codeName if file
335      *         successfully read and NAME= found, null otherwise
336      */
337     private static Triplet<String, String, String> readOsRelease() {
338         String family = null;
339         String versionId = Constants.UNKNOWN;
340         String codeName = Constants.UNKNOWN;
341         List<String> osRelease = FileUtil.readFile("/etc/os-release");
342         // Search for NAME=
343         for (String line : osRelease) {
344             if (line.startsWith("VERSION=")) {
345                 LOG.debug(OS_RELEASE_LOG, line);
346                 // remove beginning and ending '"' characters, etc from
347                 // VERSION="14.04.4 LTS, Trusty Tahr" (Ubuntu style)
348                 // or VERSION="17 (Beefy Miracle)" (os-release doc style)
349                 line = line.replace("VERSION=", "").replaceAll(DOUBLE_QUOTES, "").trim();
350                 String[] split = line.split("[()]");
351                 if (split.length <= 1) {
352                     // If no parentheses, check for Ubuntu's comma format
353                     split = line.split(", ");
354                 }
355                 if (split.length > 0) {
356                     versionId = split[0].trim();
357                 }
358                 if (split.length > 1) {
359                     codeName = split[1].trim();
360                 }
361             } else if (line.startsWith("NAME=") && family == null) {
362                 LOG.debug(OS_RELEASE_LOG, line);
363                 // remove beginning and ending '"' characters, etc from
364                 // NAME="Ubuntu"
365                 family = line.replace("NAME=", "").replaceAll(DOUBLE_QUOTES, "").trim();
366             } else if (line.startsWith("VERSION_ID=") && versionId.equals(Constants.UNKNOWN)) {
367                 LOG.debug(OS_RELEASE_LOG, line);
368                 // remove beginning and ending '"' characters, etc from
369                 // VERSION_ID="14.04"
370                 versionId = line.replace("VERSION_ID=", "").replaceAll(DOUBLE_QUOTES, "").trim();
371             }
372         }
373         return family == null ? null : new Triplet<>(family, versionId, codeName);
374     }

376     /**
377      * Attempts to execute `lsb_release -a`
378      *
379      * @return a triplet with the parsed family, versionID and codeName if the
380      *         command successfully executed and Distributor ID: or Description:
381      *         found, null otherwise
382      */
383     private static Triplet<String, String, String> execLsbRelease() {
384         String family = null;
385         String versionId = Constants.UNKNOWN;
386         String codeName = Constants.UNKNOWN;
387         // If description is of the format Distrib release x.x (Codename)
388         // that is primary, otherwise use Distributor ID: which returns the
389         // distribution concatenated, e.g., RedHat instead of Red Hat
390         for (String line : ExecutingCommand.runNative("lsb_release -a")) {
391             if (line.startsWith("Description:")) {
392                 LOG.debug(LSB_RELEASE_A_LOG, line);
393                 line = line.replace("Description:", "").trim();
394                 if (line.contains(RELEASE_DELIM)) {
395                     Triplet<String, String, String> triplet = parseRelease(line, RELEASE_DELIM);
396                     family = triplet.getA();
397                     if (versionId.equals(Constants.UNKNOWN)) {
398                         versionId = triplet.getB();
399                     }
400                     if (codeName.equals(Constants.UNKNOWN)) {
401                         codeName = triplet.getC();
402                     }
403                 }
404             } else if (line.startsWith("Distributor ID:") && family == null) {
405                 LOG.debug(LSB_RELEASE_A_LOG, line);
406                 family = line.replace("Distributor ID:", "").trim();
407             } else if (line.startsWith("Release:") && versionId.equals(Constants.UNKNOWN)) {
408                 LOG.debug(LSB_RELEASE_A_LOG, line);
409                 versionId = line.replace("Release:", "").trim();
410             } else if (line.startsWith("Codename:") && codeName.equals(Constants.UNKNOWN)) {
411                 LOG.debug(LSB_RELEASE_A_LOG, line);
412                 codeName = line.replace("Codename:", "").trim();
413             }
414         }
415         return family == null ? null : new Triplet<>(family, versionId, codeName);
416     }

418     /**
419      * Attempts to read /etc/lsb-release
420      *
421      * @return a triplet with the parsed family, versionID and codeName if file
422      *         successfully read and and DISTRIB_ID or DISTRIB_DESCRIPTION, null
423      *         otherwise
424      */
425     private static Triplet<String, String, String> readLsbRelease() {
426         String family = null;
427         String versionId = Constants.UNKNOWN;
428         String codeName = Constants.UNKNOWN;
429         List<String> osRelease = FileUtil.readFile("/etc/lsb-release");
430         // Search for NAME=
431         for (String line : osRelease) {
432             if (line.startsWith("DISTRIB_DESCRIPTION=")) {
433                 LOG.debug(LSB_RELEASE_LOG, line);
434                 line = line.replace("DISTRIB_DESCRIPTION=", "").replaceAll(DOUBLE_QUOTES, "").trim();
435                 if (line.contains(RELEASE_DELIM)) {
436                     Triplet<String, String, String> triplet = parseRelease(line, RELEASE_DELIM);
437                     family = triplet.getA();
438                     if (versionId.equals(Constants.UNKNOWN)) {
439                         versionId = triplet.getB();
440                     }
441                     if (codeName.equals(Constants.UNKNOWN)) {
442                         codeName = triplet.getC();
443                     }
444                 }
445             } else if (line.startsWith("DISTRIB_ID=") && family == null) {
446                 LOG.debug(LSB_RELEASE_LOG, line);
447                 family = line.replace("DISTRIB_ID=", "").replaceAll(DOUBLE_QUOTES, "").trim();
448             } else if (line.startsWith("DISTRIB_RELEASE=") && versionId.equals(Constants.UNKNOWN)) {
449                 LOG.debug(LSB_RELEASE_LOG, line);
450                 versionId = line.replace("DISTRIB_RELEASE=", "").replaceAll(DOUBLE_QUOTES, "").trim();
451             } else if (line.startsWith("DISTRIB_CODENAME=") && codeName.equals(Constants.UNKNOWN)) {
452                 LOG.debug(LSB_RELEASE_LOG, line);
453                 codeName = line.replace("DISTRIB_CODENAME=", "").replaceAll(DOUBLE_QUOTES, "").trim();
454             }
455         }
456         return family == null ? null : new Triplet<>(family, versionId, codeName);
457     }

459     /**
460      * Attempts to read /etc/distrib-release (for some value of distrib)
461      *
462      * @param filename
463      *            The /etc/distrib-release file
464      * @return a triplet with the parsed family, versionID and codeName if file
465      *         successfully read and " release " or " VERSION " found, null
466      *         otherwise
467      */
468     private static Triplet<String, String, String> readDistribRelease(String filename) {
469         if (new File(filename).exists()) {
470             List<String> osRelease = FileUtil.readFile(filename);
471             // Search for Distrib release x.x (Codename)
472             for (String line : osRelease) {
473                 LOG.debug("{}: {}", filename, line);
474                 if (line.contains(RELEASE_DELIM)) {
475                     // If this parses properly we're done
476                     return parseRelease(line, RELEASE_DELIM);
477                 } else if (line.contains(" VERSION ")) {
478                     // If this parses properly we're done
479                     return parseRelease(line, " VERSION ");
480                 }
481             }
482         }
483         return null;
484     }

486     /**
487      * Helper method to parse version description line style
488      *
489      * @param line
490      *            a String of the form "Distributor release x.x (Codename)"
491      * @param splitLine
492      *            A regex to split on, e.g. " release "
493      * @return a triplet with the parsed family, versionID and codeName
494      */
495     private static Triplet<String, String, String> parseRelease(String line, String splitLine) {
496         String[] split = line.split(splitLine);
497         String family = split[0].trim();
498         String versionId = Constants.UNKNOWN;
499         String codeName = Constants.UNKNOWN;
500         if (split.length > 1) {
501             split = split[1].split("[()]");
502             if (split.length > 0) {
503                 versionId = split[0].trim();
504             }
505             if (split.length > 1) {
506                 codeName = split[1].trim();
507             }
508         }
509         return new Triplet<>(family, versionId, codeName);
510     }

512     /**
513      * Looks for a collection of possible distrib-release filenames
514      *
515      * @return The first valid matching filename
516      */
517     protected static String getReleaseFilename() {
518         // Look for any /etc/*-release, *-version, and variants
519         File etc = new File("/etc");
520         // Find any *_input files in that path
521         File[] matchingFiles = etc.listFiles(//
522                 f -> (f.getName().endsWith("-release") || //
523                         f.getName().endsWith("-version") || //
524                         f.getName().endsWith("_release") || //
525                         f.getName().endsWith("_version")) //
526                         && !(f.getName().endsWith("os-release") || //
527                                 f.getName().endsWith("lsb-release") || //
528                                 f.getName().endsWith("system-release")));
529         if (matchingFiles != null && matchingFiles.length > 0) {
530             return matchingFiles[0].getPath();
531         }
532         if (new File("/etc/release").exists()) {
533             return "/etc/release";
534         }
535         // If all else fails, try this
536         return "/etc/issue";
537     }

539     /**
540      * Converts a portion of a filename (e.g. the 'redhat' in /etc/redhat-release)
541      * to a mixed case string representing the family (e.g., Red Hat)
542      *
543      * @param name
544      *            Stripped version of filename after removing /etc and -release
545      * @return Mixed case family
546      */
547     private static String filenameToFamily(String name) {
548 
549         if (name.isEmpty()) {
550             return "Solaris";
551         } else if ("issue".equalsIgnoreCase(name)) {
552             // /etc/issue will end up here
553             return "Unknown";
554         } else {
555             Properties filenameProps = FileUtil.readPropertiesFromFilename(FILENAME_PROPERTIES);
556             String family = filenameProps.getProperty(name.toLowerCase());
557             return family != null ? family : name.substring(0, 1).toUpperCase() + name.substring(1);
558         }
559     }

561     @Override
562     public List<OSService> getServices() {
563         // Get running services
564         List<OSService> services = new ArrayList<>();
565         Set<String> running = new HashSet<>();
566         for (OSProcess p : getChildProcesses(1, ProcessFiltering.ALL_PROCESSES, ProcessSorting.PID_ASC, 0)) {
567             OSService s = new OSService(p.getName(), p.getProcessID(), RUNNING);
568             services.add(s);
569             running.add(p.getName());
570         }
571         boolean systemctlFound = false;
572         List<String> systemctl = ExecutingCommand.runNative("systemctl list-unit-files");
573         for (String str : systemctl) {
574             String[] split = ParseUtil.whitespaces.split(str);
575             if (split.length >= 2 && split[0].endsWith(".service") && "enabled".equals(split[1])) {
576                 // remove .service extension
577                 String name = split[0].substring(0, split[0].length() - 8);
578                 int index = name.lastIndexOf('.');
579                 String shortName = (index < 0 || index > name.length() - 2) ? name : name.substring(index + 1);
580                 if (!running.contains(name) && !running.contains(shortName)) {
581                     OSService s = new OSService(name, 0, STOPPED);
582                     services.add(s);
583                     systemctlFound = true;
584                 }
585             }
586         }
587         if (!systemctlFound) {
588             // Get Directories for stopped services
589             File dir = new File("/etc/init");
590             if (dir.exists() && dir.isDirectory()) {
591                 for (File f : dir.listFiles((f, name) -> name.toLowerCase().endsWith(".conf"))) {
592                     // remove .conf extension
593                     String name = f.getName().substring(0, f.getName().length() - 5);
594                     int index = name.lastIndexOf('.');
595                     String shortName = (index < 0 || index > name.length() - 2) ? name : name.substring(index + 1);
596                     if (!running.contains(name) && !running.contains(shortName)) {
597                         OSService s = new OSService(name, 0, STOPPED);
598                         services.add(s);
599                     }
600                 }
601             } else {
602                 LOG.error("Directory: /etc/init does not exist");
603             }
604         }
605         return services;
606     }
