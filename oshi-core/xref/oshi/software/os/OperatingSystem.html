44  /**
45   * An operating system (OS) is the software on a computer that manages the way
46   * different programs use its hardware, and regulates the ways that a user
47   * controls the computer.
48   * <p>
49   * Considered thread safe, but see remarks for the {@link #getSessions()}
50   * method.
51   */
52  @ThreadSafe
53  public interface OperatingSystem {
54  
55      /**
56       * Constants which may be used to filter Process lists in
57       * {@link #getProcesses(Predicate, Comparator, int)},
58       * {@link #getChildProcesses(int, Predicate, Comparator, int)}, and
59       * {@link #getDescendantProcesses(int, Predicate, Comparator, int)}.
60       */
61      final class ProcessFiltering {
62          private ProcessFiltering() {
63          }
64  
65          /**
66           * No filtering.
67           */
68          public static final Predicate<OSProcess> ALL_PROCESSES = p -> true;
69          /**
70           * Exclude processes with {@link State#INVALID} process state.
71           */
72          public static final Predicate<OSProcess> VALID_PROCESS = p -> !p.getState().equals(State.INVALID);
73          /**
74           * Exclude child processes. Only include processes which are their own parent.
75           */
76          public static final Predicate<OSProcess> NO_PARENT = p -> p.getParentProcessID() == p.getProcessID();
77          /**
78           * Only incude 64-bit processes.
79           */
80          public static final Predicate<OSProcess> BITNESS_64 = p -> p.getBitness() == 64;
81          /**
82           * Only include 32-bit processes.
83           */
84          public static final Predicate<OSProcess> BITNESS_32 = p -> p.getBitness() == 32;
85      }
86  
87      /**
88       * Constants which may be used to sort Process lists in
89       * {@link #getProcesses(Predicate, Comparator, int)},
90       * {@link #getChildProcesses(int, Predicate, Comparator, int)}, and
91       * {@link #getDescendantProcesses(int, Predicate, Comparator, int)}.
92       */
93      final class ProcessSorting {
94          private ProcessSorting() {
95          }
96  
97          /**
98           * No sorting
99           */
100         public static final Comparator<OSProcess> NO_SORTING = (p1, p2) -> 0;
101         /**
102          * Sort by decreasing cumulative CPU percentage
103          */
104         public static final Comparator<OSProcess> CPU_DESC = Comparator
105                 .comparingDouble(OSProcess::getProcessCpuLoadCumulative).reversed();
106         /**
107          * Sort by decreasing Resident Set Size (RSS)
108          */
109         public static final Comparator<OSProcess> RSS_DESC = Comparator.comparingLong(OSProcess::getResidentSetSize)
110                 .reversed();
111         /**
112          * Sort by up time, newest processes first
113          */
114         public static final Comparator<OSProcess> UPTIME_ASC = Comparator.comparingLong(OSProcess::getUpTime);
115         /**
116          * Sort by up time, oldest processes first
117          */
118         public static final Comparator<OSProcess> UPTIME_DESC = UPTIME_ASC.reversed();
119         /**
120          * Sort by Process Id
121          */
122         public static final Comparator<OSProcess> PID_ASC = Comparator.comparingInt(OSProcess::getProcessID);
123         /**
124          * Sort by Parent Process Id
125          */
126         public static final Comparator<OSProcess> PARENTPID_ASC = Comparator
127                 .comparingInt(OSProcess::getParentProcessID);
128         /**
129          * Sort by Process Name (case insensitive)
130          */
131         public static final Comparator<OSProcess> NAME_ASC = Comparator.comparing(OSProcess::getName,
132                 String.CASE_INSENSITIVE_ORDER);
133     }

170     /**
171      * Gets currently running processes. No order is guaranteed.
172      *
173      * @return A list of {@link oshi.software.os.OSProcess} objects for the
174      *         specified number (or all) of currently running processes, sorted as
175      *         specified. The list may contain null elements or processes with a
176      *         state of {@link OSProcess.State#INVALID} if a process terminates
177      *         during iteration.
178      */
179     default List<OSProcess> getProcesses() {
180         return getProcesses(null, null, 0);
181     }
182 
183     /**
184      * Gets currently running processes, optionally filtering, sorting, and limited
185      * to the top "N".
186      *
187      * @param filter
188      *            An optional {@link Predicate} limiting the results to the
189      *            specified filter. Some common predicates are available in
190      *            {@link ProcessSorting}. May be {@code null} for no filtering.
191      * @param sort
192      *            An optional {@link Comparator} specifying the sorting order. Some
193      *            common comparators are available in {@link ProcessSorting}. May be
194      *            {@code null} for no sorting.
195      * @param limit
196      *            Max number of results to return, or 0 to return all results
197      * @return A list of {@link oshi.software.os.OSProcess} objects, optionally
198      *         filtered, sorted, and limited to the specified number.
199      *         <p>
200      *         The list may contain processes with a state of
201      *         {@link OSProcess.State#INVALID} if a process terminates during
202      *         iteration.
203      */
204     List<OSProcess> getProcesses(Predicate<OSProcess> filter, Comparator<OSProcess> sort, int limit);
205 
206     /**
207      * Gets information on a {@link Collection} of currently running processes. This
208      * has potentially improved performance vs. iterating individual processes.
209      *
210      * @param pids
211      *            A collection of process IDs
212      * @return A list of {@link oshi.software.os.OSProcess} objects for the
213      *         specified process ids if it is running
214      */
215     default List<OSProcess> getProcesses(Collection<Integer> pids) {
216         return pids.stream().map(this::getProcess).filter(Objects::nonNull).filter(ProcessFiltering.VALID_PROCESS)
217                 .collect(Collectors.toList());
218     }

355     /**
356      * Gets currently logged in users.
357      * <p>
358      * On macOS, Linux, and Unix systems, the default implementation uses native
359      * code (see {@code man getutxent}) that is not thread safe. OSHI's use of this
360      * code is synchronized and may be used in a multi-threaded environment without
361      * introducing any additional conflicts. Users should note, however, that other
362      * operating system code may access the same native code.
363      * <p>
364      * The {@link oshi.driver.unix.Who#queryWho()} method produces similar output
365      * parsing the output of the Posix-standard {@code who} command, and may
366      * internally employ reentrant code on some platforms. Users may opt to use this
367      * command-line variant by default using the {@code oshi.os.unix.whoCommand}
368      * configuration property.
369      *
370      * @return A list of {@link oshi.software.os.OSSession} objects representing
371      *         logged-in users
372      */
373     default List<OSSession> getSessions() {
374         return Who.queryWho();
375     }

377     /**
378      * Gets windows on the operating system's GUI desktop.
379      * <p>
380      * On Unix-like systems, reports X11 windows only, which may be limited to the
381      * current display and will not report windows used by other window managers.
382      * <p>
383      * While not a guarantee, a best effort is made to return windows in
384      * foreground-to-background order. This ordering may be used along with
385      * {@link OSDesktopWindow#getOrder()} to (probably) determine the frontmost
386      * window.
387      *
388      * @param visibleOnly
389      *            Whether to restrict the list to only windows visible to the user.
390      *            <p>
391      *            This is a best effort attempt at a reasonable definition of
392      *            visibility. Visible windows may be completely transparent.
393      * @return A list of {@link oshi.software.os.OSDesktopWindow} objects
394      *         representing the desktop windows.
395      */
396     default List<OSDesktopWindow> getDesktopWindows(boolean visibleOnly) {
397         // Default X11 implementation for Unix-like operating systems.
398         // Overridden on Windows and macOS
399         return Xwininfo.queryXWindows(visibleOnly);
400     }

402     /**
403      * A class representing the Operating System version details.
404      */
405     @Immutable
406     class OSVersionInfo {
407         private final String version;
408         private final String codeName;
409         private final String buildNumber;
410         private final String versionStr;
411 
412         public OSVersionInfo(String version, String codeName, String buildNumber) {
413             // Insider Preview Windows 11 is marked as Windows 10 build 22000
414             // Temporary code until JDK os.name matches up
415             if ("10".equals(version) && buildNumber.compareTo("22000") >= 0) {
416                 this.version = "11";
417             } else {
418                 this.version = version;
419             }
420             this.codeName = codeName;
421             this.buildNumber = buildNumber;
422 
423             StringBuilder sb = new StringBuilder(getVersion() != null ? getVersion() : Constants.UNKNOWN);
424             if (!Util.isBlank(getCodeName())) {
425                 sb.append(" (").append(getCodeName()).append(')');
426             }
427             if (!Util.isBlank(getBuildNumber())) {
428                 sb.append(" build ").append(getBuildNumber());
429             }
430             this.versionStr = sb.toString();
431         }
432 
433         /**
434          * Gets the operating system version.
435          *
436          * @return The version, if any. May be {@code null}.
437          */
438         public String getVersion() {
439             return version;
440         }
441 
442         /**
443          * Gets the operating system codename.
444          *
445          * @return The code name, if any. May be {@code null}.
446          */
447         public String getCodeName() {
448             return codeName;
449         }
450 
451         /**
452          * Gets the operating system build number.
453          *
454          * @return The build number, if any. May be {@code null}.
455          */
456         public String getBuildNumber() {
457             return buildNumber;
458         }
459 
460         @Override
461         public String toString() {
462             return this.versionStr;
463         }
