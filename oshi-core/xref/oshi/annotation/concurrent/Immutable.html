24  package oshi.annotation.concurrent;

26  import java.lang.annotation.Documented;
27  
28  /**
29   * The presence of this annotation indicates that the author believes the class
30   * to be immutable and hence inherently thread-safe. An immutable class is one
31   * where the state of an instance cannot be <i>seen</i> to change. As a result
32   * <ul>
33   * <li>All public fields must be {@code final}</li>
34   * <li>All public final reference fields are either {@code null} or refer to
35   * other immutable objects</li>
36   * <li>Constructors and methods do not publish references to any potentially
37   * mutable internal state.</li>
38   * </ul>
39   * Performance optimization may mean that instances of an immutable class may
40   * have mutable internal state. The critical point is that callers cannot tell
41   * the difference. For example {@link String} is an immutable class, despite
42   * having an internal int that is non-final but used as a cache for
43   * {@link String#hashCode()}.
44   * <p>
45   * Immutable objects are inherently thread-safe; they may be passed between
46   * threads or published without synchronization.
47   * <p>
48   * This annotation is intended for internal use in OSHI as a temporary
49   * workaround until it is available in {@code jakarta.annotations}.
50   */
51  @Documented
52  public @interface Immutable {
53  }
