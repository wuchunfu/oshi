24  package oshi.demo;
25  
26  import java.util.HashMap;
27  import java.util.List;
28  import java.util.Map;
29  import java.util.Properties;
30  
31  import oshi.SystemInfo;
32  import oshi.hardware.HardwareAbstractionLayer;
33  import oshi.hardware.NetworkIF;
34  import oshi.util.FileUtil;
35  
36  /**
37   * Uses OSHI to attempt to identify whether the user is on a Virtual Machine
38   */
39  public class DetectVM {
40  
41      private static final String OSHI_VM_MAC_ADDR_PROPERTIES = "oshi.vmmacaddr.properties";
42      private static final Properties vmMacAddressProps = FileUtil
43              .readPropertiesFromFilename(OSHI_VM_MAC_ADDR_PROPERTIES);
44  
45      // Constant for CPU vendor string
46      private static final Map<String, String> vmVendor = new HashMap<>();
47      static {
48          vmVendor.put("bhyve bhyve", "bhyve");
49          vmVendor.put("KVMKVMKVM", "KVM");
50          vmVendor.put("TCGTCGTCGTCG", "QEMU");
51          vmVendor.put("Microsoft Hv", "Microsoft Hyper-V or Windows Virtual PC");
52          vmVendor.put("lrpepyh vr", "Parallels");// (endianness mismatch of "prl hyperv ")
53          vmVendor.put("VMwareVMware", "VMware");
54          vmVendor.put("XenVMMXenVMM", "Xen HVM");
55          vmVendor.put("ACRNACRNACRN", "Project ACRN");
56          vmVendor.put("QNXQVMBSQG", "QNX Hypervisor");
57      }
58  
59      private static final String[] vmModelArray = new String[] { "Linux KVM", "Linux lguest", "OpenVZ", "Qemu",
60              "Microsoft Virtual PC", "VMWare", "linux-vserver", "Xen", "FreeBSD Jail", "VirtualBox", "Parallels",
61              "Linux Containers", "LXC" };
62  
63      /**
64       * The main method, executing the {@link #identifyVM} method.
65       *
66       * @param args
67       *            Arguments, ignored.
68       */
69      public static void main(String[] args) {
70          String vmString = identifyVM();
71  
72          if (vmString.isEmpty()) {
73              System.out.println("You do not appear to be on a Virtual Machine.");
74          } else {
75              System.out.println("You appear to be on a VM: " + vmString);
76          }
77      }
78  
79      /**
80       * The function attempts to identify which Virtual Machine (VM) based on common
81       * VM signatures in MAC address and computer model.
82       *
83       * @return A string indicating the machine's virtualization info if it can be
84       *         determined, or an emptry string otherwise.
85       */
86      public static String identifyVM() {
87          SystemInfo si = new SystemInfo();
88          HardwareAbstractionLayer hw = si.getHardware();
89          // Check CPU Vendor
90          String vendor = hw.getProcessor().getProcessorIdentifier().getVendor().trim();
91          if (vmVendor.containsKey(vendor)) {
92              return vmVendor.get(vendor);
93          }
94  
95          // Try well known MAC addresses
96          List<NetworkIF> nifs = hw.getNetworkIFs();
97          for (NetworkIF nif : nifs) {
98              String mac = nif.getMacaddr().toUpperCase();
99              String oui = mac.length() > 7 ? mac.substring(0, 8) : mac;
100             if (vmMacAddressProps.containsKey(oui)) {
101                 return vmMacAddressProps.getProperty(oui);
102             }
103         }
104 
105         // Try well known models
106         String model = hw.getComputerSystem().getModel();
107         for (String vm : vmModelArray) {
108             if (model.contains(vm)) {
109                 return vm;
110             }
111         }
112         String manufacturer = hw.getComputerSystem().getManufacturer();
113         if ("Microsoft Corporation".equals(manufacturer) && "Virtual Machine".equals(model)) {
114             return "Microsoft Hyper-V";
115         }
116 
117         // Couldn't find VM, return empty string
118         return "";
119     }
