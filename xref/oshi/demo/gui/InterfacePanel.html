51  /**
52   * Displays a interface list, such as ifconfig.
53   */
54  public class InterfacePanel extends OshiJPanel { // NOSONAR squid:S110
55  
56      private static final long serialVersionUID = 1L;
57  
58      private static final int INIT_HASH_SIZE = 100;
59      private static final String IP_ADDRESS_SEPARATOR = "; ";
60  
61      private static final String PARAMS = "Network Parameters";
62      private static final String INTERFACES = "Network Interfaces";
63      private static final String[] COLUMNS = { "Name", "Index", "Speed", "IPv4 Address", "IPv6 address", "MAC address" };
64      private static final double[] COLUMN_WIDTH_PERCENT = { 0.02, 0.02, 0.1, 0.25, 0.45, 0.15 };
65  
66      public InterfacePanel(SystemInfo si) {
67          super();
68          init(si);
69      }
70  
71      private void init(SystemInfo si) {
72          setBorder(BorderFactory.createEmptyBorder(5, 5, 5, 5));
73          setLayout(new BoxLayout(this, BoxLayout.Y_AXIS));
74  
75          JLabel paramsLabel = new JLabel(PARAMS);
76          paramsLabel.setAlignmentX(Component.CENTER_ALIGNMENT);
77          add(paramsLabel);
78  
79          JTextArea paramsArea = new JTextArea(0, 0);
80          paramsArea.setText(buildParamsText(si.getOperatingSystem()));
81          add(paramsArea);
82  
83          JLabel interfaceLabel = new JLabel(INTERFACES);
84          interfaceLabel.setAlignmentX(Component.CENTER_ALIGNMENT);
85          add(interfaceLabel);
86  
87          List<NetworkIF> networkIfList = si.getHardware().getNetworkIFs(true);
88  
89          TableModel model = new DefaultTableModel(parseInterfaces(networkIfList), COLUMNS);
90          JTable intfTable = new JTable(model);
91          JScrollPane scrollV = new JScrollPane(intfTable);
92          scrollV.setVerticalScrollBarPolicy(ScrollPaneConstants.VERTICAL_SCROLLBAR_AS_NEEDED);
93          resizeColumns(intfTable.getColumnModel());
94          add(scrollV);
95      }
96  
97      private static String buildParamsText(OperatingSystem os) {
98          NetworkParams params = os.getNetworkParams();
99          StringBuilder sb = new StringBuilder("Host Name: ").append(params.getHostName());
100         if (!params.getDomainName().isEmpty()) {
101             sb.append("\nDomain Name: ").append(params.getDomainName());
102         }
103         sb.append("\nIPv4 Default Gateway: ").append(params.getIpv4DefaultGateway());
104         if (!params.getIpv6DefaultGateway().isEmpty()) {
105             sb.append("\nIPv6 Default Gateway: ").append(params.getIpv6DefaultGateway());
106         }
107         sb.append("\nDNS Servers: ").append(getIPAddressesString(params.getDnsServers()));
108         return sb.toString();
109     }
110 
111     private static Object[][] parseInterfaces(List<NetworkIF> list) {
112         Map<NetworkIF, Integer> intfSortValueMap = new HashMap<>(INIT_HASH_SIZE);
113         for (NetworkIF intf : list) {
114             intfSortValueMap.put(intf, intf.getIndex());
115         }
116         List<Entry<NetworkIF, Integer>> intfList = new ArrayList<>(intfSortValueMap.entrySet());
117         intfList.sort(Entry.comparingByValue());
118 
119         int i = 0;
120         Object[][] intfArr = new Object[intfList.size()][COLUMNS.length];
121 
122         for (Entry<NetworkIF, Integer> e : intfList) {
123             NetworkIF intf = e.getKey();
124 
125             intfArr[i][0] = intf.getName();
126             intfArr[i][1] = intf.getIndex();
127             intfArr[i][2] = intf.getSpeed();
128             intfArr[i][3] = getIPAddressesString(intf.getIPv4addr());
129             intfArr[i][4] = getIPAddressesString(intf.getIPv6addr());
130             intfArr[i][5] = Constants.UNKNOWN.equals(intf.getMacaddr()) ? "" : intf.getMacaddr();
131 
132             i++;
133         }
134 
135         return intfArr;
136     }
137 
138     private static void resizeColumns(TableColumnModel tableColumnModel) {
139         TableColumn column;
140         int tW = tableColumnModel.getTotalColumnWidth();
141         int cantCols = tableColumnModel.getColumnCount();
142         for (int i = 0; i < cantCols; i++) {
143             column = tableColumnModel.getColumn(i);
144             int pWidth = (int) Math.round(COLUMN_WIDTH_PERCENT[i] * tW);
145             column.setPreferredWidth(pWidth);
146         }
147     }
148 
149     private static String getIPAddressesString(String[] ipAddressArr) {
150         StringBuilder sb = new StringBuilder();
151         boolean first = true;
152 
153         for (String ipAddress : ipAddressArr) {
154             if (first) {
155                 first = false;
156             } else {
157                 sb.append(IP_ADDRESS_SEPARATOR);
158             }
159             sb.append(ipAddress);
160         }
161 
162         return sb.toString();
163     }
