24  package oshi.demo;
25  
26  import oshi.SystemInfo;
27  import oshi.hardware.CentralProcessor;
28  import oshi.hardware.ComputerSystem;
29  import oshi.hardware.HardwareAbstractionLayer;
30  import oshi.software.os.OperatingSystem;
31  import oshi.util.Constants;
32  
33  /**
34   * Attempts to create a unique computer identifier. Note that serial numbers
35   * won't work on Linux without user cooperation.
36   */
37  public class ComputerID {
38      /**
39       * <p>
40       * main.
41       * </p>
42       *
43       * @param args
44       *            an array of {@link java.lang.String} objects.
45       */
46      public static void main(String[] args) {
47          String unknownHash = String.format("%08x", Constants.UNKNOWN.hashCode());
48  
49          System.out.println("Here's a unique (?) id for your computer.");
50          System.out.println(getComputerIdentifier());
51          System.out.println("If any field is " + unknownHash
52                  + " then I couldn't find a serial number or uuid, and running as sudo might change this.");
53      }
54  
55      /**
56       * Generates a Computer Identifier, which may be part of a strategy to construct
57       * a licence key. (The identifier may not be unique as in one case hashcode
58       * could be same for multiple values, and the result may differ based on whether
59       * the program is running with sudo/root permission.) The identifier string is
60       * based upon the processor serial number, vendor, processor identifier, and
61       * total processor count.
62       *
63       * @return A string containing four hyphen-delimited fields representing the
64       *         processor; the first 3 are 32-bit hexadecimal values and the last one
65       *         is an integer value.
66       */
67      public static String getComputerIdentifier() {
68          SystemInfo systemInfo = new SystemInfo();
69          OperatingSystem operatingSystem = systemInfo.getOperatingSystem();
70          HardwareAbstractionLayer hardwareAbstractionLayer = systemInfo.getHardware();
71          CentralProcessor centralProcessor = hardwareAbstractionLayer.getProcessor();
72          ComputerSystem computerSystem = hardwareAbstractionLayer.getComputerSystem();
73  
74          String vendor = operatingSystem.getManufacturer();
75          String processorSerialNumber = computerSystem.getSerialNumber();
76          String uuid = computerSystem.getHardwareUUID();
77          String processorIdentifier = centralProcessor.getProcessorIdentifier().getIdentifier();
78          int processors = centralProcessor.getLogicalProcessorCount();
79  
80          String delimiter = "-";
81  
82          return String.format("%08x", vendor.hashCode()) + delimiter
83                  + String.format("%08x", processorSerialNumber.hashCode()) + delimiter
84                  + String.format("%08x", uuid.hashCode()) + delimiter
85                  + String.format("%08x", processorIdentifier.hashCode()) + delimiter + processors;
86      }
87  }
