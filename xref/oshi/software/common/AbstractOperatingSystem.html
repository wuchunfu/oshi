53  public abstract class AbstractOperatingSystem implements OperatingSystem {

58      private final Supplier<String> manufacturer = memoize(this::queryManufacturer);
59      private final Supplier<Pair<String, OSVersionInfo>> familyVersionInfo = memoize(this::queryFamilyVersionInfo);
60      private final Supplier<Integer> bitness = memoize(this::queryPlatformBitness);

81      @Override
82      public int getBitness() {
83          return bitness.get();
84      }

86      private int queryPlatformBitness() {
87          if (Platform.is64Bit()) {
88              return 64;
89          }
90          // Initialize based on JVM Bitness. Individual OS implementations will test
91          // if 32-bit JVM running on 64-bit OS
92          int jvmBitness = System.getProperty("os.arch").contains("64") ? 64 : 32;
93          return queryBitness(jvmBitness);
94      }

97       * Backup OS-specific query to determine bitness if previous checks fail
98       *
99       * @param jvmBitness
100      *            The bitness of the JVM
101      * @return The operating system bitness
102      */
103     protected abstract int queryBitness(int jvmBitness);

105     @Override
106     public List<OSProcess> getProcesses(Predicate<OSProcess> filter, Comparator<OSProcess> sort, int limit) {
107         return queryAllProcesses().stream().filter(filter == null ? ALL_PROCESSES : filter)
108                 .sorted(sort == null ? NO_SORTING : sort).limit(limit > 0 ? limit : Long.MAX_VALUE)
109                 .collect(Collectors.toList());
110     }
111 
112     protected abstract List<OSProcess> queryAllProcesses();
113 
114     @Override
115     public List<OSProcess> getChildProcesses(int parentPid, Predicate<OSProcess> filter, Comparator<OSProcess> sort,
116             int limit) {
117         // Get this pid and its children
118         List<OSProcess> childProcs = queryChildProcesses(parentPid);
119         // Extract the parent from the list
120         OSProcess parent = childProcs.stream().filter(p -> p.getParentProcessID() == parentPid).findAny().orElse(null);
121         // Get the parent's start time
122         long parentStartTime = parent == null ? 0 : parent.getStartTime();
123         // Get children after parent
124         return queryChildProcesses(parentPid).stream().filter(filter == null ? ALL_PROCESSES : filter)
125                 .filter(p -> p.getProcessID() != parentPid && p.getStartTime() >= parentStartTime)
126                 .sorted(sort == null ? NO_SORTING : sort).limit(limit > 0 ? limit : Long.MAX_VALUE)
127                 .collect(Collectors.toList());
128     }
129 
130     protected abstract List<OSProcess> queryChildProcesses(int parentPid);
131 
132     @Override
133     public List<OSProcess> getDescendantProcesses(int parentPid, Predicate<OSProcess> filter,
134             Comparator<OSProcess> sort, int limit) {
135         // Get this pid and its descendants
136         List<OSProcess> descendantProcs = queryDescendantProcesses(parentPid);
137         // Extract the parent from the list
138         OSProcess parent = descendantProcs.stream().filter(p -> p.getParentProcessID() == parentPid).findAny()
139                 .orElse(null);
140         // Get the parent's start time
141         long parentStartTime = parent == null ? 0 : parent.getStartTime();
142         // Get descendants after parent
143         return queryDescendantProcesses(parentPid).stream().filter(filter == null ? ALL_PROCESSES : filter)
144                 .filter(p -> p.getProcessID() != parentPid && p.getStartTime() >= parentStartTime)
145                 .sorted(sort == null ? NO_SORTING : sort).limit(limit > 0 ? limit : Long.MAX_VALUE)
146                 .collect(Collectors.toList());
147     }
148 
149     protected abstract List<OSProcess> queryDescendantProcesses(int parentPid);
150 
151     /**
152      * Utility method for subclasses to take a full process list as input and return
153      * the children or descendants of a particular process. The process itself is
154      * also returned to more efficiently extract its start time for filtering
155      *
156      * @param allProcs
157      *            A collection of all processes
158      * @param parentPid
159      *            The process ID whose children or descendants to return
160      * @param allDescendants
161      *            If false, only gets immediate children of this process. If true,
162      *            gets all descendants.
163      * @return Set of children or descendants of parentPid
164      */
165     protected static Set<Integer> getChildrenOrDescendants(Collection<OSProcess> allProcs, int parentPid,
166             boolean allDescendants) {
167         Map<Integer, Integer> parentPidMap = allProcs.stream()
168                 .collect(Collectors.toMap(OSProcess::getProcessID, OSProcess::getParentProcessID));
169         return getChildrenOrDescendants(parentPidMap, parentPid, allDescendants);
170     }
171 
172     /**
173      * Utility method for subclasses to take a map of pid to parent as input and
174      * return the children or descendants of a particular process.
175      *
176      * @param parentPidMap
177      *            a map of all processes with processID as key and parentProcessID
178      *            as value
179      * @param parentPid
180      *            The process ID whose children or descendants to return
181      * @param allDescendants
182      *            If false, only gets immediate children of this process. If true,
183      *            gets all descendants.
184      * @return Set of children or descendants of parentPid, including the parent
185      */
186     protected static Set<Integer> getChildrenOrDescendants(Map<Integer, Integer> parentPidMap, int parentPid,
187             boolean allDescendants) {
188         // Set to hold results
189         Set<Integer> descendantPids = new HashSet<>();
190         descendantPids.add(parentPid);
191         // Queue for BFS algorithm
192         Queue<Integer> queue = new ArrayDeque<>();
193         queue.add(parentPid);
194         // Add children, repeating if recursive
195         do {
196             for (int pid : getChildren(parentPidMap, queue.poll())) {
197                 if (!descendantPids.contains(pid)) {
198                     descendantPids.add(pid);
199                     queue.add(pid);
200                 }
201             }
202         } while (allDescendants && !queue.isEmpty());
203         return descendantPids;
204     }
205 
206     private static Set<Integer> getChildren(Map<Integer, Integer> parentPidMap, int parentPid) {
207         return parentPidMap.entrySet().stream()
208                 .filter(e -> e.getValue().equals(parentPid) && !e.getKey().equals(parentPid)).map(Entry::getKey)
209                 .collect(Collectors.toSet());
210     }

212     @Override
213     public String toString() {
214         StringBuilder sb = new StringBuilder();
215         sb.append(getManufacturer()).append(' ').append(getFamily()).append(' ').append(getVersionInfo());
216         return sb.toString();
217     }
