42  /**
43   * Utility to query NetStat.
44   */
45  @ThreadSafe
46  public final class NetStat {
47  
48      private static final Logger LOG = LoggerFactory.getLogger(NetStat.class);
49  
50      private static final int CTL_NET = 4;
51      private static final int PF_ROUTE = 17;
52      private static final int NET_RT_IFLIST2 = 6;
53      private static final int RTM_IFINFO2 = 0x12;
54  
55      private NetStat() {
56      }
57  
58      /**
59       * Map data for network interfaces.
60       *
61       * @param index
62       *            If positive, limit the map to only return data for this interface
63       *            index. If negative, returns data for all indices.
64       *
65       * @return a map of {@link IFdata} object indexed by the interface index,
66       *         encapsulating the stats
67       */
68      public static Map<Integer, IFdata> queryIFdata(int index) {
69          // Ported from source code of "netstat -ir". See
70          // https://opensource.apple.com/source/network_cmds/network_cmds-457/netstat.tproj/if.c
71          Map<Integer, IFdata> data = new HashMap<>();
72          // Get buffer of all interface information
73          int[] mib = { CTL_NET, PF_ROUTE, 0, 0, NET_RT_IFLIST2, 0 };
74          size_t.ByReference len = new size_t.ByReference();
75          if (0 != SystemB.INSTANCE.sysctl(mib, 6, null, len, null, size_t.ZERO)) {
76              LOG.error("Didn't get buffer length for IFLIST2");
77              return data;
78          }
79          Memory buf = new Memory(len.longValue());
80          if (0 != SystemB.INSTANCE.sysctl(mib, 6, buf, len, null, size_t.ZERO)) {
81              LOG.error("Didn't get buffer for IFLIST2");
82              return data;
83          }
84          final long now = System.currentTimeMillis();
85  
86          // Iterate offset from buf's pointer up to limit of buf
87          int lim = (int) (buf.size() - new IFmsgHdr().size());
88          int offset = 0;
89          while (offset < lim) {
90              // Get pointer to current native part of buf
91              Pointer p = buf.share(offset);
92              // Cast pointer to if_msghdr
93              IFmsgHdr ifm = new IFmsgHdr(p);
94              ifm.read();
95              // Advance next
96              offset += ifm.ifm_msglen;
97              // Skip messages which are not the right format
98              if (ifm.ifm_type == RTM_IFINFO2) {
99                  // Cast pointer to if_msghdr2
100                 IFmsgHdr2 if2m = new IFmsgHdr2(p);
101                 if2m.read();
102                 if (index < 0 || index == if2m.ifm_index) {
103                     data.put((int) if2m.ifm_index,
104                             new IFdata(0xff & if2m.ifm_data.ifi_type, if2m.ifm_data.ifi_opackets,
105                                     if2m.ifm_data.ifi_ipackets, if2m.ifm_data.ifi_obytes, if2m.ifm_data.ifi_ibytes,
106                                     if2m.ifm_data.ifi_oerrors, if2m.ifm_data.ifi_ierrors, if2m.ifm_data.ifi_collisions,
107                                     if2m.ifm_data.ifi_iqdrops, if2m.ifm_data.ifi_baudrate, now));
108                     if (index >= 0) {
109                         return data;
110                     }
111                 }
112             }
113         }
114         return data;
115     }

117     /**
118      * Class to encapsulate IF data for method return
119      */
120     @Immutable
121     public static class IFdata {
122         private final int ifType;
123         private final long oPackets;
124         private final long iPackets;
125         private final long oBytes;
126         private final long iBytes;
127         private final long oErrors;
128         private final long iErrors;
129         private final long collisions;
130         private final long iDrops;
131         private final long speed;
132         private final long timeStamp;
133 
134         IFdata(int ifType, // NOSONAR squid:S00107
135                 long oPackets, long iPackets, long oBytes, long iBytes, long oErrors, long iErrors, long collisions,
136                 long iDrops, long speed, long timeStamp) {
137             this.ifType = ifType;
138             this.oPackets = oPackets & 0xffffffffL;
139             this.iPackets = iPackets & 0xffffffffL;
140             this.oBytes = oBytes & 0xffffffffL;
141             this.iBytes = iBytes & 0xffffffffL;
142             this.oErrors = oErrors & 0xffffffffL;
143             this.iErrors = iErrors & 0xffffffffL;
144             this.collisions = collisions & 0xffffffffL;
145             this.iDrops = iDrops & 0xffffffffL;
146             this.speed = speed & 0xffffffffL;
147             this.timeStamp = timeStamp;
148         }
149 
150         /**
151          * @return the ifType
152          */
153         public int getIfType() {
154             return ifType;
155         }
156 
157         /**
158          * @return the oPackets
159          */
160         public long getOPackets() {
161             return oPackets;
162         }
163 
164         /**
165          * @return the iPackets
166          */
167         public long getIPackets() {
168             return iPackets;
169         }
170 
171         /**
172          * @return the oBytes
173          */
174         public long getOBytes() {
175             return oBytes;
176         }
177 
178         /**
179          * @return the iBytes
180          */
181         public long getIBytes() {
182             return iBytes;
183         }
184 
185         /**
186          * @return the oErrors
187          */
188         public long getOErrors() {
189             return oErrors;
190         }
191 
192         /**
193          * @return the iErrors
194          */
195         public long getIErrors() {
196             return iErrors;
197         }
198 
199         /**
200          * @return the collisions
201          */
202         public long getCollisions() {
203             return collisions;
204         }
205 
206         /**
207          * @return the iDrops
208          */
209         public long getIDrops() {
210             return iDrops;
211         }
212 
213         /**
214          * @return the speed
215          */
216         public long getSpeed() {
217             return speed;
218         }
219 
220         /**
221          * @return the timeStamp
222          */
223         public long getTimeStamp() {
224             return timeStamp;
225         }
226     }
