43  /**
44   * This class represents the entire Central Processing Unit (CPU) of a computer
45   * system, which may contain one or more physical packages (sockets), one or
46   * more physical processors (cores), and one or more logical processors (what
47   * the Operating System sees, which may include hyperthreaded cores.)
48   */
49  @ThreadSafe
50  public interface CentralProcessor {
51  
52      /**
53       * The CPU's identifier strings ,including name, vendor, stepping, model, and
54       * family information (also called the signature of a CPU).
55       * <p>
56       * The Processor Identifier is primarily associated with Intel-based chips.
57       * Attempts are made to provide comparable values for other chip manufacturers.
58       *
59       * @return a {@link ProcessorIdentifier} object encapsulating CPU identifier
60       *         information.
61       */
62      ProcessorIdentifier getProcessorIdentifier();
63  
64      /**
65       * Maximum frequeny (in Hz), of the logical processors on this CPU.
66       *
67       * @return The max frequency or -1 if unknown.
68       */
69      long getMaxFreq();
70  
71      /**
72       * Attempts to return the current frequency (in Hz), of the logical processors
73       * on this CPU.
74       * <p>
75       * May not be implemented on all Operating Systems.
76       * <p>
77       * On Windows, returns an estimate based on the percent of maximum frequency. On
78       * Windows systems with more than 64 logical processors, may only return
79       * frequencies for the current processor group in the first portion of the
80       * array.
81       *
82       * @return An array of processor frequencies for each logical processor on the
83       *         system. Use the {@link #getLogicalProcessors()} to correlate these
84       *         frequencies with physical packages and processors.
85       */
86      long[] getCurrentFreq();
87  
88      /**
89       * Returns an {@code UnmodifiableList} of the CPU's logical processors. The list
90       * will be sorted in order of increasing NUMA node number, and then processor
91       * number. This order is consistent with other methods providing per-processor
92       * results.
93       *
94       * @return An {@code UnmodifiabeList} of logical processors.
95       */
96      List<LogicalProcessor> getLogicalProcessors();
97  
98      /**
99       * Returns the "recent cpu usage" for the whole system by counting ticks from
100      * {@link #getSystemCpuLoadTicks()} between the user-provided value from a
101      * previous call.
102      *
103      * @param oldTicks
104      *            A tick array from a previous call to
105      *            {@link #getSystemCpuLoadTicks()}
106      * @return CPU load between 0 and 1 (100%)
107      */
108     double getSystemCpuLoadBetweenTicks(long[] oldTicks);
109 
110     /**
111      * Get System-wide CPU Load tick counters. Returns an array with seven elements
112      * representing milliseconds spent in User (0), Nice (1), System (2), Idle (3),
113      * IOwait (4), Hardware interrupts (IRQ) (5), Software interrupts/DPC (SoftIRQ)
114      * (6), or Steal (7) states. Use
115      * {@link oshi.hardware.CentralProcessor.TickType#getIndex()} to retrieve the
116      * appropriate index. By measuring the difference between ticks across a time
117      * interval, CPU load over that interval may be calculated.
118      * <p>
119      * Note that while tick counters are in units of milliseconds, they may advance
120      * in larger increments along with (platform dependent) clock ticks. For
121      * example, by default Windows clock ticks are 1/64 of a second (about 15 or 16
122      * milliseconds) and Linux ticks are distribution and configuration dependent
123      * but usually 1/100 of a second (10 milliseconds).
124      * <p>
125      * Nice and IOWait information is not available on Windows, and IOwait and IRQ
126      * information is not available on macOS, so these ticks will always be zero.
127      * <p>
128      * To calculate overall Idle time using this method, include both Idle and
129      * IOWait ticks. Similarly, IRQ, SoftIRQ, and Steal ticks should be added to the
130      * System value to get the total. System ticks also include time executing other
131      * virtual hosts (steal).
132      *
133      * @return An array of 7 long values representing time spent in User, Nice,
134      *         System, Idle, IOwait, IRQ, SoftIRQ, and Steal states.
135      */
136     long[] getSystemCpuLoadTicks();
137 
138     /**
139      * Returns the system load average for the number of elements specified, up to
140      * 3, representing 1, 5, and 15 minutes. The system load average is the sum of
141      * the number of runnable entities queued to the available processors and the
142      * number of runnable entities running on the available processors averaged over
143      * a period of time. The way in which the load average is calculated is
144      * operating system specific but is typically a damped time-dependent average.
145      * If the load average is not available, a negative value is returned. This
146      * method is designed to provide a hint about the system load and may be queried
147      * frequently.
148      * <p>
149      * The load average may be unavailable on some platforms (e.g., Windows) where
150      * it is expensive to implement this method.
151      *
152      * @param nelem
153      *            Number of elements to return.
154      * @return an array of the system load averages for 1, 5, and 15 minutes with
155      *         the size of the array specified by nelem; or negative values if not
156      *         available.
157      */
158     double[] getSystemLoadAverage(int nelem);
159 
160     /**
161      * Returns the "recent cpu usage" for all logical processors by counting ticks
162      * from {@link #getProcessorCpuLoadTicks()} between the user-provided value from
163      * a previous call.
164      *
165      * @param oldTicks
166      *            A tick array from a previous call to
167      *            {@link #getProcessorCpuLoadTicks()}
168      * @return array of CPU load between 0 and 1 (100%) for each logical processor
169      */
170     double[] getProcessorCpuLoadBetweenTicks(long[][] oldTicks);
171 
172     /**
173      * Get Processor CPU Load tick counters. Returns a two dimensional array, with
174      * {@link #getLogicalProcessorCount()} arrays, each containing seven elements
175      * representing milliseconds spent in User (0), Nice (1), System (2), Idle (3),
176      * IOwait (4), Hardware interrupts (IRQ) (5), Software interrupts/DPC (SoftIRQ)
177      * (6), or Steal (7) states. Use
178      * {@link oshi.hardware.CentralProcessor.TickType#getIndex()} to retrieve the
179      * appropriate index. By measuring the difference between ticks across a time
180      * interval, CPU load over that interval may be calculated.
181      * <p>
182      * Note that while tick counters are in units of milliseconds, they may advance
183      * in larger increments along with (platform dependent) clock ticks. For
184      * example, by default Windows clock ticks are 1/64 of a second (about 15 or 16
185      * milliseconds) and Linux ticks are distribution and configuration dependent
186      * but usually 1/100 of a second (10 milliseconds).
187      * <p>
188      * Nice and IOwait per processor information is not available on Windows, and
189      * IOwait and IRQ information is not available on macOS, so these ticks will
190      * always be zero.
191      * <p>
192      * To calculate overall Idle time using this method, include both Idle and
193      * IOWait ticks. Similarly, IRQ, SoftIRQ and Steal ticks should be added to the
194      * System value to get the total. System ticks also include time executing other
195      * virtual hosts (steal).
196      *
197      * @return A 2D array of logicalProcessorCount x 7 long values representing time
198      *         spent in User, Nice, System, Idle, IOwait, IRQ, SoftIRQ, and Steal
199      *         states.
200      */
201     long[][] getProcessorCpuLoadTicks();
202 
203     /**
204      * Get the number of logical CPUs available for processing. This value may be
205      * higher than physical CPUs if hyperthreading is enabled.
206      *
207      * @return The number of logical CPUs available.
208      */
209     int getLogicalProcessorCount();
210 
211     /**
212      * Get the number of physical CPUs/cores available for processing.
213      *
214      * @return The number of physical CPUs available.
215      */
216     int getPhysicalProcessorCount();
217 
218     /**
219      * Get the number of packages/sockets in the system. A single package may
220      * contain multiple cores.
221      *
222      * @return The number of physical packages available.
223      */
224     int getPhysicalPackageCount();
225 
226     /**
227      * Get the number of system-wide context switches which have occurred.
228      * <p>
229      * Not available system-wide on macOS. Process- and Thread-level context
230      * switches are available from {@link OSProcess#getContextSwitches()} and
231      * {@link OSThread#getContextSwitches()}.
232      *
233      * @return The number of context switches, if this information is available; 0
234      *         otherwise.
235      */
236     long getContextSwitches();
237 
238     /**
239      * Get the number of system-wide interrupts which have occurred.
240      * <p>
241      * Not available system-wide on macOS.
242      *
243      * @return The number of interrupts, if this information is available; 0
244      *         otherwise.
245      */
246     long getInterrupts();
247 
248     /**
249      * Index of CPU tick counters in the {@link #getSystemCpuLoadTicks()} and
250      * {@link #getProcessorCpuLoadTicks()} arrays.
251      */
252     enum TickType {
253         /**
254          * CPU utilization that occurred while executing at the user level
255          * (application).
256          */
257         USER(0),
258         /**
259          * CPU utilization that occurred while executing at the user level with nice
260          * priority.
261          */
262         NICE(1),
263         /**
264          * CPU utilization that occurred while executing at the system level (kernel).
265          */
266         SYSTEM(2),
267         /**
268          * Time that the CPU or CPUs were idle and the system did not have an
269          * outstanding disk I/O request.
270          */
271         IDLE(3),
272         /**
273          * Time that the CPU or CPUs were idle during which the system had an
274          * outstanding disk I/O request.
275          */
276         IOWAIT(4),
277         /**
278          * Time that the CPU used to service hardware IRQs
279          */
280         IRQ(5),
281         /**
282          * Time that the CPU used to service soft IRQs
283          */
284         SOFTIRQ(6),
285         /**
286          * Time which the hypervisor dedicated for other guests in the system. Only
287          * supported on Linux and AIX
288          */
289         STEAL(7);
290 
291         private final int index;
292 
293         TickType(int value) {
294             this.index = value;
295         }
296 
297         /**
298          * @return The integer index of this ENUM in the processor tick arrays, which
299          *         matches the output of Linux /proc/cpuinfo
300          */
301         public int getIndex() {
302             return index;
303         }
304     }

306     /**
307      * A class representing a Logical Processor and its replationship to physical
308      * processors, physical packages, and logical groupings such as NUMA Nodes and
309      * Processor groups, useful for identifying processor topology.
310      */
311     @Immutable
312     class LogicalProcessor {
313         private final int processorNumber;
314         private final int physicalProcessorNumber;
315         private final int physicalPackageNumber;
316         private final int numaNode;
317         private final int processorGroup;
318 
319         /**
320          * @param processorNumber
321          *            the Processor number
322          * @param physicalProcessorNumber
323          *            the core number
324          * @param physicalPackageNumber
325          *            the package/socket number
326          */
327         public LogicalProcessor(int processorNumber, int physicalProcessorNumber, int physicalPackageNumber) {
328             this(processorNumber, physicalProcessorNumber, physicalPackageNumber, 0, 0);
329         }
330 
331         /**
332          * @param processorNumber
333          *            the Processor number
334          * @param physicalProcessorNumber
335          *            the core number
336          * @param physicalPackageNumber
337          *            the package/socket number
338          * @param numaNode
339          *            the NUMA node number
340          */
341         public LogicalProcessor(int processorNumber, int physicalProcessorNumber, int physicalPackageNumber,
342                 int numaNode) {
343             this(processorNumber, physicalProcessorNumber, physicalPackageNumber, numaNode, 0);
344         }
345 
346         /**
347          * @param processorNumber
348          *            the Processor number
349          * @param physicalProcessorNumber
350          *            the core number
351          * @param physicalPackageNumber
352          *            the package/socket number
353          * @param numaNode
354          *            the NUMA node number
355          * @param processorGroup
356          *            the Processor Group number
357          */
358         public LogicalProcessor(int processorNumber, int physicalProcessorNumber, int physicalPackageNumber,
359                 int numaNode, int processorGroup) {
360             this.processorNumber = processorNumber;
361             this.physicalProcessorNumber = physicalProcessorNumber;
362             this.physicalPackageNumber = physicalPackageNumber;
363             this.numaNode = numaNode;
364             this.processorGroup = processorGroup;
365         }
366 
367         /**
368          * The Logical Processor number as seen by the Operating System. Used for
369          * assigning process affinity and reporting CPU usage and other statistics.
370          *
371          * @return the processorNumber
372          */
373         public int getProcessorNumber() {
374             return processorNumber;
375         }
376 
377         /**
378          * The physical processor (core) id number assigned to this logical processor.
379          * Hyperthreaded logical processors which share the same physical processor will
380          * have the same number.
381          *
382          * @return the physicalProcessorNumber
383          */
384         public int getPhysicalProcessorNumber() {
385             return physicalProcessorNumber;
386         }
387 
388         /**
389          * The physical package (socket) id number assigned to this logical processor.
390          * Multicore CPU packages may have multiple physical processors which share the
391          * same number.
392          *
393          * @return the physicalPackageNumber
394          */
395         public int getPhysicalPackageNumber() {
396             return physicalPackageNumber;
397         }
398 
399         /**
400          * The NUMA node. If the operating system supports Non-Uniform Memory Access
401          * this identifies the node number. Set to 0 if the operating system does not
402          * support NUMA. Not supported on macOS or FreeBSD.
403          *
404          * @return the NUMA Node number
405          */
406         public int getNumaNode() {
407             return numaNode;
408         }
409 
410         /**
411          * The Processor Group. Only applies to Windows systems with more than 64
412          * logical processors. Set to 0 for other operating systems or Windows systems
413          * with 64 or fewer logical processors.
414          *
415          * @return the processorGroup
416          */
417         public int getProcessorGroup() {
418             return processorGroup;
419         }
420 
421         @Override
422         public String toString() {
423             return "LogicalProcessor [processorNumber=" + processorNumber + ", coreNumber=" + physicalProcessorNumber
424                     + ", packageNumber=" + physicalPackageNumber + ", numaNode=" + numaNode + ", processorGroup="
425                     + processorGroup + "]";
426         }
427     }

429     /**
430      * A class encapsulating ghe CPU's identifier strings ,including name, vendor,
431      * stepping, model, and family information (also called the signature of a CPU)
432      */
433     @Immutable
434     final class ProcessorIdentifier {
435         private static final String OSHI_ARCHITECTURE_PROPERTIES = "oshi.architecture.properties";
436 
437         // Provided in constructor
438         private final String cpuVendor;
439         private final String cpuName;
440         private final String cpuFamily;
441         private final String cpuModel;
442         private final String cpuStepping;
443         private final String processorID;
444         private final String cpuIdentifier;
445         private final boolean cpu64bit;
446         private final long cpuVendorFreq;
447 
448         private final Supplier<String> microArchictecture = memoize(this::queryMicroarchitecture);
449 
450         public ProcessorIdentifier(String cpuVendor, String cpuName, String cpuFamily, String cpuModel,
451                 String cpuStepping, String processorID, boolean cpu64bit) {
452             this(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit, -1L);
453         }
454 
455         public ProcessorIdentifier(String cpuVendor, String cpuName, String cpuFamily, String cpuModel,
456                 String cpuStepping, String processorID, boolean cpu64bit, long vendorFreq) {
457             this.cpuVendor = cpuVendor;
458             this.cpuName = cpuName;
459             this.cpuFamily = cpuFamily;
460             this.cpuModel = cpuModel;
461             this.cpuStepping = cpuStepping;
462             this.processorID = processorID;
463             this.cpu64bit = cpu64bit;
464 
465             // Build Identifier
466             StringBuilder sb = new StringBuilder();
467             if (cpuVendor.contentEquals("GenuineIntel")) {
468                 sb.append(cpu64bit ? "Intel64" : "x86");
469             } else {
470                 sb.append(cpuVendor);
471             }
472             sb.append(" Family ").append(cpuFamily);
473             sb.append(" Model ").append(cpuModel);
474             sb.append(" Stepping ").append(cpuStepping);
475             this.cpuIdentifier = sb.toString();
476 
477             if (vendorFreq > 0) {
478                 this.cpuVendorFreq = vendorFreq;
479             } else {
480                 // Parse Freq from name string
481                 Pattern pattern = Pattern.compile("@ (.*)$");
482                 Matcher matcher = pattern.matcher(cpuName);
483                 if (matcher.find()) {
484                     String unit = matcher.group(1);
485                     this.cpuVendorFreq = ParseUtil.parseHertz(unit);
486                 } else {
487                     this.cpuVendorFreq = -1L;
488                 }
489             }
490         }
491 
492         /**
493          * Processor vendor.
494          *
495          * @return vendor string.
496          */
497         public String getVendor() {
498             return cpuVendor;
499         }
500 
501         /**
502          * Name, eg. Intel(R) Core(TM)2 Duo CPU T7300 @ 2.00GHz
503          *
504          * @return Processor name.
505          */
506         public String getName() {
507             return cpuName;
508         }
509 
510         /**
511          * Gets the family. For non-Intel/AMD processors, returns the comparable value,
512          * such as the Architecture.
513          *
514          * @return the family
515          */
516         public String getFamily() {
517             return cpuFamily;
518         }
519 
520         /**
521          * Gets the model. For non-Intel/AMD processors, returns the comparable value,
522          * such as the Partnum.
523          *
524          * @return the model
525          */
526         public String getModel() {
527             return cpuModel;
528         }
529 
530         /**
531          * Gets the stepping. For non-Intel/AMD processors, returns the comparable
532          * value, such as the rnpn composite of Variant and Revision.
533          *
534          * @return the stepping
535          */
536         public String getStepping() {
537             return cpuStepping;
538         }
539 
540         /**
541          * Gets the Processor ID. This is a hexidecimal string representing an 8-byte
542          * value, normally obtained using the CPUID opcode with the EAX register set to
543          * 1. The first four bytes are the resulting contents of the EAX register, which
544          * is the Processor signature, represented in human-readable form by
545          * {@link #getIdentifier()} . The remaining four bytes are the contents of the
546          * EDX register, containing feature flags.
547          * <p>
548          * For processors that do not support the CPUID opcode this field is populated
549          * with a comparable hex string. For example, ARM Processors will fill the first
550          * 32 bytes with the MIDR. AIX PowerPC Processors will return the machine ID.
551          * <p>
552          * NOTE: The order of returned bytes is platform and software dependent. Values
553          * may be in either Big Endian or Little Endian order.
554          * <p>
555          * NOTE: If OSHI is unable to determine the ProcessorID from native sources, it
556          * will attempt to reconstruct one from available information in the processor
557          * identifier.
558          *
559          * @return A string representing the Processor ID
560          */
561         public String getProcessorID() {
562             return processorID;
563         }
564 
565         /**
566          * Identifier, eg. x86 Family 6 Model 15 Stepping 10. For non-Intel/AMD
567          * processors, this string is populated with comparable values.
568          *
569          * @return Processor identifier.
570          */
571         public String getIdentifier() {
572             return cpuIdentifier;
573         }
574 
575         /**
576          * Is CPU 64bit?
577          *
578          * @return True if cpu is 64bit.
579          */
580         public boolean isCpu64bit() {
581             return cpu64bit;
582         }
583 
584         /**
585          * Vendor frequency (in Hz), eg. for processor named Intel(R) Core(TM)2 Duo CPU
586          * T7300 @ 2.00GHz the vendor frequency is 2000000000.
587          *
588          * @return Processor frequency or -1 if unknown.
589          */
590         public long getVendorFreq() {
591             return cpuVendorFreq;
592         }
593 
594         /**
595          * Returns the processor's microarchitecture, if known.
596          *
597          * @return A string containing the microarchitecture if known.
598          *         {@link Constants#UNKNOWN} otherwise.
599          */
600         public String getMicroarchitecture() {
601             return microArchictecture.get();
602         }

604         private String queryMicroarchitecture() {
605             String arch = null;
606             Properties archProps = FileUtil.readPropertiesFromFilename(OSHI_ARCHITECTURE_PROPERTIES);
607             // Intel is default, no prefix
608             StringBuilder sb = new StringBuilder();
609             // AMD and ARM properties have prefix
610             String ucVendor = this.cpuVendor.toUpperCase();
611             if (ucVendor.contains("AMD")) {
612                 sb.append("amd.");
613             } else if (ucVendor.contains("ARM")) {
614                 sb.append("arm.");
615             } else if (ucVendor.contains("IBM")) {
616                 // Directly parse the name to POWER#
617                 int powerIdx = this.cpuName.indexOf("_POWER");
618                 if (powerIdx > 0) {
619                     arch = this.cpuName.substring(powerIdx + 1);
620                 }
621             } else if (ucVendor.contains("APPLE")) {
622                 sb.append("apple.");
623             }
624             if (Util.isBlank(arch) && !sb.toString().equals("arm.")) {
625                 // Append family
626                 sb.append(this.cpuFamily);
627                 arch = archProps.getProperty(sb.toString());
628             }
629 
630             if (Util.isBlank(arch)) {
631                 // Append model
632                 sb.append('.').append(this.cpuModel);
633                 arch = archProps.getProperty(sb.toString());
634             }
635 
636             if (Util.isBlank(arch)) {
637                 // Append stepping
638                 sb.append('.').append(this.cpuStepping);
639                 arch = archProps.getProperty(sb.toString());
640             }
641 
642             return Util.isBlank(arch) ? Constants.UNKNOWN : arch;
643         }
644 
645         @Override
646         public String toString() {
647             return getIdentifier();
648         }
