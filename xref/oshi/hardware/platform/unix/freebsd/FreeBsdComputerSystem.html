42  /**
43   * Hardware data obtained from dmidecode.
44   */
45  @Immutable
46  final class FreeBsdComputerSystem extends AbstractComputerSystem {
47  
48      private final Supplier<Quintet<String, String, String, String, String>> manufModelSerialUuidVers = memoize(
49              FreeBsdComputerSystem::readDmiDecode);
50  
51      @Override
52      public String getManufacturer() {
53          return manufModelSerialUuidVers.get().getA();
54      }
55  
56      @Override
57      public String getModel() {
58          return manufModelSerialUuidVers.get().getB();
59      }
60  
61      @Override
62      public String getSerialNumber() {
63          return manufModelSerialUuidVers.get().getC();
64      }
65  
66      @Override
67      public String getHardwareUUID() {
68          return manufModelSerialUuidVers.get().getD();
69      }
70  
71      @Override
72      public Firmware createFirmware() {
73          return new FreeBsdFirmware();
74      }
75  
76      @Override
77      public Baseboard createBaseboard() {
78          return new UnixBaseboard(manufModelSerialUuidVers.get().getA(), manufModelSerialUuidVers.get().getB(),
79                  manufModelSerialUuidVers.get().getC(), manufModelSerialUuidVers.get().getE());
80      }
81  
82      private static Quintet<String, String, String, String, String> readDmiDecode() {
83          String manufacturer = null;
84          String model = null;
85          String serialNumber = null;
86          String uuid = null;
87          String version = null;
88  
89          // $ sudo dmidecode -t system
90          // # dmidecode 3.0
91          // Scanning /dev/mem for entry point.
92          // SMBIOS 2.7 present.
93          //
94          // Handle 0x0001, DMI type 1, 27 bytes
95          // System Information
96          // Manufacturer: Parallels Software International Inc.
97          // Product Name: Parallels Virtual Platform
98          // Version: None
99          // Serial Number: Parallels-47 EC 38 2A 33 1B 4C 75 94 0F F7 AF 86 63 C0
100         // C4
101         // UUID: 2A38EC47-1B33-854C-940F-F7AF8663C0C4
102         // Wake-up Type: Power Switch
103         // SKU Number: Undefined
104         // Family: Parallels VM
105         //
106         // Handle 0x0016, DMI type 32, 20 bytes
107         // System Boot Information
108         // Status: No errors detected
109 
110         final String manufacturerMarker = "Manufacturer:";
111         final String productNameMarker = "Product Name:";
112         final String serialNumMarker = "Serial Number:";
113         final String uuidMarker = "UUID:";
114         final String versionMarker = "Version:";
115 
116         // Only works with root permissions but it's all we've got
117         for (final String checkLine : ExecutingCommand.runNative("dmidecode -t system")) {
118             if (checkLine.contains(manufacturerMarker)) {
119                 manufacturer = checkLine.split(manufacturerMarker)[1].trim();
120             } else if (checkLine.contains(productNameMarker)) {
121                 model = checkLine.split(productNameMarker)[1].trim();
122             } else if (checkLine.contains(serialNumMarker)) {
123                 serialNumber = checkLine.split(serialNumMarker)[1].trim();
124             } else if (checkLine.contains(uuidMarker)) {
125                 uuid = checkLine.split(uuidMarker)[1].trim();
126             } else if (checkLine.contains(versionMarker)) {
127                 version = checkLine.split(versionMarker)[1].trim();
128             }
129         }
130         // If we get to end and haven't assigned, use fallback
131         if (Util.isBlank(serialNumber)) {
132             serialNumber = querySystemSerialNumber();
133         }
134         if (Util.isBlank(uuid)) {
135             uuid = BsdSysctlUtil.sysctl("kern.hostuuid", Constants.UNKNOWN);
136         }
137         return new Quintet<>(Util.isBlank(manufacturer) ? Constants.UNKNOWN : manufacturer,
138                 Util.isBlank(model) ? Constants.UNKNOWN : model,
139                 Util.isBlank(serialNumber) ? Constants.UNKNOWN : serialNumber,
140                 Util.isBlank(uuid) ? Constants.UNKNOWN : uuid, Util.isBlank(version) ? Constants.UNKNOWN : version);
141     }
142 
143     private static String querySystemSerialNumber() {
144         String marker = "system.hardware.serial =";
145         for (String checkLine : ExecutingCommand.runNative("lshal")) {
146             if (checkLine.contains(marker)) {
147                 return ParseUtil.getSingleQuoteStringValue(checkLine);
148             }
149         }
150         return Constants.UNKNOWN;
151     }
152 }
