128     @Override
129     public boolean updateAttributes() {
130         // Open a session and create CFStrings
131         DASessionRef session = DA.DASessionCreate(CF.CFAllocatorGetDefault());
132         if (session == null) {
133             LOG.error("Unable to open session to DiskArbitration framework.");
134             return false;
135         }
136         Map<CFKey, CFStringRef> cfKeyMap = mapCFKeys();
137         // Execute the update
138         boolean diskFound = updateDiskStats(session, Fsstat.queryPartitionToMountMap(), cfKeyMap);
139         // Release the session and CFStrings
140         session.release();
141         for (CFTypeRef value : cfKeyMap.values()) {
142             value.release();
143         }
144 
145         return diskFound;
146     }

148     private boolean updateDiskStats(DASessionRef session, Map<String, String> mountPointMap,
149             Map<CFKey, CFStringRef> cfKeyMap) {
150         // Now look up the device using the BSD Name to get its
151         // statistics
152         String bsdName = getName();
153         CFMutableDictionaryRef matchingDict = IOKitUtil.getBSDNameMatchingDict(bsdName);
154         if (matchingDict != null) {
155             // search for all IOservices that match the bsd name
156             IOIterator driveListIter = IOKitUtil.getMatchingServices(matchingDict);
157             if (driveListIter != null) {
158                 // getMatchingServices releases matchingDict
159                 IORegistryEntry drive = driveListIter.next();
160                 // Should only match one drive
161                 if (drive != null) {
162                     // Should be an IOMedia object with a parent
163                     // IOBlockStorageDriver or AppleAPFSContainerScheme object
164                     // Get the properties from the parent
165                     if (drive.conformsTo("IOMedia")) {
166                         IORegistryEntry parent = drive.getParentEntry("IOService");
167                         if (parent != null && (parent.conformsTo("IOBlockStorageDriver")
168                                 || parent.conformsTo("AppleAPFSContainerScheme"))) {
169                             CFMutableDictionaryRef properties = parent.createCFProperties();
170                             // We now have a properties object with the
171                             // statistics we need on it. Fetch them
172                             Pointer result = properties.getValue(cfKeyMap.get(CFKey.STATISTICS));
173                             CFDictionaryRef statistics = new CFDictionaryRef(result);
174                             this.timeStamp = System.currentTimeMillis();
175 
176                             // Now get the stats we want
177                             result = statistics.getValue(cfKeyMap.get(CFKey.READ_OPS));
178                             CFNumberRef stat = new CFNumberRef(result);
179                             this.reads = stat.longValue();
180                             result = statistics.getValue(cfKeyMap.get(CFKey.READ_BYTES));
181                             stat.setPointer(result);
182                             this.readBytes = stat.longValue();
183 
184                             result = statistics.getValue(cfKeyMap.get(CFKey.WRITE_OPS));
185                             stat.setPointer(result);
186                             this.writes = stat.longValue();
187                             result = statistics.getValue(cfKeyMap.get(CFKey.WRITE_BYTES));
188                             stat.setPointer(result);
189                             this.writeBytes = stat.longValue();
190 
191                             // Total time is in nanoseconds. Add read+write
192                             // and convert total to ms
193                             final Pointer readTimeResult = statistics.getValue(cfKeyMap.get(CFKey.READ_TIME));
194                             final Pointer writeTimeResult = statistics.getValue(cfKeyMap.get(CFKey.WRITE_TIME));
195                             // AppleAPFSContainerScheme does not have timer statistics
196                             if (readTimeResult != null && writeTimeResult != null) {
197                                 stat.setPointer(readTimeResult);
198                                 long xferTime = stat.longValue();
199                                 stat.setPointer(writeTimeResult);
200                                 xferTime += stat.longValue();
201                                 this.transferTime = xferTime / 1_000_000L;
202                             }
203 
204                             properties.release();
205                         } else {
206                             // This is normal for FileVault drives, Fusion
207                             // drives, and other virtual bsd names
208                             LOG.debug("Unable to find block storage driver properties for {}", bsdName);
209                         }
210                         // Now get partitions for this disk.
211                         List<HWPartition> partitions = new ArrayList<>();
212 
213                         CFMutableDictionaryRef properties = drive.createCFProperties();
214                         // Partitions will match BSD Unit property
215                         Pointer result = properties.getValue(cfKeyMap.get(CFKey.BSD_UNIT));
216                         CFNumberRef bsdUnit = new CFNumberRef(result);
217                         // We need a CFBoolean that's false.
218                         // Whole disk has 'true' for Whole and 'false'
219                         // for leaf; store the boolean false
220                         result = properties.getValue(cfKeyMap.get(CFKey.LEAF));
221                         CFBooleanRef cfFalse = new CFBooleanRef(result);
222                         // create a matching dict for BSD Unit
223                         CFMutableDictionaryRef propertyDict = CF.CFDictionaryCreateMutable(CF.CFAllocatorGetDefault(),
224                                 new CFIndex(0), null, null);
225                         propertyDict.setValue(cfKeyMap.get(CFKey.BSD_UNIT), bsdUnit);
226                         propertyDict.setValue(cfKeyMap.get(CFKey.WHOLE), cfFalse);
227                         matchingDict = CF.CFDictionaryCreateMutable(CF.CFAllocatorGetDefault(), new CFIndex(0), null,
228                                 null);
229                         matchingDict.setValue(cfKeyMap.get(CFKey.IO_PROPERTY_MATCH), propertyDict);
230 
231                         // search for IOservices that match the BSD Unit
232                         // with whole=false; these are partitions
233                         IOIterator serviceIterator = IOKitUtil.getMatchingServices(matchingDict);
234                         // getMatchingServices releases matchingDict
235                         properties.release();
236                         propertyDict.release();
237 
238                         if (serviceIterator != null) {
239                             // Iterate disks
240                             IORegistryEntry sdService = IOKit.INSTANCE.IOIteratorNext(serviceIterator);
241                             while (sdService != null) {
242                                 // look up the BSD Name
243                                 String partBsdName = sdService.getStringProperty("BSD Name");
244                                 String name = partBsdName;
245                                 String type = "";
246                                 // Get the DiskArbitration dictionary for
247                                 // this partition
248                                 DADiskRef disk = DA.DADiskCreateFromBSDName(CF.CFAllocatorGetDefault(), session,
249                                         partBsdName);
250                                 if (disk != null) {
251                                     CFDictionaryRef diskInfo = DA.DADiskCopyDescription(disk);
252                                     if (diskInfo != null) {
253                                         // get volume name from its key
254                                         result = diskInfo.getValue(cfKeyMap.get(CFKey.DA_MEDIA_NAME));
255                                         type = CFUtil.cfPointerToString(result);
256                                         result = diskInfo.getValue(cfKeyMap.get(CFKey.DA_VOLUME_NAME));
257                                         if (result == null) {
258                                             name = type;
259                                         } else {
260                                             name = CFUtil.cfPointerToString(result);
261                                         }
262                                         diskInfo.release();
263                                     }
264                                     disk.release();
265                                 }
266                                 String mountPoint = mountPointMap.getOrDefault(partBsdName, "");
267                                 Long size = sdService.getLongProperty("Size");
268                                 Integer bsdMajor = sdService.getIntegerProperty("BSD Major");
269                                 Integer bsdMinor = sdService.getIntegerProperty("BSD Minor");
270                                 String uuid = sdService.getStringProperty("UUID");
271                                 partitions.add(new HWPartition(partBsdName, name, type,
272                                         uuid == null ? Constants.UNKNOWN : uuid, size == null ? 0L : size,
273                                         bsdMajor == null ? 0 : bsdMajor, bsdMinor == null ? 0 : bsdMinor, mountPoint));
274                                 // iterate
275                                 sdService.release();
276                                 sdService = IOKit.INSTANCE.IOIteratorNext(serviceIterator);
277                             }
278                             serviceIterator.release();
279                         }
280                         this.partitionList = Collections.unmodifiableList(partitions.stream()
281                                 .sorted(Comparator.comparing(HWPartition::getName)).collect(Collectors.toList()));
282                         if (parent != null) {
283                             parent.release();
284                         }
285                     } else {
286                         LOG.error("Unable to find IOMedia device or parent for {}", bsdName);
287                     }
288                     drive.release();
289                 }
290                 driveListIter.release();
291                 return true;
292             }
293         }
294         return false;
295     }

297     /**
298      * Gets the disks on this machine
299      *
300      * @return a list of {@link HWDiskStore} objects representing the disks
301      */
302     public static List<HWDiskStore> getDisks() {
303         Map<String, String> mountPointMap = Fsstat.queryPartitionToMountMap();
304         Map<CFKey, CFStringRef> cfKeyMap = mapCFKeys();
305 
306         List<HWDiskStore> diskList = new ArrayList<>();
307 
308         // Open a DiskArbitration session
309         DASessionRef session = DA.DASessionCreate(CF.CFAllocatorGetDefault());
310         if (session == null) {
311             LOG.error("Unable to open session to DiskArbitration framework.");
312             return Collections.emptyList();
313         }
314 
315         // Get IOMedia objects representing whole drives
316         List<String> bsdNames = new ArrayList<>();
317         IOIterator iter = IOKitUtil.getMatchingServices("IOMedia");
318         if (iter != null) {
319             IORegistryEntry media = iter.next();
320             while (media != null) {
321                 Boolean whole = media.getBooleanProperty("Whole");
322                 if (whole != null && whole) {
323                     DADiskRef disk = DA.DADiskCreateFromIOMedia(CF.CFAllocatorGetDefault(), session, media);
324                     bsdNames.add(DA.DADiskGetBSDName(disk));
325                     disk.release();
326                 }
327                 media.release();
328                 media = iter.next();
329             }
330             iter.release();
331         }
332 
333         // Now iterate the bsdNames
334         for (String bsdName : bsdNames) {
335             String model = "";
336             String serial = "";
337             long size = 0L;
338 
339             // Get a reference to the disk - only matching /dev/disk*
340             String path = "/dev/" + bsdName;
341 
342             // Get the DiskArbitration dictionary for this disk, which has model
343             // and size (capacity)
344             DADiskRef disk = DA.DADiskCreateFromBSDName(CF.CFAllocatorGetDefault(), session, path);
345             if (disk != null) {
346                 CFDictionaryRef diskInfo = DA.DADiskCopyDescription(disk);
347                 if (diskInfo != null) {
348                     // Parse out model and size from their respective keys
349                     Pointer result = diskInfo.getValue(cfKeyMap.get(CFKey.DA_DEVICE_MODEL));
350                     model = CFUtil.cfPointerToString(result);
351                     result = diskInfo.getValue(cfKeyMap.get(CFKey.DA_MEDIA_SIZE));
352                     CFNumberRef sizePtr = new CFNumberRef(result);
353                     size = sizePtr.longValue();
354                     diskInfo.release();
355 
356                     // Use the model as a key to get serial from IOKit
357                     if (!"Disk Image".equals(model)) {
358                         CFStringRef modelNameRef = CFStringRef.createCFString(model);
359                         CFMutableDictionaryRef propertyDict = CF.CFDictionaryCreateMutable(CF.CFAllocatorGetDefault(),
360                                 new CFIndex(0), null, null);
361                         propertyDict.setValue(cfKeyMap.get(CFKey.MODEL), modelNameRef);
362                         CFMutableDictionaryRef matchingDict = CF.CFDictionaryCreateMutable(CF.CFAllocatorGetDefault(),
363                                 new CFIndex(0), null, null);
364                         matchingDict.setValue(cfKeyMap.get(CFKey.IO_PROPERTY_MATCH), propertyDict);
365 
366                         // search for all IOservices that match the model
367                         IOIterator serviceIterator = IOKitUtil.getMatchingServices(matchingDict);
368                         // getMatchingServices releases matchingDict
369                         modelNameRef.release();
370                         propertyDict.release();
371 
372                         if (serviceIterator != null) {
373                             IORegistryEntry sdService = serviceIterator.next();
374                             while (sdService != null) {
375                                 // look up the serial number
376                                 serial = sdService.getStringProperty("Serial Number");
377                                 sdService.release();
378                                 if (serial != null) {
379                                     break;
380                                 }
381                                 // iterate
382                                 sdService.release();
383                                 sdService = serviceIterator.next();
384                             }
385                             serviceIterator.release();
386                         }
387                         if (serial == null) {
388                             serial = "";
389                         }
390                     }
391                 }
392                 disk.release();
393 
394                 // If empty, ignore
395                 if (size <= 0) {
396                     continue;
397                 }
398                 HWDiskStore diskStore = new MacHWDiskStore(bsdName, model.trim(), serial.trim(), size, session,
399                         mountPointMap, cfKeyMap);
400                 diskList.add(diskStore);
401             }
402         }
403         // Close DA session
404         session.release();
405         for (CFTypeRef value : cfKeyMap.values()) {
406             value.release();
407         }
408         return diskList;
409     }

411     /**
412      * Temporarily cache pointers to keys. The values from this map must be released
413      * after use.}
414      *
415      * @return A map of keys in the {@link CFKey} enum to corresponding
416      *         {@link CFStringRef}.
417      */
418     private static Map<CFKey, CFStringRef> mapCFKeys() {
419         Map<CFKey, CFStringRef> keyMap = new EnumMap<>(CFKey.class);
420         for (CFKey cfKey : CFKey.values()) {
421             keyMap.put(cfKey, CFStringRef.createCFString(cfKey.getKey()));
422         }
423         return keyMap;
424     }

426     /*
427      * Strings to convert to CFStringRef for pointer lookups
428      */
429     private enum CFKey {
430         IO_PROPERTY_MATCH("IOPropertyMatch"), //
431 
432         STATISTICS("Statistics"), //
433         READ_OPS("Operations (Read)"), READ_BYTES("Bytes (Read)"), READ_TIME("Total Time (Read)"), //
434         WRITE_OPS("Operations (Write)"), WRITE_BYTES("Bytes (Write)"), WRITE_TIME("Total Time (Write)"), //
435 
436         BSD_UNIT("BSD Unit"), LEAF("Leaf"), WHOLE("Whole"), //
437 
438         DA_MEDIA_NAME("DAMediaName"), DA_VOLUME_NAME("DAVolumeName"), DA_MEDIA_SIZE("DAMediaSize"), //
439         DA_DEVICE_MODEL("DADeviceModel"), MODEL("Model");
440 
441         private final String key;
442 
443         CFKey(String key) {
444             this.key = key;
445         }
446 
447         public String getKey() {
448             return this.key;
449         }
