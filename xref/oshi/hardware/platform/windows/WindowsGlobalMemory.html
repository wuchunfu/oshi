52  /**
53   * Memory obtained by Performance Info.
54   */
55  @ThreadSafe
56  final class WindowsGlobalMemory extends AbstractGlobalMemory {
57  
58      private static final Logger LOG = LoggerFactory.getLogger(WindowsGlobalMemory.class);
59  
60      private static final boolean IS_WINDOWS10_OR_GREATER = VersionHelpers.IsWindows10OrGreater();
61  
62      private final Supplier<Triplet<Long, Long, Long>> availTotalSize = memoize(WindowsGlobalMemory::readPerfInfo,
63              defaultExpiration());
64  
65      private final Supplier<VirtualMemory> vm = memoize(this::createVirtualMemory);
66  
67      @Override
68      public long getAvailable() {
69          return availTotalSize.get().getA();
70      }
71  
72      @Override
73      public long getTotal() {
74          return availTotalSize.get().getB();
75      }
76  
77      @Override
78      public long getPageSize() {
79          return availTotalSize.get().getC();
80      }
81  
82      @Override
83      public VirtualMemory getVirtualMemory() {
84          return vm.get();
85      }
86  
87      private VirtualMemory createVirtualMemory() {
88          return new WindowsVirtualMemory(this);
89      }
90  
91      @Override
92      public List<PhysicalMemory> getPhysicalMemory() {
93          List<PhysicalMemory> physicalMemoryList = new ArrayList<>();
94          if (IS_WINDOWS10_OR_GREATER) {
95              WmiResult<PhysicalMemoryProperty> bankMap = Win32PhysicalMemory.queryphysicalMemory();
96              for (int index = 0; index < bankMap.getResultCount(); index++) {
97                  String bankLabel = WmiUtil.getString(bankMap, PhysicalMemoryProperty.BANKLABEL, index);
98                  long capacity = WmiUtil.getUint64(bankMap, PhysicalMemoryProperty.CAPACITY, index);
99                  long speed = WmiUtil.getUint32(bankMap, PhysicalMemoryProperty.SPEED, index) * 1_000_000L;
100                 String manufacturer = WmiUtil.getString(bankMap, PhysicalMemoryProperty.MANUFACTURER, index);
101                 String memoryType = smBiosMemoryType(
102                         WmiUtil.getUint32(bankMap, PhysicalMemoryProperty.SMBIOSMEMORYTYPE, index));
103                 physicalMemoryList.add(new PhysicalMemory(bankLabel, capacity, speed, manufacturer, memoryType));
104             }
105         } else {
106             WmiResult<PhysicalMemoryPropertyWin8> bankMap = Win32PhysicalMemory.queryphysicalMemoryWin8();
107             for (int index = 0; index < bankMap.getResultCount(); index++) {
108                 String bankLabel = WmiUtil.getString(bankMap, PhysicalMemoryPropertyWin8.BANKLABEL, index);
109                 long capacity = WmiUtil.getUint64(bankMap, PhysicalMemoryPropertyWin8.CAPACITY, index);
110                 long speed = WmiUtil.getUint32(bankMap, PhysicalMemoryPropertyWin8.SPEED, index) * 1_000_000L;
111                 String manufacturer = WmiUtil.getString(bankMap, PhysicalMemoryPropertyWin8.MANUFACTURER, index);
112                 String memoryType = memoryType(
113                         WmiUtil.getUint16(bankMap, PhysicalMemoryPropertyWin8.MEMORYTYPE, index));
114                 physicalMemoryList.add(new PhysicalMemory(bankLabel, capacity, speed, manufacturer, memoryType));
115             }
116         }
117         return physicalMemoryList;
118     }

148             return "ROM";
149         case 11:
150             return "Flash";
151         case 12:
152             return "EEPROM";
153         case 13:
154             return "FEPROM";
155         case 14:
156             return "EPROM";
157         case 15:
158             return "CDRAM";
159         case 16:
160             return "3DRAM";
161         case 17:
162             return "SDRAM";
163         case 18:

182     /**
183      * Convert SMBIOS type number to a human readable string
184      *
185      * @param type
186      *            The SMBIOS type
187      * @return A string describing the type
188      */
189     private static String smBiosMemoryType(int type) {
190         // https://www.dmtf.org/sites/default/files/standards/documents/DSP0134_3.2.0.pdf
191         // table 76
192         switch (type) {
193         case 0x01:
194             return "Other";
195         case 0x03:
196             return "DRAM";
197         case 0x04:
198             return "EDRAM";
199         case 0x05:
200             return "VRAM";
201         case 0x06:
202             return "SRAM";
203         case 0x07:
204             return "RAM";
205         case 0x08:
206             return "ROM";
207         case 0x09:
208             return "FLASH";
209         case 0x0A:
210             return "EEPROM";
211         case 0x0B:
212             return "FEPROM";
213         case 0x0C:
214             return "EPROM";
215         case 0x0D:
216             return "CDRAM";
217         case 0x0E:
218             return "3DRAM";
219         case 0x0F:
220             return "SDRAM";
221         case 0x10:
222             return "SGRAM";
223         case 0x11:
224             return "RDRAM";
225         case 0x12:
226             return "DDR";
227         case 0x13:
228             return "DDR2";
229         case 0x14:
230             return "DDR2 FB-DIMM";
231         case 0x18:
232             return "DDR3";
233         case 0x19:
234             return "FBD2";
235         case 0x1A:
236             return "DDR4";
237         case 0x1B:
238             return "LPDDR";
239         case 0x1C:
240             return "LPDDR2";
241         case 0x1D:
242             return "LPDDR3";
243         case 0x1E:
244             return "LPDDR4";
245         case 0x1F:
246             return "Logical non-volatile device";
247         case 0x02:
248         default:
249             return "Unknown";
250         }
251     }

253     private static Triplet<Long, Long, Long> readPerfInfo() {
254         PERFORMANCE_INFORMATION performanceInfo = new PERFORMANCE_INFORMATION();
255         if (!Psapi.INSTANCE.GetPerformanceInfo(performanceInfo, performanceInfo.size())) {
256             LOG.error("Failed to get Performance Info. Error code: {}", Kernel32.INSTANCE.GetLastError());
257             return new Triplet<>(0L, 0L, 4098L);
258         }
259         long pageSize = performanceInfo.PageSize.longValue();
260         long memAvailable = pageSize * performanceInfo.PhysicalAvailable.longValue();
261         long memTotal = pageSize * performanceInfo.PhysicalTotal.longValue();
262         return new Triplet<>(memAvailable, memTotal, pageSize);
263     }
