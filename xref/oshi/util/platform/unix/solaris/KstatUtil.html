45  /**
46   * Provides access to kstat information on Solaris
47   */
48  @ThreadSafe
49  public final class KstatUtil {
50  
51      private static final Logger LOG = LoggerFactory.getLogger(KstatUtil.class);
52  
53      private static final LibKstat KS = LibKstat.INSTANCE;
54  
55      // Opens the kstat chain. Automatically closed on exit.
56      // Only one thread may access the chain at any time, so we wrap this object in
57      // the KstatChain class which locks the class until closed.
58      private static final KstatCtl KC = KS.kstat_open();
59      private static final ReentrantLock CHAIN = new ReentrantLock();
60  
61      private KstatUtil() {
62      }
63  
64      /**
65       * A copy of the Kstat chain, encapsulating a {@code kstat_ctl_t} object. Only
66       * one thread may actively use this object at any time.
67       * <p>
68       * Instantiating this object is accomplished using the
69       * {@link KstatUtil#openChain} method. It locks and updates the chain and is the
70       * equivalent of calling {@link LibKstat#kstat_open}. The control object should
71       * be closed with {@link #close}, the equivalent of calling
72       * {@link LibKstat#kstat_close}
73       */
74      public static final class KstatChain implements AutoCloseable {
75  
76          private KstatChain() {
77              CHAIN.lock();
78              update();
79          }
80  
81          /**
82           * Convenience method for {@link LibKstat#kstat_read} which gets data from the
83           * kernel for the kstat pointed to by {@code ksp}. {@code ksp.ks_data} is
84           * automatically allocated (or reallocated) to be large enough to hold all of
85           * the data. {@code ksp.ks_ndata} is set to the number of data fields,
86           * {@code ksp.ks_data_size} is set to the total size of the data, and
87           * ksp.ks_snaptime is set to the high-resolution time at which the data snapshot
88           * was taken.
89           *
90           * @param ksp
91           *            The kstat from which to retrieve data
92           * @return {@code true} if successful; {@code false} otherwise
93           */
94          public static boolean read(Kstat ksp) {
95              int retry = 0;
96              while (0 > KS.kstat_read(KC, ksp, null)) {
97                  if (LibKstat.EAGAIN != Native.getLastError() || 5 <= ++retry) {
98                      if (LOG.isDebugEnabled()) {
99                          LOG.debug("Failed to read kstat {}:{}:{}",
100                                 Native.toString(ksp.ks_module, StandardCharsets.US_ASCII), ksp.ks_instance,
101                                 Native.toString(ksp.ks_name, StandardCharsets.US_ASCII));
102                     }
103                     return false;
104                 }
105                 Util.sleep(8 << retry);
106             }
107             return true;
108         }
109 
110         /**
111          * Convenience method for {@link LibKstat#kstat_lookup}. Traverses the kstat
112          * chain, searching for a kstat with the same {@code module}, {@code instance},
113          * and {@code name} fields; this triplet uniquely identifies a kstat. If
114          * {@code module} is {@code null}, {@code instance} is -1, or {@code name} is
115          * {@code null}, then those fields will be ignored in the search.
116          *
117          * @param module
118          *            The module, or null to ignore
119          * @param instance
120          *            The instance, or -1 to ignore
121          * @param name
122          *            The name, or null to ignore
123          * @return The first match of the requested Kstat structure if found, or
124          *         {@code null}
125          */
126         public static Kstat lookup(String module, int instance, String name) {
127             return KS.kstat_lookup(KC, module, instance, name);
128         }
129 
130         /**
131          * Convenience method for {@link LibKstat#kstat_lookup}. Traverses the kstat
132          * chain, searching for all kstats with the same {@code module},
133          * {@code instance}, and {@code name} fields; this triplet uniquely identifies a
134          * kstat. If {@code module} is {@code null}, {@code instance} is -1, or
135          * {@code name} is {@code null}, then those fields will be ignored in the
136          * search.
137          *
138          * @param module
139          *            The module, or null to ignore
140          * @param instance
141          *            The instance, or -1 to ignore
142          * @param name
143          *            The name, or null to ignore
144          * @return All matches of the requested Kstat structure if found, or an empty
145          *         list otherwise
146          */
147         public static List<Kstat> lookupAll(String module, int instance, String name) {
148             List<Kstat> kstats = new ArrayList<>();
149             for (Kstat ksp = KS.kstat_lookup(KC, module, instance, name); ksp != null; ksp = ksp.next()) {
150                 if ((module == null || module.equals(Native.toString(ksp.ks_module, StandardCharsets.US_ASCII)))
151                         && (instance < 0 || instance == ksp.ks_instance)
152                         && (name == null || name.equals(Native.toString(ksp.ks_name, StandardCharsets.US_ASCII)))) {
153                     kstats.add(ksp);
154                 }
155             }
156             return kstats;
157         }
158 
159         /**
160          * Convenience method for {@link LibKstat#kstat_chain_update}. Brings this kstat
161          * header chain in sync with that of the kernel.
162          * <p>
163          * This function compares the kernel's current kstat chain ID(KCID), which is
164          * incremented every time the kstat chain changes, to this object's KCID.
165          *
166          * @return the new KCID if the kstat chain has changed, 0 if it hasn't, or -1 on
167          *         failure.
168          */
169         public static int update() {
170             return KS.kstat_chain_update(KC);
171         }
172 
173         /**
174          * Release the lock on the chain.
175          */
176         @Override
177         public void close() {
178             CHAIN.unlock();
179         }
180     }
181 
182     /**
183      * Create a copy of the Kstat chain and lock it for use by this object.
184      *
185      * @return A locked copy of the chain. It should be unlocked/released when you
186      *         are done with it with {@link KstatChain#close()}.
187      */
188     public static KstatChain openChain() {
189         return new KstatChain();
190     }
191 
192     /**
193      * Convenience method for {@link LibKstat#kstat_data_lookup} with String return
194      * values. Searches the kstat's data section for the record with the specified
195      * name. This operation is valid only for kstat types which have named data
196      * records. Currently, only the KSTAT_TYPE_NAMED and KSTAT_TYPE_TIMER kstats
197      * have named data records.
198      *
199      * @param ksp
200      *            The kstat to search
201      * @param name
202      *            The key for the name-value pair, or name of the timer as
203      *            applicable
204      * @return The value as a String.
205      */
206     public static String dataLookupString(Kstat ksp, String name) {
207         if (ksp.ks_type != LibKstat.KSTAT_TYPE_NAMED && ksp.ks_type != LibKstat.KSTAT_TYPE_TIMER) {
208             throw new IllegalArgumentException("Not a kstat_named or kstat_timer kstat.");
209         }
210         Pointer p = KS.kstat_data_lookup(ksp, name);
211         if (p == null) {
212             LOG.debug("Failed to lookup kstat value for key {}", name);
213             return "";
214         }
215         KstatNamed data = new KstatNamed(p);
216         switch (data.data_type) {
217         case LibKstat.KSTAT_DATA_CHAR:
218             return Native.toString(data.value.charc, StandardCharsets.UTF_8);
219         case LibKstat.KSTAT_DATA_INT32:
220             return Integer.toString(data.value.i32);
221         case LibKstat.KSTAT_DATA_UINT32:
222             return FormatUtil.toUnsignedString(data.value.ui32);
223         case LibKstat.KSTAT_DATA_INT64:
224             return Long.toString(data.value.i64);
225         case LibKstat.KSTAT_DATA_UINT64:
226             return FormatUtil.toUnsignedString(data.value.ui64);
227         case LibKstat.KSTAT_DATA_STRING:
228             return data.value.str.addr.getString(0);
229         default:
230             LOG.error("Unimplemented kstat data type {}", data.data_type);
231             return "";
232         }
233     }
234 
235     /**
236      * Convenience method for {@link LibKstat#kstat_data_lookup} with numeric return
237      * values. Searches the kstat's data section for the record with the specified
238      * name. This operation is valid only for kstat types which have named data
239      * records. Currently, only the KSTAT_TYPE_NAMED and KSTAT_TYPE_TIMER kstats
240      * have named data records.
241      *
242      * @param ksp
243      *            The kstat to search
244      * @param name
245      *            The key for the name-value pair, or name of the timer as
246      *            applicable
247      * @return The value as a long. If the data type is a character or string type,
248      *         returns 0 and logs an error.
249      */
250     public static long dataLookupLong(Kstat ksp, String name) {
251         if (ksp.ks_type != LibKstat.KSTAT_TYPE_NAMED && ksp.ks_type != LibKstat.KSTAT_TYPE_TIMER) {
252             throw new IllegalArgumentException("Not a kstat_named or kstat_timer kstat.");
253         }
254         Pointer p = KS.kstat_data_lookup(ksp, name);
255         if (p == null) {
256             if (LOG.isDebugEnabled()) {
257                 LOG.debug("Failed lo lookup kstat value on {}:{}:{} for key {}",
258                         Native.toString(ksp.ks_module, StandardCharsets.US_ASCII), ksp.ks_instance,
259                         Native.toString(ksp.ks_name, StandardCharsets.US_ASCII), name);
260             }
261             return 0L;
262         }
263         KstatNamed data = new KstatNamed(p);
264         switch (data.data_type) {
265         case LibKstat.KSTAT_DATA_INT32:
266             return data.value.i32;
267         case LibKstat.KSTAT_DATA_UINT32:
268             return FormatUtil.getUnsignedInt(data.value.ui32);
269         case LibKstat.KSTAT_DATA_INT64:
270             return data.value.i64;
271         case LibKstat.KSTAT_DATA_UINT64:
272             return data.value.ui64;
273         default:
274             LOG.error("Unimplemented or non-numeric kstat data type {}", data.data_type);
275             return 0L;
276         }
277     }
