27  import java.io.FileFilter;
28  import java.io.IOException;
29  import java.nio.file.Paths;
30  import java.util.ArrayList;
31  import java.util.HashMap;
32  import java.util.List;

35  import oshi.annotation.concurrent.ThreadSafe;

41  /**
42   * Sensors from WMI or Open Hardware Monitor
43   */
44  @ThreadSafe
45  final class LinuxSensors extends AbstractSensors {
46  
47      // Possible sensor types. See sysfs documentation for others, e.g. current
48      private static final String TEMP = "temp";
49      private static final String FAN = "fan";
50      private static final String VOLTAGE = "in";
51      private static final String[] SENSORS = { TEMP, FAN, VOLTAGE };
52  
53      // Base HWMON path, adds 0, 1, etc. to end for various sensors
54      private static final String HWMON = "hwmon";
55      private static final String HWMON_PATH = "/sys/class/hwmon/" + HWMON;
56      // Base THERMAL_ZONE path, adds 0, 1, etc. to end for temperature sensors
57      private static final String THERMAL_ZONE = "thermal_zone";
58      private static final String THERMAL_ZONE_PATH = "/sys/class/thermal/" + THERMAL_ZONE;
59  
60      // Initial test to see if we are running on a Pi
61      private static final boolean IS_PI = queryCpuTemperatureFromVcGenCmd() > 0;
62  
63      // Map from sensor to path. Built by constructor, so thread safe
64      private final Map<String, String> sensorsMap = new HashMap<>();
65  
66      /**
67       * <p>
68       * Constructor for LinuxSensors.
69       * </p>
70       */
71      LinuxSensors() {
72          if (!IS_PI) {
73              populateSensorsMapFromHwmon();
74              // if no temperature sensor is found in hwmon, try thermal_zone
75              if (!this.sensorsMap.containsKey(TEMP)) {
76                  populateSensorsMapFromThermalZone();
77              }
78          }
79      }
80  
81      /*
82       * Iterate over all hwmon* directories and look for sensor files, e.g.,
83       * /sys/class/hwmon/hwmon0/temp1_input
84       */
85      private void populateSensorsMapFromHwmon() {
86          for (String sensor : SENSORS) {
87              // Final to pass to anonymous class
88              final String sensorPrefix = sensor;
89              // Find any *_input files in that path
90              getSensorFilesFromPath(HWMON_PATH, sensor, f -> {
91                  try {
92                      return f.getName().startsWith(sensorPrefix) && f.getName().endsWith("_input")
93                              && FileUtil.getIntFromFile(f.getCanonicalPath()) > 0;
94                  } catch (IOException e) {
95                      return false;
96                  }
97              });
98          }
99      }
100 
101     /*
102      * Iterate over all thermal_zone* directories and look for sensor files, e.g.,
103      * /sys/class/thermal/thermal_zone0/temp
104      */
105     private void populateSensorsMapFromThermalZone() {
106         getSensorFilesFromPath(THERMAL_ZONE_PATH, TEMP, f -> f.getName().equals(TEMP));
107     }
108 
109     /**
110      * Find all sensor files in a specific path and adds them to the hwmonMap
111      *
112      * @param sensorPath
113      *            A string containing the sensor path
114      * @param sensor
115      *            A string containing the sensor
116      * @param sensorFileFilter
117      *            A FileFilter for detecting valid sensor files
118      */
119     private void getSensorFilesFromPath(String sensorPath, String sensor, FileFilter sensorFileFilter) {
120         int i = 0;
121         while (Paths.get(sensorPath + i).toFile().isDirectory()) {
122             String path = sensorPath + i;
123             File dir = new File(path);
124             File[] matchingFiles = dir.listFiles(sensorFileFilter);
125             if (matchingFiles != null && matchingFiles.length > 0) {
126                 this.sensorsMap.put(sensor, String.format("%s/%s", path, sensor));
127             }
128             i++;
129         }
130     }
131 
132     @Override
133     public double queryCpuTemperature() {
134         if (IS_PI) {
135             return queryCpuTemperatureFromVcGenCmd();
136         }
137         String tempStr = this.sensorsMap.get(TEMP);
138         if (tempStr != null) {
139             long millidegrees = 0;
140             if (tempStr.contains(HWMON)) {
141                 // First attempt should be CPU temperature at index 1, if available
142                 millidegrees = FileUtil.getLongFromFile(String.format("%s1_input", tempStr));
143                 // Should return a single line of millidegrees Celsius
144                 if (millidegrees > 0) {
145                     return millidegrees / 1000d;
146                 }
147                 // If temp1_input doesn't exist, iterate over temp2..temp6_input
148                 // and average
149                 long sum = 0;
150                 int count = 0;
151                 for (int i = 2; i <= 6; i++) {
152                     millidegrees = FileUtil.getLongFromFile(String.format("%s%d_input", tempStr, i));
153                     if (millidegrees > 0) {
154                         sum += millidegrees;
155                         count++;
156                     }
157                 }
158                 if (count > 0) {
159                     return sum / (count * 1000d);
160                 }
161             } else if (tempStr.contains(THERMAL_ZONE)) {
162                 // If temp2..temp6_input doesn't exist, try thermal_zone0
163                 millidegrees = FileUtil.getLongFromFile(tempStr);
164                 // Should return a single line of millidegrees Celsius
165                 if (millidegrees > 0) {
166                     return millidegrees / 1000d;
167                 }
168             }
169         }
170         return 0d;
171     }
172 
173     /**
174      * Retrieves temperature from Raspberry Pi
175      *
176      * @return The temperature on a Pi, 0 otherwise
177      */
178     private static double queryCpuTemperatureFromVcGenCmd() {
179         String tempStr = ExecutingCommand.getFirstAnswer("vcgencmd measure_temp");
180         // temp=50.8'C
181         if (tempStr.startsWith("temp=")) {
182             return ParseUtil.parseDoubleOrDefault(tempStr.replaceAll("[^\\d|\\.]+", ""), 0d);
183         }
184         return 0d;
185     }
186 
187     @Override
188     public int[] queryFanSpeeds() {
189         if (!IS_PI) {
190             String fanStr = this.sensorsMap.get(FAN);
191             if (fanStr != null) {
192                 List<Integer> speeds = new ArrayList<>();
193                 int fan = 1;
194                 for (;;) {
195                     String fanPath = String.format("%s%d_input", fanStr, fan);
196                     if (!new File(fanPath).exists()) {
197                         // No file found, we've reached max fans
198                         break;
199                     }
200                     // Should return a single line of RPM
201                     speeds.add(FileUtil.getIntFromFile(fanPath));
202                     // Done reading data for current fan, read next fan
203                     fan++;
204                 }
205                 int[] fanSpeeds = new int[speeds.size()];
206                 for (int i = 0; i < speeds.size(); i++) {
207                     fanSpeeds[i] = speeds.get(i);
208                 }
209                 return fanSpeeds;
210             }
211         }
212         return new int[0];
213     }
214 
215     @Override
216     public double queryCpuVoltage() {
217         if (IS_PI) {
218             return queryCpuVoltageFromVcGenCmd();
219         }
220         String voltageStr = this.sensorsMap.get(VOLTAGE);
221         if (voltageStr != null) {
222             // Should return a single line of millivolt
223             return FileUtil.getIntFromFile(String.format("%s1_input", voltageStr)) / 1000d;
224         }
225         return 0d;
226     }
227 
228     /**
229      * Retrieves voltage from Raspberry Pi
230      *
231      * @return The temperature on a Pi, 0 otherwise
232      */
233     private static double queryCpuVoltageFromVcGenCmd() {
234         // For raspberry pi
235         String voltageStr = ExecutingCommand.getFirstAnswer("vcgencmd measure_volts core");
236         // volt=1.20V
237         if (voltageStr.startsWith("volt=")) {
238             return ParseUtil.parseDoubleOrDefault(voltageStr.replaceAll("[^\\d|\\.]+", ""), 0d);
239         }
240         return 0d;
241     }
