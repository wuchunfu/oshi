24  package oshi.hardware.platform.unix.solaris;
25  
26  import java.util.ArrayList;
27  import java.util.List;
28  
29  import oshi.annotation.concurrent.Immutable;
30  import oshi.hardware.GraphicsCard;
31  import oshi.hardware.common.AbstractGraphicsCard;
32  import oshi.util.Constants;
33  import oshi.util.ExecutingCommand;
34  import oshi.util.ParseUtil;
35  
36  /**
37   * Graphics Card info obtained from prtconf
38   */
39  @Immutable
40  final class SolarisGraphicsCard extends AbstractGraphicsCard {
41  
42      private static final String PCI_CLASS_DISPLAY = "0003";
43  
44      /**
45       * Constructor for SolarisGraphicsCard
46       *
47       * @param name
48       *            The name
49       * @param deviceId
50       *            The device ID
51       * @param vendor
52       *            The vendor
53       * @param versionInfo
54       *            The version info
55       * @param vram
56       *            The VRAM
57       */
58      SolarisGraphicsCard(String name, String deviceId, String vendor, String versionInfo, long vram) {
59          super(name, deviceId, vendor, versionInfo, vram);
60      }
61  
62      /**
63       * public method used by
64       * {@link oshi.hardware.common.AbstractHardwareAbstractionLayer} to access the
65       * graphics cards.
66       *
67       * @return List of
68       *         {@link oshi.hardware.platform.unix.solaris.SolarisGraphicsCard}
69       *         objects.
70       */
71      public static List<GraphicsCard> getGraphicsCards() {
72          List<GraphicsCard> cardList = new ArrayList<>();
73          // Enumerate all devices and add if required
74          List<String> devices = ExecutingCommand.runNative("prtconf -pv");
75          if (devices.isEmpty()) {
76              return cardList;
77          }
78          String name = "";
79          String vendorId = "";
80          String productId = "";
81          String classCode = "";
82          List<String> versionInfoList = new ArrayList<>();
83          for (String line : devices) {
84              // Node 0x... identifies start of a new device. Save previous if it's a graphics
85              // card
86              if (line.contains("Node 0x")) {
87                  if (PCI_CLASS_DISPLAY.equals(classCode)) {
88                      cardList.add(new SolarisGraphicsCard(name.isEmpty() ? Constants.UNKNOWN : name,
89                              productId.isEmpty() ? Constants.UNKNOWN : productId,
90                              vendorId.isEmpty() ? Constants.UNKNOWN : vendorId,
91                              versionInfoList.isEmpty() ? Constants.UNKNOWN : String.join(", ", versionInfoList), 0L));
92                  }
93                  // Reset strings
94                  name = "";
95                  vendorId = Constants.UNKNOWN;
96                  productId = Constants.UNKNOWN;
97                  classCode = "";
98                  versionInfoList.clear();
99              } else {
100                 String[] split = line.trim().split(":", 2);
101                 if (split.length == 2) {
102                     if (split[0].equals("model")) {
103                         // This is preferred, always set it
104                         name = ParseUtil.getSingleQuoteStringValue(line);
105                     } else if (split[0].equals("name")) {
106                         // Name is backup for model if model doesn't exist, so only
107                         // put if name blank
108                         if (name.isEmpty()) {
109                             name = ParseUtil.getSingleQuoteStringValue(line);
110                         }
111                     } else if (split[0].equals("vendor-id")) {
112                         // Format: vendor-id: 00008086
113                         vendorId = "0x" + line.substring(line.length() - 4);
114                     } else if (split[0].equals("device-id")) {
115                         // Format: device-id: 00002440
116                         productId = "0x" + line.substring(line.length() - 4);
117                     } else if (split[0].equals("revision-id")) {
118                         // Format: revision-id: 00000002
119                         versionInfoList.add(line.trim());
120                     } else if (split[0].equals("class-code")) {
121                         // Format: 00030000
122                         // Display class is 0003xx, first 6 bytes of this code
123                         classCode = line.substring(line.length() - 8, line.length() - 4);
124                     }
125                 }
126             }
127         }
128         // In case we reached end before saving
129         if (PCI_CLASS_DISPLAY.equals(classCode)) {
130             cardList.add(new SolarisGraphicsCard(name.isEmpty() ? Constants.UNKNOWN : name,
131                     productId.isEmpty() ? Constants.UNKNOWN : productId,
132                     vendorId.isEmpty() ? Constants.UNKNOWN : vendorId,
133                     versionInfoList.isEmpty() ? Constants.UNKNOWN : String.join(", ", versionInfoList), 0L));
134         }
135         return cardList;
136     }
137 }
