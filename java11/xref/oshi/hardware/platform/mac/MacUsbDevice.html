61      /**
62       * Instantiates a list of {@link oshi.hardware.UsbDevice} objects, representing
63       * devices connected via a usb port (including internal devices).
64       * <p>
65       * If the value of {@code tree} is true, the top level devices returned from
66       * this method are the USB Controllers; connected hubs and devices in its device
67       * tree share that controller's bandwidth. If the value of {@code tree} is
68       * false, USB devices (not controllers) are listed in a single flat list.
69       *
70       * @param tree
71       *            If true, returns a list of controllers, which requires recursive
72       *            iteration of connected devices. If false, returns a flat list of
73       *            devices excluding controllers.
74       * @return a list of {@link oshi.hardware.UsbDevice} objects.
75       */
76      public static List<UsbDevice> getUsbDevices(boolean tree) {
77          List<UsbDevice> devices = getUsbDevices();
78          if (tree) {
79              return devices;
80          }
81          List<UsbDevice> deviceList = new ArrayList<>();
82          // Top level is controllers; they won't be added to the list, but all
83          // their connected devices will be
84          for (UsbDevice device : devices) {
85              addDevicesToList(deviceList, device.getConnectedDevices());
86          }
87          return deviceList;
88      }
89  
90      private static List<UsbDevice> getUsbDevices() {
91          // Maps to store information using RegistryEntryID as the key
92          Map<Long, String> nameMap = new HashMap<>();
93          Map<Long, String> vendorMap = new HashMap<>();
94          Map<Long, String> vendorIdMap = new HashMap<>();
95          Map<Long, String> productIdMap = new HashMap<>();
96          Map<Long, String> serialMap = new HashMap<>();
97          Map<Long, List<Long>> hubMap = new HashMap<>();
98  
99          List<Long> usbControllers = new ArrayList<>();
100         IORegistryEntry root = IOKitUtil.getRoot();
101         // Iterate over children of root in the IOUSB plane. This does not include
102         // controllers so we have to check their parents
103         IOIterator iter = root.getChildIterator(IOUSB);
104         if (iter != null) {
105             // Define keys
106             CFStringRef locationIDKey = CFStringRef.createCFString("locationID");
107             CFStringRef ioPropertyMatchKey = CFStringRef.createCFString("IOPropertyMatch");
108 
109             // Get the device directly under each controller
110             IORegistryEntry device = iter.next();
111             while (device != null) {
112                 long id = 0L;
113                 // The parent of this device in IOService plane is the controller
114                 IORegistryEntry controller = device.getParentEntry(IOSERVICE);
115                 if (controller != null) {
116                     // Unique global identifier for this controller
117                     id = controller.getRegistryEntryID();
118                     // Populate other data for the controller
119                     nameMap.put(id, controller.getName());
120                     // The only information we have in registry for this controller is
121                     // the locationID. Use that to search for matching PCI device to obtain
122                     // more information for the controller
123                     CFTypeRef ref = controller.createCFProperty(locationIDKey);
124                     if (ref != null) {
125                         getControllerIdByLocation(id, ref, locationIDKey, ioPropertyMatchKey, vendorIdMap,
126                                 productIdMap);
127                         ref.release();
128                     }
129                     controller.release();
130                 }
131                 usbControllers.add(id);
132 
133                 // Now recursively add this device and its children to the maps
134                 // id is the coontroller ID and the first parent ID
135                 addDeviceAndChildrenToMaps(device, id, nameMap, vendorMap, vendorIdMap, productIdMap, serialMap,
136                         hubMap);
137 
138                 device.release();
139                 device = iter.next();
140             }
141             locationIDKey.release();
142             ioPropertyMatchKey.release();
143             iter.release();
144         }
145         root.release();
146 
147         // Build tree and return
148         List<UsbDevice> controllerDevices = new ArrayList<>();
149         for (Long controller : usbControllers) {
150             controllerDevices.add(getDeviceAndChildren(controller, "0000", "0000", nameMap, vendorMap, vendorIdMap,
151                     productIdMap, serialMap, hubMap));
152         }
153         return controllerDevices;
154     }

156     /**
157      * Recursively populate maps with information from a USB Device and its children
158      *
159      * @param device
160      *            The device which, along with its children, should be added
161      * @param parentId
162      *            The id of the device's parent.
163      * @param nameMap
164      *            the map of names
165      * @param vendorMap
166      *            the map of vendors
167      * @param vendorIdMap
168      *            the map of vendorIds
169      * @param productIdMap
170      *            the map of productIds
171      * @param serialMap
172      *            the map of serial numbers
173      * @param hubMap
174      *            the map of hubs
175      */
176     private static void addDeviceAndChildrenToMaps(IORegistryEntry device, long parentId, Map<Long, String> nameMap,
177             Map<Long, String> vendorMap, Map<Long, String> vendorIdMap, Map<Long, String> productIdMap,
178             Map<Long, String> serialMap, Map<Long, List<Long>> hubMap) {
179 
180         // Unique global identifier for this device
181         long id = device.getRegistryEntryID();
182         // Store id as a child of parent in hubmap
183         hubMap.computeIfAbsent(parentId, x -> new ArrayList<>()).add(id);
184         // Get device name and store in map
185         nameMap.put(id, device.getName().trim());
186         // Get vendor and store in map
187         String vendor = device.getStringProperty("USB Vendor Name");
188         if (vendor != null) {
189             vendorMap.put(id, vendor.trim());
190         }
191         // Get vendorId and store in map
192         Long vendorId = device.getLongProperty("idVendor");
193         if (vendorId != null) {
194             vendorIdMap.put(id, String.format("%04x", 0xffff & vendorId));
195         }
196         // Get productId and store in map
197         Long productId = device.getLongProperty("idProduct");
198         if (productId != null) {
199             productIdMap.put(id, String.format("%04x", 0xffff & productId));
200         }
201         // Get serial and store in map
202         String serial = device.getStringProperty("USB Serial Number");
203         if (serial != null) {
204             serialMap.put(id, serial.trim());
205         }
206 
207         // Now get this device's children (if any) and recurse
208         IOIterator childIter = device.getChildIterator(IOUSB);
209         IORegistryEntry childDevice = childIter.next();
210         while (childDevice != null) {
211             addDeviceAndChildrenToMaps(childDevice, id, nameMap, vendorMap, vendorIdMap, productIdMap, serialMap,
212                     hubMap);
213 
214             childDevice.release();
215             childDevice = childIter.next();
216         }
217         childIter.release();
218     }

220     private static void addDevicesToList(List<UsbDevice> deviceList, List<UsbDevice> list) {
221         for (UsbDevice device : list) {
222             deviceList.add(
223                     new MacUsbDevice(device.getName(), device.getVendor(), device.getVendorId(), device.getProductId(),
224                             device.getSerialNumber(), device.getUniqueDeviceId(), Collections.emptyList()));
225             addDevicesToList(deviceList, device.getConnectedDevices());
226         }
227     }

229     /**
230      * Looks up vendor and product id information for a USB Host Controller by
231      * cross-referencing the location
232      *
233      * @param id
234      *            The global unique ID for the host controller used as a key for
235      *            maps
236      * @param locationId
237      *            The locationID of this controller returned from the registry
238      * @param locationIDKey
239      *            A pointer to the locationID string
240      * @param ioPropertyMatchKey
241      *            A pointer to the IOPropertyMatch string
242      * @param productIdMap
243      *            the map of productIds
244      * @param vendorIdMap
245      *            the map of vendorIds
246      */
247     private static void getControllerIdByLocation(long id, CFTypeRef locationId, CFStringRef locationIDKey,
248             CFStringRef ioPropertyMatchKey, Map<Long, String> vendorIdMap, Map<Long, String> productIdMap) {
249         // Create a matching property dictionary from the locationId
250         CFMutableDictionaryRef propertyDict = CF.CFDictionaryCreateMutable(CF.CFAllocatorGetDefault(), new CFIndex(0),
251                 null, null);
252         propertyDict.setValue(locationIDKey, locationId);
253         CFMutableDictionaryRef matchingDict = CF.CFDictionaryCreateMutable(CF.CFAllocatorGetDefault(), new CFIndex(0),
254                 null, null);
255         matchingDict.setValue(ioPropertyMatchKey, propertyDict);
256 
257         // search for all IOservices that match the locationID
258         IOIterator serviceIterator = IOKitUtil.getMatchingServices(matchingDict);
259         // getMatchingServices releases matchingDict
260         propertyDict.release();
261 
262         // Iterate matching services looking for devices whose parents have the
263         // vendor and device ids
264         boolean found = false;
265         if (serviceIterator != null) {
266             IORegistryEntry matchingService = serviceIterator.next();
267             while (matchingService != null && !found) {
268                 // Get the parent, which contains the keys we need
269                 IORegistryEntry parent = matchingService.getParentEntry(IOSERVICE);
270                 // look up the vendor-id by key
271                 // vendor-id is a byte array of 4 bytes
272                 if (parent != null) {
273                     byte[] vid = parent.getByteArrayProperty("vendor-id");
274                     if (vid != null && vid.length >= 2) {
275                         vendorIdMap.put(id, String.format("%02x%02x", vid[1], vid[0]));
276                         found = true;
277                     }
278                     // look up the device-id by key
279                     // device-id is a byte array of 4 bytes
280                     byte[] pid = parent.getByteArrayProperty("device-id");
281                     if (pid != null && pid.length >= 2) {
282                         productIdMap.put(id, String.format("%02x%02x", pid[1], pid[0]));
283                         found = true;
284                     }
285                     parent.release();
286                 }
287                 // iterate
288                 matchingService.release();
289                 matchingService = serviceIterator.next();
290             }
291             serviceIterator.release();
292         }
293     }

295     /**
296      * Recursively creates MacUsbDevices by fetching information from maps to
297      * populate fields
298      *
299      * @param registryEntryId
300      *            The device unique registry id.
301      * @param vid
302      *            The default (parent) vendor ID
303      * @param pid
304      *            The default (parent) product ID
305      * @param nameMap
306      *            the map of names
307      * @param vendorMap
308      *            the map of vendors
309      * @param vendorIdMap
310      *            the map of vendorIds
311      * @param productIdMap
312      *            the map of productIds
313      * @param serialMap
314      *            the map of serial numbers
315      * @param hubMap
316      *            the map of hubs
317      * @return A MacUsbDevice corresponding to this device
318      */
319     private static MacUsbDevice getDeviceAndChildren(Long registryEntryId, String vid, String pid,
320             Map<Long, String> nameMap, Map<Long, String> vendorMap, Map<Long, String> vendorIdMap,
321             Map<Long, String> productIdMap, Map<Long, String> serialMap, Map<Long, List<Long>> hubMap) {
322         String vendorId = vendorIdMap.getOrDefault(registryEntryId, vid);
323         String productId = productIdMap.getOrDefault(registryEntryId, pid);
324         List<Long> childIds = hubMap.getOrDefault(registryEntryId, new ArrayList<>());
325         List<UsbDevice> usbDevices = new ArrayList<>();
326         for (Long id : childIds) {
327             usbDevices.add(getDeviceAndChildren(id, vendorId, productId, nameMap, vendorMap, vendorIdMap, productIdMap,
328                     serialMap, hubMap));
329         }
330         Collections.sort(usbDevices);
331         return new MacUsbDevice(nameMap.getOrDefault(registryEntryId, vendorId + ":" + productId),
332                 vendorMap.getOrDefault(registryEntryId, ""), vendorId, productId,
333                 serialMap.getOrDefault(registryEntryId, ""), "0x" + Long.toHexString(registryEntryId), usbDevices);
334     }
