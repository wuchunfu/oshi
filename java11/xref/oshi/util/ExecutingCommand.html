24  package oshi.util;
25  
26  import java.io.BufferedReader;
27  import java.io.IOException;
28  import java.io.InputStreamReader;
29  import java.nio.charset.Charset;
30  import java.util.ArrayList;
31  import java.util.Arrays;
32  import java.util.Collections;
33  import java.util.List;
34  
35  import org.slf4j.Logger;
36  import org.slf4j.LoggerFactory;
37  
38  import com.sun.jna.Platform;
39  
40  import oshi.annotation.concurrent.ThreadSafe;
41  
42  /**
43   * A class for executing on the command line and returning the result of
44   * execution.
45   */
46  @ThreadSafe
47  public final class ExecutingCommand {
48  
49      private static final Logger LOG = LoggerFactory.getLogger(ExecutingCommand.class);
50  
51      private static final String[] DEFAULT_ENV = getDefaultEnv();
52  
53      private ExecutingCommand() {
54      }
55  
56      private static String[] getDefaultEnv() {
57          if (Platform.isWindows()) {
58              return new String[] { "LANGUAGE=C" };
59          } else {
60              return new String[] { "LC_ALL=C" };
61          }
62      }
63  
64      /**
65       * Executes a command on the native command line and returns the result. This is
66       * a convenience method to call {@link java.lang.Runtime#exec(String)} and
67       * capture the resulting output in a list of Strings. On Windows, built-in
68       * commands not associated with an executable program may require
69       * {@code cmd.exe /c} to be prepended to the command.
70       *
71       * @param cmdToRun
72       *            Command to run
73       * @return A list of Strings representing the result of the command, or empty
74       *         string if the command failed
75       */
76      public static List<String> runNative(String cmdToRun) {
77          String[] cmd = cmdToRun.split(" ");
78          return runNative(cmd);
79      }
80  
81      /**
82       * Executes a command on the native command line and returns the result line by
83       * line. This is a convenience method to call
84       * {@link java.lang.Runtime#exec(String[])} and capture the resulting output in
85       * a list of Strings. On Windows, built-in commands not associated with an
86       * executable program may require the strings {@code cmd.exe} and {@code /c} to
87       * be prepended to the array.
88       *
89       * @param cmdToRunWithArgs
90       *            Command to run and args, in an array
91       * @return A list of Strings representing the result of the command, or empty
92       *         string if the command failed
93       */
94      public static List<String> runNative(String[] cmdToRunWithArgs) {
95          return runNative(cmdToRunWithArgs, DEFAULT_ENV);
96      }
97  
98      /**
99       * Executes a command on the native command line and returns the result line by
100      * line. This is a convenience method to call
101      * {@link java.lang.Runtime#exec(String[])} and capture the resulting output in
102      * a list of Strings. On Windows, built-in commands not associated with an
103      * executable program may require the strings {@code cmd.exe} and {@code /c} to
104      * be prepended to the array.
105      *
106      * @param cmdToRunWithArgs
107      *            Command to run and args, in an array
108      * @param envp
109      *            array of strings, each element of which has environment variable
110      *            settings in the format name=value, or null if the subprocess
111      *            should inherit the environment of the current process.
112      * @return A list of Strings representing the result of the command, or empty
113      *         string if the command failed
114      */
115     public static List<String> runNative(String[] cmdToRunWithArgs, String[] envp) {
116         Process p = null;
117         try {
118             p = Runtime.getRuntime().exec(cmdToRunWithArgs, envp);
119             return getProcessOutput(p, cmdToRunWithArgs);
120         } catch (SecurityException | IOException e) {
121             LOG.trace("Couldn't run command {}: {}", Arrays.toString(cmdToRunWithArgs), e.getMessage());
122         } finally {
123             // Ensure all resources are released
124             if (p != null) {
125                 // Windows and Solaris don't close descriptors on destroy,
126                 // so we must handle separately
127                 if (Platform.isWindows() || Platform.isSolaris()) {
128                     try {
129                         p.getOutputStream().close();
130                     } catch (IOException e) {
131                         // do nothing on failure
132                     }
133                     try {
134                         p.getInputStream().close();
135                     } catch (IOException e) {
136                         // do nothing on failure
137                     }
138                     try {
139                         p.getErrorStream().close();
140                     } catch (IOException e) {
141                         // do nothing on failure
142                     }
143                 }
144                 p.destroy();
145             }
146         }
147         return Collections.emptyList();
148     }
149 
150     private static List<String> getProcessOutput(Process p, String[] cmd) {
151         ArrayList<String> sa = new ArrayList<>();
152         try (BufferedReader reader = new BufferedReader(
153                 new InputStreamReader(p.getInputStream(), Charset.defaultCharset()))) {
154             String line;
155             while ((line = reader.readLine()) != null) {
156                 sa.add(line);
157             }
158             p.waitFor();
159         } catch (IOException e) {
160             LOG.trace("Problem reading output from {}: {}", Arrays.toString(cmd), e.getMessage());
161         } catch (InterruptedException ie) {
162             LOG.trace("Interrupted while reading output from {}: {}", Arrays.toString(cmd), ie.getMessage());
163             Thread.currentThread().interrupt();
164         }
165         return sa;
166     }
167 
168     /**
169      * Return first line of response for selected command.
170      *
171      * @param cmd2launch
172      *            String command to be launched
173      * @return String or empty string if command failed
174      */
175     public static String getFirstAnswer(String cmd2launch) {
176         return getAnswerAt(cmd2launch, 0);
177     }
178 
179     /**
180      * Return response on selected line index (0-based) after running selected
181      * command.
182      *
183      * @param cmd2launch
184      *            String command to be launched
185      * @param answerIdx
186      *            int index of line in response of the command
187      * @return String whole line in response or empty string if invalid index or
188      *         running of command fails
189      */
190     public static String getAnswerAt(String cmd2launch, int answerIdx) {
191         List<String> sa = ExecutingCommand.runNative(cmd2launch);
192 
193         if (answerIdx >= 0 && answerIdx < sa.size()) {
194             return sa.get(answerIdx);
195         }
196         return "";
197     }
198 
199 }
