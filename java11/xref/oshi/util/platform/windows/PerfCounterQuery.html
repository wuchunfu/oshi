46  /**
47   * Enables queries of Performance Counters using wild cards to filter instances
48   */
49  @ThreadSafe
50  public final class PerfCounterQuery {
51  
52      private static final Logger LOG = LoggerFactory.getLogger(PerfCounterQuery.class);
53  
54      private static final boolean IS_VISTA_OR_GREATER = VersionHelpers.IsWindowsVistaOrGreater();
55  
56      // Use a thread safe set to cache failed pdh queries
57      private static final Set<String> FAILED_QUERY_CACHE = ConcurrentHashMap.newKeySet();
58  
59      // For XP, use a map to cache localization strings
60      private static final ConcurrentHashMap<String, String> LOCALIZE_CACHE = IS_VISTA_OR_GREATER ? null
61              : new ConcurrentHashMap<>();
62  
63      /*
64       * Multiple classes use these constants
65       */
66      public static final String TOTAL_INSTANCE = "_Total";
67      public static final String TOTAL_INSTANCES = "*_Total";
68      public static final String NOT_TOTAL_INSTANCE = "^" + TOTAL_INSTANCE;
69      public static final String NOT_TOTAL_INSTANCES = "^" + TOTAL_INSTANCES;
70  
71      private PerfCounterQuery() {
72      }
73  
74      /**
75       * Query the a Performance Counter using PDH, with WMI backup on failure, for
76       * values corresponding to the property enum.
77       *
78       * @param <T>
79       *            The enum type of {@code propertyEnum}
80       * @param propertyEnum
81       *            An enum which implements
82       *            {@link oshi.util.platform.windows.PerfCounterQuery.PdhCounterProperty}
83       *            and contains the WMI field (Enum value) and PDH Counter string
84       *            (instance and counter)
85       * @param perfObject
86       *            The PDH object for this counter; all counters on this object will
87       *            be refreshed at the same time
88       * @param perfWmiClass
89       *            The WMI PerfData_RawData_* class corresponding to the PDH object
90       * @return An {@link EnumMap} of the values indexed by {@code propertyEnum} on
91       *         success, or an empty map if both PDH and WMI queries failed.
92       */
93      public static <T extends Enum<T>> Map<T, Long> queryValues(Class<T> propertyEnum, String perfObject,
94              String perfWmiClass) {
95          if (!FAILED_QUERY_CACHE.contains(perfObject)) {
96              Map<T, Long> valueMap = queryValuesFromPDH(propertyEnum, perfObject);
97              if (!valueMap.isEmpty()) {
98                  return valueMap;
99              }
100             // If we are here, query failed
101             LOG.warn("Disabling further attempts to query {}.", perfObject);
102             FAILED_QUERY_CACHE.add(perfObject);
103         }
104         return queryValuesFromWMI(propertyEnum, perfWmiClass);
105     }
106 
107     /**
108      * Query the a Performance Counter using PDH for values corresponding to the
109      * property enum.
110      *
111      * @param <T>
112      *            The enum type of {@code propertyEnum}
113      * @param propertyEnum
114      *            An enum which implements
115      *            {@link oshi.util.platform.windows.PerfCounterQuery.PdhCounterProperty}
116      *            and contains the WMI field (Enum value) and PDH Counter string
117      *            (instance and counter)
118      * @param perfObject
119      *            The PDH object for this counter; all counters on this object will
120      *            be refreshed at the same time
121      * @return An {@link EnumMap} of the values indexed by {@code propertyEnum} on
122      *         success, or an empty map if the PDH query failed.
123      */
124     public static <T extends Enum<T>> Map<T, Long> queryValuesFromPDH(Class<T> propertyEnum, String perfObject) {
125         T[] props = propertyEnum.getEnumConstants();
126         // If pre-Vista, localize the perfObject
127         String perfObjectLocalized = PerfCounterQuery.localizeIfNeeded(perfObject);
128         EnumMap<T, PerfCounter> counterMap = new EnumMap<>(propertyEnum);
129         EnumMap<T, Long> valueMap = new EnumMap<>(propertyEnum);
130         try (PerfCounterQueryHandler pdhQueryHandler = new PerfCounterQueryHandler()) {
131             // Set up the query and counter handles
132             for (T prop : props) {
133                 PerfCounter counter = PerfDataUtil.createCounter(perfObjectLocalized,
134                         ((PdhCounterProperty) prop).getInstance(), ((PdhCounterProperty) prop).getCounter());
135                 counterMap.put(prop, counter);
136                 if (!pdhQueryHandler.addCounterToQuery(counter)) {
137                     return valueMap;
138                 }
139             }
140             // And then query. Zero timestamp means update failed
141             if (0 < pdhQueryHandler.updateQuery()) {
142                 for (T prop : props) {
143                     valueMap.put(prop, pdhQueryHandler.queryCounter(counterMap.get(prop)));
144                 }
145             }
146         }
147         return valueMap;
148     }

150     /**
151      * Query the a Performance Counter using WMI for values corresponding to the
152      * property enum.
153      *
154      * @param <T>
155      *            The enum type of {@code propertyEnum}
156      * @param propertyEnum
157      *            An enum which implements
158      *            {@link oshi.util.platform.windows.PerfCounterQuery.PdhCounterProperty}
159      *            and contains the WMI field (Enum value) and PDH Counter string
160      *            (instance and counter)
161      * @param wmiClass
162      *            The WMI PerfData_RawData_* class corresponding to the PDH object
163      * @return An {@link EnumMap} of the values indexed by {@code propertyEnum} if
164      *         successful, an empty map if the WMI query failed.
165      */
166     public static <T extends Enum<T>> Map<T, Long> queryValuesFromWMI(Class<T> propertyEnum, String wmiClass) {
167         WmiQuery<T> query = new WmiQuery<>(wmiClass, propertyEnum);
168         WmiResult<T> result = Objects.requireNonNull(WmiQueryHandler.createInstance()).queryWMI(query);
169         EnumMap<T, Long> valueMap = new EnumMap<>(propertyEnum);
170         if (result.getResultCount() > 0) {
171             for (T prop : propertyEnum.getEnumConstants()) {
172                 switch (result.getCIMType(prop)) {
173                 case Wbemcli.CIM_UINT16:
174                     valueMap.put(prop, (long) WmiUtil.getUint16(result, prop, 0));
175                     break;
176                 case Wbemcli.CIM_UINT32:
177                     valueMap.put(prop, WmiUtil.getUint32asLong(result, prop, 0));
178                     break;
179                 case Wbemcli.CIM_UINT64:
180                     valueMap.put(prop, WmiUtil.getUint64(result, prop, 0));
181                     break;
182                 case Wbemcli.CIM_DATETIME:
183                     valueMap.put(prop, WmiUtil.getDateTime(result, prop, 0).toInstant().toEpochMilli());
184                     break;
185                 default:
186                     throw new ClassCastException("Unimplemented CIM Type Mapping.");
187                 }
188             }
189         }
190         return valueMap;
191     }

193     /**
194      * Localize a PerfCounter string. English counter names should normally be in
195      * {@code HKEY_LOCAL_MACHINE\SOFTWARE\Microsoft\Windows
196      * NT\CurrentVersion\Perflib\009\Counter}, but language manipulations may delete
197      * the {@code 009} index. In this case we can assume English must be the
198      * language and continue. We may still fail to match the name if the assumption
199      * is wrong but it's better than nothing.
200      *
201      * @param perfObject
202      *            A String to localize
203      * @return The localized string if localization successful, or the original
204      *         string otherwise.
205      */
206     public static String localizeIfNeeded(String perfObject) {
207         return IS_VISTA_OR_GREATER ? perfObject
208                 : LOCALIZE_CACHE.computeIfAbsent(perfObject, PerfCounterQuery::localizeUsingPerfIndex);
209     }
210 
211     private static String localizeUsingPerfIndex(String perfObject) {
212         String localized = perfObject;
213         try {
214             localized = PdhUtil.PdhLookupPerfNameByIndex(null, PdhUtil.PdhLookupPerfIndexByEnglishName(perfObject));
215         } catch (Win32Exception e) {
216             LOG.warn(
217                     "Unable to locate English counter names in registry Perflib 009. Assuming English counters. Error {}. {}",
218                     String.format("0x%x", e.getHR().intValue()),
219                     "See https://support.microsoft.com/en-us/help/300956/how-to-manually-rebuild-performance-counter-library-values");
220         } catch (PdhException e) {
221             LOG.warn("Unable to localize {} performance counter.  Error {}.", perfObject,
222                     String.format("0x%x", e.getErrorCode()));
223         }
224         if (localized.isEmpty()) {
225             return perfObject;
226         }
227         LOG.debug("Localized {} to {}", perfObject, localized);
228         return localized;
229     }
