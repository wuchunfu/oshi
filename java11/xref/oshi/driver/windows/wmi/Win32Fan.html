24  package oshi.driver.windows.wmi;
25  
26  import java.util.Objects;
27  
28  import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiQuery; //NOSONAR squid:S1191
29  import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiResult;
30  
31  import oshi.annotation.concurrent.ThreadSafe;
32  import oshi.util.platform.windows.WmiQueryHandler;
33  
34  /**
35   * Utility to query WMI class {@code Win32_Fan}
36   */
37  @ThreadSafe
38  public final class Win32Fan {
39  
40      private static final String WIN32_FAN = "Win32_Fan";
41  
42      /**
43       * Fan speed property.
44       */
45      public enum SpeedProperty {
46          DESIREDSPEED;
47      }
48  
49      private Win32Fan() {
50      }
51  
52      /**
53       * Queries the fan speed.
54       *
55       * @return Currently requested fan speed, defined in revolutions per minute,
56       *         when a variable speed fan is supported.
57       */
58      public static WmiResult<SpeedProperty> querySpeed() {
59          WmiQuery<SpeedProperty> fanQuery = new WmiQuery<>(WIN32_FAN, SpeedProperty.class);
60          return Objects.requireNonNull(WmiQueryHandler.createInstance()).queryWMI(fanQuery);
61      }
62  }
