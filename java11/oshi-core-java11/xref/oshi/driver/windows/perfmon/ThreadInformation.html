24  package oshi.driver.windows.perfmon;
25  
26  import java.util.List;
27  import java.util.Map;
28  
29  import oshi.annotation.concurrent.ThreadSafe;
30  import oshi.util.platform.windows.PerfCounterQuery;
31  import oshi.util.platform.windows.PerfCounterWildcardQuery;
32  import oshi.util.platform.windows.PerfCounterWildcardQuery.PdhCounterWildcardProperty;
33  import oshi.util.tuples.Pair;
34  
35  /**
36   * Utility to query Thread Information performance counter
37   */
38  @ThreadSafe
39  public final class ThreadInformation {
40  
41      private static final String THREAD = "Thread";
42      private static final String WIN32_PERF_RAW_DATA_PERF_PROC_THREAD = "Win32_PerfRawData_PerfProc_Thread WHERE NOT Name LIKE \"%_Total\"";
43  
44      /**
45       * Thread performance counters
46       */
47      public enum ThreadPerformanceProperty implements PdhCounterWildcardProperty {
48          // First element defines WMI instance name field and PDH instance filter
49          NAME(PerfCounterQuery.NOT_TOTAL_INSTANCES),
50          // Remaining elements define counters
51          PERCENTUSERTIME("% User Time"), //
52          PERCENTPRIVILEGEDTIME("% Privileged Time"), //
53          ELAPSEDTIME("Elapsed Time"), //
54          PRIORITYCURRENT("Priority Current"), //
55          STARTADDRESS("Start Address"), //
56          THREADSTATE("Thread State"), //
57          THREADWAITREASON("Thread Wait Reason"), // 5 is SUSPENDED
58          IDPROCESS("ID Process"), //
59          IDTHREAD("ID Thread"), //
60          CONTEXTSWITCHESPERSEC("Context Switches/sec");
61  
62          private final String counter;
63  
64          ThreadPerformanceProperty(String counter) {
65              this.counter = counter;
66          }
67  
68          @Override
69          public String getCounter() {
70              return counter;
71          }
72      }
73  
74      private ThreadInformation() {
75      }
76  
77      /**
78       * Returns thread counters.
79       *
80       * @return Thread counters for each thread.
81       */
82      public static Pair<List<String>, Map<ThreadPerformanceProperty, List<Long>>> queryThreadCounters() {
83          return PerfCounterWildcardQuery.queryInstancesAndValues(ThreadPerformanceProperty.class, THREAD,
84                  WIN32_PERF_RAW_DATA_PERF_PROC_THREAD);
85      }
86  }
