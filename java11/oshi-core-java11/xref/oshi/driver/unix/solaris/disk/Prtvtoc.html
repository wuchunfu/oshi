24  package oshi.driver.unix.solaris.disk;
25  
26  import java.util.ArrayList;
27  import java.util.List;
28  
29  import oshi.annotation.concurrent.ThreadSafe;
30  import oshi.hardware.HWPartition;
31  import oshi.util.Constants;
32  import oshi.util.ExecutingCommand;
33  import oshi.util.ParseUtil;
34  
35  /**
36   * Utility to query iostat
37   */
38  @ThreadSafe
39  public final class Prtvtoc {
40  
41      private static final String PRTVTOC_DEV_DSK = "prtvtoc /dev/dsk/";
42  
43      private Prtvtoc() {
44      }
45  
46      public static List<HWPartition> queryPartitions(String mount, int major) {
47          List<HWPartition> partList = new ArrayList<>();
48          // This requires sudo permissions; will result in "permission denied"
49          // otherwise in which case we return empty partition list
50          List<String> prtvotc = ExecutingCommand.runNative(PRTVTOC_DEV_DSK + mount);
51          // Sample output - see man prtvtoc
52          if (prtvotc.size() > 1) {
53              int bytesPerSector = 0;
54              String[] split;
55              // We have a result, parse partition table
56              for (String line : prtvotc) {
57                  // If line starts with asterisk we ignore except for the one
58                  // specifying bytes per sector
59                  if (line.startsWith("*")) {
60                      if (line.endsWith("bytes/sector")) {
61                          split = ParseUtil.whitespaces.split(line);
62                          if (split.length > 0) {
63                              bytesPerSector = ParseUtil.parseIntOrDefault(split[1], 0);
64                          }
65                      }
66                  } else if (bytesPerSector > 0) {
67                      // If bytes/sector is still 0, these are not real partitions so
68                      // ignore.
69                      // Lines without asterisk have 6 or 7 whitespaces-split values
70                      // representing (last field optional):
71                      // Partition Tag Flags Sector Count Sector Mount
72                      split = ParseUtil.whitespaces.split(line.trim());
73                      // Partition 2 is always the whole disk so we ignore it
74                      if (split.length >= 6 && !"2".equals(split[0])) {
75                          // First field is partition number
76                          String identification = mount + "s" + split[0];
77                          // major already defined as method param
78                          int minor = ParseUtil.parseIntOrDefault(split[0], 0);
79                          // Second field is tag. Parse:
80                          String name;
81                          switch (ParseUtil.parseIntOrDefault(split[1], 0)) {
82                          case 0x01:
83                          case 0x18:
84                              name = "boot";
85                              break;
86                          case 0x02:
87                              name = "root";
88                              break;
89                          case 0x03:
90                              name = "swap";
91                              break;
92                          case 0x04:
93                              name = "usr";
94                              break;
95                          case 0x05:
96                              name = "backup";
97                              break;
98                          case 0x06:
99                              name = "stand";
100                             break;
101                         case 0x07:
102                             name = "var";
103                             break;
104                         case 0x08:
105                             name = "home";
106                             break;
107                         case 0x09:
108                             name = "altsctr";
109                             break;
110                         case 0x0a:
111                             name = "cache";
112                             break;
113                         case 0x0b:
114                             name = "reserved";
115                             break;
116                         case 0x0c:
117                             name = "system";
118                             break;
119                         case 0x0e:
120                             name = "public region";
121                             break;
122                         case 0x0f:
123                             name = "private region";
124                             break;
125                         default:
126                             name = Constants.UNKNOWN;
127                             break;
128                         }
129                         // Third field is flags.
130                         String type;
131                         // First character writable, second is mountable
132                         switch (split[2]) {
133                         case "00":
134                             type = "wm";
135                             break;
136                         case "10":
137                             type = "rm";
138                             break;
139                         case "01":
140                             type = "wu";
141                             break;
142                         default:
143                             type = "ru";
144                             break;
145                         }
146                         // Fifth field is sector count
147                         long partSize = bytesPerSector * ParseUtil.parseLongOrDefault(split[4], 0L);
148                         // Seventh field (if present) is mount point
149                         String mountPoint = "";
150                         if (split.length > 6) {
151                             mountPoint = split[6];
152                         }
153                         partList.add(
154                                 new HWPartition(identification, name, type, "", partSize, major, minor, mountPoint));
155                     }
156                 }
157             }
158         }
159         return partList;
160     }
161 }
