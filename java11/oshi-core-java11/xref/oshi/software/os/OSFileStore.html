24  package oshi.software.os;
25  
26  import oshi.annotation.concurrent.ThreadSafe;
27  
28  /**
29   * A FileStore represents a storage pool, device, partition, volume, concrete
30   * file system or other implementation specific means of file storage. This
31   * object carries the same interpretation as core Java's
32   * {@link java.nio.file.FileStore} class, with additional information.
33   */
34  @ThreadSafe
35  public interface OSFileStore {
36  
37      /**
38       * Name of the File System
39       *
40       * @return The file system name
41       */
42      String getName();
43  
44      /**
45       * Volume name of the File System
46       *
47       * @return The volume name of the file system
48       */
49      String getVolume();
50  
51      /**
52       * Label of the File System
53       *
54       * @return The volume label of the file system. Only relevant on Windows and on
55       *         Linux, if assigned; otherwise defaults to the FileSystem name. On
56       *         other operating systems is redundant with the name.
57       */
58      String getLabel();
59  
60      /**
61       * Logical volume of the File System
62       *
63       * Provides an optional alternative volume identifier for the file system. Only
64       * supported on Linux, provides symlink value via '/dev/mapper/' (used with LVM
65       * file systems).
66       *
67       * @return The logical volume of the file system
68       */
69      String getLogicalVolume();
70  
71      /**
72       * Mountpoint of the File System
73       *
74       * @return The mountpoint of the file system
75       */
76      String getMount();
77  
78      /**
79       * Description of the File System
80       *
81       * @return The file system description
82       */
83      String getDescription();
84  
85      /**
86       * Type of the File System (FAT, NTFS, etx2, ext4, etc)
87       *
88       * @return The file system type
89       */
90      String getType();
91  
92      /**
93       * Filesystem options
94       *
95       * @return A comma-deimited string of options
96       */
97      String getOptions();
98  
99      /**
100      * UUID/GUID of the File System
101      *
102      * @return The file system UUID/GUID
103      */
104     String getUUID();
105 
106     /**
107      * Free space on the drive. This space is unallocated but may require elevated
108      * permissions to write.
109      *
110      * @return Free space on the drive (in bytes)
111      */
112     long getFreeSpace();
113 
114     /**
115      * Usable space on the drive. This is space available to unprivileged users.
116      *
117      * @return Usable space on the drive (in bytes)
118      */
119     long getUsableSpace();
120 
121     /**
122      * Total space/capacity of the drive.
123      *
124      * @return Total capacity of the drive (in bytes)
125      */
126     long getTotalSpace();
127 
128     /**
129      * Usable / free inodes on the drive. Not applicable on Windows.
130      *
131      * @return Usable / free inodes on the drive (count), or -1 if unimplemented
132      */
133     long getFreeInodes();
134 
135     /**
136      * Total / maximum number of inodes of the filesystem. Not applicable on
137      * Windows.
138      *
139      * @return Total / maximum number of inodes of the filesystem (count), or -1 if
140      *         unimplemented
141      */
142     long getTotalInodes();
143 
144     /**
145      * Make a best effort to update all the statistics about the file store without
146      * needing to recreate the file store list. This method provides for more
147      * frequent periodic updates of file store statistics.
148      *
149      * @return True if the update was (probably) successful, false if the disk was
150      *         not found
151      */
152     boolean updateAttributes();
153 }
