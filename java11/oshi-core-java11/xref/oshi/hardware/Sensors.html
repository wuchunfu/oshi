24  package oshi.hardware;
25  
26  import oshi.annotation.concurrent.ThreadSafe;
27  
28  /**
29   * Sensors include hardware sensors to monitor temperature, fan speed, and other
30   * information.
31   * <p>
32   * Drivers may or may not exist to collect this data depending on the installed
33   * hardware and Operating System. In addition, software-hardware communication
34   * may suffer intermittent errors when attempting to access this information.
35   * Users should expect, test for, and handle zero values and/or empty arrays
36   * which will result if the OS is unable to provide the information.
37   * <p>
38   * Windows information is retrieved via Windows Management Instrumentation
39   * (WMI). Unfortunately, most hardware providers do not publish values to WMI.
40   * Oshi attempts to retrieve values from
41   * <a href="https://openhardwaremonitor.org/">Open Hardware Monitor</a> if it is
42   * running, in preference to the Microsoft API, which may require elevated
43   * permissions and still may provide no results or unchanging results depending
44   * on the motherboard manufacturer.
45   */
46  @ThreadSafe
47  public interface Sensors {
48      /**
49       * CPU Temperature
50       *
51       * @return CPU Temperature in degrees Celsius if available, 0 or
52       *         {@link Double#NaN} otherwise.
53       *         <p>
54       *         On Windows, if not running Open Hardware Monitor, requires elevated
55       *         permissions and hardware BIOS that supports publishing to WMI. In
56       *         this case, returns the temperature of the "Thermal Zone" which may be
57       *         different than CPU temperature obtained from other sources. In
58       *         addition, some motherboards may only refresh this value on certain
59       *         events.
60       */
61      double getCpuTemperature();
62  
63      /**
64       * Fan speeds
65       *
66       * @return Speed in rpm for all fans. May return empty array if no fans detected
67       *         or 0 fan speed if unable to measure fan speed.
68       */
69      int[] getFanSpeeds();
70  
71      /**
72       * CPU Voltage
73       *
74       * @return CPU Voltage in Volts if available, 0 otherwise.
75       */
76      double getCpuVoltage();
77  }
