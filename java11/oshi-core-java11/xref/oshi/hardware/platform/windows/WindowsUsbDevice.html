36  import oshi.annotation.concurrent.Immutable;

44  /**
45   * Windows Usb Device
46   */
47  @Immutable
48  public class WindowsUsbDevice extends AbstractUsbDevice {
49  
50      private static final GUID GUID_DEVINTERFACE_USB_HOST_CONTROLLER = new GUID(
51              "{3ABF6F2D-71C4-462A-8A92-1E6861E6AF27}");
52  
53      public WindowsUsbDevice(String name, String vendor, String vendorId, String productId, String serialNumber,
54              String uniqueDeviceId, List<UsbDevice> connectedDevices) {
55          super(name, vendor, vendorId, productId, serialNumber, uniqueDeviceId, connectedDevices);
56      }
57  
58      /**
59       * Instantiates a list of {@link oshi.hardware.UsbDevice} objects, representing
60       * devices connected via a usb port (including internal devices).
61       * <p>
62       * If the value of {@code tree} is true, the top level devices returned from
63       * this method are the USB Controllers; connected hubs and devices in its device
64       * tree share that controller's bandwidth. If the value of {@code tree} is
65       * false, USB devices (not controllers) are listed in a single flat list.
66       *
67       * @param tree
68       *            If true, returns a list of controllers, which requires recursive
69       *            iteration of connected devices. If false, returns a flat list of
70       *            devices excluding controllers.
71       * @return a list of {@link oshi.hardware.UsbDevice} objects.
72       */
73      public static List<UsbDevice> getUsbDevices(boolean tree) {
74          List<UsbDevice> devices = queryUsbDevices();
75          if (tree) {
76              return devices;
77          }
78          List<UsbDevice> deviceList = new ArrayList<>();
79          // Top level is controllers; they won't be added to the list, but all
80          // their connected devices will be
81          for (UsbDevice device : devices) {
82              // Recursively add all child devices
83              addDevicesToList(deviceList, device.getConnectedDevices());
84          }
85          return deviceList;
86      }
87  
88      private static void addDevicesToList(List<UsbDevice> deviceList, List<UsbDevice> list) {
89          for (UsbDevice device : list) {
90              deviceList.add(new WindowsUsbDevice(device.getName(), device.getVendor(), device.getVendorId(),
91                      device.getProductId(), device.getSerialNumber(), device.getUniqueDeviceId(),
92                      Collections.emptyList()));
93              addDevicesToList(deviceList, device.getConnectedDevices());
94          }
95      }
96  
97      private static List<UsbDevice> queryUsbDevices() {
98          Quintet<Set<Integer>, Map<Integer, Integer>, Map<Integer, String>, Map<Integer, String>, Map<Integer, String>> controllerDevices = DeviceTree
99                  .queryDeviceTree(GUID_DEVINTERFACE_USB_HOST_CONTROLLER);
100         Map<Integer, Integer> parentMap = controllerDevices.getB();
101         Map<Integer, String> nameMap = controllerDevices.getC();
102         Map<Integer, String> deviceIdMap = controllerDevices.getD();
103         Map<Integer, String> mfgMap = controllerDevices.getE();
104 
105         List<UsbDevice> usbDevices = new ArrayList<>();
106         // recursively build results
107         for (Integer controllerDevice : controllerDevices.getA()) {
108             WindowsUsbDevice deviceAndChildren = queryDeviceAndChildren(controllerDevice, parentMap, nameMap,
109                     deviceIdMap, mfgMap, "0000", "0000", "");
110             if (deviceAndChildren != null) {
111                 usbDevices.add(deviceAndChildren);
112             }
113         }
114         return usbDevices;
115     }
116 
117     private static WindowsUsbDevice queryDeviceAndChildren(Integer device, Map<Integer, Integer> parentMap,
118             Map<Integer, String> nameMap, Map<Integer, String> deviceIdMap, Map<Integer, String> mfgMap, String vid,
119             String pid, String parentSerial) {
120         // Parse vendor and product IDs from the device ID
121         // If this doesn't work, use the IDs from the parent
122         String vendorId = vid;
123         String productId = pid;
124         String serial = parentSerial;
125         Triplet<String, String, String> idsAndSerial = ParseUtil
126                 .parseDeviceIdToVendorProductSerial(deviceIdMap.get(device));
127         if (idsAndSerial != null) {
128             vendorId = idsAndSerial.getA();
129             productId = idsAndSerial.getB();
130             serial = idsAndSerial.getC();
131             if (serial.isEmpty() && vendorId.equals(vid) && productId.equals(pid)) {
132                 serial = parentSerial;
133             }
134         }
135         // Iterate the parent map looking for children
136         Set<Integer> childDeviceSet = parentMap.entrySet().stream().filter(e -> e.getValue().equals(device))
137                 .map(Entry::getKey).collect(Collectors.toSet());
138         // Recursively find those children and put in a list
139         List<UsbDevice> childDevices = new ArrayList<>();
140         for (Integer child : childDeviceSet) {
141             WindowsUsbDevice deviceAndChildren = queryDeviceAndChildren(child, parentMap, nameMap, deviceIdMap, mfgMap,
142                     vendorId, productId, serial);
143             if (deviceAndChildren != null) {
144                 childDevices.add(deviceAndChildren);
145             }
146         }
147         Collections.sort(childDevices);
148         // Finally construct the object and return
149         if (nameMap.containsKey(device)) {
150             String name = nameMap.get(device);
151             if (name.isEmpty()) {
152                 name = vendorId + ":" + productId;
153             }
154             String deviceId = deviceIdMap.get(device);
155             String mfg = mfgMap.get(device);
156             return new WindowsUsbDevice(name, mfg, vendorId, productId, serial, deviceId, childDevices);
157         }
158         return null;
159     }
