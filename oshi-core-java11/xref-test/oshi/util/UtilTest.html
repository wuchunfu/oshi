24  package oshi.util;
25  
26  import static org.hamcrest.MatcherAssert.assertThat;
27  import static org.hamcrest.Matchers.greaterThan;
28  import static org.hamcrest.Matchers.is;
29  
30  import org.junit.jupiter.api.Test;
31  
32  /**
33   * Test general utility methods
34   */
35  class UtilTest {
36  
37      @Test
38      void testSleep() {
39          // Windows counters may be up to 1/64 second (16ms) off
40          long now = System.nanoTime();
41          Util.sleep(100);
42          assertThat(System.nanoTime() - now, is(greaterThan(84_375_000L)));
43      }
44  
45      @Test
46      void testWildcardMatch() {
47          assertThat("Test should not match est", Util.wildcardMatch("Test", "est"), is(false));
48          assertThat("Test should match ^est", Util.wildcardMatch("Test", "^est"), is(true));
49          assertThat("Test should not match ^^est", Util.wildcardMatch("Test", "^^est"), is(false));
50          assertThat("Test should match ?est", Util.wildcardMatch("Test", "?est"), is(true));
51          assertThat("Test should not match ^?est", Util.wildcardMatch("Test", "^?est"), is(false));
52          assertThat("Test should match *est", Util.wildcardMatch("Test", "*est"), is(true));
53          assertThat("Test should not match ^*est", Util.wildcardMatch("Test", "^*est"), is(false));
54  
55          assertThat("Test should not match T?t", Util.wildcardMatch("Test", "T?t"), is(false));
56          assertThat("Test should match T??t", Util.wildcardMatch("Test", "T??t"), is(true));
57          assertThat("Test should match T*t", Util.wildcardMatch("Test", "T*t"), is(true));
58  
59          assertThat("Test should not match Tes", Util.wildcardMatch("Test", "Tes"), is(false));
60          assertThat("Test should match Tes?", Util.wildcardMatch("Test", "Tes?"), is(true));
61          assertThat("Test should match Tes*", Util.wildcardMatch("Test", "Tes*"), is(true));
62  
63          assertThat("Test should not match Te?", Util.wildcardMatch("Test", "Te?"), is(false));
64          assertThat("Test should match Te*", Util.wildcardMatch("Test", "Te*"), is(true));
65      }
66  
67      @Test
68      void testIsBlank() {
69          assertThat("\"\" should be Blank", Util.isBlank(""), is(true));
70          assertThat("null should be Blank", Util.isBlank(null), is(true));
71          assertThat("\"Not blank\" should not be Blank", Util.isBlank("Not blank"), is(false));
72      }
73  
74      @Test
75      void testIsBlankOrUnknown() {
76          assertThat("\"\" should be Blank", Util.isBlankOrUnknown(""), is(true));
77          assertThat("null should be Blank", Util.isBlankOrUnknown(null), is(true));
78          assertThat("unknown should be unknown", Util.isBlankOrUnknown(Constants.UNKNOWN), is(true));
79          assertThat("\"Not blank\" should not be Blank", Util.isBlankOrUnknown("Not blank"), is(false));
80      }
81  }
