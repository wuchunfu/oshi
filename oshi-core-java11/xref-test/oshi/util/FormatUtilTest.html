24  package oshi.util;
25  
26  import static org.hamcrest.MatcherAssert.assertThat;
27  import static org.hamcrest.Matchers.is;
28  
29  import java.text.DecimalFormatSymbols;
30  
31  import org.junit.jupiter.api.Test;
32  
33  /**
34   * The Class FormatUtilTest.
35   */
36  class FormatUtilTest {
37  
38      private static char DECIMAL_SEPARATOR = new DecimalFormatSymbols().getDecimalSeparator();
39  
40      /**
41       * Test format bytes.
42       */
43      @Test
44      void testFormatBytes() {
45          assertThat("format 0 bytes", FormatUtil.formatBytes(0), is("0 bytes"));
46          assertThat("format byte", FormatUtil.formatBytes(1), is("1 byte"));
47          assertThat("format bytes", FormatUtil.formatBytes(532), is("532 bytes"));
48          assertThat("format KiByte", FormatUtil.formatBytes(1024), is("1 KiB"));
49          assertThat("format GiByte", FormatUtil.formatBytes(1024 * 1024 * 1024), is("1 GiB"));
50          assertThat("format TiByte", FormatUtil.formatBytes(1099511627776L), is("1 TiB"));
51      }
52  
53      /**
54       * Test format bytes with decimal separator.
55       */
56      @Test
57      void testFormatBytesWithDecimalSeparator() {
58          String expected1 = "1" + DECIMAL_SEPARATOR + "3 KiB";
59          String expected2 = "2" + DECIMAL_SEPARATOR + "3 MiB";
60          String expected3 = "2" + DECIMAL_SEPARATOR + "2 GiB";
61          String expected4 = "1" + DECIMAL_SEPARATOR + "1 TiB";
62          String expected5 = "1" + DECIMAL_SEPARATOR + "1 PiB";
63          String expected6 = "1" + DECIMAL_SEPARATOR + "1 EiB";
64          assertThat("format KiBytes with decimal separator", FormatUtil.formatBytes(1340), is(expected1));
65          assertThat("format MiBytes with decimal separator", FormatUtil.formatBytes(2400016), is(expected2));
66          assertThat("format GiBytes with decimal separator", FormatUtil.formatBytes(2400000000L), is(expected3));
67          assertThat("format TiBytes with decimal separator", FormatUtil.formatBytes(1099511627776L + 109951162777L),
68                  is(expected4));
69          assertThat("format PiBytes with decimal separator",
70                  FormatUtil.formatBytes(1125899906842624L + 112589990684262L), is(expected5));
71          assertThat("format EiBytes with decimal separator",
72                  FormatUtil.formatBytes(1152921504606846976L + 115292150460684698L), is(expected6));
73      }
74  
75      /**
76       * Test format decimal bytes.
77       */
78      @Test
79      void testFormatBytesDecimal() {
80          assertThat("format 0 bytesDecimal", FormatUtil.formatBytesDecimal(0), is("0 bytes"));
81          assertThat("format byteDecimal", FormatUtil.formatBytesDecimal(1), is("1 byte"));
82          assertThat("format bytesDecimal", FormatUtil.formatBytesDecimal(532), is("532 bytes"));
83          assertThat("format KbytesDecimal", FormatUtil.formatBytesDecimal(1000), is("1 KB"));
84          assertThat("format GbytesDecimal", FormatUtil.formatBytesDecimal(1000 * 1000 * 1000), is("1 GB"));
85          assertThat("format TbytesDecimal", FormatUtil.formatBytesDecimal(1000000000000L), is("1 TB"));
86      }
87  
88      /**
89       * Test format decimal bytes with decimal separator.
90       */
91      @Test
92      void testFormatBytesDecimalWithDecimalSeparator() {
93          String expected1 = "1" + DECIMAL_SEPARATOR + "3 KB";
94          String expected2 = "2" + DECIMAL_SEPARATOR + "3 MB";
95          String expected3 = "2" + DECIMAL_SEPARATOR + "2 GB";
96          String expected4 = "1" + DECIMAL_SEPARATOR + "1 TB";
97          String expected5 = "3" + DECIMAL_SEPARATOR + "4 PB";
98          String expected6 = "5" + DECIMAL_SEPARATOR + "6 EB";
99          assertThat("format KBytes with decimal separator", FormatUtil.formatBytesDecimal(1300), is(expected1));
100         assertThat("format MBytes with decimal separator", FormatUtil.formatBytesDecimal(2300000), is(expected2));
101         assertThat("format GBytes with decimal separator", FormatUtil.formatBytesDecimal(2200000000L), is(expected3));
102         assertThat("format TBytes with decimal separator", FormatUtil.formatBytesDecimal(1100000000000L),
103                 is(expected4));
104         assertThat("format PBytes with decimal separator", FormatUtil.formatBytesDecimal(3400000000000000L),
105                 is(expected5));
106         assertThat("format EBytes with decimal separator", FormatUtil.formatBytesDecimal(5600000000000000000L),
107                 is(expected6));
108     }
109 
110     /**
111      * Test format hertz.
112      */
113     @Test
114     void testFormatHertz() {
115         assertThat("format zero Hertz", FormatUtil.formatHertz(0), is("0 Hz"));
116         assertThat("format one Hertz", FormatUtil.formatHertz(1), is("1 Hz"));
117         assertThat("format many Hertz", FormatUtil.formatHertz(999), is("999 Hz"));
118         assertThat("format KHertz", FormatUtil.formatHertz(1000), is("1 KHz"));
119         assertThat("format MHertz", FormatUtil.formatHertz(1000 * 1000), is("1 MHz"));
120         assertThat("format GHertz", FormatUtil.formatHertz(1000 * 1000 * 1000), is("1 GHz"));
121         assertThat("format THertz", FormatUtil.formatHertz(1000L * 1000L * 1000L * 1000L), is("1 THz"));
122     }
123 
124     /**
125      * Test format elapsed secs
126      */
127     @Test
128     void testFormatElapsedSecs() {
129         assertThat("format 0 elapsed seconds", FormatUtil.formatElapsedSecs(0), is("0 days, 00:00:00"));
130         assertThat("format many elapsed seconds", FormatUtil.formatElapsedSecs(12345), is("0 days, 03:25:45"));
131         assertThat("format elapsed day in seconds", FormatUtil.formatElapsedSecs(123456), is("1 days, 10:17:36"));
132         assertThat("format elapsed days in seconds", FormatUtil.formatElapsedSecs(1234567), is("14 days, 06:56:07"));
133     }

135     /**
136      * Test unsigned int to long.
137      */
138     @Test
139     void testGetUnsignedInt() {
140         assertThat("unsigned int", FormatUtil.getUnsignedInt(-1), is(4294967295L));
141     }
142 
143     /**
144      * Test unsigned string
145      */
146     @Test
147     void testToUnsignedString() {
148         assertThat("Integer to unsigned string", FormatUtil.toUnsignedString(0x00000001), is("1"));
149         assertThat("Big Integer to unsigned string", FormatUtil.toUnsignedString(0x80000000), is("2147483648"));
150         assertThat("INT_MAX to unsigned string", FormatUtil.toUnsignedString(0xffffffff), is("4294967295"));
151 
152         assertThat("Long to unsigned string", FormatUtil.toUnsignedString(0x0000000000000001L), is("1"));
153         assertThat("Big Long to unsigned string", FormatUtil.toUnsignedString(0x8000000000000000L),
154                 is("9223372036854775808"));
155         assertThat("LONG_MAX to unsigned string", FormatUtil.toUnsignedString(0xffffffffffffffffL),
156                 is("18446744073709551615"));
157     }

159     /**
160      * Test format error
161      */
162     @Test
163     void testFormatError() {
164         assertThat("Format error code", FormatUtil.formatError(-1234567000), is("0xB66A00A8"));
165     }
166 
167     /**
168      * Test round to int
169      */
170     @Test
171     void testRoundToInt() {
172         assertThat("Improper rounding pi", FormatUtil.roundToInt(Math.PI), is(3));
173         assertThat("Improper rounding e", FormatUtil.roundToInt(Math.E), is(3));
174         assertThat("Improper rounding 0", FormatUtil.roundToInt(0d), is(0));
175         assertThat("Improper rounding 1", FormatUtil.roundToInt(1d), is(1));
176     }
