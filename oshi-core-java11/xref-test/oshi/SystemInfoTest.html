169     private static void printOperatingSystem(final OperatingSystem os) {
170         oshi.add(String.valueOf(os));
171         oshi.add("Booted: " + Instant.ofEpochSecond(os.getSystemBootTime()));
172         oshi.add("Uptime: " + FormatUtil.formatElapsedSecs(os.getSystemUptime()));
173         oshi.add("Running with" + (os.isElevated() ? "" : "out") + " elevated permissions.");
174         oshi.add("Sessions:");
175         for (OSSession s : os.getSessions()) {
176             oshi.add(" " + s.toString());
177         }
178     }

203     private static void printCpu(CentralProcessor processor) {
204         oshi.add("Context Switches/Interrupts: " + processor.getContextSwitches() + " / " + processor.getInterrupts());
205 
206         long[] prevTicks = processor.getSystemCpuLoadTicks();
207         long[][] prevProcTicks = processor.getProcessorCpuLoadTicks();
208         oshi.add("CPU, IOWait, and IRQ ticks @ 0 sec:" + Arrays.toString(prevTicks));
209         // Wait a second...
210         Util.sleep(1000);
211         long[] ticks = processor.getSystemCpuLoadTicks();
212         oshi.add("CPU, IOWait, and IRQ ticks @ 1 sec:" + Arrays.toString(ticks));
213         long user = ticks[TickType.USER.getIndex()] - prevTicks[TickType.USER.getIndex()];
214         long nice = ticks[TickType.NICE.getIndex()] - prevTicks[TickType.NICE.getIndex()];
215         long sys = ticks[TickType.SYSTEM.getIndex()] - prevTicks[TickType.SYSTEM.getIndex()];
216         long idle = ticks[TickType.IDLE.getIndex()] - prevTicks[TickType.IDLE.getIndex()];
217         long iowait = ticks[TickType.IOWAIT.getIndex()] - prevTicks[TickType.IOWAIT.getIndex()];
218         long irq = ticks[TickType.IRQ.getIndex()] - prevTicks[TickType.IRQ.getIndex()];
219         long softirq = ticks[TickType.SOFTIRQ.getIndex()] - prevTicks[TickType.SOFTIRQ.getIndex()];
220         long steal = ticks[TickType.STEAL.getIndex()] - prevTicks[TickType.STEAL.getIndex()];
221         long totalCpu = user + nice + sys + idle + iowait + irq + softirq + steal;
222 
223         oshi.add(String.format(
224                 "User: %.1f%% Nice: %.1f%% System: %.1f%% Idle: %.1f%% IOwait: %.1f%% IRQ: %.1f%% SoftIRQ: %.1f%% Steal: %.1f%%",
225                 100d * user / totalCpu, 100d * nice / totalCpu, 100d * sys / totalCpu, 100d * idle / totalCpu,
226                 100d * iowait / totalCpu, 100d * irq / totalCpu, 100d * softirq / totalCpu, 100d * steal / totalCpu));
227         oshi.add(String.format("CPU load: %.1f%%", processor.getSystemCpuLoadBetweenTicks(prevTicks) * 100));
228         double[] loadAverage = processor.getSystemLoadAverage(3);
229         oshi.add("CPU load averages:" + (loadAverage[0] < 0 ? " N/A" : String.format(" %.2f", loadAverage[0]))
230                 + (loadAverage[1] < 0 ? " N/A" : String.format(" %.2f", loadAverage[1]))
231                 + (loadAverage[2] < 0 ? " N/A" : String.format(" %.2f", loadAverage[2])));
232         // per core CPU
233         StringBuilder procCpu = new StringBuilder("CPU load per processor:");
234         double[] load = processor.getProcessorCpuLoadBetweenTicks(prevProcTicks);
235         for (double avg : load) {
236             procCpu.append(String.format(" %.1f%%", avg * 100));
237         }
238         oshi.add(procCpu.toString());
239         long freq = processor.getProcessorIdentifier().getVendorFreq();
240         if (freq > 0) {
241             oshi.add("Vendor Frequency: " + FormatUtil.formatHertz(freq));
242         }
243         freq = processor.getMaxFreq();
244         if (freq > 0) {
245             oshi.add("Max Frequency: " + FormatUtil.formatHertz(freq));
246         }
247         long[] freqs = processor.getCurrentFreq();
248         if (freqs[0] > 0) {
249             StringBuilder sb = new StringBuilder("Current Frequencies: ");
250             for (int i = 0; i < freqs.length; i++) {
251                 if (i > 0) {
252                     sb.append(", ");
253                 }
254                 sb.append(FormatUtil.formatHertz(freqs[i]));
255             }
256             oshi.add(sb.toString());
257         }
258     }

260     private static void printProcesses(OperatingSystem os, GlobalMemory memory) {
261         OSProcess myProc = os.getProcess(os.getProcessId());
262         // current process will never be null. Other code should check for null here
263         oshi.add(
264                 "My PID: " + myProc.getProcessID() + " with affinity " + Long.toBinaryString(myProc.getAffinityMask()));
265         oshi.add("Processes: " + os.getProcessCount() + ", Threads: " + os.getThreadCount());
266         // Sort by highest CPU
267         List<OSProcess> procs = os.getProcesses(ProcessFiltering.ALL_PROCESSES, ProcessSorting.CPU_DESC, 5);
268         oshi.add("   PID  %CPU %MEM       VSZ       RSS Name");
269         for (int i = 0; i < procs.size() && i < 5; i++) {
270             OSProcess p = procs.get(i);
271             oshi.add(String.format(" %5d %5.1f %4.1f %9s %9s %s", p.getProcessID(),
272                     100d * (p.getKernelTime() + p.getUserTime()) / p.getUpTime(),
273                     100d * p.getResidentSetSize() / memory.getTotal(), FormatUtil.formatBytes(p.getVirtualSize()),
274                     FormatUtil.formatBytes(p.getResidentSetSize()), p.getName()));
275         }
276         OSProcess p = os.getProcess(os.getProcessId());
277         oshi.add("Current process arguments: ");
278         for (String s : p.getArguments()) {
279             oshi.add("  " + s);
280         }
281         oshi.add("Current process environment: ");
282         for (Entry<String, String> e : p.getEnvironmentVariables().entrySet()) {
283             oshi.add("  " + e.getKey() + "=" + e.getValue());
284         }
285     }

287     private static void printServices(OperatingSystem os) {
288         oshi.add("Services: ");
289         oshi.add("   PID   State   Name");
290         // DO 5 each of running and stopped
291         int i = 0;
292         for (OSService s : os.getServices()) {
293             if (s.getState().equals(OSService.State.RUNNING) && i++ < 5) {
294                 oshi.add(String.format(" %5d  %7s  %s", s.getProcessID(), s.getState(), s.getName()));
295             }
296         }
297         i = 0;
298         for (OSService s : os.getServices()) {
299             if (s.getState().equals(OSService.State.STOPPED) && i++ < 5) {
300                 oshi.add(String.format(" %5d  %7s  %s", s.getProcessID(), s.getState(), s.getName()));
301             }
302         }
303     }

342     private static void printFileSystem(FileSystem fileSystem) {
343         oshi.add("File System:");
344 
345         oshi.add(String.format(" File Descriptors: %d/%d", fileSystem.getOpenFileDescriptors(),
346                 fileSystem.getMaxFileDescriptors()));
347 
348         for (OSFileStore fs : fileSystem.getFileStores()) {
349             long usable = fs.getUsableSpace();
350             long total = fs.getTotalSpace();
351             oshi.add(String.format(
352                     " %s (%s) [%s] %s of %s free (%.1f%%), %s of %s files free (%.1f%%) is %s "
353                             + (fs.getLogicalVolume() != null && fs.getLogicalVolume().length() > 0 ? "[%s]" : "%s")
354                             + " and is mounted at %s",
355                     fs.getName(), fs.getDescription().isEmpty() ? "file system" : fs.getDescription(), fs.getType(),
356                     FormatUtil.formatBytes(usable), FormatUtil.formatBytes(fs.getTotalSpace()), 100d * usable / total,
357                     FormatUtil.formatValue(fs.getFreeInodes(), ""), FormatUtil.formatValue(fs.getTotalInodes(), ""),
358                     100d * fs.getFreeInodes() / fs.getTotalInodes(), fs.getVolume(), fs.getLogicalVolume(),
359                     fs.getMount()));
360         }
361     }
