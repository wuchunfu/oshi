24  package oshi.hardware.common;
25  
26  import java.util.Collections;
27  import java.util.List;
28  
29  import oshi.annotation.concurrent.Immutable;
30  import oshi.hardware.UsbDevice;
31  
32  /**
33   * A USB device
34   */
35  @Immutable
36  public abstract class AbstractUsbDevice implements UsbDevice {
37  
38      private final String name;
39      private final String vendor;
40      private final String vendorId;
41      private final String productId;
42      private final String serialNumber;
43      private final String uniqueDeviceId;
44      private final List<UsbDevice> connectedDevices;
45  
46      protected AbstractUsbDevice(String name, String vendor, String vendorId, String productId, String serialNumber,
47              String uniqueDeviceId, List<UsbDevice> connectedDevices) {
48          this.name = name;
49          this.vendor = vendor;
50          this.vendorId = vendorId;
51          this.productId = productId;
52          this.serialNumber = serialNumber;
53          this.uniqueDeviceId = uniqueDeviceId;
54          this.connectedDevices = Collections.unmodifiableList(connectedDevices);
55      }
56  
57      @Override
58      public String getName() {
59          return this.name;
60      }
61  
62      @Override
63      public String getVendor() {
64          return this.vendor;
65      }
66  
67      @Override
68      public String getVendorId() {
69          return this.vendorId;
70      }
71  
72      @Override
73      public String getProductId() {
74          return this.productId;
75      }
76  
77      @Override
78      public String getSerialNumber() {
79          return this.serialNumber;
80      }
81  
82      @Override
83      public String getUniqueDeviceId() {
84          return this.uniqueDeviceId;
85      }
86  
87      @Override
88      public List<UsbDevice> getConnectedDevices() {
89          return this.connectedDevices;
90      }
91  
92      @Override
93      public int compareTo(UsbDevice usb) {
94          // Naturally sort by device name
95          return getName().compareTo(usb.getName());
96      }
97  
98      @Override
99      public String toString() {
100         return indentUsb(this, 1);
101     }
102 
103     /**
104      * Helper method for indenting chained USB devices
105      *
106      * @param usbDevice
107      *            A USB device to print
108      * @param indent
109      *            number of spaces to indent
110      * @return The device toString, indented
111      */
112     private static String indentUsb(UsbDevice usbDevice, int indent) {
113         String indentFmt = indent > 4 ? String.format("%%%ds|-- ", indent - 4) : String.format("%%%ds", indent);
114         StringBuilder sb = new StringBuilder(String.format(indentFmt, ""));
115         sb.append(usbDevice.getName());
116         if (!usbDevice.getVendor().isEmpty()) {
117             sb.append(" (").append(usbDevice.getVendor()).append(')');
118         }
119         if (!usbDevice.getSerialNumber().isEmpty()) {
120             sb.append(" [s/n: ").append(usbDevice.getSerialNumber()).append(']');
121         }
122         for (UsbDevice connected : usbDevice.getConnectedDevices()) {
123             sb.append('\n').append(indentUsb(connected, indent + 4));
124         }
125         return sb.toString();
126     }
127 }
