69      /**
70       * Create a Processor
71       */
72      protected AbstractCentralProcessor() {
73          // Populate logical processor array.
74          this.logicalProcessors = Collections.unmodifiableList(initProcessorCounts());
75          // Init processor counts
76          Set<String> physProcPkgs = new HashSet<>();
77          Set<Integer> physPkgs = new HashSet<>();
78          for (LogicalProcessor logProc : this.logicalProcessors) {
79              int pkg = logProc.getPhysicalPackageNumber();
80              physProcPkgs.add(logProc.getPhysicalProcessorNumber() + ":" + pkg);
81              physPkgs.add(pkg);
82          }
83          this.logicalProcessorCount = this.logicalProcessors.size();
84          this.physicalProcessorCount = physProcPkgs.size();
85          this.physicalPackageCount = physPkgs.size();
86      }

190     @Override
191     public double getSystemCpuLoadBetweenTicks(long[] oldTicks) {
192         if (oldTicks.length != TickType.values().length) {
193             throw new IllegalArgumentException(
194                     "Tick array " + oldTicks.length + " should have " + TickType.values().length + " elements");
195         }
196         long[] ticks = getSystemCpuLoadTicks();
197         // Calculate total
198         long total = 0;
199         for (int i = 0; i < ticks.length; i++) {
200             total += ticks[i] - oldTicks[i];
201         }
202         // Calculate idle from difference in idle and IOwait
203         long idle = ticks[TickType.IDLE.getIndex()] + ticks[TickType.IOWAIT.getIndex()]
204                 - oldTicks[TickType.IDLE.getIndex()] - oldTicks[TickType.IOWAIT.getIndex()];
205         LOG.trace("Total ticks: {}  Idle ticks: {}", total, idle);
206 
207         return total > 0 && idle >= 0 ? (double) (total - idle) / total : 0d;
208     }
209 
210     @Override
211     public double[] getProcessorCpuLoadBetweenTicks(long[][] oldTicks) {
212         if (oldTicks.length != this.logicalProcessorCount || oldTicks[0].length != TickType.values().length) {
213             throw new IllegalArgumentException(
214                     "Tick array " + oldTicks.length + " should have " + this.logicalProcessorCount
215                             + " arrays, each of which has " + TickType.values().length + " elements");
216         }
217         long[][] ticks = getProcessorCpuLoadTicks();
218         double[] load = new double[this.logicalProcessorCount];
219         for (int cpu = 0; cpu < this.logicalProcessorCount; cpu++) {
220             long total = 0;
221             for (int i = 0; i < ticks[cpu].length; i++) {
222                 total += ticks[cpu][i] - oldTicks[cpu][i];
223             }
224             // Calculate idle from difference in idle and IOwait
225             long idle = ticks[cpu][TickType.IDLE.getIndex()] + ticks[cpu][TickType.IOWAIT.getIndex()]
226                     - oldTicks[cpu][TickType.IDLE.getIndex()] - oldTicks[cpu][TickType.IOWAIT.getIndex()];
227             LOG.trace("CPU: {}  Total ticks: {}  Idle ticks: {}", cpu, total, idle);
228             // update
229             load[cpu] = total > 0 && idle >= 0 ? (double) (total - idle) / total : 0d;
230         }
231         return load;
232     }

249     /**
250      * Creates a Processor ID by encoding the stepping, model, family, and feature
251      * flags.
252      *
253      * @param stepping
254      *            The CPU stepping
255      * @param model
256      *            The CPU model
257      * @param family
258      *            The CPU family
259      * @param flags
260      *            A space-delimited list of CPU feature flags
261      * @return The Processor ID string
262      */
263     protected static String createProcessorID(String stepping, String model, String family, String[] flags) {
264         long processorIdBytes = 0L;
265         long steppingL = ParseUtil.parseLongOrDefault(stepping, 0L);
266         long modelL = ParseUtil.parseLongOrDefault(model, 0L);
267         long familyL = ParseUtil.parseLongOrDefault(family, 0L);
268         // 3:0 – Stepping
269         processorIdBytes |= steppingL & 0xf;
270         // 19:16,7:4 – Model
271         processorIdBytes |= (modelL & 0x0f) << 4;
272         processorIdBytes |= (modelL & 0xf0) << 16;
273         // 27:20,11:8 – Family
274         processorIdBytes |= (familyL & 0x0f) << 8;
275         processorIdBytes |= (familyL & 0xf0) << 20;
276         // 13:12 – Processor Type, assume 0
277         for (String flag : flags) {
278             switch (flag) { // NOSONAR squid:S1479
279             case "fpu":
280                 processorIdBytes |= 1L << 32;
281                 break;
282             case "vme":
283                 processorIdBytes |= 1L << 33;
284                 break;
285             case "de":
286                 processorIdBytes |= 1L << 34;
287                 break;
288             case "pse":
289                 processorIdBytes |= 1L << 35;
290                 break;
291             case "tsc":
292                 processorIdBytes |= 1L << 36;
293                 break;
294             case "msr":
295                 processorIdBytes |= 1L << 37;
296                 break;
297             case "pae":
298                 processorIdBytes |= 1L << 38;
299                 break;
300             case "mce":
301                 processorIdBytes |= 1L << 39;
302                 break;
303             case "cx8":
304                 processorIdBytes |= 1L << 40;
305                 break;
306             case "apic":
307                 processorIdBytes |= 1L << 41;
308                 break;
309             case "sep":
310                 processorIdBytes |= 1L << 43;
311                 break;
312             case "mtrr":
313                 processorIdBytes |= 1L << 44;
314                 break;
315             case "pge":
316                 processorIdBytes |= 1L << 45;
317                 break;
318             case "mca":
319                 processorIdBytes |= 1L << 46;
320                 break;
321             case "cmov":
322                 processorIdBytes |= 1L << 47;
323                 break;
324             case "pat":
325                 processorIdBytes |= 1L << 48;
326                 break;
327             case "pse-36":
328                 processorIdBytes |= 1L << 49;
329                 break;
330             case "psn":
331                 processorIdBytes |= 1L << 50;
332                 break;
333             case "clfsh":
334                 processorIdBytes |= 1L << 51;
335                 break;
336             case "ds":
337                 processorIdBytes |= 1L << 53;
338                 break;
339             case "acpi":
340                 processorIdBytes |= 1L << 54;
341                 break;
342             case "mmx":
343                 processorIdBytes |= 1L << 55;
344                 break;
345             case "fxsr":
346                 processorIdBytes |= 1L << 56;
347                 break;
348             case "sse":
349                 processorIdBytes |= 1L << 57;
350                 break;
351             case "sse2":
352                 processorIdBytes |= 1L << 58;
353                 break;
354             case "ss":
355                 processorIdBytes |= 1L << 59;
356                 break;
357             case "htt":
358                 processorIdBytes |= 1L << 60;
359                 break;
360             case "tm":
361                 processorIdBytes |= 1L << 61;
362                 break;
363             case "ia64":
364                 processorIdBytes |= 1L << 62;
365                 break;
366             case "pbe":
367                 processorIdBytes |= 1L << 63;
368                 break;
369             default:
370                 break;
371             }
372         }
373         return String.format("%016X", processorIdBytes);
374     }

376     @Override
377     public String toString() {
378         StringBuilder sb = new StringBuilder(getProcessorIdentifier().getName());
379         sb.append("\n ").append(getPhysicalPackageCount()).append(" physical CPU package(s)");
380         sb.append("\n ").append(getPhysicalProcessorCount()).append(" physical CPU core(s)");
381         sb.append("\n ").append(getLogicalProcessorCount()).append(" logical CPU(s)");
382         sb.append('\n').append("Identifier: ").append(getProcessorIdentifier().getIdentifier());
383         sb.append('\n').append("ProcessorID: ").append(getProcessorIdentifier().getProcessorID());
384         sb.append('\n').append("Microarchitecture: ").append(getProcessorIdentifier().getMicroarchitecture());
385         return sb.toString();
386     }
