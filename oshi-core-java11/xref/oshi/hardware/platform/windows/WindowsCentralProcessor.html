74       * Initializes Class variables
75       */
76      @Override
77      protected ProcessorIdentifier queryProcessorId() {
78          String cpuVendor = "";
79          String cpuName = "";
80          String cpuIdentifier = "";
81          String cpuFamily = "";
82          String cpuModel = "";
83          String cpuStepping = "";
84          long cpuVendorFreq = 0L;
85          String processorID;
86          boolean cpu64bit = false;
87  
88          final String cpuRegistryRoot = "HARDWARE\\DESCRIPTION\\System\\CentralProcessor\\";
89          String[] processorIds = Advapi32Util.registryGetKeys(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryRoot);
90          if (processorIds.length > 0) {
91              String cpuRegistryPath = cpuRegistryRoot + processorIds[0];
92              cpuVendor = Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
93                      "VendorIdentifier");
94              cpuName = Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
95                      "ProcessorNameString");
96              cpuIdentifier = Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
97                      "Identifier");
98              try {
99                  cpuVendorFreq = Advapi32Util.registryGetIntValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath, "~MHz")
100                         * 1_000_000L;
101             } catch (Win32Exception e) {
102                 // Leave as 0, parse the identifier as backup
103             }
104         }
105         if (!cpuIdentifier.isEmpty()) {
106             cpuFamily = parseIdentifier(cpuIdentifier, "Family");
107             cpuModel = parseIdentifier(cpuIdentifier, "Model");
108             cpuStepping = parseIdentifier(cpuIdentifier, "Stepping");
109         }
110         SYSTEM_INFO sysinfo = new SYSTEM_INFO();
111         Kernel32.INSTANCE.GetNativeSystemInfo(sysinfo);
112         int processorArchitecture = sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue();
113         if (processorArchitecture == 9 // PROCESSOR_ARCHITECTURE_AMD64
114                 || processorArchitecture == 12 // PROCESSOR_ARCHITECTURE_ARM64
115                 || processorArchitecture == 6) { // PROCESSOR_ARCHITECTURE_IA64
116             cpu64bit = true;
117         }
118         WmiResult<ProcessorIdProperty> processorId = Win32Processor.queryProcessorId();
119         if (processorId.getResultCount() > 0) {
120             processorID = WmiUtil.getString(processorId, ProcessorIdProperty.PROCESSORID, 0);
121         } else {
122             processorID = createProcessorID(cpuStepping, cpuModel, cpuFamily,
123                     cpu64bit ? new String[] { "ia64" } : new String[0]);
124         }
125         return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
126                 cpuVendorFreq);
127     }

129     /**
130      * Parses identifier string
131      *
132      * @param identifier
133      *            the full identifier string
134      * @param key
135      *            the key to retrieve
136      * @return the string following id
137      */
138     private static String parseIdentifier(String identifier, String key) {
139         String[] idSplit = ParseUtil.whitespaces.split(identifier);
140         boolean found = false;
141         for (String s : idSplit) {
142             // If key string found, return next value
143             if (found) {
144                 return s;
145             }
146             found = s.equals(key);
147         }
148         // If key string not found, return empty string
149         return "";
150     }

152     @Override
153     protected List<LogicalProcessor> initProcessorCounts() {
154         if (VersionHelpers.IsWindows7OrGreater()) {
155             List<LogicalProcessor> logProcs = LogicalProcessorInformation.getLogicalProcessorInformationEx();
156             // Save numaNode,Processor lookup for future PerfCounter instance lookup
157             // The processor number is based on the Processor Group, so we keep a separate
158             // index by NUMA node.
159             int curNode = -1;
160             int procNum = 0;
161             // 0-indexed list of all lps for array lookup
162             int lp = 0;
163             this.numaNodeProcToLogicalProcMap = new HashMap<>();
164             for (LogicalProcessor logProc : logProcs) {
165                 int node = logProc.getNumaNode();
166                 // This list is grouped by NUMA node so a change in node will reset this counter
167                 if (node != curNode) {
168                     curNode = node;
169                     procNum = 0;
170                 }
171                 numaNodeProcToLogicalProcMap.put(String.format("%d,%d", logProc.getNumaNode(), procNum++), lp++);
172             }
173             return logProcs;
174         } else {
175             return LogicalProcessorInformation.getLogicalProcessorInformation();
176         }
177     }

179     @Override
180     public long[] querySystemCpuLoadTicks() {
181         // To get load in processor group scenario, we need perfmon counters, but the
182         // _Total instance is an average rather than total (scaled) number of ticks
183         // which matches GetSystemTimes() results. We can just query the per-processor
184         // ticks and add them up. Calling the get() method gains the benefit of
185         // synchronizing this output with the memoized result of per-processor ticks as
186         // well.
187         long[] ticks = new long[TickType.values().length];
188         // Sum processor ticks
189         long[][] procTicks = getProcessorCpuLoadTicks();
190         for (int i = 0; i < ticks.length; i++) {
191             for (long[] procTick : procTicks) {
192                 ticks[i] += procTick[i];
193             }
194         }
195         return ticks;
196     }

198     @Override
199     public long[] queryCurrentFreq() {
200         if (VersionHelpers.IsWindows7OrGreater()) {
201             Pair<List<String>, Map<ProcessorFrequencyProperty, List<Long>>> instanceValuePair = ProcessorInformation
202                     .queryFrequencyCounters();
203             List<String> instances = instanceValuePair.getA();
204             Map<ProcessorFrequencyProperty, List<Long>> valueMap = instanceValuePair.getB();
205             List<Long> percentMaxList = valueMap.get(ProcessorFrequencyProperty.PERCENTOFMAXIMUMFREQUENCY);
206             if (!instances.isEmpty()) {
207                 long maxFreq = this.getMaxFreq();
208                 long[] freqs = new long[getLogicalProcessorCount()];
209                 for (int p = 0; p < instances.size(); p++) {
210                     int cpu = instances.get(p).contains(",")
211                             ? numaNodeProcToLogicalProcMap.getOrDefault(instances.get(p), 0)
212                             : ParseUtil.parseIntOrDefault(instances.get(p), 0);
213                     if (cpu >= getLogicalProcessorCount()) {
214                         continue;
215                     }
216                     freqs[cpu] = percentMaxList.get(cpu) * maxFreq / 100L;
217                 }
218                 return freqs;
219             }
220         }
221         // If <Win7 or anything failed in PDH/WMI, use the native call
222         return queryNTPower(2); // Current is field index 2
223     }

225     @Override
226     public long queryMaxFreq() {
227         long[] freqs = queryNTPower(1); // Max is field index 1
228         return Arrays.stream(freqs).max().orElse(-1L);
229     }

231     /**
232      * Call CallNTPowerInformation for Processor information and return an array of
233      * the specified index
234      *
235      * @param fieldIndex
236      *            The field, in order as defined in the
237      *            {@link PowrProf#PROCESSOR_INFORMATION} structure.
238      * @return The array of values.
239      */
240     private long[] queryNTPower(int fieldIndex) {
241         ProcessorPowerInformation ppi = new ProcessorPowerInformation();
242         long[] freqs = new long[getLogicalProcessorCount()];
243         int bufferSize = ppi.size() * freqs.length;
244         Memory mem = new Memory(bufferSize);
245         if (0 != PowrProf.INSTANCE.CallNtPowerInformation(POWER_INFORMATION_LEVEL.ProcessorInformation, null, 0, mem,
246                 bufferSize)) {
247             LOG.error("Unable to get Processor Information");
248             Arrays.fill(freqs, -1L);
249             return freqs;
250         }
251         for (int i = 0; i < freqs.length; i++) {
252             ppi = new ProcessorPowerInformation(mem.share(i * (long) ppi.size()));
253             if (fieldIndex == 1) { // Max
254                 freqs[i] = ppi.maxMhz * 1_000_000L;
255             } else if (fieldIndex == 2) { // Current
256                 freqs[i] = ppi.currentMhz * 1_000_000L;
257             } else {
258                 freqs[i] = -1L;
259             }
260         }
261         return freqs;
262     }

264     @Override
265     public double[] getSystemLoadAverage(int nelem) {
266         if (nelem < 1 || nelem > 3) {
267             throw new IllegalArgumentException("Must include from one to three elements.");
268         }
269         double[] average = new double[nelem];
270         // Windows doesn't have load average
271         for (int i = 0; i < average.length; i++) {
272             average[i] = -1;
273         }
274         return average;
275     }

277     @Override
278     public long[][] queryProcessorCpuLoadTicks() {
279         Pair<List<String>, Map<ProcessorTickCountProperty, List<Long>>> instanceValuePair = ProcessorInformation
280                 .queryProcessorCounters();
281         List<String> instances = instanceValuePair.getA();
282         Map<ProcessorTickCountProperty, List<Long>> valueMap = instanceValuePair.getB();
283         List<Long> systemList = valueMap.get(ProcessorTickCountProperty.PERCENTPRIVILEGEDTIME);
284         List<Long> userList = valueMap.get(ProcessorTickCountProperty.PERCENTUSERTIME);
285         List<Long> irqList = valueMap.get(ProcessorTickCountProperty.PERCENTINTERRUPTTIME);
286         List<Long> softIrqList = valueMap.get(ProcessorTickCountProperty.PERCENTDPCTIME);
287         // % Processor Time is actually Idle time
288         List<Long> idleList = valueMap.get(ProcessorTickCountProperty.PERCENTPROCESSORTIME);
289 
290         long[][] ticks = new long[getLogicalProcessorCount()][TickType.values().length];
291         if (instances.isEmpty() || systemList == null || userList == null || irqList == null || softIrqList == null
292                 || idleList == null) {
293             return ticks;
294         }
295         for (int p = 0; p < instances.size(); p++) {
296             int cpu = instances.get(p).contains(",") ? numaNodeProcToLogicalProcMap.getOrDefault(instances.get(p), 0)
297                     : ParseUtil.parseIntOrDefault(instances.get(p), 0);
298             if (cpu >= getLogicalProcessorCount()) {
299                 continue;
300             }
301             ticks[cpu][TickType.SYSTEM.getIndex()] = systemList.get(cpu);
302             ticks[cpu][TickType.USER.getIndex()] = userList.get(cpu);
303             ticks[cpu][TickType.IRQ.getIndex()] = irqList.get(cpu);
304             ticks[cpu][TickType.SOFTIRQ.getIndex()] = softIrqList.get(cpu);
305             ticks[cpu][TickType.IDLE.getIndex()] = idleList.get(cpu);
306 
307             // Additional decrement to avoid double counting in the
308             // total array
309             ticks[cpu][TickType.SYSTEM.getIndex()] -= ticks[cpu][TickType.IRQ.getIndex()]
310                     + ticks[cpu][TickType.SOFTIRQ.getIndex()];
311 
312             // Raw value is cumulative 100NS-ticks
313             // Divide by 10_000 to get milliseconds
314             ticks[cpu][TickType.SYSTEM.getIndex()] /= 10_000L;
315             ticks[cpu][TickType.USER.getIndex()] /= 10_000L;
316             ticks[cpu][TickType.IRQ.getIndex()] /= 10_000L;
317             ticks[cpu][TickType.SOFTIRQ.getIndex()] /= 10_000L;
318             ticks[cpu][TickType.IDLE.getIndex()] /= 10_000L;
319         }
320         // Skipping nice and IOWait, they'll stay 0
321         return ticks;
322     }
