130     @Override
131     public boolean updateAttributes() {
132         String index = null;
133         List<HWPartition> partitions = getPartitions();
134         if (!partitions.isEmpty()) {
135             // If a partition exists on this drive, the major property
136             // corresponds to the disk index, so use it.
137             index = Integer.toString(partitions.get(0).getMajor());
138         } else if (getName().startsWith(PHYSICALDRIVE_PREFIX)) {
139             // If no partition exists, Windows reliably uses a name to match the
140             // disk index. That said, the skeptical person might wonder why a
141             // disk has read/write statistics without a partition, and wonder
142             // why this branch is even relevant as an option. The author of this
143             // comment does not have an answer for this valid question.
144             index = getName().substring(PHYSICALDRIVE_PREFIX.length(), getName().length());
145         } else {
146             // The author of this comment cannot fathom a circumstance in which
147             // the code reaches this point, but just in case it does, here's the
148             // correct response. If you get this log warning, the circumstances
149             // would be of great interest to the project's maintainers.
150             LOG.warn("Couldn't match index for {}", getName());
151             return false;
152         }
153         DiskStats stats = queryReadWriteStats(index);
154         if (stats.readMap.containsKey(index)) {
155             this.reads = stats.readMap.getOrDefault(index, 0L);
156             this.readBytes = stats.readByteMap.getOrDefault(index, 0L);
157             this.writes = stats.writeMap.getOrDefault(index, 0L);
158             this.writeBytes = stats.writeByteMap.getOrDefault(index, 0L);
159             this.currentQueueLength = stats.queueLengthMap.getOrDefault(index, 0L);
160             this.transferTime = stats.diskTimeMap.getOrDefault(index, 0L);
161             this.timeStamp = stats.timeStamp;
162             return true;
163         } else {
164             return false;
165         }
166     }

168     /**
169      * Gets the disks on this machine
170      *
171      * @return a list of {@link HWDiskStore} objects representing the disks
172      */
173     public static List<HWDiskStore> getDisks() {
174         WmiQueryHandler h = WmiQueryHandler.createInstance();
175         boolean comInit = false;
176         try {
177             comInit = h.initCOM();
178             List<HWDiskStore> result;
179             result = new ArrayList<>();
180             DiskStats stats = queryReadWriteStats(null);
181             PartitionMaps maps = queryPartitionMaps(h);
182 
183             WmiResult<DiskDriveProperty> vals = Win32DiskDrive.queryDiskDrive(h);
184             for (int i = 0; i < vals.getResultCount(); i++) {
185                 WindowsHWDiskStore ds = new WindowsHWDiskStore(WmiUtil.getString(vals, DiskDriveProperty.NAME, i),
186                         String.format("%s %s", WmiUtil.getString(vals, DiskDriveProperty.MODEL, i),
187                                 WmiUtil.getString(vals, DiskDriveProperty.MANUFACTURER, i)).trim(),
188                         // Most vendors store serial # as a hex string; convert
189                         ParseUtil.hexStringToString(WmiUtil.getString(vals, DiskDriveProperty.SERIALNUMBER, i)),
190                         WmiUtil.getUint64(vals, DiskDriveProperty.SIZE, i));
191 
192                 String index = Integer.toString(WmiUtil.getUint32(vals, DiskDriveProperty.INDEX, i));
193                 ds.reads = stats.readMap.getOrDefault(index, 0L);
194                 ds.readBytes = stats.readByteMap.getOrDefault(index, 0L);
195                 ds.writes = stats.writeMap.getOrDefault(index, 0L);
196                 ds.writeBytes = stats.writeByteMap.getOrDefault(index, 0L);
197                 ds.currentQueueLength = stats.queueLengthMap.getOrDefault(index, 0L);
198                 // DiskTime (sum of readTime+writeTime) slightly overestimates actual transfer
199                 // time because it includes waiting time in the queue and can exceed 100%.
200                 // However, alternative calculations require use of a timestamp with 1/64-second
201                 // resolution producing unacceptable variation in what should be a monotonically
202                 // increasing counter. See extended discussion and experiments here:
203                 // https://github.com/oshi/oshi/issues/1504
204                 ds.transferTime = stats.diskTimeMap.getOrDefault(index, 0L);
205                 ds.timeStamp = stats.timeStamp;
206                 // Get partitions
207                 List<HWPartition> partitions = new ArrayList<>();
208                 List<String> partList = maps.driveToPartitionMap.get(ds.getName());
209                 if (partList != null && !partList.isEmpty()) {
210                     for (String part : partList) {
211                         if (maps.partitionMap.containsKey(part)) {
212                             partitions.addAll(maps.partitionMap.get(part));
213                         }
214                     }
215                 }
216                 ds.partitionList = Collections.unmodifiableList(partitions.stream()
217                         .sorted(Comparator.comparing(HWPartition::getName)).collect(Collectors.toList()));
218                 // Add to list
219                 result.add(ds);
220             }
221             return result;
222         } catch (COMException e) {
223             LOG.warn("COM exception: {}", e.getMessage());
224             return Collections.emptyList();
225         } finally {
226             if (comInit) {
227                 h.unInitCOM();
228             }
229         }
230     }

232     /**
233      * Gets disk stats for the specified index. If the index is null, populates all
234      * the maps
235      *
236      * @param index
237      *            The index to populate/update maps for
238      * @return An object encapsulating maps with the stats
239      */
240     private static DiskStats queryReadWriteStats(String index) {
241         // Create object to hold and return results
242         DiskStats stats = new DiskStats();
243         Pair<List<String>, Map<PhysicalDiskProperty, List<Long>>> instanceValuePair = PhysicalDisk.queryDiskCounters();
244         List<String> instances = instanceValuePair.getA();
245         Map<PhysicalDiskProperty, List<Long>> valueMap = instanceValuePair.getB();
246         stats.timeStamp = System.currentTimeMillis();
247         List<Long> readList = valueMap.get(PhysicalDiskProperty.DISKREADSPERSEC);
248         List<Long> readByteList = valueMap.get(PhysicalDiskProperty.DISKREADBYTESPERSEC);
249         List<Long> writeList = valueMap.get(PhysicalDiskProperty.DISKWRITESPERSEC);
250         List<Long> writeByteList = valueMap.get(PhysicalDiskProperty.DISKWRITEBYTESPERSEC);
251         List<Long> queueLengthList = valueMap.get(PhysicalDiskProperty.CURRENTDISKQUEUELENGTH);
252         List<Long> diskTimeList = valueMap.get(PhysicalDiskProperty.PERCENTDISKTIME);
253 
254         if (instances.isEmpty() || readList == null || readByteList == null || writeList == null
255                 || writeByteList == null || queueLengthList == null || diskTimeList == null) {
256             return stats;
257         }
258         for (int i = 0; i < instances.size(); i++) {
259             String name = getIndexFromName(instances.get(i));
260             // If index arg passed, only update passed arg
261             if (index != null && !index.equals(name)) {
262                 continue;
263             }
264             stats.readMap.put(name, readList.get(i));
265             stats.readByteMap.put(name, readByteList.get(i));
266             stats.writeMap.put(name, writeList.get(i));
267             stats.writeByteMap.put(name, writeByteList.get(i));
268             stats.queueLengthMap.put(name, queueLengthList.get(i));
269             stats.diskTimeMap.put(name, diskTimeList.get(i) / 10_000L);
270         }
271         return stats;
272     }

274     private static PartitionMaps queryPartitionMaps(WmiQueryHandler h) {
275         // Create object to hold and return results
276         PartitionMaps maps = new PartitionMaps();
277 
278         // For Regexp matching DeviceIDs
279         Matcher mAnt;
280         Matcher mDep;
281 
282         // Map drives to partitions
283         WmiResult<DriveToPartitionProperty> drivePartitionMap = Win32DiskDriveToDiskPartition.queryDriveToPartition(h);
284         for (int i = 0; i < drivePartitionMap.getResultCount(); i++) {
285             mAnt = DEVICE_ID.matcher(WmiUtil.getRefString(drivePartitionMap, DriveToPartitionProperty.ANTECEDENT, i));
286             mDep = DEVICE_ID.matcher(WmiUtil.getRefString(drivePartitionMap, DriveToPartitionProperty.DEPENDENT, i));
287             if (mAnt.matches() && mDep.matches()) {
288                 maps.driveToPartitionMap.computeIfAbsent(mAnt.group(1).replace("\\\\", "\\"), x -> new ArrayList<>())
289                         .add(mDep.group(1));
290             }
291         }
292 
293         // Map partitions to logical disks
294         WmiResult<DiskToPartitionProperty> diskPartitionMap = Win32LogicalDiskToPartition.queryDiskToPartition(h);
295         for (int i = 0; i < diskPartitionMap.getResultCount(); i++) {
296             mAnt = DEVICE_ID.matcher(WmiUtil.getRefString(diskPartitionMap, DiskToPartitionProperty.ANTECEDENT, i));
297             mDep = DEVICE_ID.matcher(WmiUtil.getRefString(diskPartitionMap, DiskToPartitionProperty.DEPENDENT, i));
298             long size = WmiUtil.getUint64(diskPartitionMap, DiskToPartitionProperty.ENDINGADDRESS, i)
299                     - WmiUtil.getUint64(diskPartitionMap, DiskToPartitionProperty.STARTINGADDRESS, i) + 1L;
300             if (mAnt.matches() && mDep.matches()) {
301                 if (maps.partitionToLogicalDriveMap.containsKey(mAnt.group(1))) {
302                     maps.partitionToLogicalDriveMap.get(mAnt.group(1)).add(new Pair<>(mDep.group(1) + "\\", size));
303                 } else {
304                     List<Pair<String, Long>> list = new ArrayList<>();
305                     list.add(new Pair<>(mDep.group(1) + "\\", size));
306                     maps.partitionToLogicalDriveMap.put(mAnt.group(1), list);
307                 }
308             }
309         }
310 
311         // Next, get all partitions and create objects
312         WmiResult<DiskPartitionProperty> hwPartitionQueryMap = Win32DiskPartition.queryPartition(h);
313         for (int i = 0; i < hwPartitionQueryMap.getResultCount(); i++) {
314             String deviceID = WmiUtil.getString(hwPartitionQueryMap, DiskPartitionProperty.DEVICEID, i);
315             List<Pair<String, Long>> logicalDrives = maps.partitionToLogicalDriveMap.get(deviceID);
316             if (logicalDrives == null) {
317                 continue;
318             }
319             for (int j = 0; j < logicalDrives.size(); j++) {
320                 Pair<String, Long> logicalDrive = logicalDrives.get(j);
321                 if (logicalDrive != null && !logicalDrive.getA().isEmpty()) {
322                     char[] volumeChr = new char[GUID_BUFSIZE];
323                     Kernel32.INSTANCE.GetVolumeNameForVolumeMountPoint(logicalDrive.getA(), volumeChr, GUID_BUFSIZE);
324                     String uuid = ParseUtil.parseUuidOrDefault(new String(volumeChr).trim(), "");
325                     HWPartition pt = new HWPartition(
326                             WmiUtil.getString(hwPartitionQueryMap, DiskPartitionProperty.NAME, i),
327                             WmiUtil.getString(hwPartitionQueryMap, DiskPartitionProperty.TYPE, i),
328                             WmiUtil.getString(hwPartitionQueryMap, DiskPartitionProperty.DESCRIPTION, i), uuid,
329                             logicalDrive.getB(),
330                             WmiUtil.getUint32(hwPartitionQueryMap, DiskPartitionProperty.DISKINDEX, i),
331                             WmiUtil.getUint32(hwPartitionQueryMap, DiskPartitionProperty.INDEX, i),
332                             logicalDrive.getA());
333                     if (maps.partitionMap.containsKey(deviceID)) {
334                         maps.partitionMap.get(deviceID).add(pt);
335                     } else {
336                         List<HWPartition> ptlist = new ArrayList<>();
337                         ptlist.add(pt);
338                         maps.partitionMap.put(deviceID, ptlist);
339                     }
340                 }
341             }
342         }
343         return maps;
344     }

346     /**
347      * Parse a drive name like "0 C:" to just the index "0"
348      *
349      * @param s
350      *            A drive name to parse
351      * @return The first space-delimited value
352      */
353     private static String getIndexFromName(String s) {
354         if (s.isEmpty()) {
355             return s;
356         }
357         return s.split("\\s")[0];
358     }
