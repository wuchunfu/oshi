24  package oshi.hardware.platform.unix.freebsd;
25  
26  import java.util.ArrayList;
27  import java.util.Collections;
28  import java.util.List;
29  
30  import oshi.annotation.concurrent.Immutable;
31  import oshi.hardware.GraphicsCard;
32  import oshi.hardware.common.AbstractGraphicsCard;
33  import oshi.util.Constants;
34  import oshi.util.ExecutingCommand;
35  import oshi.util.ParseUtil;
36  
37  /**
38   * Graphics Card info obtained from pciconf
39   */
40  @Immutable
41  final class FreeBsdGraphicsCard extends AbstractGraphicsCard {
42  
43      private static final String PCI_CLASS_DISPLAY = "0x03";
44  
45      /**
46       * Constructor for FreeBsdGraphicsCard
47       *
48       * @param name
49       *            The name
50       * @param deviceId
51       *            The device ID
52       * @param vendor
53       *            The vendor
54       * @param versionInfo
55       *            The version info
56       * @param vram
57       *            The VRAM
58       */
59      FreeBsdGraphicsCard(String name, String deviceId, String vendor, String versionInfo, long vram) {
60          super(name, deviceId, vendor, versionInfo, vram);
61      }
62  
63      /**
64       * public method used by
65       * {@link oshi.hardware.common.AbstractHardwareAbstractionLayer} to access the
66       * graphics cards.
67       *
68       * @return List of
69       *         {@link oshi.hardware.platform.unix.freebsd.FreeBsdGraphicsCard}
70       *         objects.
71       */
72      public static List<GraphicsCard> getGraphicsCards() {
73          List<GraphicsCard> cardList = new ArrayList<>();
74          // Enumerate all devices and add if required
75          List<String> devices = ExecutingCommand.runNative("pciconf -lv");
76          if (devices.isEmpty()) {
77              return Collections.emptyList();
78          }
79          String name = Constants.UNKNOWN;
80          String vendorId = Constants.UNKNOWN;
81          String productId = Constants.UNKNOWN;
82          String classCode = "";
83          String versionInfo = Constants.UNKNOWN;
84          for (String line : devices) {
85              if (line.contains("class=0x")) {
86                  // Identifies start of a new device. Save previous if it's a graphics card
87                  if (PCI_CLASS_DISPLAY.equals(classCode)) {
88                      cardList.add(new FreeBsdGraphicsCard(name.isEmpty() ? Constants.UNKNOWN : name,
89                              productId.isEmpty() ? Constants.UNKNOWN : productId,
90                              vendorId.isEmpty() ? Constants.UNKNOWN : vendorId,
91                              versionInfo.isEmpty() ? Constants.UNKNOWN : versionInfo, 0L));
92                  }
93                  // Parse this line
94                  String[] split = ParseUtil.whitespaces.split(line);
95                  for (String s : split) {
96                      String[] keyVal = s.split("=");
97                      if (keyVal.length > 1) {
98                          if (keyVal[0].equals("class") && keyVal[1].length() >= 4) {
99                              // class=0x030000
100                             classCode = keyVal[1].substring(0, 4);
101                         } else if (keyVal[0].equals("chip") && keyVal[1].length() >= 10) {
102                             // chip=0x3ea08086
103                             productId = keyVal[1].substring(0, 6);
104                             vendorId = "0x" + keyVal[1].substring(6, 10);
105                         } else if (keyVal[0].contains("rev")) {
106                             // rev=0x00
107                             versionInfo = s;
108                         }
109                     }
110                 }
111                 // Reset name
112                 name = Constants.UNKNOWN;
113             } else {
114                 String[] split = line.trim().split("=", 2);
115                 if (split.length == 2) {
116                     String key = split[0].trim();
117                     if (key.equals("vendor")) {
118                         vendorId = ParseUtil.getSingleQuoteStringValue(line)
119                                 + (vendorId.equals(Constants.UNKNOWN) ? "" : " (" + vendorId + ")");
120                     } else if (key.equals("device")) {
121                         name = ParseUtil.getSingleQuoteStringValue(line);
122                     }
123                 }
124             }
125         }
126         // In case we reached end before saving
127         if (PCI_CLASS_DISPLAY.equals(classCode)) {
128             cardList.add(new FreeBsdGraphicsCard(name.isEmpty() ? Constants.UNKNOWN : name,
129                     productId.isEmpty() ? Constants.UNKNOWN : productId,
130                     vendorId.isEmpty() ? Constants.UNKNOWN : vendorId,
131                     versionInfo.isEmpty() ? Constants.UNKNOWN : versionInfo, 0L));
132         }
133         return cardList;
134     }
135 }
