46          super(name, vendor, vendorId, productId, serialNumber, uniqueDeviceId, connectedDevices);

49      /**
50       * Instantiates a list of {@link oshi.hardware.UsbDevice} objects, representing
51       * devices connected via a usb port (including internal devices).
52       * <p>
53       * If the value of {@code tree} is true, the top level devices returned from
54       * this method are the USB Controllers; connected hubs and devices in its device
55       * tree share that controller's bandwidth. If the value of {@code tree} is
56       * false, USB devices (not controllers) are listed in a single flat list.
57       *
58       * @param tree
59       *            If true, returns a list of controllers, which requires recursive
60       *            iteration of connected devices. If false, returns a flat list of
61       *            devices excluding controllers.
62       * @return a list of {@link oshi.hardware.UsbDevice} objects.
63       */
64      public static List<UsbDevice> getUsbDevices(boolean tree) {
65          List<UsbDevice> devices = getUsbDevices();
66          if (tree) {
67              return devices;
68          }
69          List<UsbDevice> deviceList = new ArrayList<>();
70          // Top level is controllers; they won't be added to the list, but all
71          // their connected devices will be
72          for (UsbDevice device : devices) {
73              deviceList.add(new FreeBsdUsbDevice(device.getName(), device.getVendor(), device.getVendorId(),
74                      device.getProductId(), device.getSerialNumber(), device.getUniqueDeviceId(),
75                      Collections.emptyList()));
76              addDevicesToList(deviceList, device.getConnectedDevices());
77          }
78          return deviceList;
79      }

81      private static List<UsbDevice> getUsbDevices() {
82          // Maps to store information using node # as the key
83          Map<String, String> nameMap = new HashMap<>();
84          Map<String, String> vendorMap = new HashMap<>();
85          Map<String, String> vendorIdMap = new HashMap<>();
86          Map<String, String> productIdMap = new HashMap<>();
87          Map<String, String> serialMap = new HashMap<>();
88          Map<String, String> parentMap = new HashMap<>();
89          Map<String, List<String>> hubMap = new HashMap<>();
90  
91          // Enumerate all devices and build information maps. This will build the
92          // entire device tree; we will identify the controllers as the parents
93          // of the usbus entries and eventually only populate the returned
94          // results with those
95          List<String> devices = ExecutingCommand.runNative("lshal");
96          if (devices.isEmpty()) {
97              return Collections.emptyList();
98          }
99          // For each item enumerated, store information in the maps
100         String key = "";
101         List<String> usBuses = new ArrayList<>();
102         for (String line : devices) {
103             // udi = ... identifies start of a new tree
104             if (line.startsWith("udi =")) {
105                 // Remove indent for key
106                 key = ParseUtil.getSingleQuoteStringValue(line);
107             } else if (!key.isEmpty()) {
108                 // We are currently processing for node identified by key. Save
109                 // approrpriate variables to maps.
110                 line = line.trim();
111                 if (!line.isEmpty()) {
112                     if (line.startsWith("freebsd.driver =")
113                             && "usbus".equals(ParseUtil.getSingleQuoteStringValue(line))) {
114                         usBuses.add(key);
115                     } else if (line.contains(".parent =")) {
116                         String parent = ParseUtil.getSingleQuoteStringValue(line);
117                         // If this is interface of parent, skip
118                         if (key.replace(parent, "").startsWith("_if")) {
119                             continue;
120                         }
121                         // Store parent for later usbus-skipping
122                         parentMap.put(key, parent);
123                         // Add this key to the parent's hubmap list
124                         hubMap.computeIfAbsent(parent, x -> new ArrayList<>()).add(key);
125                     } else if (line.contains(".product =")) {
126                         nameMap.put(key, ParseUtil.getSingleQuoteStringValue(line));
127                     } else if (line.contains(".vendor =")) {
128                         vendorMap.put(key, ParseUtil.getSingleQuoteStringValue(line));
129                     } else if (line.contains(".serial =")) {
130                         String serial = ParseUtil.getSingleQuoteStringValue(line);
131                         serialMap.put(key,
132                                 serial.startsWith("0x") ? ParseUtil.hexStringToString(serial.replace("0x", ""))
133                                         : serial);
134                     } else if (line.contains(".vendor_id =")) {
135                         vendorIdMap.put(key, String.format("%04x", ParseUtil.getFirstIntValue(line)));
136                     } else if (line.contains(".product_id =")) {
137                         productIdMap.put(key, String.format("%04x", ParseUtil.getFirstIntValue(line)));
138                     }
139                 }
140             }
141         }
142 
143         // Build tree and return
144         List<UsbDevice> controllerDevices = new ArrayList<>();
145         for (String usbus : usBuses) {
146             // Skip the usbuses: make their parents the controllers and replace
147             // parents' children with the buses' children
148             String parent = parentMap.get(usbus);
149             hubMap.put(parent, hubMap.get(usbus));
150             controllerDevices.add(getDeviceAndChildren(parent, "0000", "0000", nameMap, vendorMap, vendorIdMap,
151                     productIdMap, serialMap, hubMap));
152         }
153         return controllerDevices;
154     }
155 
156     private static void addDevicesToList(List<UsbDevice> deviceList, List<UsbDevice> list) {
157         for (UsbDevice device : list) {
158             deviceList.add(device);
159             addDevicesToList(deviceList, device.getConnectedDevices());
160         }
161     }

163     /**
164      * Recursively creates FreeBsdUsbDevices by fetching information from maps to
165      * populate fields
166      *
167      * @param devPath
168      *            The device node path.
169      * @param vid
170      *            The default (parent) vendor ID
171      * @param pid
172      *            The default (parent) product ID
173      * @param nameMap
174      *            the map of names
175      * @param vendorMap
176      *            the map of vendors
177      * @param vendorIdMap
178      *            the map of vendorIds
179      * @param productIdMap
180      *            the map of productIds
181      * @param serialMap
182      *            the map of serial numbers
183      * @param hubMap
184      *            the map of hubs
185      * @return A SolarisUsbDevice corresponding to this device
186      */
187     private static FreeBsdUsbDevice getDeviceAndChildren(String devPath, String vid, String pid,
188             Map<String, String> nameMap, Map<String, String> vendorMap, Map<String, String> vendorIdMap,
189             Map<String, String> productIdMap, Map<String, String> serialMap, Map<String, List<String>> hubMap) {
190         String vendorId = vendorIdMap.getOrDefault(devPath, vid);
191         String productId = productIdMap.getOrDefault(devPath, pid);
192         List<String> childPaths = hubMap.getOrDefault(devPath, new ArrayList<>());
193         List<UsbDevice> usbDevices = new ArrayList<>();
194         for (String path : childPaths) {
195             usbDevices.add(getDeviceAndChildren(path, vendorId, productId, nameMap, vendorMap, vendorIdMap,
196                     productIdMap, serialMap, hubMap));
197         }
198         Collections.sort(usbDevices);
199         return new FreeBsdUsbDevice(nameMap.getOrDefault(devPath, vendorId + ":" + productId),
200                 vendorMap.getOrDefault(devPath, ""), vendorId, productId, serialMap.getOrDefault(devPath, ""), devPath,
201                 usbDevices);
202     }
