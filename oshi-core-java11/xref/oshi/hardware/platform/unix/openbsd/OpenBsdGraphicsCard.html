24  package oshi.hardware.platform.unix.openbsd;
25  
26  import java.util.ArrayList;
27  import java.util.Collections;
28  import java.util.List;
29  import java.util.regex.Matcher;
30  import java.util.regex.Pattern;
31  
32  import oshi.annotation.concurrent.Immutable;
33  import oshi.hardware.GraphicsCard;
34  import oshi.hardware.common.AbstractGraphicsCard;
35  import oshi.util.Constants;
36  import oshi.util.ExecutingCommand;
37  
38  /**
39   * Graphics Card info obtained from pciconf
40   */
41  @Immutable
42  final class OpenBsdGraphicsCard extends AbstractGraphicsCard {
43  
44      private static final String PCI_CLASS_DISPLAY = "Class: 03 Display";
45      private static final Pattern PCI_DUMP_HEADER = Pattern.compile(" \\d+:\\d+:\\d+: (.+)");
46  
47      /**
48       * Constructor for OpenBsdGraphicsCard
49       *
50       * @param name
51       *            The name
52       * @param deviceId
53       *            The device ID
54       * @param vendor
55       *            The vendor
56       * @param versionInfo
57       *            The version info
58       * @param vram
59       *            The VRAM
60       */
61      OpenBsdGraphicsCard(String name, String deviceId, String vendor, String versionInfo, long vram) {
62          super(name, deviceId, vendor, versionInfo, vram);
63      }
64  
65      /**
66       * public method used by
67       * {@link oshi.hardware.common.AbstractHardwareAbstractionLayer} to access the
68       * graphics cards.
69       *
70       * @return List of
71       *         {@link oshi.hardware.platform.unix.freebsd.OpenBsdGraphicsCard}
72       *         objects.
73       */
74      public static List<GraphicsCard> getGraphicsCards() {
75          List<GraphicsCard> cardList = new ArrayList<>();
76          // Enumerate all devices and add if required
77          List<String> devices = ExecutingCommand.runNative("pcidump -v");
78          if (devices.isEmpty()) {
79              return Collections.emptyList();
80          }
81          String name = "";
82          String vendorId = "";
83          String productId = "";
84          boolean classCodeFound = false;
85          String versionInfo = "";
86          for (String line : devices) {
87              Matcher m = PCI_DUMP_HEADER.matcher(line);
88              if (m.matches()) {
89                  // Identifies start of a new device. Save previous if it's a graphics card
90                  if (classCodeFound) {
91                      cardList.add(new OpenBsdGraphicsCard(name.isEmpty() ? Constants.UNKNOWN : name,
92                              productId.isEmpty() ? "0x0000" : productId, vendorId.isEmpty() ? "0x0000" : vendorId,
93                              versionInfo.isEmpty() ? Constants.UNKNOWN : versionInfo, 0L));
94                  }
95                  // Device name is the captured pattern
96                  name = m.group(1);
97                  // Reset values
98                  vendorId = "";
99                  productId = "";
100                 classCodeFound = false;
101                 versionInfo = "";
102             } else {
103                 int idx;
104                 // Look for:
105                 // 0x0000: Vendor ID: 1ab8, Product ID: 4005
106                 // 0x0008: Class: 03 Display, Subclass: 00 VGA
107                 // ....... Interface: 00, Revision: 00
108                 if (!classCodeFound) {
109                     idx = line.indexOf("Vendor ID: ");
110                     if (idx >= 0 && line.length() >= idx + 15) {
111                         vendorId = "0x" + line.substring(idx + 11, idx + 15);
112                     }
113                     idx = line.indexOf("Product ID: ");
114                     if (idx >= 0 && line.length() >= idx + 16) {
115                         productId = "0x" + line.substring(idx + 12, idx + 16);
116                     }
117                     if (line.contains(PCI_CLASS_DISPLAY)) {
118                         classCodeFound = true;
119                     }
120                 } else if (versionInfo.isEmpty()) {
121                     idx = line.indexOf("Revision: ");
122                     if (idx >= 0) {
123                         versionInfo = line.substring(idx);
124                     }
125                 }
126             }
127         }
128         // In case we reached end before saving
129         if (classCodeFound) {
130             cardList.add(new OpenBsdGraphicsCard(name.isEmpty() ? Constants.UNKNOWN : name,
131                     productId.isEmpty() ? "0x0000" : productId, vendorId.isEmpty() ? "0x0000" : vendorId,
132                     versionInfo.isEmpty() ? Constants.UNKNOWN : versionInfo, 0L));
133         }
134         return cardList;
135     }
136 }
