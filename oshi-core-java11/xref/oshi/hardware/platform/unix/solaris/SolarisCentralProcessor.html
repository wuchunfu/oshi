50      @Override
51      protected ProcessorIdentifier queryProcessorId() {
52          String cpuVendor = "";
53          String cpuName = "";
54          String cpuFamily = "";
55          String cpuModel = "";
56          String cpuStepping = "";
57          long cpuFreq = 0L;
58  
59          // Get first result
60          try (KstatChain kc = KstatUtil.openChain()) {
61              Kstat ksp = KstatChain.lookup(CPU_INFO, -1, null);
62              // Set values
63              if (ksp != null && KstatChain.read(ksp)) {
64                  cpuVendor = KstatUtil.dataLookupString(ksp, "vendor_id");
65                  cpuName = KstatUtil.dataLookupString(ksp, "brand");
66                  cpuFamily = KstatUtil.dataLookupString(ksp, "family");
67                  cpuModel = KstatUtil.dataLookupString(ksp, "model");
68                  cpuStepping = KstatUtil.dataLookupString(ksp, "stepping");
69                  cpuFreq = KstatUtil.dataLookupLong(ksp, "clock_MHz") * 1_000_000L;
70              }
71          }
72          boolean cpu64bit = "64".equals(ExecutingCommand.getFirstAnswer("isainfo -b").trim());
73          String processorID = getProcessorID(cpuStepping, cpuModel, cpuFamily);
74  
75          return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
76                  cpuFreq);
77      }

79      @Override
80      protected List<LogicalProcessor> initProcessorCounts() {
81          Map<Integer, Integer> numaNodeMap = mapNumaNodes();
82          List<LogicalProcessor> logProcs = new ArrayList<>();
83          try (KstatChain kc = KstatUtil.openChain()) {
84              List<Kstat> kstats = KstatChain.lookupAll(CPU_INFO, -1, null);
85  
86              for (Kstat ksp : kstats) {
87                  if (ksp != null && KstatChain.read(ksp)) {
88                      int procId = logProcs.size(); // 0-indexed
89                      String chipId = KstatUtil.dataLookupString(ksp, "chip_id");
90                      String coreId = KstatUtil.dataLookupString(ksp, "core_id");
91                      LogicalProcessor logProc = new LogicalProcessor(procId, ParseUtil.parseIntOrDefault(coreId, 0),
92                              ParseUtil.parseIntOrDefault(chipId, 0), numaNodeMap.getOrDefault(procId, 0));
93                      logProcs.add(logProc);
94                  }
95              }
96          }
97          if (logProcs.isEmpty()) {
98              logProcs.add(new LogicalProcessor(0, 0, 0));
99          }
100         return logProcs;
101     }

103     private static Map<Integer, Integer> mapNumaNodes() {
104         // Get numa node info from lgrpinfo
105         Map<Integer, Integer> numaNodeMap = new HashMap<>();
106         int lgroup = 0;
107         for (String line : ExecutingCommand.runNative("lgrpinfo -c leaves")) {
108             // Format:
109             // lgroup 0 (root):
110             // CPUs: 0 1
111             // CPUs: 0-7
112             // CPUs: 0-3 6 7 12 13
113             // CPU: 0
114             // CPU: 1
115             if (line.startsWith("lgroup")) {
116                 lgroup = ParseUtil.getFirstIntValue(line);
117             } else if (line.contains("CPUs:") || line.contains("CPU:")) {
118                 for (Integer cpu : ParseUtil.parseHyphenatedIntList(line.split(":")[1])) {
119                     numaNodeMap.put(cpu, lgroup);
120                 }
121             }
122         }
123         return numaNodeMap;
124     }

126     @Override
127     public long[] querySystemCpuLoadTicks() {
128         long[] ticks = new long[TickType.values().length];
129         // Average processor ticks
130         long[][] procTicks = getProcessorCpuLoadTicks();
131         for (int i = 0; i < ticks.length; i++) {
132             for (long[] procTick : procTicks) {
133                 ticks[i] += procTick[i];
134             }
135             ticks[i] /= procTicks.length;
136         }
137         return ticks;
138     }

140     @Override
141     public long[] queryCurrentFreq() {
142         long[] freqs = new long[getLogicalProcessorCount()];
143         Arrays.fill(freqs, -1);
144         try (KstatChain kc = KstatUtil.openChain()) {
145             for (int i = 0; i < freqs.length; i++) {
146                 for (Kstat ksp : KstatChain.lookupAll(CPU_INFO, i, null)) {
147                     if (KstatChain.read(ksp)) {
148                         freqs[i] = KstatUtil.dataLookupLong(ksp, "current_clock_Hz");
149                     }
150                 }
151             }
152         }
153         return freqs;
154     }

156     @Override
157     public long queryMaxFreq() {
158         long max = -1L;
159         try (KstatChain kc = KstatUtil.openChain()) {
160             for (Kstat ksp : KstatChain.lookupAll(CPU_INFO, 0, null)) {
161                 if (KstatChain.read(ksp)) {
162                     String suppFreq = KstatUtil.dataLookupString(ksp, "supported_frequencies_Hz");
163                     if (!suppFreq.isEmpty()) {
164                         for (String s : suppFreq.split(":")) {
165                             long freq = ParseUtil.parseLongOrDefault(s, -1L);
166                             if (max < freq) {
167                                 max = freq;
168                             }
169                         }
170                     }
171                 }
172             }
173         }
174         return max;
175     }

177     @Override
178     public double[] getSystemLoadAverage(int nelem) {
179         if (nelem < 1 || nelem > 3) {
180             throw new IllegalArgumentException("Must include from one to three elements.");
181         }
182         double[] average = new double[nelem];
183         int retval = SolarisLibc.INSTANCE.getloadavg(average, nelem);
184         if (retval < nelem) {
185             for (int i = Math.max(retval, 0); i < average.length; i++) {
186                 average[i] = -1d;
187             }
188         }
189         return average;
190     }

192     @Override
193     public long[][] queryProcessorCpuLoadTicks() {
194         long[][] ticks = new long[getLogicalProcessorCount()][TickType.values().length];
195         int cpu = -1;
196         try (KstatChain kc = KstatUtil.openChain()) {
197             for (Kstat ksp : KstatChain.lookupAll("cpu", -1, "sys")) {
198                 // This is a new CPU
199                 if (++cpu >= ticks.length) {
200                     // Shouldn't happen
201                     break;
202                 }
203                 if (KstatChain.read(ksp)) {
204                     ticks[cpu][TickType.IDLE.getIndex()] = KstatUtil.dataLookupLong(ksp, "cpu_ticks_idle");
205                     ticks[cpu][TickType.SYSTEM.getIndex()] = KstatUtil.dataLookupLong(ksp, "cpu_ticks_kernel");
206                     ticks[cpu][TickType.USER.getIndex()] = KstatUtil.dataLookupLong(ksp, "cpu_ticks_user");
207                 }
208             }
209         }
210         return ticks;
211     }

213     /**
214      * Fetches the ProcessorID by encoding the stepping, model, family, and feature
215      * flags.
216      *
217      * @param stepping
218      *            The stepping
219      * @param model
220      *            The model
221      * @param family
222      *            The family
223      * @return The Processor ID string
224      */
225     private static String getProcessorID(String stepping, String model, String family) {
226         List<String> isainfo = ExecutingCommand.runNative("isainfo -v");
227         StringBuilder flags = new StringBuilder();
228         for (String line : isainfo) {
229             if (line.startsWith("32-bit")) {
230                 break;
231             } else if (!line.startsWith("64-bit")) {
232                 flags.append(' ').append(line.trim());
233             }
234         }
235         return createProcessorID(stepping, model, family, ParseUtil.whitespaces.split(flags.toString().toLowerCase()));
236     }

238     @Override
239     public long queryContextSwitches() {
240         long swtch = 0;
241         List<String> kstat = ExecutingCommand.runNative("kstat -p cpu_stat:::/pswitch\\\\|inv_swtch/");
242         for (String s : kstat) {
243             swtch += ParseUtil.parseLastLong(s, 0L);
244         }
245         return swtch;
246     }

248     @Override
249     public long queryInterrupts() {
250         long intr = 0;
251         List<String> kstat = ExecutingCommand.runNative("kstat -p cpu_stat:::/intr/");
252         for (String s : kstat) {
253             intr += ParseUtil.parseLastLong(s, 0L);
254         }
255         return intr;
256     }
