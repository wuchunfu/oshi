24  package oshi.hardware.platform.unix.solaris;
25  
26  import java.net.NetworkInterface;
27  import java.util.ArrayList;
28  import java.util.List;
29  
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  
33  import com.sun.jna.platform.unix.solaris.LibKstat.Kstat; // NOSONAR
34  
35  import oshi.annotation.concurrent.ThreadSafe;
36  import oshi.hardware.NetworkIF;
37  import oshi.hardware.common.AbstractNetworkIF;
38  import oshi.util.platform.unix.solaris.KstatUtil;
39  import oshi.util.platform.unix.solaris.KstatUtil.KstatChain;
40  
41  /**
42   * SolarisNetworks class.
43   */
44  @ThreadSafe
45  public final class SolarisNetworkIF extends AbstractNetworkIF {
46  
47      private static final Logger LOG = LoggerFactory.getLogger(SolarisNetworkIF.class);
48  
49      private long bytesRecv;
50      private long bytesSent;
51      private long packetsRecv;
52      private long packetsSent;
53      private long inErrors;
54      private long outErrors;
55      private long inDrops;
56      private long collisions;
57      private long speed;
58      private long timeStamp;
59  
60      public SolarisNetworkIF(NetworkInterface netint) throws InstantiationException {
61          super(netint);
62          updateAttributes();
63      }
64  
65      /**
66       * Gets all network interfaces on this machine
67       *
68       * @param includeLocalInterfaces
69       *            include local interfaces in the result
70       * @return A list of {@link NetworkIF} objects representing the interfaces
71       */
72      public static List<NetworkIF> getNetworks(boolean includeLocalInterfaces) {
73          List<NetworkIF> ifList = new ArrayList<>();
74          for (NetworkInterface ni : getNetworkInterfaces(includeLocalInterfaces)) {
75              try {
76                  ifList.add(new SolarisNetworkIF(ni));
77              } catch (InstantiationException e) {
78                  LOG.debug("Network Interface Instantiation failed: {}", e.getMessage());
79              }
80          }
81          return ifList;
82      }
83  
84      @Override
85      public long getBytesRecv() {
86          return this.bytesRecv;
87      }
88  
89      @Override
90      public long getBytesSent() {
91          return this.bytesSent;
92      }
93  
94      @Override
95      public long getPacketsRecv() {
96          return this.packetsRecv;
97      }
98  
99      @Override
100     public long getPacketsSent() {
101         return this.packetsSent;
102     }
103 
104     @Override
105     public long getInErrors() {
106         return this.inErrors;
107     }
108 
109     @Override
110     public long getOutErrors() {
111         return this.outErrors;
112     }
113 
114     @Override
115     public long getInDrops() {
116         return this.inDrops;
117     }
118 
119     @Override
120     public long getCollisions() {
121         return this.collisions;
122     }
123 
124     @Override
125     public long getSpeed() {
126         return this.speed;
127     }
128 
129     @Override
130     public long getTimeStamp() {
131         return this.timeStamp;
132     }
133 
134     @Override
135     public boolean updateAttributes() {
136         try (KstatChain kc = KstatUtil.openChain()) {
137             Kstat ksp = KstatChain.lookup("link", -1, getName());
138             if (ksp == null) { // Solaris 10 compatibility
139                 ksp = KstatChain.lookup(null, -1, getName());
140             }
141             if (ksp != null && KstatChain.read(ksp)) {
142                 this.bytesSent = KstatUtil.dataLookupLong(ksp, "obytes64");
143                 this.bytesRecv = KstatUtil.dataLookupLong(ksp, "rbytes64");
144                 this.packetsSent = KstatUtil.dataLookupLong(ksp, "opackets64");
145                 this.packetsRecv = KstatUtil.dataLookupLong(ksp, "ipackets64");
146                 this.outErrors = KstatUtil.dataLookupLong(ksp, "oerrors");
147                 this.inErrors = KstatUtil.dataLookupLong(ksp, "ierrors");
148                 this.collisions = KstatUtil.dataLookupLong(ksp, "collisions");
149                 this.inDrops = KstatUtil.dataLookupLong(ksp, "dl_idrops");
150                 this.speed = KstatUtil.dataLookupLong(ksp, "ifspeed");
151                 // Snap time in ns; convert to ms
152                 this.timeStamp = ksp.ks_snaptime / 1_000_000L;
153                 return true;
154             }
155         }
156         return false;
157     }
158 }
