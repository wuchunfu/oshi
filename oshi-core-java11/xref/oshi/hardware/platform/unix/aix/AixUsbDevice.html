26  import java.util.ArrayList;
27  import java.util.Arrays;
28  import java.util.Collections;
29  import java.util.List;
30  import java.util.function.Supplier;
31  
32  import oshi.annotation.concurrent.Immutable;
33  import oshi.hardware.UsbDevice;
34  import oshi.hardware.common.AbstractUsbDevice;
35  import oshi.util.Constants;
36  import oshi.util.ParseUtil;
37  
38  /**
39   * AIX Usb Device
40   */
41  @Immutable
42  public class AixUsbDevice extends AbstractUsbDevice {
43  
44      public AixUsbDevice(String name, String vendor, String vendorId, String productId, String serialNumber,
45              String uniqueDeviceId, List<UsbDevice> connectedDevices) {
46          super(name, vendor, vendorId, productId, serialNumber, uniqueDeviceId, connectedDevices);
47      }
48  
49      /**
50       * Instantiates a list of {@link oshi.hardware.UsbDevice} objects, representing
51       * devices connected via a usb port (including internal devices).
52       * <p>
53       * If the value of {@code tree} is true, the top level devices returned from
54       * this method are the USB Controllers; connected hubs and devices in its device
55       * tree share that controller's bandwidth. If the value of {@code tree} is
56       * false, USB devices (not controllers) are listed in a single flat list.
57       *
58       * @param tree
59       *            If true, returns a list of controllers, which requires recursive
60       *            iteration of connected devices. If false, returns a flat list of
61       *            devices excluding controllers.
62       * @param lscfg
63       *            A memoized lscfg list
64       * @return a list of {@link oshi.hardware.UsbDevice} objects.
65       */
66      public static List<UsbDevice> getUsbDevices(boolean tree, Supplier<List<String>> lscfg) {
67          List<UsbDevice> deviceList = new ArrayList<>();
68          for (String line : lscfg.get()) {
69              String s = line.trim();
70              if (s.startsWith("usb")) {
71                  String[] split = ParseUtil.whitespaces.split(s, 3);
72                  if (split.length == 3) {
73                      deviceList.add(new AixUsbDevice(split[2], Constants.UNKNOWN, Constants.UNKNOWN, Constants.UNKNOWN,
74                              Constants.UNKNOWN, split[0], Collections.emptyList()));
75                  }
76              }
77          }
78          if (tree) {
79              return Arrays.asList(new AixUsbDevice("USB Controller", "", "0000", "0000", "", "", deviceList));
80          }
81          return deviceList;
82      }
83  }
