40  /**
41   * A Power Source
42   */
43  @ThreadSafe
44  public final class LinuxPowerSource extends AbstractPowerSource {
45  
46      private static final String PS_PATH = "/sys/class/power_supply/";
47  
48      public LinuxPowerSource(String psName, String psDeviceName, double psRemainingCapacityPercent,
49              double psTimeRemainingEstimated, double psTimeRemainingInstant, double psPowerUsageRate, double psVoltage,
50              double psAmperage, boolean psPowerOnLine, boolean psCharging, boolean psDischarging,
51              CapacityUnits psCapacityUnits, int psCurrentCapacity, int psMaxCapacity, int psDesignCapacity,
52              int psCycleCount, String psChemistry, LocalDate psManufactureDate, String psManufacturer,
53              String psSerialNumber, double psTemperature) {
54          super(psName, psDeviceName, psRemainingCapacityPercent, psTimeRemainingEstimated, psTimeRemainingInstant,
55                  psPowerUsageRate, psVoltage, psAmperage, psPowerOnLine, psCharging, psDischarging, psCapacityUnits,
56                  psCurrentCapacity, psMaxCapacity, psDesignCapacity, psCycleCount, psChemistry, psManufactureDate,
57                  psManufacturer, psSerialNumber, psTemperature);
58      }
59  
60      /**
61       * Gets Battery Information
62       *
63       * @return An array of PowerSource objects representing batteries, etc.
64       */
65      public static List<PowerSource> getPowerSources() {
66          String psName;
67          String psDeviceName;
68          double psRemainingCapacityPercent = -1d;
69          double psTimeRemainingEstimated = -1d; // -1 = unknown, -2 = unlimited
70          double psTimeRemainingInstant = -1d;
71          double psPowerUsageRate = 0d;
72          double psVoltage = -1d;
73          double psAmperage = 0d;
74          boolean psPowerOnLine = false;
75          boolean psCharging = false;
76          boolean psDischarging = false;
77          CapacityUnits psCapacityUnits = CapacityUnits.RELATIVE;
78          int psCurrentCapacity = -1;
79          int psMaxCapacity = -1;
80          int psDesignCapacity = -1;
81          int psCycleCount = -1;
82          String psChemistry;
83          LocalDate psManufactureDate = null;
84          String psManufacturer;
85          String psSerialNumber;
86          double psTemperature = 0d;
87  
88          // Get list of power source names
89          File f = new File(PS_PATH);
90          String[] psNames = f.list();
91          List<PowerSource> psList = new ArrayList<>();
92          // Empty directory will give null rather than empty array, so fix
93          if (psNames != null) {
94              // For each power source, output various info
95              for (String name : psNames) {
96                  // Skip if name is ADP* or AC* (AC power supply)
97                  if (!name.startsWith("ADP") && !name.startsWith("AC")) {
98                      // Skip if can't read uevent file
99                      List<String> psInfo;
100                     psInfo = FileUtil.readFile(PS_PATH + name + "/uevent", false);
101                     if (psInfo.isEmpty()) {
102                         continue;
103                     }
104                     Map<String, String> psMap = new HashMap<>();
105                     for (String line : psInfo) {
106                         String[] split = line.split("=");
107                         if (split.length > 1 && !split[1].isEmpty()) {
108                             psMap.put(split[0], split[1]);
109                         }
110                     }
111                     psName = psMap.getOrDefault("POWER_SUPPLY_NAME", name);
112                     String status = psMap.get("POWER_SUPPLY_STATUS");
113                     psCharging = "Charging".equals(status);
114                     psDischarging = "Discharging".equals(status);
115                     if (psMap.containsKey("POWER_SUPPLY_CAPACITY")) {
116                         psRemainingCapacityPercent = ParseUtil.parseIntOrDefault(psMap.get("POWER_SUPPLY_CAPACITY"),
117                                 -100) / 100d;
118                     }
119                     if (psMap.containsKey("POWER_SUPPLY_ENERGY_NOW")) {
120                         psCurrentCapacity = ParseUtil.parseIntOrDefault(psMap.get("POWER_SUPPLY_ENERGY_NOW"), -1);
121                     } else if (psMap.containsKey("POWER_SUPPLY_CHARGE_NOW")) {
122                         psCurrentCapacity = ParseUtil.parseIntOrDefault(psMap.get("POWER_SUPPLY_CHARGE_NOW"), -1);
123                     }
124                     if (psMap.containsKey("POWER_SUPPLY_ENERGY_FULL")) {
125                         psCurrentCapacity = ParseUtil.parseIntOrDefault(psMap.get("POWER_SUPPLY_ENERGY_FULL"), 1);
126                     } else if (psMap.containsKey("POWER_SUPPLY_CHARGE_FULL")) {
127                         psCurrentCapacity = ParseUtil.parseIntOrDefault(psMap.get("POWER_SUPPLY_CHARGE_FULL"), 1);
128                     }
129                     if (psMap.containsKey("POWER_SUPPLY_ENERGY_FULL_DESIGN")) {
130                         psMaxCapacity = ParseUtil.parseIntOrDefault(psMap.get("POWER_SUPPLY_ENERGY_FULL_DESIGN"), 1);
131                     } else if (psMap.containsKey("POWER_SUPPLY_CHARGE_FULL_DESIGN")) {
132                         psMaxCapacity = ParseUtil.parseIntOrDefault(psMap.get("POWER_SUPPLY_CHARGE_FULL_DESIGN"), 1);
133                     }
134                     if (psMap.containsKey("POWER_SUPPLY_VOLTAGE_NOW")) {
135                         psVoltage = ParseUtil.parseIntOrDefault(psMap.get("POWER_SUPPLY_VOLTAGE_NOW"), -1);
136                     }
137                     if (psMap.containsKey("POWER_SUPPLY_POWER_NOW")) {
138                         psPowerUsageRate = ParseUtil.parseIntOrDefault(psMap.get("POWER_SUPPLY_POWER_NOW"), -1);
139                     }
140                     if (psVoltage > 0) {
141                         psAmperage = psPowerUsageRate / psVoltage;
142                     }
143                     if (psMap.containsKey("POWER_SUPPLY_CYCLE_COUNT")) {
144                         psCycleCount = ParseUtil.parseIntOrDefault(psMap.get("POWER_SUPPLY_CYCLE_COUNT"), -1);
145                     }
146                     psChemistry = psMap.getOrDefault("POWER_SUPPLY_TECHNOLOGY", Constants.UNKNOWN);
147                     psDeviceName = psMap.getOrDefault("POWER_SUPPLY_MODEL_NAME", Constants.UNKNOWN);
148                     psManufacturer = psMap.getOrDefault("POWER_SUPPLY_MANUFACTURER", Constants.UNKNOWN);
149                     psSerialNumber = psMap.getOrDefault("POWER_SUPPLY_SERIAL_NUMBER", Constants.UNKNOWN);
150                     if (ParseUtil.parseIntOrDefault(psMap.get("POWER_SUPPLY_PRESENT"), 1) > 0) {
151                         psList.add(new LinuxPowerSource(psName, psDeviceName, psRemainingCapacityPercent,
152                                 psTimeRemainingEstimated, psTimeRemainingInstant, psPowerUsageRate, psVoltage,
153                                 psAmperage, psPowerOnLine, psCharging, psDischarging, psCapacityUnits,
154                                 psCurrentCapacity, psMaxCapacity, psDesignCapacity, psCycleCount, psChemistry,
155                                 psManufactureDate, psManufacturer, psSerialNumber, psTemperature));
156                     }
157                 }
158             }
159         }
160         return psList;
161     }
