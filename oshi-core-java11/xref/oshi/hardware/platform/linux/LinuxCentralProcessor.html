56      @Override
57      protected ProcessorIdentifier queryProcessorId() {
58          String cpuVendor = "";
59          String cpuName = "";
60          String cpuFamily = "";
61          String cpuModel = "";
62          String cpuStepping = "";
63          String processorID;
64          long cpuFreq = 0L;
65          boolean cpu64bit = false;
66  
67          StringBuilder armStepping = new StringBuilder(); // For ARM equivalent
68          String[] flags = new String[0];
69          List<String> cpuInfo = FileUtil.readFile(CPUINFO);
70          for (String line : cpuInfo) {
71              String[] splitLine = ParseUtil.whitespacesColonWhitespace.split(line);
72              if (splitLine.length < 2) {
73                  // special case
74                  if (line.startsWith("CPU architecture: ")) {
75                      cpuFamily = line.replace("CPU architecture: ", "").trim();
76                  }
77                  continue;
78              }
79              switch (splitLine[0]) {
80              case "vendor_id":
81              case "CPU implementer":
82                  cpuVendor = splitLine[1];
83                  break;
84              case "model name":
85              case "Processor": // for Orange Pi
86                  cpuName = splitLine[1];
87                  break;
88              case "flags":
89                  flags = splitLine[1].toLowerCase().split(" ");
90                  for (String flag : flags) {
91                      if ("lm".equals(flag)) {
92                          cpu64bit = true;
93                          break;
94                      }
95                  }
96                  break;
97              case "stepping":
98                  cpuStepping = splitLine[1];
99                  break;
100             case "CPU variant":
101                 if (!armStepping.toString().startsWith("r")) {
102                     armStepping.insert(0, "r" + splitLine[1]);
103                 }
104                 break;
105             case "CPU revision":
106                 if (!armStepping.toString().contains("p")) {
107                     armStepping.append('p').append(splitLine[1]);
108                 }
109                 break;
110             case "model":
111             case "CPU part":
112                 cpuModel = splitLine[1];
113                 break;
114             case "cpu family":
115                 cpuFamily = splitLine[1];
116                 break;
117             case "cpu MHz":
118                 cpuFreq = ParseUtil.parseHertz(splitLine[1]);
119                 break;
120             default:
121                 // Do nothing
122             }
123         }
124         if (cpuName.contains("Hz")) {
125             // if Name contains CPU vendor frequency, ignore cpuinfo and use it
126             cpuFreq = -1L;
127         } else {
128             // Try lshw and use it in preference to cpuinfo
129             long cpuCapacity = Lshw.queryCpuCapacity();
130             if (cpuCapacity > cpuFreq) {
131                 cpuFreq = cpuCapacity;
132             }
133         }
134         if (cpuStepping.isEmpty()) {
135             cpuStepping = armStepping.toString();
136         }
137         processorID = getProcessorID(cpuVendor, cpuStepping, cpuModel, cpuFamily, flags);
138         if (cpuVendor.startsWith("0x")) {
139             List<String> lscpu = ExecutingCommand.runNative("lscpu");
140             for (String line : lscpu) {
141                 if (line.startsWith("Architecture:")) {
142                     cpuVendor = line.replace("Architecture:", "").trim();
143                 }
144             }
145         }
146         return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit,
147                 cpuFreq);
148     }

150     @Override
151     protected List<LogicalProcessor> initProcessorCounts() {
152         Map<Integer, Integer> numaNodeMap = mapNumaNodes();
153         List<String> procCpu = FileUtil.readFile(CPUINFO);
154         List<LogicalProcessor> logProcs = new ArrayList<>();
155         int currentProcessor = 0;
156         int currentCore = 0;
157         int currentPackage = 0;
158         boolean first = true;
159         for (String cpu : procCpu) {
160             // Count logical processors
161             if (cpu.startsWith("processor")) {
162                 if (!first) {
163                     logProcs.add(new LogicalProcessor(currentProcessor, currentCore, currentPackage,
164                             numaNodeMap.getOrDefault(currentProcessor, 0)));
165                 } else {
166                     first = false;
167                 }
168                 currentProcessor = ParseUtil.parseLastInt(cpu, 0);
169             } else if (cpu.startsWith("core id") || cpu.startsWith("cpu number")) {
170                 // Count unique combinations of core id and physical id.
171                 currentCore = ParseUtil.parseLastInt(cpu, 0);
172             } else if (cpu.startsWith("physical id")) {
173                 currentPackage = ParseUtil.parseLastInt(cpu, 0);
174             }
175         }
176         logProcs.add(new LogicalProcessor(currentProcessor, currentCore, currentPackage,
177                 numaNodeMap.getOrDefault(currentProcessor, 0)));
178 
179         return logProcs;
180     }

182     private static Map<Integer, Integer> mapNumaNodes() {
183         Map<Integer, Integer> numaNodeMap = new HashMap<>();
184         // Get numa node info from lscpu
185         List<String> lscpu = ExecutingCommand.runNative("lscpu -p=cpu,node");
186         // Format:
187         // # comment lines starting with #
188         // # then comma-delimited cpu,node
189         // 0,0
190         // 1,0
191         for (String line : lscpu) {
192             if (line.startsWith("#")) {
193                 continue;
194             }
195             String[] split = line.split(",");
196             if (split.length == 2) {
197                 numaNodeMap.put(ParseUtil.parseIntOrDefault(split[0], 0), ParseUtil.parseIntOrDefault(split[1], 0));
198             }
199         }
200         return numaNodeMap;
201     }

203     @Override
204     public long[] querySystemCpuLoadTicks() {
205         // convert the Linux Jiffies to Milliseconds.
206         long[] ticks = CpuStat.getSystemCpuLoadTicks();
207         // In rare cases, /proc/stat reading fails. If so, try again.
208         if (LongStream.of(ticks).sum() == 0) {
209             ticks = CpuStat.getSystemCpuLoadTicks();
210         }
211         long hz = LinuxOperatingSystem.getHz();
212         for (int i = 0; i < ticks.length; i++) {
213             ticks[i] = ticks[i] * 1000L / hz;
214         }
215         return ticks;
216     }

218     @Override
219     public long[] queryCurrentFreq() {
220         String cpuFreqPath = GlobalConfig.get(CPUFREQ_PATH, "");
221         long[] freqs = new long[getLogicalProcessorCount()];
222         // Attempt to fill array from cpu-freq source
223         long max = 0L;
224         for (int i = 0; i < freqs.length; i++) {
225             freqs[i] = FileUtil.getLongFromFile(cpuFreqPath + "/cpu" + i + "/cpufreq/scaling_cur_freq");
226             if (freqs[i] == 0) {
227                 freqs[i] = FileUtil.getLongFromFile(cpuFreqPath + "/cpu" + i + "/cpufreq/cpuinfo_cur_freq");
228             }
229             if (max < freqs[i]) {
230                 max = freqs[i];
231             }
232         }
233         if (max > 0L) {
234             // If successful, array is filled with values in KHz.
235             for (int i = 0; i < freqs.length; i++) {
236                 freqs[i] *= 1000L;
237             }
238             return freqs;
239         }
240         // If unsuccessful, try from /proc/cpuinfo
241         Arrays.fill(freqs, -1);
242         List<String> cpuInfo = FileUtil.readFile(CPUINFO);
243         int proc = 0;
244         for (String s : cpuInfo) {
245             if (s.toLowerCase().contains("cpu mhz")) {
246                 freqs[proc] = Math.round(ParseUtil.parseLastDouble(s, 0d) * 1_000_000d);
247                 if (++proc >= freqs.length) {
248                     break;
249                 }
250             }
251         }
252         return freqs;
253     }

255     @Override
256     public long queryMaxFreq() {
257         String cpuFreqPath = GlobalConfig.get(CPUFREQ_PATH, "");
258         long max = Arrays.stream(this.getCurrentFreq()).max().orElse(-1L);
259         // Max of current freq, if populated, is in units of Hz, convert to kHz
260         if (max > 0) {
261             max /= 1000L;
262         }
263         // Iterating CPUs only gets the existing policy, so we need to iterate the
264         // policy directories to find the system-wide policy max
265         File cpufreqdir = new File(cpuFreqPath + "/cpufreq");
266         File[] policies = cpufreqdir.listFiles();
267         if (policies != null) {
268             for (int i = 0; i < policies.length; i++) {
269                 File f = policies[i];
270                 if (f.getName().startsWith("policy")) {
271                     long freq = FileUtil.getLongFromFile(cpuFreqPath + "/cpufreq/" + f.getName() + "/scaling_max_freq");
272                     if (freq == 0) {
273                         freq = FileUtil.getLongFromFile(cpuFreqPath + "/cpufreq/" + f.getName() + "/cpuinfo_max_freq");
274                     }
275                     if (max < freq) {
276                         max = freq;
277                     }
278                 }
279             }
280         }
281         if (max > 0L) {
282             // If successful, value is in KHz.
283             max *= 1000L;
284             // Cpufreq result assumes intel pstates and is unreliable for AMD processors.
285             // Check lshw as a backup
286             long lshwMax = Lshw.queryCpuCapacity();
287             return lshwMax > max ? lshwMax : max;
288         }
289         return -1L;
290     }

292     @Override
293     public double[] getSystemLoadAverage(int nelem) {
294         if (nelem < 1 || nelem > 3) {
295             throw new IllegalArgumentException("Must include from one to three elements.");
296         }
297         double[] average = new double[nelem];
298         int retval = LinuxLibc.INSTANCE.getloadavg(average, nelem);
299         if (retval < nelem) {
300             for (int i = Math.max(retval, 0); i < average.length; i++) {
301                 average[i] = -1d;
302             }
303         }
304         return average;
305     }

307     @Override
308     public long[][] queryProcessorCpuLoadTicks() {
309         long[][] ticks = CpuStat.getProcessorCpuLoadTicks(getLogicalProcessorCount());
310         // In rare cases, /proc/stat reading fails. If so, try again.
311         // In theory we should check all of them, but on failure we can expect all 0's
312         // so we only need to check for processor 0
313         if (LongStream.of(ticks[0]).sum() == 0) {
314             ticks = CpuStat.getProcessorCpuLoadTicks(getLogicalProcessorCount());
315         }
316         // convert the Linux Jiffies to Milliseconds.
317         long hz = LinuxOperatingSystem.getHz();
318         for (int i = 0; i < ticks.length; i++) {
319             for (int j = 0; j < ticks[i].length; j++) {
320                 ticks[i][j] = ticks[i][j] * 1000L / hz;
321             }
322         }
323         return ticks;
324     }

326     /**
327      * Fetches the ProcessorID from dmidecode (if possible with root permissions),
328      * the cpuid command (if installed) or by encoding the stepping, model, family,
329      * and feature flags.
330      *
331      * @param vendor
332      *            The vendor
333      * @param stepping
334      *            The stepping
335      * @param model
336      *            The model
337      * @param family
338      *            The family
339      * @param flags
340      *            The flags
341      * @return The Processor ID string
342      */
343     private static String getProcessorID(String vendor, String stepping, String model, String family, String[] flags) {
344         boolean procInfo = false;
345         String marker = "Processor Information";
346         for (String checkLine : ExecutingCommand.runNative("dmidecode -t 4")) {
347             if (!procInfo && checkLine.contains(marker)) {
348                 marker = "ID:";
349                 procInfo = true;
350             } else if (procInfo && checkLine.contains(marker)) {
351                 return checkLine.split(marker)[1].trim();
352             }
353         }
354         // If we've gotten this far, dmidecode failed. Try cpuid.
355         marker = "eax=";
356         for (String checkLine : ExecutingCommand.runNative("cpuid -1r")) {
357             if (checkLine.contains(marker) && checkLine.trim().startsWith("0x00000001")) {
358                 String eax = "";
359                 String edx = "";
360                 for (String register : ParseUtil.whitespaces.split(checkLine)) {
361                     if (register.startsWith("eax=")) {
362                         eax = ParseUtil.removeMatchingString(register, "eax=0x");
363                     } else if (register.startsWith("edx=")) {
364                         edx = ParseUtil.removeMatchingString(register, "edx=0x");
365                     }
366                 }
367                 return edx + eax;
368             }
369         }
370         // If we've gotten this far, dmidecode failed. Encode arguments
371         if (vendor.startsWith("0x")) {
372             return createMIDR(vendor, stepping, model, family) + "00000000";
373         }
374         return createProcessorID(stepping, model, family, flags);
375     }

377     /**
378      * Creates the MIDR, the ARM equivalent of CPUID ProcessorID
379      *
380      * @param vendor
381      *            the CPU implementer
382      * @param stepping
383      *            the "rnpn" variant and revision
384      * @param model
385      *            the partnum
386      * @param family
387      *            the architecture
388      * @return A 32-bit hex string for the MIDR
389      */
390     private static String createMIDR(String vendor, String stepping, String model, String family) {
391         int midrBytes = 0;
392         // Build 32-bit MIDR
393         if (stepping.startsWith("r") && stepping.contains("p")) {
394             String[] rev = stepping.substring(1).split("p");
395             // 3:0 – Revision: last n in rnpn
396             midrBytes |= ParseUtil.parseLastInt(rev[1], 0);
397             // 23:20 - Variant: first n in rnpn
398             midrBytes |= ParseUtil.parseLastInt(rev[0], 0) << 20;
399         }
400         // 15:4 - PartNum = model
401         midrBytes |= ParseUtil.parseLastInt(model, 0) << 4;
402         // 19:16 - Architecture = family
403         midrBytes |= ParseUtil.parseLastInt(family, 0) << 16;
404         // 31:24 - Implementer = vendor
405         midrBytes |= ParseUtil.parseLastInt(vendor, 0) << 24;
406 
407         return String.format("%08X", midrBytes);
408     }
