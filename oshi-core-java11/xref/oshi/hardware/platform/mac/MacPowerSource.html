48  /**
49   * A Power Source
50   */
51  @ThreadSafe
52  public final class MacPowerSource extends AbstractPowerSource {
53  
54      private static final CoreFoundation CF = CoreFoundation.INSTANCE;
55      private static final IOKit IO = IOKit.INSTANCE;
56  
57      public MacPowerSource(String psName, String psDeviceName, double psRemainingCapacityPercent,
58              double psTimeRemainingEstimated, double psTimeRemainingInstant, double psPowerUsageRate, double psVoltage,
59              double psAmperage, boolean psPowerOnLine, boolean psCharging, boolean psDischarging,
60              CapacityUnits psCapacityUnits, int psCurrentCapacity, int psMaxCapacity, int psDesignCapacity,
61              int psCycleCount, String psChemistry, LocalDate psManufactureDate, String psManufacturer,
62              String psSerialNumber, double psTemperature) {
63          super(psName, psDeviceName, psRemainingCapacityPercent, psTimeRemainingEstimated, psTimeRemainingInstant,
64                  psPowerUsageRate, psVoltage, psAmperage, psPowerOnLine, psCharging, psDischarging, psCapacityUnits,
65                  psCurrentCapacity, psMaxCapacity, psDesignCapacity, psCycleCount, psChemistry, psManufactureDate,
66                  psManufacturer, psSerialNumber, psTemperature);
67      }
68  
69      /**
70       * Gets Battery Information.
71       *
72       * @return An array of PowerSource objects representing batteries, etc.
73       */
74      public static List<PowerSource> getPowerSources() {
75          String psDeviceName = Constants.UNKNOWN;
76          double psTimeRemainingInstant = 0d;
77          double psPowerUsageRate = 0d;
78          double psVoltage = -1d;
79          double psAmperage = 0d;
80          boolean psPowerOnLine = false;
81          boolean psCharging = false;
82          boolean psDischarging = false;
83          CapacityUnits psCapacityUnits = CapacityUnits.RELATIVE;
84          int psCurrentCapacity = 0;
85          int psMaxCapacity = 1;
86          int psDesignCapacity = 1;
87          int psCycleCount = -1;
88          String psChemistry = Constants.UNKNOWN;
89          LocalDate psManufactureDate = null;
90          String psManufacturer = Constants.UNKNOWN;
91          String psSerialNumber = Constants.UNKNOWN;
92          double psTemperature = 0d;
93  
94          // Mac PowerSource information comes from two sources: the IOKit's IOPS
95          // functions (which, in theory, return an array of objects but in most cases
96          // should return one), and the IORegistry's entry for AppleSmartBattery, which
97          // always returns one object.
98          //
99          // We start by fetching the registry information, which will be replicated
100         // across all IOPS entries if there are more than one.
101 
102         IORegistryEntry smartBattery = IOKitUtil.getMatchingService("AppleSmartBattery");
103         if (smartBattery != null) {
104             String s = smartBattery.getStringProperty("DeviceName");
105             if (s != null) {
106                 psDeviceName = s;
107             }
108             s = smartBattery.getStringProperty("Manufacturer");
109             if (s != null) {
110                 psManufacturer = s;
111             }
112             s = smartBattery.getStringProperty("BatterySerialNumber");
113             if (s != null) {
114                 psSerialNumber = s;
115             }
116 
117             Integer temp = smartBattery.getIntegerProperty("ManufactureDate");
118             if (temp != null) {
119                 // Bits 0...4 => day (value 1-31; 5 bits)
120                 // Bits 5...8 => month (value 1-12; 4 bits)
121                 // Bits 9...15 => years since 1980 (value 0-127; 7 bits)
122                 int day = temp & 0x1f;
123                 int month = (temp >> 5) & 0xf;
124                 int year80 = (temp >> 9) & 0x7f;
125                 psManufactureDate = LocalDate.of(1980 + year80, month, day);
126             }
127 
128             temp = smartBattery.getIntegerProperty("DesignCapacity");
129             if (temp != null) {
130                 psDesignCapacity = temp;
131             }
132             temp = smartBattery.getIntegerProperty("MaxCapacity");
133             if (temp != null) {
134                 psMaxCapacity = temp;
135             }
136             temp = smartBattery.getIntegerProperty("CurrentCapacity");
137             if (temp != null) {
138                 psCurrentCapacity = temp;
139             }
140             psCapacityUnits = CapacityUnits.MAH;
141 
142             temp = smartBattery.getIntegerProperty("TimeRemaining");
143             if (temp != null) {
144                 psTimeRemainingInstant = temp * 60d;
145             }
146             temp = smartBattery.getIntegerProperty("CycleCount");
147             if (temp != null) {
148                 psCycleCount = temp;
149             }
150             temp = smartBattery.getIntegerProperty("Temperature");
151             if (temp != null) {
152                 psTemperature = temp / 100d;
153             }
154             temp = smartBattery.getIntegerProperty("Voltage");
155             if (temp != null) {
156                 psVoltage = temp / 1000d;
157             }
158             temp = smartBattery.getIntegerProperty("Amperage");
159             if (temp != null) {
160                 psAmperage = temp;
161             }
162             psPowerUsageRate = psVoltage * psAmperage;
163 
164             Boolean bool = smartBattery.getBooleanProperty("ExternalConnected");
165             if (bool != null) {
166                 psPowerOnLine = bool;
167             }
168             bool = smartBattery.getBooleanProperty("IsCharging");
169             if (bool != null) {
170                 psCharging = bool;
171             }
172             psDischarging = !psCharging;
173 
174             smartBattery.release();
175         }
176 
177         // Get the blob containing current power source state
178         CFTypeRef powerSourcesInfo = IO.IOPSCopyPowerSourcesInfo();
179         CFArrayRef powerSourcesList = IO.IOPSCopyPowerSourcesList(powerSourcesInfo);
180         int powerSourcesCount = powerSourcesList.getCount();
181 
182         // Get time remaining
183         // -1 = unknown, -2 = unlimited
184         double psTimeRemainingEstimated = IO.IOPSGetTimeRemainingEstimate();
185 
186         CFStringRef nameKey = CFStringRef.createCFString("Name");
187         CFStringRef isPresentKey = CFStringRef.createCFString("Is Present");
188         CFStringRef currentCapacityKey = CFStringRef.createCFString("Current Capacity");
189         CFStringRef maxCapacityKey = CFStringRef.createCFString("Max Capacity");
190         // For each power source, output various info
191         List<PowerSource> psList = new ArrayList<>(powerSourcesCount);
192         for (int ps = 0; ps < powerSourcesCount; ps++) {
193             // Get the dictionary for that Power Source
194             Pointer pwrSrcPtr = powerSourcesList.getValueAtIndex(ps);
195             CFTypeRef powerSource = new CFTypeRef();
196             powerSource.setPointer(pwrSrcPtr);
197             CFDictionaryRef dictionary = IO.IOPSGetPowerSourceDescription(powerSourcesInfo, powerSource);
198 
199             // Get values from dictionary (See IOPSKeys.h)
200             // Skip if not present
201             Pointer result = dictionary.getValue(isPresentKey);
202             if (result != null) {
203                 CFBooleanRef isPresentRef = new CFBooleanRef(result);
204                 if (0 != CF.CFBooleanGetValue(isPresentRef)) {
205                     // Get name
206                     result = dictionary.getValue(nameKey);
207                     String psName = CFUtil.cfPointerToString(result);
208                     // Remaining Capacity = current / max
209                     double currentCapacity = 0d;
210                     if (dictionary.getValueIfPresent(currentCapacityKey, null)) {
211                         result = dictionary.getValue(currentCapacityKey);
212                         CFNumberRef cap = new CFNumberRef(result);
213                         currentCapacity = cap.intValue();
214                     }
215                     double maxCapacity = 1d;
216                     if (dictionary.getValueIfPresent(maxCapacityKey, null)) {
217                         result = dictionary.getValue(maxCapacityKey);
218                         CFNumberRef cap = new CFNumberRef(result);
219                         maxCapacity = cap.intValue();
220                     }
221                     double psRemainingCapacityPercent = Math.min(1d, currentCapacity / maxCapacity);
222                     // Add to list
223                     psList.add(new MacPowerSource(psName, psDeviceName, psRemainingCapacityPercent,
224                             psTimeRemainingEstimated, psTimeRemainingInstant, psPowerUsageRate, psVoltage, psAmperage,
225                             psPowerOnLine, psCharging, psDischarging, psCapacityUnits, psCurrentCapacity, psMaxCapacity,
226                             psDesignCapacity, psCycleCount, psChemistry, psManufactureDate, psManufacturer,
227                             psSerialNumber, psTemperature));
228                 }
229             }
230         }
231         isPresentKey.release();
232         nameKey.release();
233         currentCapacityKey.release();
234         maxCapacityKey.release();
235         // Release the blob
236         powerSourcesList.release();
237         powerSourcesInfo.release();
238 
239         return psList;
240     }
