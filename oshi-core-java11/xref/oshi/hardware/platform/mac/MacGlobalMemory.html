51  /**
52   * Memory obtained by host_statistics (vm_stat) and sysctl.
53   */
54  @ThreadSafe
55  final class MacGlobalMemory extends AbstractGlobalMemory {
56  
57      private static final Logger LOG = LoggerFactory.getLogger(MacGlobalMemory.class);
58  
59      private final Supplier<Long> available = memoize(this::queryVmStats, defaultExpiration());
60  
61      private final Supplier<Long> total = memoize(MacGlobalMemory::queryPhysMem);
62  
63      private final Supplier<Long> pageSize = memoize(MacGlobalMemory::queryPageSize);
64  
65      private final Supplier<VirtualMemory> vm = memoize(this::createVirtualMemory);
66  
67      @Override
68      public long getAvailable() {
69          return available.get();
70      }
71  
72      @Override
73      public long getTotal() {
74          return total.get();
75      }
76  
77      @Override
78      public long getPageSize() {
79          return pageSize.get();
80      }
81  
82      @Override
83      public VirtualMemory getVirtualMemory() {
84          return vm.get();
85      }
86  
87      @Override
88      public List<PhysicalMemory> getPhysicalMemory() {
89          List<PhysicalMemory> pmList = new ArrayList<>();
90          List<String> sp = ExecutingCommand.runNative("system_profiler SPMemoryDataType");
91          int bank = 0;
92          String bankLabel = Constants.UNKNOWN;
93          long capacity = 0L;
94          long speed = 0L;
95          String manufacturer = Constants.UNKNOWN;
96          String memoryType = Constants.UNKNOWN;
97          for (String line : sp) {
98              if (line.trim().startsWith("BANK")) {
99                  // Save previous bank
100                 if (bank++ > 0) {
101                     pmList.add(new PhysicalMemory(bankLabel, capacity, speed, manufacturer, memoryType));
102                 }
103                 bankLabel = line.trim();
104                 int colon = bankLabel.lastIndexOf(':');
105                 if (colon > 0) {
106                     bankLabel = bankLabel.substring(0, colon - 1);
107                 }
108             } else if (bank > 0) {
109                 String[] split = line.trim().split(":");
110                 if (split.length == 2) {
111                     switch (split[0]) {
112                     case "Size":
113                         capacity = ParseUtil.parseDecimalMemorySizeToBinary(split[1].trim());
114                         break;
115                     case "Type":
116                         memoryType = split[1].trim();
117                         break;
118                     case "Speed":
119                         speed = ParseUtil.parseHertz(split[1]);
120                         break;
121                     case "Manufacturer":
122                         manufacturer = split[1].trim();
123                         break;
124                     default:
125                         break;
126                     }
127                 }
128             }
129         }
130         pmList.add(new PhysicalMemory(bankLabel, capacity, speed, manufacturer, memoryType));
131 
132         return pmList;
133     }
134 
135     private long queryVmStats() {
136         VMStatistics vmStats = new VMStatistics();
137         if (0 != SystemB.INSTANCE.host_statistics(SystemB.INSTANCE.mach_host_self(), SystemB.HOST_VM_INFO, vmStats,
138                 new IntByReference(vmStats.size() / SystemB.INT_SIZE))) {
139             LOG.error("Failed to get host VM info. Error code: {}", Native.getLastError());
140             return 0L;
141         }
142         return (vmStats.free_count + vmStats.inactive_count) * getPageSize();
143     }
144 
145     private static long queryPhysMem() {
146         return SysctlUtil.sysctl("hw.memsize", 0L);
147     }
148 
149     private static long queryPageSize() {
150         LongByReference pPageSize = new LongByReference();
151         if (0 == SystemB.INSTANCE.host_page_size(SystemB.INSTANCE.mach_host_self(), pPageSize)) {
152             return pPageSize.getValue();
153         }
154         LOG.error("Failed to get host page size. Error code: {}", Native.getLastError());
155         return 4098L;
156     }
157 
158     private VirtualMemory createVirtualMemory() {
159         return new MacVirtualMemory(this);
160     }
