76      @Override
77      public TcpStats getTCPv4Stats() {
78          MIB_TCPSTATS stats = new MIB_TCPSTATS();
79          IPHLP.GetTcpStatisticsEx(stats, AF_INET);
80          return new TcpStats(stats.dwCurrEstab, stats.dwActiveOpens, stats.dwPassiveOpens, stats.dwAttemptFails,
81                  stats.dwEstabResets, stats.dwOutSegs, stats.dwInSegs, stats.dwRetransSegs, stats.dwInErrs,
82                  stats.dwOutRsts);
83      }

108     @Override
109     public List<IPConnection> getConnections() {
110         if (IS_VISTA_OR_GREATER) {
111             List<IPConnection> conns = new ArrayList<>();
112             conns.addAll(queryTCPv4Connections());
113             conns.addAll(queryTCPv6Connections());
114             conns.addAll(queryUDPv4Connections());
115             conns.addAll(queryUDPv6Connections());
116             return conns;
117         }
118         return Collections.emptyList();
119     }
120 
121     private static List<IPConnection> queryTCPv4Connections() {
122         List<IPConnection> conns = new ArrayList<>();
123         // Get size needed
124         IntByReference sizePtr = new IntByReference();
125         IPHLP.GetExtendedTcpTable(null, sizePtr, false, AF_INET, TCP_TABLE_OWNER_PID_ALL, 0);
126         // Get buffer and populate table
127         int size;
128         Memory buf;
129         do {
130             size = sizePtr.getValue();
131             buf = new Memory(size);
132             IPHLP.GetExtendedTcpTable(buf, sizePtr, false, AF_INET, TCP_TABLE_OWNER_PID_ALL, 0);
133             // In case size changes and buffer was too small, repeat
134         } while (size < sizePtr.getValue());
135         MIB_TCPTABLE_OWNER_PID tcpTable = new MIB_TCPTABLE_OWNER_PID(buf);
136         for (int i = 0; i < tcpTable.dwNumEntries; i++) {
137             MIB_TCPROW_OWNER_PID row = tcpTable.table[i];
138             conns.add(new IPConnection("tcp4", ParseUtil.parseIntToIP(row.dwLocalAddr),
139                     ParseUtil.bigEndian16ToLittleEndian(row.dwLocalPort), ParseUtil.parseIntToIP(row.dwRemoteAddr),
140                     ParseUtil.bigEndian16ToLittleEndian(row.dwRemotePort), stateLookup(row.dwState), 0, 0,
141                     row.dwOwningPid));
142         }
143         return conns;
144     }

217     private static TcpState stateLookup(int state) {
218         switch (state) {
219         case 1:
220         case 12:
221             return CLOSED;
222         case 2:
223             return LISTEN;
224         case 3:
225             return SYN_SENT;
226         case 4:
227             return SYN_RECV;
228         case 5:
229             return ESTABLISHED;
230         case 6:
231             return FIN_WAIT_1;
232         case 7:
233             return FIN_WAIT_2;
234         case 8:
235             return CLOSE_WAIT;
236         case 9:
237             return CLOSING;
238         case 10:
239             return LAST_ACK;
240         case 11:
241             return TIME_WAIT;
242         default:
243             return UNKNOWN;
244         }
