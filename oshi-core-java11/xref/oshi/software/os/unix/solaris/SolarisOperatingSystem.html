86      @Override
87      public Pair<String, OSVersionInfo> queryFamilyVersionInfo() {
88          String[] split = ParseUtil.whitespaces.split(ExecutingCommand.getFirstAnswer("uname -rv"));
89          String version = split[0];
90          String buildNumber = null;
91          if (split.length > 1) {
92              buildNumber = split[1];
93          }
94          return new Pair<>("SunOS", new OSVersionInfo(version, "Solaris", buildNumber));
95      }

120     @Override
121     public OSProcess getProcess(int pid) {
122         List<OSProcess> procs = getProcessListFromPS("ps -o " + PS_COMMAND_ARGS + " -p " + pid, pid);
123         if (procs.isEmpty()) {
124             return null;
125         }
126         return procs.get(0);

148     private static List<OSProcess> queryAllProcessesFromPS() {
149         return getProcessListFromPS("ps -eo " + PS_COMMAND_ARGS, -1);
150     }

152     private static List<OSProcess> getProcessListFromPS(String psCommand, int pid) {
153         List<OSProcess> procs = new ArrayList<>();
154         List<String> procList = ExecutingCommand.runNative(psCommand);
155         if (procList.size() > 1) {
156             // Get a map by pid of prstat output
157             List<String> prstatList = pid < 0 ? ExecutingCommand.runNative("prstat -v 1 1")
158                     : ExecutingCommand.runNative("prstat -v -p " + pid + " 1 1");
159             Map<String, String> prstatRowMap = new HashMap<>();
160             for (String s : prstatList) {
161                 String row = s.trim();
162                 int idx = row.indexOf(' ');
163                 if (idx > 0) {
164                     prstatRowMap.put(row.substring(0, idx), row);
165                 }
166             }
167             // remove header row and iterate proc list
168             procList.remove(0);
169             for (String proc : procList) {
170                 Map<PsKeywords, String> psMap = ParseUtil.stringToEnumMap(PsKeywords.class, proc.trim(), ' ');
171                 // Check if last (thus all) value populated
172                 if (psMap.containsKey(PsKeywords.ARGS)) {
173                     String pidStr = psMap.get(PsKeywords.PID);
174                     Map<PrstatKeywords, String> prstatMap = ParseUtil.stringToEnumMap(PrstatKeywords.class,
175                             prstatRowMap.getOrDefault(pidStr, ""), ' ');
176                     procs.add(new SolarisOSProcess(pid < 0 ? ParseUtil.parseIntOrDefault(pidStr, 0) : pid, psMap,
177                             prstatMap));
178                 }
179             }
180         }
181         return procs;
182     }

209     private static long querySystemUptime() {
210         try (KstatChain kc = KstatUtil.openChain()) {
211             Kstat ksp = KstatChain.lookup("unix", 0, "system_misc");
212             if (ksp != null) {
213                 // Snap Time is in nanoseconds; divide for seconds
214                 return ksp.ks_snaptime / 1_000_000_000L;
215             }
216         }
217         return 0L;
218     }

225     private static long querySystemBootTime() {
226         try (KstatChain kc = KstatUtil.openChain()) {
227             Kstat ksp = KstatChain.lookup("unix", 0, "system_misc");
228             if (ksp != null && KstatChain.read(ksp)) {
229                 return KstatUtil.dataLookupLong(ksp, "boot_time");
230             }
231         }
232         return System.currentTimeMillis() / 1000L - querySystemUptime();
233     }

240     @Override
241     public OSService[] getServices() {
242         List<OSService> services = new ArrayList<>();
243         // Get legacy RC service name possibilities
244         List<String> legacySvcs = new ArrayList<>();
245         File dir = new File("/etc/init.d");
246         File[] listFiles;
247         if (dir.exists() && dir.isDirectory() && (listFiles = dir.listFiles()) != null) {
248             for (File f : listFiles) {
249                 legacySvcs.add(f.getName());
250             }
251         }
252         // Iterate service list
253         List<String> svcs = ExecutingCommand.runNative("svcs -p");
254         /*-
255          Output:
256          STATE          STIME    FRMI
257          legacy_run     23:56:49 lrc:/etc/rc2_d/S47pppd
258          legacy_run     23:56:49 lrc:/etc/rc2_d/S81dodatadm_udaplt
259          legacy_run     23:56:49 lrc:/etc/rc2_d/S89PRESERVE
260          online         23:56:25 svc:/system/early-manifest-import:default
261          online         23:56:25 svc:/system/svc/restarter:default
262                         23:56:24       13 svc.startd
263                         ...
264          */
265         for (String line : svcs) {
266             if (line.startsWith("online")) {
267                 int delim = line.lastIndexOf(":/");
268                 if (delim > 0) {
269                     String name = line.substring(delim + 1);
270                     if (name.endsWith(":default")) {
271                         name = name.substring(0, name.length() - 8);
272                     }
273                     services.add(new OSService(name, 0, STOPPED));
274                 }
275             } else if (line.startsWith(" ")) {
276                 String[] split = ParseUtil.whitespaces.split(line.trim());
277                 if (split.length == 3) {
278                     services.add(new OSService(split[2], ParseUtil.parseIntOrDefault(split[1], 0), RUNNING));
279                 }
280             } else if (line.startsWith("legacy_run")) {
281                 for (String svc : legacySvcs) {
282                     if (line.endsWith(svc)) {
283                         services.add(new OSService(svc, 0, STOPPED));
284                         break;
285                     }
286                 }
287             }
288         }
289         return services.toArray(new OSService[0]);
290     }
