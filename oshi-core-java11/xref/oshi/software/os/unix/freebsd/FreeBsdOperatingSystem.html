57  /**
58   * FreeBSD is a free and open-source Unix-like operating system descended from
59   * the Berkeley Software Distribution (BSD), which was based on Research Unix.
60   * The first version of FreeBSD was released in 1993. In 2005, FreeBSD was the
61   * most popular open-source BSD operating system, accounting for more than
62   * three-quarters of all installed simply, permissively licensed BSD systems.
63   */

71      /*
72       * Package-private for use by FreeBsdOSProcess
73       */
74      enum PsKeywords {
75          STATE, PID, PPID, USER, UID, GROUP, GID, NLWP, PRI, VSZ, RSS, ETIMES, SYSTIME, TIME, COMM, MAJFLT, MINFLT,
76          NVCSW, NIVCSW, ARGS; // ARGS must always be last
77      }
78  
79      static final String PS_COMMAND_ARGS = Arrays.stream(PsKeywords.values()).map(Enum::name).map(String::toLowerCase)
80              .collect(Collectors.joining(","));

84          return "Unix/BSD";
85      }

87      @Override
88      public Pair<String, OSVersionInfo> queryFamilyVersionInfo() {
89          String family = BsdSysctlUtil.sysctl("kern.ostype", "FreeBSD");
90  
91          String version = BsdSysctlUtil.sysctl("kern.osrelease", "");
92          String versionInfo = BsdSysctlUtil.sysctl("kern.version", "");
93          String buildNumber = versionInfo.split(":")[0].replace(family, "").replace(version, "").trim();
94  
95          return new Pair<>(family, new OSVersionInfo(version, null, buildNumber));
96      }

98      @Override
99      protected int queryBitness(int jvmBitness) {
100         if (jvmBitness < 64 && ExecutingCommand.getFirstAnswer("uname -m").indexOf("64") == -1) {
101             return jvmBitness;
102         }
103         return 64;
104     }

126     @Override
127     public List<OSProcess> queryChildProcesses(int parentPid) {
128         List<OSProcess> allProcs = queryAllProcesses();
129         Set<Integer> descendantPids = getChildrenOrDescendants(allProcs, parentPid, false);

140     @Override
141     public OSProcess getProcess(int pid) {
142         List<OSProcess> procs = getProcessListFromPS(pid);
143         if (procs.isEmpty()) {
144             return null;
145         }
146         return procs.get(0);
147     }
148 
149     private static List<OSProcess> getProcessListFromPS(int pid) {
150         List<OSProcess> procs = new ArrayList<>();
151         String psCommand = "ps -awwxo " + PS_COMMAND_ARGS;
152         if (pid >= 0) {
153             psCommand += " -p " + pid;
154         }
155         List<String> procList = ExecutingCommand.runNative(psCommand);
156         if (procList.isEmpty() || procList.size() < 2) {
157             return procs;
158         }
159         // remove header row
160         procList.remove(0);
161         // Fill list
162         for (String proc : procList) {
163             Map<PsKeywords, String> psMap = ParseUtil.stringToEnumMap(PsKeywords.class, proc.trim(), ' ');
164             // Check if last (thus all) value populated
165             if (psMap.containsKey(PsKeywords.ARGS)) {
166                 procs.add(new FreeBsdOSProcess(
167                         pid < 0 ? ParseUtil.parseIntOrDefault(psMap.get(PsKeywords.PID), 0) : pid, psMap));
168             }
169         }
170         return procs;
171     }

178     @Override
179     public int getProcessCount() {
180         List<String> procList = ExecutingCommand.runNative("ps -axo pid");
181         if (!procList.isEmpty()) {
182             // Subtract 1 for header
183             return procList.size() - 1;
184         }
185         return 0;
186     }

188     @Override
189     public int getThreadCount() {
190         int threads = 0;
191         for (String proc : ExecutingCommand.runNative("ps -axo nlwp")) {
192             threads += ParseUtil.parseIntOrDefault(proc.trim(), 0);
193         }
194         return threads;
195     }

207     private static long querySystemBootTime() {
208         Timeval tv = new Timeval();
209         if (!BsdSysctlUtil.sysctl("kern.boottime", tv) || tv.tv_sec == 0) {
210             // Usually this works. If it doesn't, fall back to text parsing.
211             // Boot time will be the first consecutive string of digits.
212             return ParseUtil.parseLongOrDefault(
213                     ExecutingCommand.getFirstAnswer("sysctl -n kern.boottime").split(",")[0].replaceAll("\\D", ""),
214                     System.currentTimeMillis() / 1000);
215         }
216         // tv now points to a 128-bit timeval structure for boot time.
217         // First 8 bytes are seconds, second 8 bytes are microseconds (we ignore)
218         return tv.tv_sec;
219     }

226     @Override
227     public OSService[] getServices() {
228         // Get running services
229         List<OSService> services = new ArrayList<>();
230         Set<String> running = new HashSet<>();
231         for (OSProcess p : getChildProcesses(1, ProcessFiltering.ALL_PROCESSES, ProcessSorting.PID_ASC, 0)) {
232             OSService s = new OSService(p.getName(), p.getProcessID(), RUNNING);
233             services.add(s);
234             running.add(p.getName());
235         }
236         // Get Directories for stopped services
237         File dir = new File("/etc/rc.d");
238         File[] listFiles;
239         if (dir.exists() && dir.isDirectory() && (listFiles = dir.listFiles()) != null) {
240             for (File f : listFiles) {
241                 String name = f.getName();
242                 if (!running.contains(name)) {
243                     OSService s = new OSService(name, 0, STOPPED);
244                     services.add(s);
245                 }
246             }
247         } else {
248             LOG.error("Directory: /etc/init does not exist");
249         }
250         return services.toArray(new OSService[0]);
251     }
