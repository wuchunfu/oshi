24  package oshi.software.os.unix.freebsd;
25  
26  import static oshi.util.Memoizer.defaultExpiration;
27  import static oshi.util.Memoizer.memoize;
28  
29  import java.util.function.Supplier;
30  
31  import com.sun.jna.Memory; // NOSONAR squid:S1191
32  
33  import oshi.annotation.concurrent.ThreadSafe;
34  import oshi.driver.unix.NetStat;
35  import oshi.jna.platform.unix.CLibrary.BsdTcpstat;
36  import oshi.jna.platform.unix.CLibrary.BsdUdpstat;
37  import oshi.software.common.AbstractInternetProtocolStats;
38  import oshi.util.ParseUtil;
39  import oshi.util.platform.unix.freebsd.BsdSysctlUtil;
40  import oshi.util.tuples.Pair;
41  
42  /**
43   * Internet Protocol Stats implementation
44   */
45  @ThreadSafe
46  public class FreeBsdInternetProtocolStats extends AbstractInternetProtocolStats {
47  
48      private Supplier<Pair<Long, Long>> establishedv4v6 = memoize(NetStat::queryTcpnetstat, defaultExpiration());
49      private Supplier<BsdTcpstat> tcpstat = memoize(FreeBsdInternetProtocolStats::queryTcpstat, defaultExpiration());
50      private Supplier<BsdUdpstat> udpstat = memoize(FreeBsdInternetProtocolStats::queryUdpstat, defaultExpiration());
51  
52      @Override
53      public TcpStats getTCPv4Stats() {
54          BsdTcpstat tcp = tcpstat.get();
55          return new TcpStats(establishedv4v6.get().getA(), ParseUtil.unsignedIntToLong(tcp.tcps_connattempt),
56                  ParseUtil.unsignedIntToLong(tcp.tcps_accepts), ParseUtil.unsignedIntToLong(tcp.tcps_conndrops),
57                  ParseUtil.unsignedIntToLong(tcp.tcps_drops), ParseUtil.unsignedIntToLong(tcp.tcps_sndpack),
58                  ParseUtil.unsignedIntToLong(tcp.tcps_rcvpack), ParseUtil.unsignedIntToLong(tcp.tcps_sndrexmitpack),
59                  ParseUtil.unsignedIntToLong(
60                          tcp.tcps_rcvbadsum + tcp.tcps_rcvbadoff + tcp.tcps_rcvmemdrop + tcp.tcps_rcvshort),
61                  0L);
62      }
63  
64      @Override
65      public UdpStats getUDPv4Stats() {
66          BsdUdpstat stat = udpstat.get();
67          return new UdpStats(ParseUtil.unsignedIntToLong(stat.udps_opackets),
68                  ParseUtil.unsignedIntToLong(stat.udps_ipackets), ParseUtil.unsignedIntToLong(stat.udps_noportmcast),
69                  ParseUtil.unsignedIntToLong(stat.udps_hdrops + stat.udps_badsum + stat.udps_badlen));
70      }
71  
72      @Override
73      public UdpStats getUDPv6Stats() {
74          BsdUdpstat stat = udpstat.get();
75          return new UdpStats(ParseUtil.unsignedIntToLong(stat.udps_snd6_swcsum),
76                  ParseUtil.unsignedIntToLong(stat.udps_rcv6_swcsum), 0L, 0L);
77      }
78  
79      private static BsdTcpstat queryTcpstat() {
80          BsdTcpstat ft = new BsdTcpstat();
81          Memory m = BsdSysctlUtil.sysctl("net.inet.tcp.stats");
82          if (m != null && m.size() >= 128) {
83              ft.tcps_connattempt = m.getInt(0);
84              ft.tcps_accepts = m.getInt(4);
85              ft.tcps_drops = m.getInt(12);
86              ft.tcps_conndrops = m.getInt(16);
87              ft.tcps_sndpack = m.getInt(64);
88              ft.tcps_sndrexmitpack = m.getInt(72);
89              ft.tcps_rcvpack = m.getInt(104);
90              ft.tcps_rcvbadsum = m.getInt(112);
91              ft.tcps_rcvbadoff = m.getInt(116);
92              ft.tcps_rcvmemdrop = m.getInt(120);
93              ft.tcps_rcvshort = m.getInt(124);
94          }
95          return ft;
96      }
97  
98      private static BsdUdpstat queryUdpstat() {
99          BsdUdpstat ut = new BsdUdpstat();
100         Memory m = BsdSysctlUtil.sysctl("net.inet.udp.stats");
101         if (m != null && m.size() >= 1644) {
102             ut.udps_ipackets = m.getInt(0);
103             ut.udps_hdrops = m.getInt(4);
104             ut.udps_badsum = m.getInt(8);
105             ut.udps_badlen = m.getInt(12);
106             ut.udps_opackets = m.getInt(36);
107             ut.udps_noportmcast = m.getInt(48);
108             ut.udps_rcv6_swcsum = m.getInt(64);
109             ut.udps_snd6_swcsum = m.getInt(80);
110         }
111         return ut;
112     }
113 }
