61  /**
62   * The Mac File System contains {@link oshi.software.os.OSFileStore}s which are
63   * a storage pool, device, partition, volume, concrete file system or other
64   * implementation specific means of file storage. In macOS, these are found in
65   * the /Volumes directory.

138     @Override
139     public List<OSFileStore> getFileStores(boolean localOnly) {
140         // List of file systems
141         return getFileStoreMatching(null, localOnly);
142     }

144     // Called by MacOSFileStore
145     static List<OSFileStore> getFileStoreMatching(String nameToMatch) {
146         return getFileStoreMatching(nameToMatch, false);
147     }

149     private static List<OSFileStore> getFileStoreMatching(String nameToMatch, boolean localOnly) {
150         List<OSFileStore> fsList = new ArrayList<>();
151 
152         // Use getfsstat to find fileSystems
153         // Query with null to get total # required
154         int numfs = SystemB.INSTANCE.getfsstat64(null, 0, 0);
155         if (numfs > 0) {
156             // Open a DiskArbitration session to get VolumeName of file systems
157             // with bsd names
158             DASessionRef session = DiskArbitration.INSTANCE
159                     .DASessionCreate(CoreFoundation.INSTANCE.CFAllocatorGetDefault());
160             if (session == null) {
161                 LOG.error("Unable to open session to DiskArbitration framework.");
162             } else {
163                 CFStringRef daVolumeNameKey = CFStringRef.createCFString("DAVolumeName");
164 
165                 // Create array to hold results
166                 Statfs[] fs = new Statfs[numfs];
167                 // Fill array with results
168                 numfs = SystemB.INSTANCE.getfsstat64(fs, numfs * new Statfs().size(), SystemB.MNT_NOWAIT);
169                 for (int f = 0; f < numfs; f++) {
170                     // Mount on name will match mounted path, e.g. /Volumes/foo
171                     // Mount to name will match canonical path., e.g., /dev/disk0s2
172                     // Byte arrays are null-terminated strings
173 
174                     // Get volume and path name, and type
175                     String volume = Native.toString(fs[f].f_mntfromname, StandardCharsets.UTF_8);
176                     String path = Native.toString(fs[f].f_mntonname, StandardCharsets.UTF_8);
177                     String type = Native.toString(fs[f].f_fstypename, StandardCharsets.UTF_8);
178                     // Skip non-local drives if requested, skip system types
179                     final int flags = fs[f].f_flags;
180 
181                     // Skip non-local drives if requested, and exclude pseudo file systems
182                     if ((localOnly && (flags & MNT_LOCAL) == 0) || !path.equals("/")
183                             && (PSEUDO_FS_TYPES.contains(type) || FileSystemUtil.isFileStoreExcluded(path, volume,
184                                     FS_PATH_INCLUDES, FS_PATH_EXCLUDES, FS_VOLUME_INCLUDES, FS_VOLUME_EXCLUDES))) {
185                         continue;
186                     }
187 
188                     String description = "Volume";
189                     if (LOCAL_DISK.matcher(volume).matches()) {
190                         description = "Local Disk";
191                     } else if (volume.startsWith("localhost:") || volume.startsWith("//") || volume.startsWith("smb://")
192                             || NETWORK_FS_TYPES.contains(type)) {
193                         description = "Network Drive";
194                     }
195                     String name = "";
196                     File file = new File(path);
197                     if (name.isEmpty()) {
198                         name = file.getName();
199                         // getName() for / is still blank, so:
200                         if (name.isEmpty()) {
201                             name = file.getPath();
202                         }
203                     }
204                     if (nameToMatch != null && !nameToMatch.equals(name)) {
205                         continue;
206                     }
207 
208                     StringBuilder options = new StringBuilder((MNT_RDONLY & flags) == 0 ? "rw" : "ro");
209                     String moreOptions = OPTIONS_MAP.entrySet().stream().filter(e -> (e.getKey() & flags) > 0)
210                             .map(Map.Entry::getValue).collect(Collectors.joining(","));
211                     if (!moreOptions.isEmpty()) {
212                         options.append(',').append(moreOptions);
213                     }
214 
215                     String uuid = "";
216                     // Use volume to find DiskArbitration volume name and search for
217                     // the registry entry for UUID
218                     String bsdName = volume.replace("/dev/disk", "disk");
219                     if (bsdName.startsWith("disk")) {
220                         // Get the DiskArbitration dictionary for this disk,
221                         // which has volumename
222                         DADiskRef disk = DiskArbitration.INSTANCE.DADiskCreateFromBSDName(
223                                 CoreFoundation.INSTANCE.CFAllocatorGetDefault(), session, volume);
224                         if (disk != null) {
225                             CFDictionaryRef diskInfo = DiskArbitration.INSTANCE.DADiskCopyDescription(disk);
226                             if (diskInfo != null) {
227                                 // get volume name from its key
228                                 Pointer result = diskInfo.getValue(daVolumeNameKey);
229                                 name = CFUtil.cfPointerToString(result);
230                                 diskInfo.release();
231                             }
232                             disk.release();
233                         }
234                         // Search for bsd name in IOKit registry for UUID
235                         CFMutableDictionaryRef matchingDict = IOKitUtil.getBSDNameMatchingDict(bsdName);
236                         if (matchingDict != null) {
237                             // search for all IOservices that match the bsd name
238                             IOIterator fsIter = IOKitUtil.getMatchingServices(matchingDict);
239                             if (fsIter != null) {
240                                 // getMatchingServices releases matchingDict
241                                 // Should only match one logical drive
242                                 IORegistryEntry fsEntry = fsIter.next();
243                                 if (fsEntry != null && fsEntry.conformsTo("IOMedia")) {
244                                     // Now get the UUID
245                                     uuid = fsEntry.getStringProperty("UUID");
246                                     if (uuid != null) {
247                                         uuid = uuid.toLowerCase();
248                                     }
249                                     fsEntry.release();
250                                 }
251                                 fsIter.release();
252                             }
253                         }
254                     }
255 
256                     fsList.add(new MacOSFileStore(name, volume, name, path, options.toString(),
257                             uuid == null ? "" : uuid, "", description, type, file.getFreeSpace(), file.getUsableSpace(),
258                             file.getTotalSpace(), fs[f].f_ffree, fs[f].f_files));
259                 }
260                 daVolumeNameKey.release();
261                 // Close DA session
262                 session.release();
263             }
264         }
265         return fsList;
266     }
