153     private Pair<List<String>, Map<String, String>> queryArgsAndEnvironment() {
154         // Set up return objects
155         List<String> args = new ArrayList<>();
156         // API does not specify any particular order of entries, but it is reasonable to
157         // maintain whatever order the OS provided to the end user
158         Map<String, String> env = new LinkedHashMap<>();
159 
160         // Get command line via sysctl
161         int[] mib = new int[3];
162         mib[0] = 1; // CTL_KERN
163         mib[1] = 49; // KERN_PROCARGS2
164         mib[2] = getProcessID();
165         // Allocate memory for arguments
166         Memory procargs = new Memory(ARGMAX);
167         procargs.clear();
168         size_t.ByReference size = new size_t.ByReference(ARGMAX);
169         // Fetch arguments
170         if (0 == SystemB.INSTANCE.sysctl(mib, mib.length, procargs, size, null, size_t.ZERO)) {
171             // Procargs contains an int representing total # of args, followed by a
172             // null-terminated execpath string and then the arguments, each
173             // null-terminated (possible multiple consecutive nulls),
174             // The execpath string is also the first arg.
175             // Following this is an int representing total # of env, followed by
176             // null-terminated envs in similar format
177             int nargs = procargs.getInt(0);
178             // Sanity check
179             if (nargs > 0 && nargs <= 1024) {
180                 // Skip first int (containing value of nargs)
181                 long offset = SystemB.INT_SIZE;
182                 // Skip exec_command, as
183                 offset += procargs.getString(offset).length();
184                 // Iterate character by character using offset
185                 // Build each arg and add to list
186                 while (offset < size.longValue()) {
187                     // Advance through additional nulls
188                     while (procargs.getByte(offset) == 0) {
189                         if (++offset >= size.longValue()) {
190                             break;
191                         }
192                     }
193                     // Grab a string. This should go until the null terminator
194                     String arg = procargs.getString(offset);
195                     if (nargs-- > 0) {
196                         // If we havent found nargs yet, it's an arg
197                         args.add(arg);
198                     } else {
199                         // otherwise it's an env
200                         int idx = arg.indexOf('=');
201                         if (idx > 0) {
202                             env.put(arg.substring(0, idx), arg.substring(idx + 1));
203                         }
204                     }
205                     // Advance offset to next null
206                     offset += arg.length();
207                 }
208             }
209         } else {
210             LOG.warn(
211                     "Failed sysctl call for process arguments (kern.procargs2), process {} may not exist. Error code: {}",
212                     getProcessID(), Native.getLastError());
213         }
214         return new Pair<>(Collections.unmodifiableList(args), Collections.unmodifiableMap(env));
215     }

257     @Override
258     public List<OSThread> getThreadDetails() {
259         long now = System.currentTimeMillis();
260         List<OSThread> details = new ArrayList<>();
261         List<ThreadStats> stats = ThreadInfo.queryTaskThreads(getProcessID());
262         for (ThreadStats stat : stats) {
263             // For long running threads the start time calculation can overestimate
264             long start = now - stat.getUpTime();
265             if (start < this.getStartTime()) {
266                 start = this.getStartTime();
267             }
268             details.add(new MacOSThread(getProcessID(), stat.getThreadId(), stat.getState(), stat.getSystemTime(),
269                     stat.getUserTime(), start, now - start, stat.getPriority()));
270         }
271         return details;
272     }

329     @Override
330     public long getAffinityMask() {
331         // macOS doesn't do affinity. Return a bitmask of the current processors.
332         int logicalProcessorCount = SysctlUtil.sysctl("hw.logicalcpu", 1);
333         return logicalProcessorCount < 64 ? (1L << logicalProcessorCount) - 1 : -1L;
334     }

351     @Override
352     public boolean updateAttributes() {
353         long now = System.currentTimeMillis();
354         ProcTaskAllInfo taskAllInfo = new ProcTaskAllInfo();
355         if (0 > SystemB.INSTANCE.proc_pidinfo(getProcessID(), SystemB.PROC_PIDTASKALLINFO, 0, taskAllInfo,
356                 taskAllInfo.size()) || taskAllInfo.ptinfo.pti_threadnum < 1) {
357             this.state = INVALID;
358             return false;
359         }
360         Pointer buf = new Memory(SystemB.PROC_PIDPATHINFO_MAXSIZE);
361         if (0 < SystemB.INSTANCE.proc_pidpath(getProcessID(), buf, SystemB.PROC_PIDPATHINFO_MAXSIZE)) {
362             this.path = buf.getString(0).trim();
363             // Overwrite name with last part of path
364             String[] pathSplit = this.path.split("/");
365             if (pathSplit.length > 0) {
366                 this.name = pathSplit[pathSplit.length - 1];
367             }
368         }
369         if (this.name.isEmpty()) {
370             // pbi_comm contains first 16 characters of name
371             this.name = Native.toString(taskAllInfo.pbsd.pbi_comm, StandardCharsets.UTF_8);
372         }
373 
374         switch (taskAllInfo.pbsd.pbi_status) {
375         case SSLEEP:
376             this.state = SLEEPING;
377             break;
378         case SWAIT:
379             this.state = WAITING;
380             break;
381         case SRUN:
382             this.state = RUNNING;
383             break;
384         case SIDL:
385             this.state = NEW;
386             break;
387         case SZOMB:
388             this.state = ZOMBIE;
389             break;
390         case SSTOP:
391             this.state = STOPPED;
392             break;
393         default:
394             this.state = OTHER;
395             break;
396         }
397         this.parentProcessID = taskAllInfo.pbsd.pbi_ppid;
398         this.userID = Integer.toString(taskAllInfo.pbsd.pbi_uid);
399         Passwd pwuid = SystemB.INSTANCE.getpwuid(taskAllInfo.pbsd.pbi_uid);
400         if (pwuid != null) {
401             this.user = pwuid.pw_name;
402         }
403         this.groupID = Integer.toString(taskAllInfo.pbsd.pbi_gid);
404         Group grgid = SystemB.INSTANCE.getgrgid(taskAllInfo.pbsd.pbi_gid);
405         if (grgid != null) {
406             this.group = grgid.gr_name;
407         }
408         this.threadCount = taskAllInfo.ptinfo.pti_threadnum;
409         this.priority = taskAllInfo.ptinfo.pti_priority;
410         this.virtualSize = taskAllInfo.ptinfo.pti_virtual_size;
411         this.residentSetSize = taskAllInfo.ptinfo.pti_resident_size;
412         this.kernelTime = taskAllInfo.ptinfo.pti_total_system / 1_000_000L;
413         this.userTime = taskAllInfo.ptinfo.pti_total_user / 1_000_000L;
414         this.startTime = taskAllInfo.pbsd.pbi_start_tvsec * 1000L + taskAllInfo.pbsd.pbi_start_tvusec / 1000L;
415         this.upTime = now - this.startTime;
416         this.openFiles = taskAllInfo.pbsd.pbi_nfiles;
417         this.bitness = (taskAllInfo.pbsd.pbi_flags & P_LP64) == 0 ? 32 : 64;
418         this.majorFaults = taskAllInfo.ptinfo.pti_pageins;
419         // testing using getrusage confirms pti_faults includes both major and minor
420         this.minorFaults = taskAllInfo.ptinfo.pti_faults - taskAllInfo.ptinfo.pti_pageins; // NOSONAR squid:S2184
421         this.contextSwitches = taskAllInfo.ptinfo.pti_csw;
422         if (this.minorVersion >= 9) {
423             RUsageInfoV2 rUsageInfoV2 = new RUsageInfoV2();
424             if (0 == SystemB.INSTANCE.proc_pid_rusage(getProcessID(), SystemB.RUSAGE_INFO_V2, rUsageInfoV2)) {
425                 this.bytesRead = rUsageInfoV2.ri_diskio_bytesread;
426                 this.bytesWritten = rUsageInfoV2.ri_diskio_byteswritten;
427             }
428         }
429         VnodePathInfo vpi = new VnodePathInfo();
430         if (0 < SystemB.INSTANCE.proc_pidinfo(getProcessID(), SystemB.PROC_PIDVNODEPATHINFO, 0, vpi, vpi.size())) {
431             this.currentWorkingDirectory = Native.toString(vpi.pvi_cdir.vip_path, StandardCharsets.US_ASCII);
432         }
433         return true;
434     }
