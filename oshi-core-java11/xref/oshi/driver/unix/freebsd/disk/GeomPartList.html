24  package oshi.driver.unix.freebsd.disk;
25  
26  import java.util.ArrayList;
27  import java.util.Comparator;
28  import java.util.HashMap;
29  import java.util.List;
30  import java.util.Map;
31  import java.util.stream.Collectors;
32  
33  import oshi.annotation.concurrent.ThreadSafe;
34  import oshi.hardware.HWPartition;
35  import oshi.util.Constants;
36  import oshi.util.ExecutingCommand;
37  import oshi.util.ParseUtil;
38  
39  /**
40   * Utility to query geom part list
41   */
42  @ThreadSafe
43  public final class GeomPartList {
44  
45      private static final String GEOM_PART_LIST = "geom part list";
46      private static final String STAT_FILESIZE = "stat -f %i /dev/";
47  
48      private GeomPartList() {
49      }
50  
51      /**
52       * Queries partition data using geom, mount, and stat commands
53       *
54       * @return A map with disk name as the key and a List of partitions as the value
55       */
56      public static Map<String, List<HWPartition>> queryPartitions() {
57          Map<String, String> mountMap = Mount.queryPartitionToMountMap();
58          // Map of device name to partitions, to be returned
59          Map<String, List<HWPartition>> partitionMap = new HashMap<>();
60          // The Disk Store associated with a partition, key to the map
61          String diskName = null;
62          // List to hold partitions, will be added as value to the map
63          List<HWPartition> partList = new ArrayList<>();
64          // Parameters needed for constructor.
65          String partName = null; // Non-null identifies a valid partition
66          String identification = Constants.UNKNOWN;
67          String type = Constants.UNKNOWN;
68          String uuid = Constants.UNKNOWN;
69          long size = 0;
70          String mountPoint = "";
71  
72          List<String> geom = ExecutingCommand.runNative(GEOM_PART_LIST);
73          for (String line : geom) {
74              line = line.trim();
75              // Marks the DiskStore device for a partition.
76              if (line.startsWith("Geom name:")) {
77                  // Save any previous partition list in the map
78                  if (diskName != null && !partList.isEmpty()) {
79                      // Store map (old diskName)
80                      partitionMap.put(diskName, partList);
81                      // Reset the list
82                      partList = new ArrayList<>();
83                  }
84                  // Now use new diskName
85                  diskName = line.substring(line.lastIndexOf(' ') + 1);
86              }
87              // If we don't have a valid store, don't bother parsing anything
88              if (diskName != null) {
89                  // Marks the beginning of partition data
90                  if (line.contains("Name:")) {
91                      // Add the current partition to the list, if any
92                      if (partName != null) {
93                          // FreeBSD Major # is 0.
94                          // Minor # is filesize of /dev entry.
95                          int minor = ParseUtil
96                                  .parseIntOrDefault(ExecutingCommand.getFirstAnswer(STAT_FILESIZE + partName), 0);
97                          partList.add(new HWPartition(identification, partName, type, uuid, size, 0, minor, mountPoint));
98                          partName = null;
99                          identification = Constants.UNKNOWN;
100                         type = Constants.UNKNOWN;
101                         uuid = Constants.UNKNOWN;
102                         size = 0;
103                     }
104                     // Verify new entry is a partition
105                     // (will happen in 'providers' section)
106                     String part = line.substring(line.lastIndexOf(' ') + 1);
107                     if (part.startsWith(diskName)) {
108                         partName = part;
109                         identification = part;
110                         mountPoint = mountMap.getOrDefault(part, "");
111                     }
112                 }
113                 // If we don't have a valid partition, don't parse anything until we do.
114                 if (partName != null) {
115                     String[] split = ParseUtil.whitespaces.split(line);
116                     if (split.length >= 2) {
117                         if (line.startsWith("Mediasize:")) {
118                             size = ParseUtil.parseLongOrDefault(split[1], 0L);
119                         } else if (line.startsWith("rawuuid:")) {
120                             uuid = split[1];
121                         } else if (line.startsWith("type:")) {
122                             type = split[1];
123                         }
124                     }
125                 }
126             }
127         }
128         if (diskName != null) {
129             // Process last partition
130             if (partName != null) {
131                 int minor = ParseUtil.parseIntOrDefault(ExecutingCommand.getFirstAnswer(STAT_FILESIZE + partName), 0);
132                 partList.add(new HWPartition(identification, partName, type, uuid, size, 0, minor, mountPoint));
133             }
134             // Process last diskstore
135             if (!partList.isEmpty()) {
136                 partList = partList.stream().sorted(Comparator.comparing(HWPartition::getName))
137                         .collect(Collectors.toList());
138                 partitionMap.put(diskName, partList);
139             }
140         }
141         return partitionMap;
142     }
143 }
