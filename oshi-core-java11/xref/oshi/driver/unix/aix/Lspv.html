24  package oshi.driver.unix.aix;
25  
26  import java.util.ArrayList;
27  import java.util.Collections;
28  import java.util.HashMap;
29  import java.util.List;
30  import java.util.Map;
31  import java.util.Map.Entry;
32  
33  import oshi.annotation.concurrent.ThreadSafe;
34  import oshi.hardware.HWPartition;
35  import oshi.util.ExecutingCommand;
36  import oshi.util.ParseUtil;
37  import oshi.util.tuples.Pair;
38  
39  /**
40   * Utility to query lspv
41   */
42  @ThreadSafe
43  public final class Lspv {
44  
45      private Lspv() {
46      }
47  
48      /**
49       * Query {@code lspv} to get partition info
50       *
51       * @param device
52       *            The disk to get the volumes from
53       * @param majMinMap
54       *            A map of device name to a pair with major and minor numbers.
55       *
56       * @return A pair containing the model and serial number for the device, or null
57       *         if not found
58       */
59      public static List<HWPartition> queryLogicalVolumes(String device, Map<String, Pair<Integer, Integer>> majMinMap) {
60          /*-
61           $ lspv -L hdisk0
62          PHYSICAL VOLUME:    hdisk0                   VOLUME GROUP:     rootvg
63          PV IDENTIFIER:      000acfde95524f85 VG IDENTIFIER     000acfde00004c000000000395525276
64          PV STATE:           active
65          STALE PARTITIONS:   0                        ALLOCATABLE:      yes
66          PP SIZE:            128 megabyte(s)          LOGICAL VOLUMES:  12
67          TOTAL PPs:          271 (34688 megabytes)    VG DESCRIPTORS:   2
68          FREE PPs:           227 (29056 megabytes)    HOT SPARE:        no
69          USED PPs:           44 (5632 megabytes)      MAX REQUEST:      256 kilobytes
70          FREE DISTRIBUTION:  54..46..19..54..54
71          USED DISTRIBUTION:  01..08..35..00..00
72           */
73          String stateMarker = "PV STATE:";
74          String sizeMarker = "PP SIZE:";
75          long ppSize = 0L; // All physical partitions are the same size
76          for (String s : ExecutingCommand.runNative("lspv -L " + device)) {
77              if (s.startsWith(stateMarker)) {
78                  if (!s.contains("active")) {
79                      return Collections.emptyList();
80                  }
81              } else if (s.contains(sizeMarker)) {
82                  ppSize = ParseUtil.getFirstIntValue(s);
83              }
84          }
85          if (ppSize == 0L) {
86              return Collections.emptyList();
87          }
88          // Convert to megabytes
89          ppSize <<= 20;
90          /*-
91           $ lspv -p hdisk0
92          hdisk0:
93          PP RANGE  STATE   REGION        LV NAME             TYPE       MOUNT POINT
94          1-1     used    outer edge    hd5                 boot       N/A
95          2-55    free    outer edge
96          56-59    used    outer middle  hd6                 paging     N/A
97          60-61    used    outer middle  livedump            jfs2       /var/adm/ras/livedump
98          62-62    used    outer middle  loglv01             jfslog     N/A
99          63-63    used    outer middle  lv01                jfs        N/A
100         64-109   free    outer middle
101         110-110   used    center        hd8                 jfs2log    N/A
102         111-112   used    center        hd4                 jfs2       /
103         113-128   used    center        hd2                 jfs2       /usr
104         129-131   used    center        hd9var              jfs2       /var
105         132-132   used    center        hd3                 jfs2       /tmp
106         133-133   used    center        hd9var              jfs2       /var
107         134-136   used    center        hd10opt             jfs2       /opt
108         137-137   used    center        hd11admin           jfs2       /admin
109         138-140   used    center        hd2                 jfs2       /usr
110         141-141   used    center        hd3                 jfs2       /tmp
111         142-142   used    center        hd4                 jfs2       /
112         143-143   used    center        hd9var              jfs2       /var
113         144-144   used    center        hd2                 jfs2       /usr
114         145-163   free    center
115         164-217   free    inner middle
116         218-271   free    inner edge
117          */
118         Map<String, String> mountMap = new HashMap<>();
119         Map<String, String> typeMap = new HashMap<>();
120         Map<String, Integer> ppMap = new HashMap<>();
121         for (String s : ExecutingCommand.runNative("lspv -p " + device)) {
122             String[] split = ParseUtil.whitespaces.split(s.trim());
123             if (split.length >= 6 && "used".equals(split[1])) {
124                 // Region may have two words, so count from end
125                 String name = split[split.length - 3];
126                 mountMap.put(name, split[split.length - 1]);
127                 typeMap.put(name, split[split.length - 2]);
128                 int ppCount = 1 + ParseUtil.getNthIntValue(split[0], 2) - ParseUtil.getNthIntValue(split[0], 1);
129                 ppMap.put(name, ppCount + ppMap.getOrDefault(name, 0));
130             }
131         }
132         List<HWPartition> partitions = new ArrayList<>();
133         for (Entry<String, String> entry : mountMap.entrySet()) {
134             String mount = "N/A".equals(entry.getValue()) ? "" : entry.getValue();
135             // All maps should have same keys
136             String name = entry.getKey();
137             String type = typeMap.get(name);
138             long size = ppSize * ppMap.get(name);
139             Pair<Integer, Integer> majMin = majMinMap.get(name);
140             int major = majMin == null ? ParseUtil.getFirstIntValue(name) : majMin.getA();
141             int minor = majMin == null ? ParseUtil.getFirstIntValue(name) : majMin.getB();
142             partitions.add(new HWPartition(name, name, type, "", size, major, minor, mount));
143         }
144         return partitions;
145     }
146 }
