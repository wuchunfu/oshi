24  package oshi.driver.unix.aix.perfstat;
25  
26  import com.sun.jna.platform.unix.aix.Perfstat; // NOSONAR squid:S1191
27  import com.sun.jna.platform.unix.aix.Perfstat.perfstat_memory_total_t;
28  
29  import oshi.annotation.concurrent.ThreadSafe;
30  
31  /**
32   * Utility to query performance stats for memory
33   */
34  @ThreadSafe
35  public final class PerfstatMemory {
36  
37      private static final Perfstat PERF = Perfstat.INSTANCE;
38  
39      private PerfstatMemory() {
40      }
41  
42      /**
43       * Queries perfstat_memory_total for total memory usage statistics
44       *
45       * @return usage statistics
46       */
47      public static perfstat_memory_total_t queryMemoryTotal() {
48          perfstat_memory_total_t memory = new perfstat_memory_total_t();
49          int ret = PERF.perfstat_memory_total(null, memory, memory.size(), 1);
50          if (ret > 0) {
51              return memory;
52          }
53          return new perfstat_memory_total_t();
54      }
55  }
