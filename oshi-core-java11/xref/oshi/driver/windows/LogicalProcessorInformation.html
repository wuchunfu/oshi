42  /**
43   * Utility to query Logical Processor Information
44   */
45  @ThreadSafe
46  public final class LogicalProcessorInformation {
47  
48      private LogicalProcessorInformation() {
49      }
50  
51      /**
52       * Get a list of logical processors on this machine. Requires Windows 7 and
53       * higher.
54       *
55       * @return A list of logical processors
56       */
57      public static List<LogicalProcessor> getLogicalProcessorInformationEx() {
58          // Collect a list of logical processors on each physical core and
59          // package. These will be 64-bit bitmasks.
60          SYSTEM_LOGICAL_PROCESSOR_INFORMATION_EX[] procInfo = Kernel32Util
61                  .getLogicalProcessorInformationEx(WinNT.LOGICAL_PROCESSOR_RELATIONSHIP.RelationAll);
62          // Used to cross-reference a processor to package pr core
63          List<GROUP_AFFINITY[]> packages = new ArrayList<>();
64          List<GROUP_AFFINITY> cores = new ArrayList<>();
65          // Used to iterate
66          List<NUMA_NODE_RELATIONSHIP> numaNodes = new ArrayList<>();
67  
68          for (int i = 0; i < procInfo.length; i++) {
69              switch (procInfo[i].relationship) {
70              case LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorPackage:
71                  // could assign a package to more than one processor group
72                  packages.add(((PROCESSOR_RELATIONSHIP) procInfo[i]).groupMask);
73                  break;
74              case LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorCore:
75                  // for Core, groupCount is always 1
76                  cores.add(((PROCESSOR_RELATIONSHIP) procInfo[i]).groupMask[0]);
77                  break;
78              case LOGICAL_PROCESSOR_RELATIONSHIP.RelationNumaNode:
79                  numaNodes.add((NUMA_NODE_RELATIONSHIP) procInfo[i]);
80                  break;
81              default:
82                  // Ignore Group and Cache info
83                  break;
84              }
85          }
86          // Windows doesn't define core and package numbers, so we sort the lists
87          // so core and package numbers increment consistently with processor
88          // numbers/bitmasks, ordered in groups
89          cores.sort(Comparator.comparing(c -> c.group * 64L + c.mask.longValue()));
90          // if package in multiple groups will still use first group for sorting
91          packages.sort(Comparator.comparing(p -> p[0].group * 64L + p[0].mask.longValue()));
92  
93          // Iterate Logical Processors and use bitmasks to match packages, cores,
94          // and NUMA nodes. Perfmon instances are numa node + processor number, so we
95          // iterate by numa node so the returned list will properly index perfcounter
96          // numa/proc-per-numa indices with all numa nodes grouped together
97          numaNodes.sort(Comparator.comparing(n -> n.nodeNumber));
98          List<LogicalProcessor> logProcs = new ArrayList<>();
99          for (NUMA_NODE_RELATIONSHIP node : numaNodes) {
100             int nodeNum = node.nodeNumber;
101             int group = node.groupMask.group;
102             long mask = node.groupMask.mask.longValue();
103             // Processor numbers are uniquely identified by processor group and processor
104             // number on that group, which matches the bitmask.
105             int lowBit = Long.numberOfTrailingZeros(mask);
106             int hiBit = 63 - Long.numberOfLeadingZeros(mask);
107             for (int lp = lowBit; lp <= hiBit; lp++) {
108                 if ((mask & (1L << lp)) != 0) {
109                     LogicalProcessor logProc = new LogicalProcessor(lp, getMatchingCore(cores, group, lp),
110                             getMatchingPackage(packages, group, lp), nodeNum, group);
111                     logProcs.add(logProc);
112                 }
113             }
114         }
115         return logProcs;
116     }
117 
118     private static int getMatchingPackage(List<GROUP_AFFINITY[]> packages, int g, int lp) {
119         for (int i = 0; i < packages.size(); i++) {
120             for (int j = 0; j < packages.get(i).length; j++) {
121                 if ((packages.get(i)[j].mask.longValue() & (1L << lp)) != 0 && packages.get(i)[j].group == g) {
122                     return i;
123                 }
124             }
125         }
126         return 0;
127     }
128 
129     private static int getMatchingCore(List<GROUP_AFFINITY> cores, int g, int lp) {
130         for (int j = 0; j < cores.size(); j++) {
131             if ((cores.get(j).mask.longValue() & (1L << lp)) != 0 && cores.get(j).group == g) {
132                 return j;
133             }
134         }
135         return 0;
136     }

142     /**
143      * Get a list of logical processors on this machine
144      *
145      * @return A list of logical processors
146      */
147     public static List<LogicalProcessor> getLogicalProcessorInformation() {
148         // Collect a list of logical processors on each physical core and
149         // package.
150         List<Long> packageMaskList = new ArrayList<>();
151         List<Long> coreMaskList = new ArrayList<>();
152         WinNT.SYSTEM_LOGICAL_PROCESSOR_INFORMATION[] processors = Kernel32Util.getLogicalProcessorInformation();
153         for (SYSTEM_LOGICAL_PROCESSOR_INFORMATION proc : processors) {
154             if (proc.relationship == WinNT.LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorPackage) {
155                 packageMaskList.add(proc.processorMask.longValue());
156             } else if (proc.relationship == WinNT.LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorCore) {
157                 coreMaskList.add(proc.processorMask.longValue());
158             }
159         }
160         // Sort the list (natural ordering) so core and package numbers
161         // increment as expected.
162         coreMaskList.sort(null);
163         packageMaskList.sort(null);
164 
165         // Assign logical processors to cores and packages
166         List<LogicalProcessor> logProcs = new ArrayList<>();
167         for (int core = 0; core < coreMaskList.size(); core++) {
168             long coreMask = coreMaskList.get(core);
169             // Lowest and Highest set bits, indexing from 0
170             int lowBit = Long.numberOfTrailingZeros(coreMask);
171             int hiBit = 63 - Long.numberOfLeadingZeros(coreMask);
172             // Create logical processors for this core
173             for (int i = lowBit; i <= hiBit; i++) {
174                 if ((coreMask & (1L << i)) != 0) {
175                     LogicalProcessor logProc = new LogicalProcessor(i, core,
176                             LogicalProcessorInformation.getBitMatchingPackageNumber(packageMaskList, i));
177                     logProcs.add(logProc);
178                 }
179             }
180         }
181         return logProcs;
182     }
183 
184     /**
185      * Iterate over the package mask list and find a matching mask index
186      *
187      * @param packageMaskList
188      *            The list of bitmasks to iterate
189      * @param logProc
190      *            The bit to find matching mask
191      * @return The index of the list which matched the bit
192      */
193     private static int getBitMatchingPackageNumber(List<Long> packageMaskList, int logProc) {
194         for (int i = 0; i < packageMaskList.size(); i++) {
195             if ((packageMaskList.get(i).longValue() & (1L << logProc)) != 0) {
196                 return i;
197             }
198         }
199         return 0;
200     }
