24  package oshi.driver.windows.perfmon;
25  
26  import java.util.List;
27  import java.util.Map;
28  
29  import oshi.annotation.concurrent.ThreadSafe;
30  import oshi.util.platform.windows.PerfCounterQuery;
31  import oshi.util.platform.windows.PerfCounterWildcardQuery;
32  import oshi.util.platform.windows.PerfCounterWildcardQuery.PdhCounterWildcardProperty;
33  import oshi.util.tuples.Pair;
34  
35  /**
36   * Utility to query Process Information performance counter
37   */
38  @ThreadSafe
39  public final class ProcessInformation {
40  
41      private static final String WIN32_PERFPROC_PROCESS = "Win32_PerfRawData_PerfProc_Process";
42      private static final String PROCESS = "Process";
43      private static final String WIN32_PERFPROC_PROCESS_WHERE_NOT_NAME_LIKE_TOTAL = WIN32_PERFPROC_PROCESS
44              + " WHERE NOT Name LIKE \"%_Total\"";
45  
46      /**
47       * Process performance counters
48       */
49      public enum ProcessPerformanceProperty implements PdhCounterWildcardProperty {
50          // First element defines WMI instance name field and PDH instance filter
51          NAME(PerfCounterQuery.NOT_TOTAL_INSTANCES),
52          // Remaining elements define counters
53          PRIORITYBASE("Priority Base"), //
54          ELAPSEDTIME("Elapsed Time"), //
55          IDPROCESS("ID Process"), //
56          CREATINGPROCESSID("Creating Process ID"), //
57          IOREADBYTESPERSEC("IO Read Bytes/sec"), //
58          IOWRITEBYTESPERSEC("IO Write Bytes/sec"), //
59          PRIVATEBYTES("Working Set - Private"), //
60          PAGEFAULTSPERSEC("Page Faults/sec");
61  
62          private final String counter;
63  
64          ProcessPerformanceProperty(String counter) {
65              this.counter = counter;
66          }
67  
68          @Override
69          public String getCounter() {
70              return counter;
71          }
72      }
73  
74      /**
75       * Handle performance counters
76       */
77      public enum HandleCountProperty implements PdhCounterWildcardProperty {
78          // First element defines WMI instance name field and PDH instance filter
79          NAME(PerfCounterQuery.TOTAL_INSTANCE),
80          // Remaining elements define counters
81          HANDLECOUNT("Handle Count");
82  
83          private final String counter;
84  
85          HandleCountProperty(String counter) {
86              this.counter = counter;
87          }
88  
89          @Override
90          public String getCounter() {
91              return counter;
92          }
93      }
94  
95      private ProcessInformation() {
96      }
97  
98      /**
99       * Returns process counters.
100      *
101      * @return Process counters for each process.
102      */
103     public static Pair<List<String>, Map<ProcessPerformanceProperty, List<Long>>> queryProcessCounters() {
104         return PerfCounterWildcardQuery.queryInstancesAndValues(ProcessPerformanceProperty.class, PROCESS,
105                 WIN32_PERFPROC_PROCESS_WHERE_NOT_NAME_LIKE_TOTAL);
106     }
107 
108     /**
109      * Returns handle counters
110      *
111      * @return Process handle counters for each process.
112      */
113     public static Pair<List<String>, Map<HandleCountProperty, List<Long>>> queryHandles() {
114         return PerfCounterWildcardQuery.queryInstancesAndValues(HandleCountProperty.class, PROCESS,
115                 WIN32_PERFPROC_PROCESS);
116     }
117 }
