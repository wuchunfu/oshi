41  /**
42   * Utility to read process data from HKEY_PERFORMANCE_DATA information with
43   * backup from Performance Counters or WMI
44   */
45  @ThreadSafe
46  public final class ProcessPerformanceData {
47  
48      private static final String PROCESS = "Process";
49      public static final String WIN_HKEY_PERFDATA = "oshi.os.windows.hkeyperfdata";
50      private static final boolean PERFDATA = GlobalConfig.get(WIN_HKEY_PERFDATA, true);
51  
52      private ProcessPerformanceData() {
53      }
54  
55      /**
56       * Query the registry for process performance counters
57       *
58       * @param pids
59       *            An optional collection of process IDs to filter the list to. May
60       *            be null for no filtering.
61       * @return A map with Process ID as the key and a {@link PerfCounterBlock}
62       *         object populated with performance counter information if successful,
63       *         or null otherwise.
64       */
65      public static Map<Integer, PerfCounterBlock> buildProcessMapFromRegistry(Collection<Integer> pids) {
66          // Grab the data from the registry.
67          Triplet<List<Map<ProcessPerformanceProperty, Object>>, Long, Long> processData = null;
68          if (PERFDATA) {
69              processData = HkeyPerformanceDataUtil.readPerfDataFromRegistry(PROCESS, ProcessPerformanceProperty.class);
70          }
71          if (processData == null) {
72              return null;
73          }
74          List<Map<ProcessPerformanceProperty, Object>> processInstanceMaps = processData.getA();
75          long now = processData.getC(); // 1970 epoch
76  
77          // Create a map and fill it
78          Map<Integer, PerfCounterBlock> processMap = new HashMap<>();
79          // Iterate instances.
80          for (Map<ProcessPerformanceProperty, Object> processInstanceMap : processInstanceMaps) {
81              int pid = ((Integer) processInstanceMap.get(ProcessPerformanceProperty.IDPROCESS)).intValue();
82              String name = (String) processInstanceMap.get(ProcessPerformanceProperty.NAME);
83              if ((pids == null || pids.contains(pid)) && !"_Total".equals(name)) {
84                  // Field name is elapsed time but the value is the process start time
85                  long ctime = (Long) processInstanceMap.get(ProcessPerformanceProperty.ELAPSEDTIME);
86                  // if creation time value is less than current millis, it's in 1970 epoch,
87                  // otherwise it's 1601 epoch and we must convert
88                  if (ctime > now) {
89                      ctime = WinBase.FILETIME.filetimeToDate((int) (ctime >> 32), (int) (ctime & 0xffffffffL)).getTime();
90                  }
91                  long upTime = now - ctime;
92                  if (upTime < 1L) {
93                      upTime = 1L;
94                  }
95                  processMap.put(pid,
96                          new PerfCounterBlock(name,
97                                  (Integer) processInstanceMap.get(ProcessPerformanceProperty.CREATINGPROCESSID),
98                                  (Integer) processInstanceMap.get(ProcessPerformanceProperty.PRIORITYBASE),
99                                  (Long) processInstanceMap.get(ProcessPerformanceProperty.PRIVATEBYTES), ctime, upTime,
100                                 (Long) processInstanceMap.get(ProcessPerformanceProperty.IOREADBYTESPERSEC),
101                                 (Long) processInstanceMap.get(ProcessPerformanceProperty.IOWRITEBYTESPERSEC),
102                                 (Integer) processInstanceMap.get(ProcessPerformanceProperty.PAGEFAULTSPERSEC)));
103             }
104         }
105         return processMap;
106     }

108     /**
109      * Query PerfMon for process performance counters
110      *
111      * @param pids
112      *            An optional collection of process IDs to filter the list to. May
113      *            be null for no filtering.
114      * @return A map with Process ID as the key and a {@link PerfCounterBlock}
115      *         object populated with performance counter information.
116      */
117     public static Map<Integer, PerfCounterBlock> buildProcessMapFromPerfCounters(Collection<Integer> pids) {
118         Map<Integer, PerfCounterBlock> processMap = new HashMap<>();
119         Pair<List<String>, Map<ProcessPerformanceProperty, List<Long>>> instanceValues = ProcessInformation
120                 .queryProcessCounters();
121         long now = System.currentTimeMillis(); // 1970 epoch
122         List<String> instances = instanceValues.getA();
123         Map<ProcessPerformanceProperty, List<Long>> valueMap = instanceValues.getB();
124         List<Long> pidList = valueMap.get(ProcessPerformanceProperty.IDPROCESS);
125         List<Long> ppidList = valueMap.get(ProcessPerformanceProperty.CREATINGPROCESSID);
126         List<Long> priorityList = valueMap.get(ProcessPerformanceProperty.PRIORITYBASE);
127         List<Long> ioReadList = valueMap.get(ProcessPerformanceProperty.IOREADBYTESPERSEC);
128         List<Long> ioWriteList = valueMap.get(ProcessPerformanceProperty.IOWRITEBYTESPERSEC);
129         List<Long> workingSetSizeList = valueMap.get(ProcessPerformanceProperty.PRIVATEBYTES);
130         List<Long> elapsedTimeList = valueMap.get(ProcessPerformanceProperty.ELAPSEDTIME);
131         List<Long> pageFaultsList = valueMap.get(ProcessPerformanceProperty.PAGEFAULTSPERSEC);
132 
133         for (int inst = 0; inst < instances.size(); inst++) {
134             int pid = pidList.get(inst).intValue();
135             if (pids == null || pids.contains(pid)) {
136                 // Field name is elapsed time but the value is the process start time
137                 long ctime = elapsedTimeList.get(inst);
138                 // if creation time value is less than current millis, it's in 1970 epoch,
139                 // otherwise it's 1601 epoch and we must convert
140                 if (ctime > now) {
141                     ctime = WinBase.FILETIME.filetimeToDate((int) (ctime >> 32), (int) (ctime & 0xffffffffL)).getTime();
142                 }
143                 long upTime = now - ctime;
144                 if (upTime < 1L) {
145                     upTime = 1L;
146                 }
147                 processMap.put(pid,
148                         new PerfCounterBlock(instances.get(inst), ppidList.get(inst).intValue(),
149                                 priorityList.get(inst).intValue(), workingSetSizeList.get(inst), ctime, upTime,
150                                 ioReadList.get(inst), ioWriteList.get(inst), pageFaultsList.get(inst).intValue()));
151             }
152         }
153         return processMap;
154     }

156     /**
157      * Class to encapsulate data from the registry performance counter block
158      */
159     @Immutable
160     public static class PerfCounterBlock {
161         private final String name;
162         private final int parentProcessID;
163         private final int priority;
164         private final long residentSetSize;
165         private final long startTime;
166         private final long upTime;
167         private final long bytesRead;
168         private final long bytesWritten;
169         private final int pageFaults;
170 
171         public PerfCounterBlock(String name, int parentProcessID, int priority, long residentSetSize, long startTime,
172                 long upTime, long bytesRead, long bytesWritten, int pageFaults) {
173             this.name = name;
174             this.parentProcessID = parentProcessID;
175             this.priority = priority;
176             this.residentSetSize = residentSetSize;
177             this.startTime = startTime;
178             this.upTime = upTime;
179             this.bytesRead = bytesRead;
180             this.bytesWritten = bytesWritten;
181             this.pageFaults = pageFaults;
182         }
183 
184         /**
185          * @return the name
186          */
187         public String getName() {
188             return name;
189         }
190 
191         /**
192          * @return the parentProcessID
193          */
194         public int getParentProcessID() {
195             return parentProcessID;
196         }
197 
198         /**
199          * @return the priority
200          */
201         public int getPriority() {
202             return priority;
203         }
204 
205         /**
206          * @return the residentSetSize
207          */
208         public long getResidentSetSize() {
209             return residentSetSize;
210         }
211 
212         /**
213          * @return the startTime
214          */
215         public long getStartTime() {
216             return startTime;
217         }
218 
219         /**
220          * @return the upTime
221          */
222         public long getUpTime() {
223             return upTime;
224         }
225 
226         /**
227          * @return the bytesRead
228          */
229         public long getBytesRead() {
230             return bytesRead;
231         }
232 
233         /**
234          * @return the bytesWritten
235          */
236         public long getBytesWritten() {
237             return bytesWritten;
238         }
239 
240         /**
241          * @return the pageFaults
242          */
243         public long getPageFaults() {
244             return pageFaults;
245         }
246     }
