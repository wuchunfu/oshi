51  /**
52   * Utility to read process data from HKEY_PERFORMANCE_DATA information with
53   * backup from Performance Counters or WMI
54   */
55  @ThreadSafe
56  public final class SessionWtsData {
57  
58      private static final int WTS_ACTIVE = 0;
59      private static final int WTS_CLIENTADDRESS = 14;
60      private static final int WTS_SESSIONINFO = 24;
61      private static final int WTS_CLIENTPROTOCOLTYPE = 16;
62  
63      private static final boolean IS_VISTA_OR_GREATER = VersionHelpers.IsWindowsVistaOrGreater();
64  
65      private static final Wtsapi32 WTS = Wtsapi32.INSTANCE;
66  
67      private SessionWtsData() {
68      }
69  
70      public static List<OSSession> queryUserSessions() {
71          List<OSSession> sessions = new ArrayList<>();
72          if (IS_VISTA_OR_GREATER) {
73              PointerByReference ppSessionInfo = new PointerByReference();
74              IntByReference pCount = new IntByReference();
75              if (WTS.WTSEnumerateSessions(Wtsapi32.WTS_CURRENT_SERVER_HANDLE, 0, 1, ppSessionInfo, pCount)) {
76                  Pointer pSessionInfo = ppSessionInfo.getValue();
77                  if (pCount.getValue() > 0) {
78                      WTS_SESSION_INFO sessionInfoRef = new WTS_SESSION_INFO(pSessionInfo);
79                      WTS_SESSION_INFO[] sessionInfo = (WTS_SESSION_INFO[]) sessionInfoRef.toArray(pCount.getValue());
80                      for (WTS_SESSION_INFO session : sessionInfo) {
81                          if (session.State == WTS_ACTIVE) {
82                              // Use session id to fetch additional session information
83                              PointerByReference ppBuffer = new PointerByReference();
84                              IntByReference pBytes = new IntByReference();
85                              WTS.WTSQuerySessionInformation(Wtsapi32.WTS_CURRENT_SERVER_HANDLE, session.SessionId,
86                                      WTS_CLIENTPROTOCOLTYPE, ppBuffer, pBytes);
87                              Pointer pBuffer = ppBuffer.getValue(); // pointer to USHORT
88                              short protocolType = pBuffer.getShort(0); // 0 = console, 2 = RDP
89                              WTS.WTSFreeMemory(pBuffer);
90                              // We've already got console from registry, only test RDP
91                              if (protocolType > 0) {
92                                  // DEVICE
93                                  String device = session.pWinStationName;
94                                  // USER and LOGIN TIME
95                                  WTS.WTSQuerySessionInformation(Wtsapi32.WTS_CURRENT_SERVER_HANDLE, session.SessionId,
96                                          WTS_SESSIONINFO, ppBuffer, pBytes);
97                                  pBuffer = ppBuffer.getValue(); // returns WTSINFO
98                                  WTSINFO wtsInfo = new WTSINFO(pBuffer);
99                                  // Temporary due to broken LARGE_INTEGER, remove in JNA 5.6.0
100                                 long logonTime = new WinBase.FILETIME(
101                                         new WinNT.LARGE_INTEGER(wtsInfo.LogonTime.getValue())).toTime();
102                                 String userName = wtsInfo.getUserName();
103                                 WTS.WTSFreeMemory(pBuffer);
104                                 // HOST
105                                 WTS.WTSQuerySessionInformation(Wtsapi32.WTS_CURRENT_SERVER_HANDLE, session.SessionId,
106                                         WTS_CLIENTADDRESS, ppBuffer, pBytes);
107                                 pBuffer = ppBuffer.getValue(); // returns WTS_CLIENT_ADDRESS
108                                 WTS_CLIENT_ADDRESS addr = new WTS_CLIENT_ADDRESS(pBuffer);
109                                 WTS.WTSFreeMemory(pBuffer);
110                                 String host = "::";
111                                 if (addr.AddressFamily == IPHlpAPI.AF_INET) {
112                                     try {
113                                         host = InetAddress.getByAddress(Arrays.copyOfRange(addr.Address, 2, 6))
114                                                 .getHostAddress();
115                                     } catch (UnknownHostException e) {
116                                         // If array is not length of 4, shouldn't happen
117                                         host = "Illegal length IP Array";
118                                     }
119                                 } else if (addr.AddressFamily == IPHlpAPI.AF_INET6) {
120                                     // Get ints for address parsing
121                                     int[] ipArray = convertBytesToInts(addr.Address);
122                                     host = ParseUtil.parseUtAddrV6toIP(ipArray);
123                                 }
124                                 sessions.add(new OSSession(userName, device, logonTime, host));
125                             }
126                         }
127                     }
128                 }
129                 WTS.WTSFreeMemory(pSessionInfo);
130             }
131         }
132         return sessions;
133     }
134 
135     /**
136      * Per WTS_INFO_CLASS docs, the IP address is offset by two bytes from the start
137      * of the Address member of the WTS_CLIENT_ADDRESS structure. Also contrary to
138      * docs, IPv4 is not a null terminated string.
139      * <p>
140      * This method converts the byte[20] to an int[4] parseable by existing code
141      *
142      * @param address
143      *            The 20-byte array from the WTS_CLIENT_ADDRESS structure
144      * @return A 4-int array for {@link ParseUtil#parseUtAddrV6toIP}
145      */
146     private static int[] convertBytesToInts(byte[] address) {
147         IntBuffer intBuf = ByteBuffer.wrap(Arrays.copyOfRange(address, 2, 18)).order(ByteOrder.BIG_ENDIAN)
148                 .asIntBuffer();
149         int[] array = new int[intBuf.remaining()];
150         intBuf.get(array);
151         return array;
152     }
