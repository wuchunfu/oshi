24  package oshi.driver.windows.wmi;
25  
26  import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiQuery; //NOSONAR squid:S1191
27  import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiResult;
28  
29  import oshi.annotation.concurrent.ThreadSafe;
30  import oshi.util.platform.windows.WmiQueryHandler;
31  
32  /**
33   * Utility to query WMI class {@code Win32_Processor}
34   */
35  @ThreadSafe
36  public final class Win32Processor {
37  
38      private static final String WIN32_PROCESSOR = "Win32_Processor";
39  
40      /**
41       * Processor voltage properties.
42       */
43      public enum VoltProperty {
44          CURRENTVOLTAGE, VOLTAGECAPS;
45      }
46  
47      /**
48       * Processor ID property
49       */
50      public enum ProcessorIdProperty {
51          PROCESSORID;
52      }
53  
54      /**
55       * Processor bitness property
56       */
57      public enum BitnessProperty {
58          ADDRESSWIDTH;
59      }
60  
61      private Win32Processor() {
62      }
63  
64      /**
65       * Returns processor voltage.
66       *
67       * @return Current voltage of the processor. If the eighth bit is set, bits 0-6
68       *         contain the voltage multiplied by 10. If the eighth bit is not set,
69       *         then the bit setting in VoltageCaps represents the voltage value.
70       */
71      public static WmiResult<VoltProperty> queryVoltage() {
72          WmiQuery<VoltProperty> voltQuery = new WmiQuery<>(WIN32_PROCESSOR, VoltProperty.class);
73          return WmiQueryHandler.createInstance().queryWMI(voltQuery);
74      }
75  
76      /**
77       * Returns processor ID.
78       *
79       * @return Processor information that describes the processor features. For an
80       *         x86 class CPU, the field format depends on the processor support of
81       *         the CPUID instruction. If the instruction is supported, the property
82       *         contains 2 (two) DWORD formatted values. The first is an offset of
83       *         08h-0Bh, which is the EAX value that a CPUID instruction returns with
84       *         input EAX set to 1. The second is an offset of 0Ch-0Fh, which is the
85       *         EDX value that the instruction returns. Only the first two bytes of
86       *         the property are significant and contain the contents of the DX
87       *         register at CPU reset—all others are set to 0 (zero), and the
88       *         contents are in DWORD format.
89       */
90      public static WmiResult<ProcessorIdProperty> queryProcessorId() {
91          WmiQuery<ProcessorIdProperty> idQuery = new WmiQuery<>(WIN32_PROCESSOR, ProcessorIdProperty.class);
92          return WmiQueryHandler.createInstance().queryWMI(idQuery);
93      }
94  
95      /**
96       * Returns address width.
97       *
98       * @return On a 32-bit operating system, the value is 32 and on a 64-bit
99       *         operating system it is 64.
100      */
101     public static WmiResult<BitnessProperty> queryBitness() {
102         WmiQuery<BitnessProperty> bitnessQuery = new WmiQuery<>(WIN32_PROCESSOR, BitnessProperty.class);
103         return WmiQueryHandler.createInstance().queryWMI(bitnessQuery);
104     }
105 }
