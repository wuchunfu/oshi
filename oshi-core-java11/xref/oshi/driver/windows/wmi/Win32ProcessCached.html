24  package oshi.driver.windows.wmi;
25  
26  import static oshi.util.Memoizer.memoize;
27  
28  import java.util.HashMap;
29  import java.util.Map;
30  import java.util.concurrent.locks.ReentrantLock;
31  import java.util.function.Supplier;
32  
33  import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiResult; // NOSONAR squid:S1191
34  
35  import oshi.annotation.concurrent.GuardedBy;
36  import oshi.annotation.concurrent.ThreadSafe;
37  import oshi.driver.windows.wmi.Win32Process.CommandLineProperty;
38  import oshi.util.platform.windows.WmiUtil;
39  import oshi.util.tuples.Pair;
40  
41  /**
42   * Utility to query WMI class {@code Win32_Process} using cache
43   */
44  @ThreadSafe
45  public final class Win32ProcessCached {
46  
47      private static final Supplier<Win32ProcessCached> INSTANCE = memoize(Win32ProcessCached::createInstance);
48  
49      // Use a map to cache command line queries
50      @GuardedBy("commandLineCacheLock")
51      private final Map<Integer, Pair<Long, String>> commandLineCache = new HashMap<>();
52      private final ReentrantLock commandLineCacheLock = new ReentrantLock();
53  
54      private Win32ProcessCached() {
55      }
56  
57      /**
58       * Get the singleton instance of this class, instantiating the map which caches
59       * command lines.
60       *
61       * @return the singleton instance
62       */
63      public static Win32ProcessCached getInstance() {
64          return INSTANCE.get();
65      }
66  
67      private static Win32ProcessCached createInstance() {
68          return new Win32ProcessCached();
69      }
70  
71      /**
72       * Gets the process command line, while also querying and caching command lines
73       * for all running processes if the specified process is not in the cache.
74       * <p>
75       * When iterating over a process list, the WMI overhead of querying each single
76       * command line can quickly exceed the time it takes to query all command lines.
77       * This method permits access to cached queries from a previous call,
78       * significantly improving aggregate performance.
79       *
80       * @param processId
81       *            The process ID for which to return the command line.
82       * @param startTime
83       *            The start time of the process, in milliseconds since the 1970
84       *            epoch. If this start time is after the time this process was
85       *            previously queried, the prior entry will be deemed invalid and the
86       *            cache refreshed.
87       * @return The command line of the specified process. If the command line is
88       *         cached from a previous call and the start time is prior to the time
89       *         it was cached, this method will quickly return the cached value.
90       *         Otherwise, will refresh the cache with all running processes prior to
91       *         returning, which may incur some latency.
92       *         <p>
93       *         May return a command line from the cache even after a process has
94       *         terminated. Otherwise will return the empty string.
95       */
96      public String getCommandLine(int processId, long startTime) {
97          // We could use synchronized method but this is more clear
98          commandLineCacheLock.lock();
99          try {
100             // See if this process is in the cache already
101             Pair<Long, String> pair = commandLineCache.get(processId);
102             // Valid process must have been started before map insertion
103             if (pair != null && startTime < pair.getA()) {
104                 // Entry is valid, return it!
105                 return pair.getB();
106             } else {
107                 // Invalid entry, rebuild cache
108                 // Processes started after this time will be invalid
109                 long now = System.currentTimeMillis();
110                 // Gets all processes. Takes ~200ms
111                 WmiResult<CommandLineProperty> commandLineAllProcs = Win32Process.queryCommandLines(null);
112                 // Stale processes use resources. Periodically clear before building
113                 // Use a threshold of map size > 2x # of processes
114                 if (commandLineCache.size() > commandLineAllProcs.getResultCount() * 2) {
115                     commandLineCache.clear();
116                 }
117                 // Iterate results and put in map. Save value for current PID along the way
118                 String result = "";
119                 for (int i = 0; i < commandLineAllProcs.getResultCount(); i++) {
120                     int pid = WmiUtil.getUint32(commandLineAllProcs, CommandLineProperty.PROCESSID, i);
121                     String cl = WmiUtil.getString(commandLineAllProcs, CommandLineProperty.COMMANDLINE, i);
122                     commandLineCache.put(pid, new Pair<>(now, cl));
123                     if (pid == processId) {
124                         result = cl;
125                     }
126                 }
127                 return result;
128             }
129         } finally {
130             commandLineCacheLock.unlock();
131         }
132     }
133 }
