24  package oshi.jna.platform.mac;
25  
26  import com.sun.jna.Native; // NOSONAR squid:S1191
27  import com.sun.jna.Structure;
28  import com.sun.jna.Structure.FieldOrder;
29  import com.sun.jna.Union;
30  
31  import oshi.jna.platform.unix.CLibrary;
32  
33  /**
34   * System class. This class should be considered non-API as it may be removed
35   * if/when its code is incorporated into the JNA project.
36   */
37  public interface SystemB extends com.sun.jna.platform.mac.SystemB, CLibrary {
38  
39      SystemB INSTANCE = Native.load("System", SystemB.class);
40  
41      int PROC_PIDLISTFDS = 1;
42      int PROX_FDTYPE_SOCKET = 2;
43      int PROC_PIDFDSOCKETINFO = 3;
44      int TSI_T_NTIMERS = 4;
45      int SOCKINFO_IN = 1;
46      int SOCKINFO_TCP = 2;
47  
48      int UTX_USERSIZE = 256;
49      int UTX_LINESIZE = 32;
50      int UTX_IDSIZE = 4;
51      int UTX_HOSTSIZE = 256;
52  
53      int AF_INET = 2; // The Internet Protocol version 4 (IPv4) address family.
54      int AF_INET6 = 30; // The Internet Protocol version 6 (IPv6) address family.
55  
56      /**
57       * Mac connection info
58       */
59      @FieldOrder({ "ut_user", "ut_id", "ut_line", "ut_pid", "ut_type", "ut_tv", "ut_host", "ut_pad" })
60      class MacUtmpx extends Structure {
61          public byte[] ut_user = new byte[UTX_USERSIZE]; // login name
62          public byte[] ut_id = new byte[UTX_IDSIZE]; // id
63          public byte[] ut_line = new byte[UTX_LINESIZE]; // tty name
64          public int ut_pid; // process id creating the entry
65          public short ut_type; // type of this entry
66          public Timeval ut_tv; // time entry was created
67          public byte[] ut_host = new byte[UTX_HOSTSIZE]; // host name
68          public byte[] ut_pad = new byte[16]; // reserved for future use
69      }
70  
71      /**
72       * Mac file descriptor info
73       */
74      @FieldOrder({ "proc_fd", "proc_fdtype" })
75      class ProcFdInfo extends Structure {
76          public int proc_fd;
77          public int proc_fdtype;
78      }
79  
80      /**
81       * Mac internet socket info
82       */
83      @FieldOrder({ "insi_fport", "insi_lport", "insi_gencnt", "insi_flags", "insi_flow", "insi_vflag", "insi_ip_ttl",
84              "rfu_1", "insi_faddr", "insi_laddr", "insi_v4", "insi_v6" })
85      class InSockInfo extends Structure {
86          public int insi_fport; /* foreign port */
87          public int insi_lport; /* local port */
88          public long insi_gencnt; /* generation count of this instance */
89          public int insi_flags; /* generic IP/datagram flags */
90          public int insi_flow;
91  
92          public byte insi_vflag; /* ini_IPV4 or ini_IPV6 */
93          public byte insi_ip_ttl; /* time to live proto */
94          public int rfu_1; /* reserved */
95          /* protocol dependent part, v4 only in last element */
96          public int[] insi_faddr = new int[4]; /* foreign host table entry */
97          public int[] insi_laddr = new int[4]; /* local host table entry */
98          public byte insi_v4; /* type of service */
99          public byte[] insi_v6 = new byte[9];
100     }
101 
102     /**
103      * Mac TCP socket info
104      */
105     @FieldOrder({ "tcpsi_ini", "tcpsi_state", "tcpsi_timer", "tcpsi_mss", "tcpsi_flags", "rfu_1", "tcpsi_tp" })
106     class TcpSockInfo extends Structure {
107         public InSockInfo tcpsi_ini;
108         public int tcpsi_state;
109         public int[] tcpsi_timer = new int[TSI_T_NTIMERS];
110         public int tcpsi_mss;
111         public int tcpsi_flags;
112         public int rfu_1; /* reserved */
113         public long tcpsi_tp; /* opaque handle of TCP protocol control block */
114     }
115 
116     /**
117      * Mack IP Socket Info
118      */
119     @FieldOrder({ "soi_stat", "soi_so", "soi_pcb", "soi_type", "soi_protocol", "soi_family", "soi_options",
120             "soi_linger", "soi_state", "soi_qlen", "soi_incqlen", "soi_qlimit", "soi_timeo", "soi_error", "soi_oobmark",
121             "soi_rcv", "soi_snd", "soi_kind", "rfu_1", "soi_proto" })
122     class SocketInfo extends Structure {
123         public long[] soi_stat = new long[17]; // vinfo_stat
124         public long soi_so; /* opaque handle of socket */
125         public long soi_pcb; /* opaque handle of protocol control block */
126         public int soi_type;
127         public int soi_protocol;
128         public int soi_family;
129         public short soi_options;
130         public short soi_linger;
131         public short soi_state;
132         public short soi_qlen;
133         public short soi_incqlen;
134         public short soi_qlimit;
135         public short soi_timeo;
136         public short soi_error;
137         public int soi_oobmark;
138         public int[] soi_rcv = new int[6]; // sockbuf_info
139         public int[] soi_snd = new int[6]; // sockbuf_info
140         public int soi_kind;
141         public int rfu_1; /* reserved */
142         public Pri soi_proto;
143     }
144 
145     /**
146      * Mac file info
147      */
148     @FieldOrder({ "fi_openflags", "fi_status", "fi_offset", "fi_type", "fi_guardflags" })
149     class ProcFileInfo extends Structure {
150         public int fi_openflags;
151         public int fi_status;
152         public long fi_offset;
153         public int fi_type;
154         public int fi_guardflags;
155     }
156 
157     /**
158      * Mac socket info
159      */
160     @FieldOrder({ "pfi", "psi" })
161     class SocketFdInfo extends Structure {
162         public ProcFileInfo pfi;
163         public SocketInfo psi;
164     }
165 
166     /**
167      * Union for TCP or internet socket info
168      */
169     class Pri extends Union {
170         public InSockInfo pri_in;
171         public TcpSockInfo pri_tcp;
172         // max element is 524 bytes
173         public byte[] max_size = new byte[524];
174     }
175 
176     /**
177      * Reads a line from the current file position in the utmp file. It returns a
178      * pointer to a structure containing the fields of the line.
179      * <p>
180      * Not thread safe
181      *
182      * @return a {@link MacUtmpx} on success, and NULL on failure (which includes
183      *         the "record not found" case)
184      */
185     MacUtmpx getutxent();
186 
187     int proc_pidfdinfo(int pid, int fd, int flavor, Structure buffer, int buffersize);
188 }
