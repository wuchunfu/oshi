24  package oshi.jna.platform.unix.solaris;
25  
26  import com.sun.jna.Native; // NOSONAR squid:S1191
27  import com.sun.jna.NativeLong;
28  import com.sun.jna.Structure;
29  import com.sun.jna.Structure.FieldOrder;
30  
31  import oshi.jna.platform.unix.CLibrary;
32  
33  /**
34   * C library. This class should be considered non-API as it may be removed
35   * if/when its code is incorporated into the JNA project.
36   */
37  public interface SolarisLibc extends CLibrary {
38  
39      SolarisLibc INSTANCE = Native.load("c", SolarisLibc.class);
40  
41      int UTX_USERSIZE = 32;
42      int UTX_LINESIZE = 32;
43      int UTX_IDSIZE = 4;
44      int UTX_HOSTSIZE = 257;
45  
46      /**
47       * Connection info
48       */
49      @FieldOrder({ "ut_user", "ut_id", "ut_line", "ut_pid", "ut_type", "ut_tv", "ut_session", "ut_syslen", "ut_host" })
50      class SolarisUtmpx extends Structure {
51          public byte[] ut_user = new byte[UTX_USERSIZE]; // user login name
52          public byte[] ut_id = new byte[UTX_IDSIZE]; // etc/inittab id (usually line #)
53          public byte[] ut_line = new byte[UTX_LINESIZE]; // device name
54          public int ut_pid; // process id
55          public short ut_type; // type of entry
56          public Timeval ut_tv; // time entry was made
57          public int ut_session; // session ID, used for windowing
58          public short ut_syslen; // significant length of ut_host including terminating null
59          public byte[] ut_host = new byte[UTX_HOSTSIZE]; // host name
60      }
61  
62      /**
63       * Part of utmpx structure
64       */
65      @FieldOrder({ "e_termination", "e_exit" })
66      class Exit_status extends Structure {
67          public short e_termination; // Process termination status
68          public short e_exit; // Process exit status
69      }
70  
71      /**
72       * 64-bit timeval required for utmpx structure
73       */
74      @FieldOrder({ "tv_sec", "tv_usec" })
75      class Timeval extends Structure {
76          public NativeLong tv_sec; // seconds
77          public NativeLong tv_usec; // microseconds
78      }
79  
80      /**
81       * Reads a line from the current file position in the utmp file. It returns a
82       * pointer to a structure containing the fields of the line.
83       * <p>
84       * Not thread safe
85       *
86       * @return a {@link SolarisUtmpx} on success, and NULL on failure (which
87       *         includes the "record not found" case)
88       */
89      SolarisUtmpx getutxent();
90  }
