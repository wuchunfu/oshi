27  import com.sun.jna.Pointer;
28  import com.sun.jna.Structure;
29  import com.sun.jna.Structure.FieldOrder;

31  import oshi.jna.platform.unix.CLibrary;
32  
33  /**
34   * C library. This class should be considered non-API as it may be removed
35   * if/when its code is incorporated into the JNA project.
36   */
37  public interface OpenBsdLibc extends CLibrary {
38      OpenBsdLibc INSTANCE = Native.load(null, OpenBsdLibc.class);
39  
40      int CTL_KERN = 1; // "high kernel": proc, limits
41      int CTL_VM = 1; // "high kernel": proc, limits
42      int CTL_HW = 6; // generic cpu/io
43      int CTL_MACHDEP = 7; // machine dependent
44      int CTL_VFS = 10; // VFS sysctl's
45  
46      int KERN_OSTYPE = 1; // string: system version
47      int KERN_OSRELEASE = 2; // string: system release
48      int KERN_OSREV = 3; // int: system revision
49      int KERN_VERSION = 4; // string: compile time info
50      int KERN_MAXVNODES = 5; // int: max vnodes
51      int KERN_MAXPROC = 6; // int: max processes
52      int KERN_ARGMAX = 8; // int: max arguments to exec
53      int KERN_CPTIME = 40; // array: cp_time
54      int KERN_CPTIME2 = 71; // array: cp_time2
55  
56      int VM_UVMEXP = 4; // struct uvmexp
57  
58      int HW_MACHINE = 1; // string: machine class
59      int HW_MODEL = 2; // string: specific machine model
60      int HW_PAGESIZE = 7; // int: software page size
61      int HW_CPUSPEED = 12; // get CPU frequency
62      int HW_NCPUFOUND = 21; // CPU found (includes offline)
63      int HW_SMT = 24; // enable SMT/HT/CMT
64      int HW_NCPUONLINE = 25; // number of cpus being used
65  
66      int VFS_GENERIC = 0; // generic filesystem information
67      int VFS_BCACHESTAT = 3; // struct: buffer cache statistics given as next argument
68  
69      /*
70       * CPU state indices
71       */
72      int CPUSTATES = 5;
73      int CP_USER = 0;
74      int CP_NICE = 1;
75      int CP_SYS = 2;
76      int CP_INTR = 3; // 4 on 6.4 and later
77      int CP_IDLE = 4; // 5 on 6.4 and later
78  
79      int UINT64_SIZE = Native.getNativeSize(long.class);
80      int INT_SIZE = Native.getNativeSize(int.class);
81  
82      /**
83       * OpenBSD Cache stats for memory
84       */
85      @FieldOrder({ "numbufs", "numbufpages", "numdirtypages", "numcleanpages", "pendingwrites", "pendingreads",
86              "numwrites", "numreads", "cachehits", "busymapped", "dmapages", "highpages", "delwribufs", "kvaslots",
87              "kvaslots_avail", "highflips", "highflops", "dmaflips" })
88      class Bcachestats extends Structure {
89          public long numbufs; // number of buffers allocated
90          public long numbufpages; // number of pages in buffer cache
91          public long numdirtypages; // number of dirty free pages
92          public long numcleanpages; // number of clean free pages
93          public long pendingwrites; // number of pending writes
94          public long pendingreads; // number of pending reads
95          public long numwrites; // total writes started
96          public long numreads; // total reads started
97          public long cachehits; // total reads found in cache
98          public long busymapped; // number of busy and mapped buffers
99          public long dmapages; // dma reachable pages in buffer cache
100         public long highpages; // pages above dma region
101         public long delwribufs; // delayed write buffers
102         public long kvaslots; // kva slots total
103         public long kvaslots_avail; // available kva slots
104         public long highflips; // total flips to above DMA
105         public long highflops; // total failed flips to above DMA
106         public long dmaflips; // total flips from high to DMA
107 
108         public Bcachestats(Pointer p) {
109             super(p);
110             read();
111         }
112     }
113 
114     /**
115      * Return type for BSD sysctl kern.boottime
116      */
117     @FieldOrder({ "tv_sec", "tv_usec" })
118     class Timeval extends Structure {
119         public long tv_sec; // seconds
120         public long tv_usec; // microseconds
121     }
122 }
