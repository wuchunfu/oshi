47  /**
48   * Helper class to centralize the boilerplate portions of PDH counter setup and
49   * allow applications to easily add, query, and remove counters.
50   */
51  @ThreadSafe
52  public final class PerfDataUtil {
53  
54      private static final Logger LOG = LoggerFactory.getLogger(PerfDataUtil.class);
55  
56      private static final DWORD_PTR PZERO = new DWORD_PTR(0);
57      private static final DWORDByReference PDH_FMT_RAW = new DWORDByReference(new DWORD(Pdh.PDH_FMT_RAW));
58      private static final Pdh PDH = Pdh.INSTANCE;
59  
60      private static final boolean IS_VISTA_OR_GREATER = VersionHelpers.IsWindowsVistaOrGreater();
61  
62      /**
63       * Encapsulates the three string components of a performance counter
64       */
65      @Immutable
66      public static class PerfCounter {
67          private String object;
68          private String instance;
69          private String counter;
70  
71          public PerfCounter(String objectName, String instanceName, String counterName) {
72              this.object = objectName;
73              this.instance = instanceName;
74              this.counter = counterName;
75          }
76  
77          /**
78           * @return Returns the object.
79           */
80          public String getObject() {
81              return object;
82          }
83  
84          /**
85           * @return Returns the instance.
86           */
87          public String getInstance() {
88              return instance;
89          }
90  
91          /**
92           * @return Returns the counter.
93           */
94          public String getCounter() {
95              return counter;
96          }
97  
98          /**
99           * Returns the path for this counter
100          *
101          * @return A string representing the counter path
102          */
103         public String getCounterPath() {
104             StringBuilder sb = new StringBuilder();
105             sb.append('\\').append(object);
106             if (instance != null) {
107                 sb.append('(').append(instance).append(')');
108             }
109             sb.append('\\').append(counter);
110             return sb.toString();
111         }
112     }
113 
114     private PerfDataUtil() {
115     }
116 
117     /**
118      * Create a Performance Counter
119      *
120      * @param object
121      *            The object/path for the counter
122      * @param instance
123      *            The instance of the counter, or null if no instance
124      * @param counter
125      *            The counter name
126      * @return A PerfCounter object encapsulating the object, instance, and counter
127      */
128     public static PerfCounter createCounter(String object, String instance, String counter) {
129         return new PerfCounter(object, instance, counter);
130     }
131 
132     /**
133      * Update a query and get the timestamp
134      *
135      * @param query
136      *            The query to update all counters in
137      * @return The update timestamp of the first counter in the query
138      */
139     public static long updateQueryTimestamp(WinNT.HANDLEByReference query) {
140         LONGLONGByReference pllTimeStamp = new LONGLONGByReference();
141         int ret = IS_VISTA_OR_GREATER ? PDH.PdhCollectQueryDataWithTime(query.getValue(), pllTimeStamp)
142                 : PDH.PdhCollectQueryData(query.getValue());
143         // Due to race condition, initial update may fail with PDH_NO_DATA.
144         int retries = 0;
145         while (ret == PdhMsg.PDH_NO_DATA && retries++ < 3) {
146             // Exponential fallback.
147             Util.sleep(1 << retries);
148             ret = IS_VISTA_OR_GREATER ? PDH.PdhCollectQueryDataWithTime(query.getValue(), pllTimeStamp)
149                     : PDH.PdhCollectQueryData(query.getValue());
150         }
151         if (ret != WinError.ERROR_SUCCESS) {
152             if (LOG.isWarnEnabled()) {
153                 LOG.warn("Failed to update counter. Error code: {}", String.format(FormatUtil.formatError(ret)));
154             }
155             return 0L;
156         }
157         // Perf Counter timestamp is in local time
158         return IS_VISTA_OR_GREATER ? ParseUtil.filetimeToUtcMs(pllTimeStamp.getValue().longValue(), true)
159                 : System.currentTimeMillis();
160     }

162     /**
163      * Open a pdh query
164      *
165      * @param q
166      *            pointer to the query
167      * @return true if successful
168      */
169     public static boolean openQuery(HANDLEByReference q) {
170         int ret = PDH.PdhOpenQuery(null, PZERO, q);
171         if (ret != WinError.ERROR_SUCCESS) {
172             if (LOG.isErrorEnabled()) {
173                 LOG.error("Failed to open PDH Query. Error code: {}", String.format(FormatUtil.formatError(ret)));
174             }
175             return false;
176         }
177         return true;
178     }
179 
180     /**
181      * Close a pdh query
182      *
183      * @param q
184      *            pointer to the query
185      * @return true if successful
186      */
187     public static boolean closeQuery(HANDLEByReference q) {
188         return WinError.ERROR_SUCCESS == PDH.PdhCloseQuery(q.getValue());
189     }
190 
191     /**
192      * Get value of pdh counter
193      *
194      * @param counter
195      *            The counter to get the value of
196      * @return long value of the counter, or negative value representing an error
197      *         code
198      */
199     public static long queryCounter(WinNT.HANDLEByReference counter) {
200         PDH_RAW_COUNTER counterValue = new PDH_RAW_COUNTER();
201         int ret = PDH.PdhGetRawCounterValue(counter.getValue(), PDH_FMT_RAW, counterValue);
202         if (ret != WinError.ERROR_SUCCESS) {
203             if (LOG.isWarnEnabled()) {
204                 LOG.warn("Failed to get counter. Error code: {}", String.format(FormatUtil.formatError(ret)));
205             }
206             return ret;
207         }
208         return counterValue.FirstValue;
209     }
210 
211     /**
212      * Adds a pdh counter to a query
213      *
214      * @param query
215      *            Pointer to the query to add the counter
216      * @param path
217      *            String name of the PerfMon counter. For Vista+, must be in
218      *            English. Must localize this path for pre-Vista.
219      * @param p
220      *            Pointer to the counter
221      * @return true if successful
222      */
223     public static boolean addCounter(WinNT.HANDLEByReference query, String path, WinNT.HANDLEByReference p) {
224         int ret = IS_VISTA_OR_GREATER ? PDH.PdhAddEnglishCounter(query.getValue(), path, PZERO, p)
225                 : PDH.PdhAddCounter(query.getValue(), path, PZERO, p);
226         if (ret != WinError.ERROR_SUCCESS) {
227             if (LOG.isWarnEnabled()) {
228                 LOG.warn("Failed to add PDH Counter: {}, Error code: {}", path,
229                         String.format(FormatUtil.formatError(ret)));
230             }
231             return false;
232         }
233         return true;
234     }
235 
236     /**
237      * Remove a pdh counter
238      *
239      * @param p
240      *            pointer to the counter
241      * @return true if successful
242      */
243     public static boolean removeCounter(HANDLEByReference p) {
244         return WinError.ERROR_SUCCESS == PDH.PdhRemoveCounter(p.getValue());
245     }
